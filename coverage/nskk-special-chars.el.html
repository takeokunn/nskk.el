66(defconst nskk-special-chars--hiragana-katakana-offset 96
67 &quot;ひらがなとカタカナのUnicodeコードポイントのオフセット。
68ひらがな「あ」(U+3042) とカタカナ「ア」(U+30A2) の差分。&quot;)

70(defun nskk-hiragana-to-katakana (str)
71 &quot;ひらがな文字列 STR をカタカナに変換する。
72
73ひらがな以外の文字はそのまま返される。
74
75例:
76 (nskk-hiragana-to-katakana \&quot;あいうえお\&quot;) ;; =&gt; \&quot;アイウエオ\&quot;
77 (nskk-hiragana-to-katakana \&quot;こんにちは\&quot;) ;; =&gt; \&quot;コンニチハ\&quot;
78 (nskk-hiragana-to-katakana \&quot;aBc\&quot;) ;; =&gt; \&quot;aBc\&quot; (変換なし)&quot;
79 (let ((result &quot;&quot;))
80 (dotimes (i (length str))
81 (let ((char (aref str i)))
82 (if (and (&gt;= char ?ぁ) (&lt;= char ?ん))
83 ;; ひらがなの場合、カタカナに変換
84 (setq result (concat result
85 (char-to-string
86 (+ char nskk-special-chars--hiragana-katakana-offset))))
87 ;; それ以外はそのまま
88 (setq result (concat result (char-to-string char))))))
89 result))

135(defun nskk-capitalize-string (str)
136 &quot;STR の最初の文字を大文字に変換する。
137
138日本語文字列の場合は何もしない。
139
140例:
141 (nskk-capitalize-string \&quot;hello\&quot;) ;; =&gt; \&quot;Hello\&quot;
142 (nskk-capitalize-string \&quot;こんにちは\&quot;) ;; =&gt; \&quot;こんにちは\&quot;&quot;
143 (if (string-empty-p str)
144 str
145 (let ((first-char (aref str 0)))
146 (if (and (&gt;= first-char ?a) (&lt;= first-char ?z))
147 (concat (upcase (substring str 0 1))
148 (substring str 1))
149 str))))

192(defun nskk-katakana-to-halfwidth (str)
193 &quot;全角カタカナ文字列 STR を半角カタカナ（JIS X 0201）に変換する。
194
195濁音・半濁音は2文字（基底文字+濁点/半濁点）に分解される。
196
197例:
198 (nskk-katakana-to-halfwidth \&quot;アイウエオ\&quot;) ;; =&gt; \&quot;ｱｲｳｴｵ\&quot;
199 (nskk-katakana-to-halfwidth \&quot;ガギグ\&quot;) ;; =&gt; \&quot;ｶﾞｷﾞｸﾞ\&quot;&quot;
200 (let ((result &quot;&quot;))
201 (dotimes (i (length str))
202 (let* ((char (char-to-string (aref str i)))
203 (halfwidth (cdr (assoc char nskk-special-chars--halfwidth-katakana-map))))
204 (if halfwidth
205 (setq result (concat result halfwidth))
206 ;; マップにない文字はそのまま
207 (setq result (concat result char)))))
208 result))

225(defun nskk-normalize-input (input)
226 &quot;INPUT を正規化する。
227
228全角英数字を半角に変換するなどの前処理を行う。
229
230例:
231 (nskk-normalize-input \&quot;ＡＢＣ\&quot;) ;; =&gt; \&quot;ABC\&quot;&quot;
232 ;; 全角英数字を半角に変換
233 (let ((result &quot;&quot;))
234 (dotimes (i (length input))
235 (let ((char (aref input i)))
236 (cond
237 ;; 全角英字 (A-Z)
238 ((and (&gt;= char ?Ａ) (&lt;= char ?Ｚ))
239 (setq result (concat result (char-to-string (- char (- ?Ａ ?A))))))
240 ;; 全角英字 (a-z)
241 ((and (&gt;= char ?ａ) (&lt;= char ?ｚ))
242 (setq result (concat result (char-to-string (- char (- ?ａ ?a))))))
243 ;; 全角数字 (0-9)
244 ((and (&gt;= char ?０) (&lt;= char ?９))
245 (setq result (concat result (char-to-string (- char (- ?０ ?0))))))
246 ;; その他はそのまま
247 (t
248 (setq result (concat result (char-to-string char)))))))
249 result))

268(defun nskk-string-type (str)
269 &quot;STR の文字種を判定する。
270
271返り値:
272 - `hiragana' : すべてひらがな
273 - `katakana' : すべてカタカナ
274 - `kanji' : 漢字を含む
275 - `ascii' : すべてASCII
276 - `mixed' : 混在
277
278例:
279 (nskk-string-type \&quot;あいうえお\&quot;) ;; =&gt; hiragana
280 (nskk-string-type \&quot;アイウエオ\&quot;) ;; =&gt; katakana
281 (nskk-string-type \&quot;漢字\&quot;) ;; =&gt; kanji
282 (nskk-string-type \&quot;abc\&quot;) ;; =&gt; ascii
283 (nskk-string-type \&quot;あaい\&quot;) ;; =&gt; mixed&quot;
284 (if (string-empty-p str)
285 'empty
286 (let ((has-hiragana nil)
287 (has-katakana nil)
288 (has-kanji nil)
289 (has-ascii nil))
290 (dotimes (i (length str))
291 (let ((char (aref str i)))
292 (cond
293 ((nskk-is-hiragana-char char) (setq has-hiragana t))
294 ((nskk-is-katakana-char char) (setq has-katakana t))
295 ((and (&gt;= char ?一) (&lt;= char ?龿)) (setq has-kanji t))
296 ((&lt; char 128) (setq has-ascii t)))))
297 (cond
298 (has-kanji 'kanji)
299 ((and has-hiragana (not has-katakana) (not has-ascii)) 'hiragana)
300 ((and has-katakana (not has-hiragana) (not has-ascii)) 'katakana)
301 ((and has-ascii (not has-hiragana) (not has-katakana) (not has-kanji)) 'ascii)
302 (t 'mixed)))))
