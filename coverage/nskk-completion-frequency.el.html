124;;;###autoload
125(defun nskk-completion-frequency-search (index query &rest args)
126 &quot;頻度ベース補完検索を実行する。
127
128引数:
129 INDEX - nskk-dict-index構造体
130 QUERY - 検索クエリ（部分一致または前方一致）
131
132キーワード引数:
133 :limit - 最大結果数（デフォルト: nskk-completion-frequency-default-limit）
134 :okuri-type - 送り仮名タイプ（'okuri-ari/'okuri-nasi/nil）
135 :method - 頻度計算方法（デフォルト: nskk-completion-frequency-method）
136 :decay - 時間減衰を有効化（デフォルト: nskk-completion-frequency-decay-enabled）
137 :match-type - マッチタイプ（'prefix/'partial、デフォルト: 'prefix）
138
139戻り値:
140 nskk-completion-frequency-result構造体のリスト（頻度順）&quot;
141 (unless (stringp query)
142 (error &quot;Query must be a string: %s&quot; query))
143
144 (let* ((limit (or (plist-get args :limit)
145 nskk-completion-frequency-default-limit))
146 (okuri-type (plist-get args :okuri-type))
147 (method (or (plist-get args :method)
148 nskk-completion-frequency-method))
149 (decay (if (plist-member args :decay)
150 (plist-get args :decay)
151 nskk-completion-frequency-decay-enabled))
152 (match-type (or (plist-get args :match-type) 'prefix))
153 (results nil)
154 (tables (cond
155 ((eq okuri-type 'okuri-ari)
156 (list (nskk-dict-index-okuri-ari-table index)))
157 ((eq okuri-type 'okuri-nasi)
158 (list (nskk-dict-index-okuri-nasi-table index)))
159 (t
160 (list (nskk-dict-index-okuri-nasi-table index)
161 (nskk-dict-index-okuri-ari-table index))))))
162
163 ;; 各テーブルから候補を収集
164 (dolist (table tables)
165 (maphash
166 (lambda (midashi entry)
167 (when (nskk-completion-frequency--match-p query midashi match-type)
168 (let* ((freq (nskk-dict-entry-frequency entry))
169 (last-used (nskk-dict-entry-last-used entry))
170 (adjusted (nskk-completion-frequency--calculate-adjusted-frequency
171 freq last-used method decay)))
172 (push (nskk-completion-frequency-result--create
173 :midashi midashi
174 :entry entry
175 :frequency freq
176 :adjusted-freq adjusted
177 :last-used last-used)
178 results))))
179 table))
180
181 ;; 調整後頻度でソート
182 (setq results (sort results
183 (lambda (a b)
184 (&gt; (nskk-completion-frequency-result-adjusted-freq a)
185 (nskk-completion-frequency-result-adjusted-freq b)))))
186
187 ;; ランクを設定
188 (let ((rank 1))
189 (dolist (result results)
190 (setf (nskk-completion-frequency-result-rank result) rank)
191 (setq rank (1+ rank))))
192
193 ;; limit適用
194 (when (and limit (&gt; (length results) limit))
195 (setq results (seq-take results limit)))
196
197 results))

201(defun nskk-completion-frequency--match-p (query midashi match-type)
202 &quot;クエリと見出し語がマッチするか判定する（内部関数）。
203
204引数:
205 QUERY - 検索クエリ
206 MIDASHI - 見出し語
207 MATCH-TYPE - マッチタイプ（'prefix/'partial）
208
209戻り値:
210 マッチする場合t、しない場合nil&quot;
211 (pcase match-type
212 ('prefix
213 (string-prefix-p query midashi))
214 ('partial
215 (string-match-p (regexp-quote query) midashi))
216 (_
217 (string-prefix-p query midashi))))

221(defun nskk-completion-frequency--calculate-adjusted-frequency (freq last-used method decay)
222 &quot;調整後頻度を計算する（内部関数）。
223
224引数:
225 FREQ - 生の頻度
226 LAST-USED - 最終使用時刻（float-time形式、nilの場合は未使用）
227 METHOD - 頻度計算方法
228 DECAY - 時間減衰を有効化するか
229
230戻り値:
231 調整後頻度（浮動小数点数）&quot;
232 (pcase method
233 ('simple
234 ;; 単純な使用回数
235 (float freq))
236
237 ('weighted
238 ;; 時間減衰を考慮した重み付け
239 (if (and decay last-used)
240 (nskk-completion-frequency--apply-time-decay freq last-used)
241 (float freq)))
242
243 ('lfu
244 ;; LFU: 頻度のみを考慮（時間は無視）
245 (float freq))
246
247 ('lru
248 ;; LRU風: 最終使用時刻を重視
249 (if last-used
250 (let ((elapsed (- (float-time) last-used)))
251 ;; 最近使用したほど高スコア
252 (+ (* freq 0.3)
253 (* (/ 1.0 (+ 1.0 (/ elapsed 86400.0))) 100.0)))
254 (* freq 0.3)))
255
256 (_
257 (float freq))))

259(defun nskk-completion-frequency--apply-time-decay (freq last-used)
260 &quot;時間減衰を適用する（内部関数）。
261
262引数:
263 FREQ - 生の頻度
264 LAST-USED - 最終使用時刻（float-time形式）
265
266処理:
267 経過時間に基づいて頻度を減衰させる
268 adjusted = freq * (decay-factor ^ (elapsed / decay-unit))
269
270戻り値:
271 減衰後頻度（浮動小数点数）&quot;
272 (if (null last-used)
273 (float freq)
274 (let* ((now (float-time))
275 (elapsed (- now last-used))
276 (units (/ elapsed nskk-completion-frequency-decay-unit))
277 (decay-multiplier (expt nskk-completion-frequency-decay-factor units)))
278 (* freq decay-multiplier))))

283(defun nskk-completion-frequency-update-stats (index midashi &optional okuri-type)
284 &quot;見出し語の使用統計を更新する。
285
286引数:
287 INDEX - nskk-dict-index構造体
288 MIDASHI - 見出し語
289 OKURI-TYPE - 送り仮名タイプ（'okuri-ari/'okuri-nasi、省略時は自動判定）
290
291処理:
292 1. エントリを検索
293 2. 頻度をインクリメント
294 3. 最終使用時刻を更新
295
296戻り値:
297 更新成功時t、エントリが見つからない場合nil&quot;
298 (let* ((okuri (or okuri-type
299 (nskk-completion-frequency--infer-okuri-type midashi)))
300 (table (if (eq okuri 'okuri-ari)
301 (nskk-dict-index-okuri-ari-table index)
302 (nskk-dict-index-okuri-nasi-table index)))
303 (entry (gethash midashi table)))
304 (when entry
305 ;; 頻度をインクリメント
306 (cl-incf (nskk-dict-entry-frequency entry))
307 ;; 最終使用時刻を更新
308 (setf (nskk-dict-entry-last-used entry) (float-time))
309 t)))

311(defun nskk-completion-frequency--infer-okuri-type (midashi)
312 &quot;見出し語から送り仮名タイプを推測する（内部関数）。
313
314引数:
315 MIDASHI - 見出し語
316
317戻り値:
318 'okuri-ari または 'okuri-nasi&quot;
319 (if (and (&gt; (length midashi) 0)
320 (let ((last-char (aref midashi (1- (length midashi)))))
321 (and (&gt;= last-char ?a) (&lt;= last-char ?z))))
322 'okuri-ari
323 'okuri-nasi))

328(defun nskk-completion-frequency-top-entries (index n &optional okuri-type)
329 &quot;頻度の高いエントリをN件取得する。
330
331引数:
332 INDEX - nskk-dict-index構造体
333 N - 取得件数
334 OKURI-TYPE - 送り仮名タイプ（'okuri-ari/'okuri-nasi/nil）
335
336戻り値:
337 ((midashi . frequency) ...) のリスト&quot;
338 (let ((entries nil)
339 (tables (cond
340 ((eq okuri-type 'okuri-ari)
341 (list (nskk-dict-index-okuri-ari-table index)))
342 ((eq okuri-type 'okuri-nasi)
343 (list (nskk-dict-index-okuri-nasi-table index)))
344 (t
345 (list (nskk-dict-index-okuri-nasi-table index)
346 (nskk-dict-index-okuri-ari-table index))))))
347
348 ;; 全エントリを収集
349 (dolist (table tables)
350 (maphash (lambda (midashi entry)
351 (push (cons midashi (nskk-dict-entry-frequency entry)) entries))
352 table))
353
354 ;; 頻度でソート
355 (setq entries (sort entries (lambda (a b) (&gt; (cdr a) (cdr b)))))
356
357 ;; 上位N件を返す
358 (seq-take entries n)))

361(defun nskk-completion-frequency-statistics (index &optional okuri-type)
362 &quot;頻度統計情報を取得する。
363
364引数:
365 INDEX - nskk-dict-index構造体
366 OKURI-TYPE - 送り仮名タイプ（'okuri-ari/'okuri-nasi/nil）
367
368戻り値:
369 plist形式の統計情報
370 :total-entries - 総エントリ数
371 :used-entries - 使用されたエントリ数
372 :total-frequency - 総使用頻度
373 :avg-frequency - 平均頻度
374 :max-frequency - 最大頻度
375 :median-frequency - 中央値頻度&quot;
376 (let ((frequencies nil)
377 (used-count 0)
378 (total-freq 0)
379 (tables (cond
380 ((eq okuri-type 'okuri-ari)
381 (list (nskk-dict-index-okuri-ari-table index)))
382 ((eq okuri-type 'okuri-nasi)
383 (list (nskk-dict-index-okuri-nasi-table index)))
384 (t
385 (list (nskk-dict-index-okuri-nasi-table index)
386 (nskk-dict-index-okuri-ari-table index))))))
387
388 ;; 頻度を収集
389 (dolist (table tables)
390 (maphash (lambda (_ entry)
391 (let ((freq (nskk-dict-entry-frequency entry)))
392 (push freq frequencies)
393 (setq total-freq (+ total-freq freq))
394 (when (&gt; freq 0)
395 (setq used-count (1+ used-count)))))
396 table))
397
398 (let* ((total-entries (length frequencies))
399 (avg-freq (if (&gt; total-entries 0)
400 (/ (float total-freq) total-entries)
401 0.0))
402 (sorted-freqs (sort frequencies #'&gt;))
403 (max-freq (if sorted-freqs (car sorted-freqs) 0))
404 (median-freq (if sorted-freqs
405 (nth (/ (length sorted-freqs) 2) sorted-freqs)
406 0)))
407
408 (list :total-entries total-entries
409 :used-entries used-count
410 :total-frequency total-freq
411 :avg-frequency avg-freq
412 :max-frequency max-freq
413 :median-frequency median-freq))))
