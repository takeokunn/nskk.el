27;; このファイルはSKK辞書の統合検索システムを実装します。
28;;
29;; サポートする検索タイプ:
30;; - 完全一致検索 (exact): O(1) 平均
31;; - 前方一致検索 (prefix): O(k + n) - トライ木を活用
32;; - 部分一致検索 (partial): O(n)
33;; - ファジー検索 (fuzzy): O(n * m) - Levenshtein距離
34;;
35;; 特徴:
36;; - トライ木を活用した高速前方一致検索
37;; - カスタマイズ可能なソート順
38;; - 送り仮名タイプの柔軟な指定
39;; - パフォーマンス要件を満たす実装
40;;
41;; パフォーマンス目標:
42;; - 完全一致: &lt; 0.1ms
43;; - 前方一致（100件）: &lt; 1ms
44;; - 部分一致（1000件）: &lt; 50ms
45;; - ファジー検索（1000件）: &lt; 100ms
46;;
47;; 使用例:
48;;
49;; (require 'nskk-search)
50;;
51;; ;; 完全一致検索
52;; (nskk-search index &quot;かんじ&quot; 'exact)
53;; ;; =&gt; nskk-dict-entry
54;;
55;; ;; 前方一致検索
56;; (nskk-search index &quot;かん&quot; 'prefix nil 10)
57;; ;; =&gt; ((&quot;かん&quot; . entry1) (&quot;かんじ&quot; . entry2) ...)
58;;
59;; ;; ファジー検索
60;; (nskk-search index &quot;かんじ&quot; 'fuzzy nil 5)
61;; ;; =&gt; ((&quot;かんじ&quot; . entry1 . 0) (&quot;かんき&quot; . entry2 . 1) ...)
62

116(defun nskk-search (index query &optional search-type okuri-type limit)
117 &quot;辞書INDEXからQUERYを検索する。
118
119引数:
120 INDEX - nskk-dict-index構造体
121 QUERY - 検索クエリ文字列
122 SEARCH-TYPE - 検索タイプ（'exact/'prefix/'partial/'fuzzy、デフォルト'exact）
123 OKURI-TYPE - 送り仮名タイプ（'okuri-ari/'okuri-nasi/nil=両方、デフォルトnil）
124 LIMIT - 結果の最大数（nilの場合無制限）
125
126戻り値:
127 検索タイプに応じた結果
128 - exact: nskk-dict-entry または nil
129 - prefix/partial/fuzzy: ((key . entry) ...) のリスト
130
131エラー:
132 - クエリが空文字列の場合: nskk-dict-search-invalid-query
133 - インデックスが無効の場合: nskk-dict-search-invalid-index&quot;
134 ;; 入力検証
135 (unless (nskk-dict-index-p index)
136 (signal 'nskk-dict-search-invalid-index (list index)))
137 (unless (and (stringp query) (&gt; (length query) 0))
138 (signal 'nskk-dict-search-invalid-query (list query)))
139
140 ;; 検索タイプのデフォルト値
141 (setq search-type (or search-type 'exact))
142
143 ;; 検索タイプに応じてディスパッチ
144 (pcase search-type
145 ('exact
146 (nskk-search-exact index query okuri-type))
147 ('prefix
148 (nskk-search-prefix index query okuri-type limit))
149 ('partial
150 (nskk-search-partial index query okuri-type limit))
151 ('fuzzy
152 (nskk-search-fuzzy index query okuri-type limit))
153 (_
154 (signal 'nskk-dict-search-invalid-query
155 (list (format &quot;Unknown search type: %s&quot; search-type))))))

159(defun nskk-search-exact (index query okuri-type)
160 &quot;完全一致検索を実行する。
161
162引数:
163 INDEX - nskk-dict-index構造体
164 QUERY - 検索クエリ文字列
165 OKURI-TYPE - 送り仮名タイプ（'okuri-ari/'okuri-nasi/nil）
166
167処理:
168 1. 送り仮名タイプを判定（指定がない場合）
169 2. 対応するハッシュテーブルから検索
170 3. エントリを返す
171
172計算量: O(1) 平均
173
174戻り値: nskk-dict-entry または nil&quot;
175 (cond
176 ((eq okuri-type 'okuri-ari)
177 (gethash query (nskk-dict-index-okuri-ari-table index)))
178 ((eq okuri-type 'okuri-nasi)
179 (gethash query (nskk-dict-index-okuri-nasi-table index)))
180 (t
181 ;; 送り仮名タイプが指定されていない場合、両方を検索
182 ;; まず送り仮名なしを優先
183 (or (gethash query (nskk-dict-index-okuri-nasi-table index))
184 (gethash query (nskk-dict-index-okuri-ari-table index))))))

188(defun nskk-search-prefix (index query okuri-type limit)
189 &quot;前方一致検索を実行する。
190
191引数:
192 INDEX - nskk-dict-index構造体
193 QUERY - 検索クエリ文字列
194 OKURI-TYPE - 送り仮名タイプ（'okuri-ari/'okuri-nasi/nil）
195 LIMIT - 結果の最大数（nilの場合無制限）
196
197処理:
198 1. トライ木を使用した高速検索
199 2. 送り仮名タイプに応じたトライ木を選択
200 3. nskk-trie-prefix-searchを呼び出す
201 4. 結果をソート（頻度順、五十音順など）
202
203計算量: O(k + n) (k=prefix長, n=結果数)
204
205戻り値: ((key . entry) ...) のリスト&quot;
206 (let ((results nil)
207 (tries (cond
208 ((eq okuri-type 'okuri-ari)
209 (list (nskk-dict-index-trie-ari index)))
210 ((eq okuri-type 'okuri-nasi)
211 (list (nskk-dict-index-trie-nasi index)))
212 (t
213 ;; 両方から検索（送り仮名なしを優先）
214 (list (nskk-dict-index-trie-nasi index)
215 (nskk-dict-index-trie-ari index))))))
216
217 ;; 各トライ木から検索
218 (dolist (trie tries)
219 (when trie
220 (let ((trie-results (nskk-trie-prefix-search trie query limit)))
221 (setq results (append results trie-results)))))
222
223 ;; 重複を除去（同じキーが送り仮名ありとなしに存在する場合）
224 (setq results (nskk-search--remove-duplicates results))
225
226 ;; limit適用（複数トライ木から取得した場合）
227 (when (and limit (&gt; (length results) limit))
228 (setq results (seq-take results limit)))
229
230 ;; ソート
231 (nskk-search--sort-results results)))
232
233(defun nskk-search--remove-duplicates (results)
234 &quot;検索結果から重複を除去する（内部関数）。
235
236引数:
237 RESULTS - ((key . entry) ...) のリスト
238
239戻り値:
240 重複除去後のリスト（最初に出現したものを保持）&quot;
241 (let ((seen (make-hash-table :test 'equal))
242 (unique nil))
243 (dolist (result results)
244 (let ((key (car result)))
245 (unless (gethash key seen)
246 (puthash key t seen)
247 (push result unique))))
248 (nreverse unique)))

252(defun nskk-search-partial (index query okuri-type limit)
253 &quot;部分一致検索を実行する。
254
255引数:
256 INDEX - nskk-dict-index構造体
257 QUERY - 検索クエリ文字列
258 OKURI-TYPE - 送り仮名タイプ（'okuri-ari/'okuri-nasi/nil）
259 LIMIT - 結果の最大数（nilの場合無制限）
260
261処理:
262 1. 全エントリを走査
263 2. 見出し語にクエリが含まれるか判定
264 3. マッチしたエントリを収集
265 4. 頻度でソート
266
267計算量: O(n) (n=エントリ数)
268
269戻り値: ((key . entry) ...) のリスト
270
271注意: 大規模辞書では遅い可能性があるため、limitの使用を推奨&quot;
272 (let ((results nil)
273 (count 0)
274 (tables (cond
275 ((eq okuri-type 'okuri-ari)
276 (list (nskk-dict-index-okuri-ari-table index)))
277 ((eq okuri-type 'okuri-nasi)
278 (list (nskk-dict-index-okuri-nasi-table index)))
279 (t
280 (list (nskk-dict-index-okuri-nasi-table index)
281 (nskk-dict-index-okuri-ari-table index))))))
282
283 ;; 各テーブルを走査
284 (catch 'limit-reached
285 (dolist (table tables)
286 (maphash (lambda (key entry)
287 (when (string-match-p (regexp-quote query) key)
288 (push (cons key entry) results)
289 (cl-incf count)
290 (when (and limit (&gt;= count limit))
291 (throw 'limit-reached nil))))
292 table)))
293
294 ;; ソート
295 (nskk-search--sort-results results)))

299(defun nskk-search-fuzzy (index query okuri-type limit)
300 &quot;ファジー検索を実行する（基礎実装）。
301
302引数:
303 INDEX - nskk-dict-index構造体
304 QUERY - 検索クエリ文字列
305 OKURI-TYPE - 送り仮名タイプ（'okuri-ari/'okuri-nasi/nil）
306 LIMIT - 結果の最大数（nilの場合無制限）
307
308処理:
309 1. Levenshtein距離を計算
310 2. 距離が閾値以下のエントリを収集
311 3. 距離でソート
312
313計算量: O(n * m) (n=エントリ数, m=クエリ長)
314
315戻り値: ((key . entry . distance) ...) のリスト&quot;
316 (let ((results nil)
317 (tables (cond
318 ((eq okuri-type 'okuri-ari)
319 (list (nskk-dict-index-okuri-ari-table index)))
320 ((eq okuri-type 'okuri-nasi)
321 (list (nskk-dict-index-okuri-nasi-table index)))
322 (t
323 (list (nskk-dict-index-okuri-nasi-table index)
324 (nskk-dict-index-okuri-ari-table index))))))
325
326 ;; 各テーブルを走査し、Levenshtein距離を計算
327 (dolist (table tables)
328 (maphash (lambda (key entry)
329 (let ((distance (nskk-search--levenshtein-distance query key)))
330 (when (&lt;= distance nskk-search-fuzzy-threshold)
331 (push (cons key (cons entry distance)) results))))
332 table))
333
334 ;; 距離でソート（昇順）
335 (setq results (sort results
336 (lambda (a b)
337 (&lt; (cddr a) (cddr b)))))
338
339 ;; limit適用
340 (when (and limit (&gt; (length results) limit))
341 (setq results (seq-take results limit)))
342
343 results))

345(defun nskk-search--levenshtein-distance (s1 s2)
346 &quot;2つの文字列のLevenshtein距離を計算する（内部関数）。
347
348引数:
349 S1 - 文字列1
350 S2 - 文字列2
351
352アルゴリズム: 動的計画法
353計算量: O(|s1| * |s2|)
354
355戻り値: Levenshtein距離（整数）&quot;
356 (let* ((len1 (length s1))
357 (len2 (length s2))
358 ;; 動的計画法のテーブル
359 (dp (make-vector (1+ len1) nil)))
360
361 ;; テーブルを初期化
362 (dotimes (i (1+ len1))
363 (aset dp i (make-vector (1+ len2) 0)))
364
365 ;; 初期値設定
366 (dotimes (i (1+ len1))
367 (aset (aref dp i) 0 i))
368 (dotimes (j (1+ len2))
369 (aset (aref dp 0) j j))
370
371 ;; 動的計画法で距離を計算
372 (dotimes (i len1)
373 (dotimes (j len2)
374 (let ((cost (if (char-equal (aref s1 i) (aref s2 j)) 0 1)))
375 (aset (aref dp (1+ i)) (1+ j)
376 (min
377 ;; 挿入
378 (1+ (aref (aref dp i) (1+ j)))
379 ;; 削除
380 (1+ (aref (aref dp (1+ i)) j))
381 ;; 置換
382 (+ (aref (aref dp i) j) cost))))))
383
384 ;; 結果を返す
385 (aref (aref dp len1) len2)))

407(defun nskk-search-sort-by-frequency (results)
408 &quot;検索結果を使用頻度でソートする。
409
410引数:
411 RESULTS - ((key . entry) ...) のリスト
412
413処理:
414 エントリのfrequencyフィールドで降順ソート
415
416戻り値: ソート済みリスト&quot;
417 (sort results
418 (lambda (a b)
419 (let ((freq-a (nskk-dict-entry-frequency (cdr a)))
420 (freq-b (nskk-dict-entry-frequency (cdr b))))
421 (&gt; freq-a freq-b)))))
