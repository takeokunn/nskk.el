135;;;###autoload
136(defun nskk-custom-annotation-add (midashi candidate annotation)
137 &quot;MIDASHI の CANDIDATE に対して ANNOTATION を追加する。&quot;
138 (interactive
139 (list (read-string &quot;見出し語: &quot;)
140 (read-string &quot;候補: &quot;)
141 (read-string &quot;注釈: &quot;)))
142
143 (run-hook-with-args 'nskk-custom-annotation-before-add-hook
144 midashi candidate annotation)
145
146 (let ((key (cons midashi candidate)))
147 (puthash key annotation nskk-custom-annotation--table)
148 (setq nskk-custom-annotation--modified t))
149
150 (run-hook-with-args 'nskk-custom-annotation-after-add-hook
151 midashi candidate annotation)
152
153 (when nskk-custom-annotation-auto-save
154 (nskk-custom-annotation-save))
155
156 (message &quot;Added annotation for %s/%s: %s&quot; midashi candidate annotation))
157
158;;;###autoload
159(defun nskk-custom-annotation-get (midashi candidate)
160 &quot;MIDASHI の CANDIDATE に対する注釈を取得する。
161注釈がない場合は nil を返す。&quot;
162 (let ((key (cons midashi candidate)))
163 (gethash key nskk-custom-annotation--table)))

165;;;###autoload
166(defun nskk-custom-annotation-edit (midashi candidate new-annotation)
167 &quot;MIDASHI の CANDIDATE の注釈を NEW-ANNOTATION に変更する。&quot;
168 (interactive
169 (let* ((midashi (read-string &quot;見出し語: &quot;))
170 (candidate (read-string &quot;候補: &quot;))
171 (current (nskk-custom-annotation-get midashi candidate))
172 (new (read-string &quot;新しい注釈: &quot; current)))
173 (list midashi candidate new)))
174
175 (if (nskk-custom-annotation-get midashi candidate)
176 (progn
177 (nskk-custom-annotation-add midashi candidate new-annotation)
178 (message &quot;Updated annotation for %s/%s&quot; midashi candidate))
179 (message &quot;No annotation found for %s/%s&quot; midashi candidate)))

181;;;###autoload
182(defun nskk-custom-annotation-remove (midashi candidate)
183 &quot;MIDASHI の CANDIDATE から注釈を削除する。&quot;
184 (interactive
185 (list (read-string &quot;見出し語: &quot;)
186 (read-string &quot;候補: &quot;)))
187
188 (run-hook-with-args 'nskk-custom-annotation-before-remove-hook
189 midashi candidate)
190
191 (let ((key (cons midashi candidate)))
192 (when (gethash key nskk-custom-annotation--table)
193 (remhash key nskk-custom-annotation--table)
194 (setq nskk-custom-annotation--modified t)
195 (message &quot;Removed annotation for %s/%s&quot; midashi candidate)
196
197 (run-hook-with-args 'nskk-custom-annotation-after-remove-hook
198 midashi candidate)
199
200 (when nskk-custom-annotation-auto-save
201 (nskk-custom-annotation-save))
202 t)))

213(defun nskk-custom-annotation-add-from-template (midashi candidate template-name text)
214 &quot;MIDASHI の CANDIDATE に対して、TEMPLATE-NAME を使用して注釈を追加する。&quot;
215 (interactive
216 (list (read-string &quot;見出し語: &quot;)
217 (read-string &quot;候補: &quot;)
218 (intern (completing-read &quot;テンプレート: &quot;
219 (mapcar #'car nskk-custom-annotation-templates)))
220 (read-string &quot;内容: &quot;)))
221
222 (when nskk-custom-annotation-enable-templates
223 (if-let ((template (alist-get template-name nskk-custom-annotation-templates)))
224 (let ((annotation (format template text)))

226 (error &quot;Template not found: %s&quot; template-name))))

240;;;###autoload
241(defun nskk-custom-annotation-generate (midashi candidate)
242 &quot;MIDASHI の CANDIDATE に対して注釈を自動生成する。
243`nskk-custom-annotation-generate-hooks' に登録された関数を順次実行し、
244最初に非nilを返した注釈を使用する。&quot;
245 (interactive
246 (list (read-string &quot;見出し語: &quot;)
247 (read-string &quot;候補: &quot;)))
248
249 (let ((annotation nil))
250 (run-hook-with-args-until-success
251 'nskk-custom-annotation-generate-hooks
252 midashi candidate)
253
254 (when annotation
255 (nskk-custom-annotation-add midashi candidate annotation)
256 annotation)))

260;;;###autoload
261(defun nskk-custom-annotation-save (&optional file)
262 &quot;カスタム注釈を FILE に保存する。
263FILE が指定されない場合は `nskk-custom-annotation-file' を使用する。&quot;
264 (interactive)
265 (let ((file (or file nskk-custom-annotation-file)))
266 ;; ディレクトリを作成
267 (let ((dir (file-name-directory file)))
268 (unless (file-directory-p dir)
269 (make-directory dir t)))
270
271 ;; ファイルに書き込み
272 (with-temp-buffer
273 (insert &quot;;; -*- mode: emacs-lisp; coding: utf-8 -*-\n&quot;)
274 (insert &quot;;; NSKK Custom Annotations\n&quot;)
275 (insert (format &quot;;; Saved: %s\n\n&quot;
276 (format-time-string &quot;%Y-%m-%d %H:%M:%S&quot;)))
277
278 (insert &quot;(\n&quot;)
279 (maphash (lambda (key value)
280 (let ((midashi (car key))
281 (candidate (cdr key)))
282 (insert (format &quot; ((%S . %S) . %S)\n&quot;
283 midashi candidate value))))
284 nskk-custom-annotation--table)
285 (insert &quot;)\n&quot;)
286
287 (write-region (point-min) (point-max) file nil 'silent))
288
289 (setq nskk-custom-annotation--modified nil)
290 (message &quot;Saved %d annotations to %s&quot;
291 (hash-table-count nskk-custom-annotation--table)
292 file)))

294;;;###autoload
295(defun nskk-custom-annotation-load (&optional file)
296 &quot;FILE からカスタム注釈を読み込む。
297FILE が指定されない場合は `nskk-custom-annotation-file' を使用する。&quot;
298 (interactive)
299 (let ((file (or file nskk-custom-annotation-file)))
300 (when (file-exists-p file)
301 (let ((data (with-temp-buffer
302 (insert-file-contents file)
303 (goto-char (point-min))
304 ;; コメント行をスキップ
305 (while (looking-at &quot;^;;&quot;)
306 (forward-line 1))
307 (read (current-buffer)))))
308
309 ;; ハッシュテーブルをクリア
310 (clrhash nskk-custom-annotation--table)
311
312 ;; データを読み込み
313 (dolist (entry data)
314 (let* ((key (car entry))
315 (value (cdr entry))
316 (midashi (car key))
317 (candidate (cdr key)))
318 (puthash key value nskk-custom-annotation--table)))
319
320 (setq nskk-custom-annotation--modified nil)
321 (message &quot;Loaded %d annotations from %s&quot;
322 (hash-table-count nskk-custom-annotation--table)
323 file)))))

342;;;###autoload
343(defun nskk-custom-annotation-export (file)
344 &quot;カスタム注釈を FILE にエクスポートする（SKK辞書形式）。&quot;
345 (interactive &quot;FExport to: &quot;)
346 (with-temp-buffer
347 (insert &quot;;; -*- coding: utf-8 -*-\n&quot;)
348 (insert &quot;;; NSKK Custom Annotations (SKK format)\n\n&quot;)
349
350 ;; 見出し語ごとにグループ化
351 (let ((grouped (make-hash-table :test 'equal)))
352 (maphash (lambda (key value)
353 (let ((midashi (car key))
354 (candidate (cdr key)))
355 (push (cons candidate value)
356 (gethash midashi grouped))))
357 nskk-custom-annotation--table)
358
359 ;; SKK形式で出力
360 (maphash (lambda (midashi candidates)
361 (insert midashi &quot; /&quot;)
362 (dolist (cand (nreverse candidates))
363 (insert (car cand))
364 (when (cdr cand)
365 (insert &quot;;&quot; (cdr cand)))
366 (insert &quot;/&quot;))
367 (insert &quot;\n&quot;))
368 grouped))
369
370 (write-region (point-min) (point-max) file))
371 (message &quot;Exported annotations to %s&quot; file))

375;;;###autoload
376(defun nskk-custom-annotation-statistics ()
377 &quot;カスタム注釈の統計情報を返す。&quot;
378 (let ((total (hash-table-count nskk-custom-annotation--table))
379 (midashi-count (make-hash-table :test 'equal))
380 (avg-length 0))
381
382 ;; 見出し語数をカウント
383 (maphash (lambda (key _value)
384 (let ((midashi (car key)))
385 (puthash midashi t midashi-count)))
386 nskk-custom-annotation--table)
387
388 ;; 平均注釈長を計算
389 (when (&gt; total 0)
390 (let ((total-length 0))
391 (maphash (lambda (_key value)
392 (setq total-length (+ total-length (length value))))
393 nskk-custom-annotation--table)
394 (setq avg-length (/ total-length total))))
395
396 (list :total total
397 :unique-midashi (hash-table-count midashi-count)
398 :avg-length avg-length
399 :modified nskk-custom-annotation--modified)))
