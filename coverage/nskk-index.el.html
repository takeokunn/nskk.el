27;; このファイルはNSKKの統合インデックスシステムを実装します。
28;;
29;; 目的:
30;; - nskk-dict-struct、nskk-trie、nskk-cacheを統合
31;; - 効率的なインデックス構築アルゴリズム
32;; - 複数辞書の統合インデックス
33;; - Phase 3の並列化に向けた準備
34;;
35;; 特徴:
36;; 1. **統合レイヤー**: 既存モジュールを橋渡し
37;; 2. **キャッシュ統合**: 検索結果を自動キャッシュ
38;; 3. **増分更新**: 全体再構築を避ける
39;; 4. **バッチ処理**: 複数エントリの一括更新
40;; 5. **並列化準備**: パーティション分割戦略
41;;
42;; パフォーマンス目標:
43;; - 検索時間: &lt; 10ms（10万エントリ）
44;; - 構築時間: &lt; 500ms（10万エントリ）
45;; - 増分更新: &lt; 1ms/エントリ
46;; - メモリ使用: &lt; 50MB（10万エントリ）
47;; - キャッシュヒット率: &gt; 80%
48;;
49;; 使用例:
50;;
51;; (require 'nskk-index)
52;;
53;; ;; インデックス作成
54;; (let ((index (nskk-index-create :cache-capacity 1000)))
55;; ;; 辞書から構築
56;; (nskk-index-build index dict-struct)
57;;
58;; ;; 検索
59;; (nskk-index-search index &quot;かん&quot; 'prefix nil 10)
60;;
61;; ;; 増分更新
62;; (nskk-index-update index entry)
63;;
64;; ;; 統計情報
65;; (nskk-index-stats index))

276(defun nskk-index-search (index query &optional search-type limit)
277 &quot;インデックスから検索する（キャッシュ統合）。
278
279引数:
280 INDEX - nskk-index構造体
281 QUERY - 検索クエリ
282 SEARCH-TYPE - 検索タイプ（'exact/'prefix/'partial/'fuzzy、デフォルト'exact）
283 LIMIT - 結果の最大数
284
285処理:
286 1. キャッシュキーを生成
287 2. キャッシュから検索
288 3. ヒットしなければnskk-searchを呼び出し
289 4. 結果をキャッシュに保存
290 5. 統計情報を更新
291
292戻り値:
293 検索結果（nskk-searchと同じ形式）&quot;
294 (unless (nskk-index-dict-struct index)
295 (error &quot;Index not built yet&quot;))
296
297 ;; 統計情報を更新
298 (cl-incf (nskk-index-total-searches index))
299
300 ;; キャッシュキーを生成
301 (let* ((cache-key (nskk-index--make-cache-key query search-type limit))
302 (cache (nskk-index-cache index))
303 (cached-result (when cache (nskk-cache-get cache cache-key))))
304
305 (if cached-result
306 ;; キャッシュヒット
307 (progn
308 (cl-incf (nskk-index-cache-hits index))
309 cached-result)
310 ;; キャッシュミス：実際に検索
311 (cl-incf (nskk-index-cache-misses index))
312 (let ((result (nskk-search (nskk-index-dict-struct index)
313 query
314 search-type
315 nil ; okuri-type
316 limit)))
317 ;; 結果をキャッシュに保存
318 (when cache
319 (nskk-cache-put cache cache-key result))
320 result))))

350(defun nskk-index-update (index entry)
351 &quot;インデックスにエントリを追加・更新する（増分更新）。
352
353引数:
354 INDEX - nskk-index構造体
355 ENTRY - nskk-dict-entry構造体
356
357処理:
358 1. 辞書構造のハッシュテーブルに追加
359 2. トライ木に挿入
360 3. キャッシュを無効化（該当エントリのみ）
361 4. エントリ数を更新
362
363戻り値:
364 INDEX（更新後）&quot;
365 (unless (nskk-index-dict-struct index)
366 (error &quot;Index not built yet&quot;))
367
368 (let* ((dict-struct (nskk-index-dict-struct index))
369 (midashi (nskk-dict-entry-midashi entry))
370 (okuri-type (nskk-dict-entry-okuri-type entry))
371 (table (if (eq okuri-type 'okuri-ari)
372 (nskk-dict-index-okuri-ari-table dict-struct)
373 (nskk-dict-index-okuri-nasi-table dict-struct)))
374 (trie (if (eq okuri-type 'okuri-ari)
375 (nskk-dict-index-trie-ari dict-struct)
376 (nskk-dict-index-trie-nasi dict-struct)))
377 (is-new (not (gethash midashi table))))
378
379 ;; ハッシュテーブルに追加
380 (puthash midashi entry table)
381
382 ;; トライ木に挿入
383 (when trie
384 (nskk-trie-insert trie midashi entry))
385
386 ;; キャッシュを無効化（該当エントリのみ）
387 (let ((cache (nskk-index-cache index)))
388 (when cache
389 (nskk-cache-invalidate cache (nskk-index--make-cache-key midashi 'exact nil))))
390
391 ;; エントリ数を更新（新規の場合のみ）
392 (when is-new
393 (cl-incf (nskk-index-total-entries index)))
394
395 index))

398(defun nskk-index-update-batch (index entries)
399 &quot;複数のエントリを一括更新する（バッチ処理）。
400
401引数:
402 INDEX - nskk-index構造体
403 ENTRIES - nskk-dict-entryのリスト
404
405処理:
406 1. 各エントリを順次更新
407 2. キャッシュクリア（バッチ処理後）
408
409戻り値:
410 INDEX（更新後）
411
412パフォーマンス: キャッシュ無効化を最後に一度だけ実行&quot;
413 (unless (nskk-index-dict-struct index)
414 (error &quot;Index not built yet&quot;))
415
416 ;; 各エントリを更新（キャッシュ無効化は抑制）
417 (let ((dict-struct (nskk-index-dict-struct index))
418 (new-count 0))
419 (dolist (entry entries)
420 (let* ((midashi (nskk-dict-entry-midashi entry))
421 (okuri-type (nskk-dict-entry-okuri-type entry))
422 (table (if (eq okuri-type 'okuri-ari)
423 (nskk-dict-index-okuri-ari-table dict-struct)
424 (nskk-dict-index-okuri-nasi-table dict-struct)))
425 (trie (if (eq okuri-type 'okuri-ari)
426 (nskk-dict-index-trie-ari dict-struct)
427 (nskk-dict-index-trie-nasi dict-struct)))
428 (is-new (not (gethash midashi table))))
429
430 ;; ハッシュテーブルに追加
431 (puthash midashi entry table)
432
433 ;; トライ木に挿入
434 (when trie
435 (nskk-trie-insert trie midashi entry))
436
437 ;; 新規エントリ数をカウント
438 (when is-new
439 (cl-incf new-count))))
440
441 ;; エントリ数を一括更新
442 (cl-incf (nskk-index-total-entries index) new-count))
443
444 ;; キャッシュをクリア（バッチ処理後）
445 (let ((cache (nskk-index-cache index)))
446 (when cache
447 (nskk-cache-clear cache)))
448
449 index)

534(defun nskk-index-optimize (index)
535 &quot;インデックスを最適化する。
536
537引数:
538 INDEX - nskk-index構造体
539
540処理:
541 1. キャッシュ容量を調整（ヒット率に基づく）
542 2. 辞書構造を圧縮（将来実装）
543 3. 最適化フラグを設定
544
545戻り値:
546 INDEX（最適化後）&quot;
547 (when nskk-index-enable-auto-optimize
548 (let* ((stats (nskk-index-stats index))
549 (hit-rate (plist-get stats :cache-hit-rate))
550 (cache (nskk-index-cache index)))
551
552 ;; キャッシュヒット率が低い場合、容量を増やす
553 (when (and cache (&lt; hit-rate 0.5))
554 ;; 新しいキャッシュを作成（容量を1.5倍）
555 (let* ((old-capacity (plist-get (nskk-cache-stats cache) :capacity))
556 (new-capacity (floor (* old-capacity 1.5)))
557 (cache-type (plist-get (nskk-cache-stats cache) :type))
558 (new-cache (nskk-cache-create cache-type new-capacity)))
559 (setf (nskk-index-cache index) new-cache)))
560
561 ;; 最適化フラグを設定
562 (setf (nskk-index-optimized index) t)))
563
564 index)

712 (unless (or (null cache)
713 (nskk-cache-lru-p cache)
714 (nskk-cache-lfu-p cache))
715 (push &quot;Invalid cache structure&quot; errors)))

718 (nreverse errors)
719 t)))
