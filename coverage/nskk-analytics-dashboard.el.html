126(defun nskk-analytics-dashboard--build-header ()
127 &quot;ダッシュボードヘッダーを構築する。
128
129戻り値: フォーマット済み文字列&quot;
130 (let* ((metrics (or nskk-analytics-dashboard--live-metrics
131 (nskk-analytics-collect-metrics)))
132 (patterns (or nskk-analytics-pattern--patterns
133 (nskk-analytics-detect-patterns))))
134
135 (setq nskk-analytics-dashboard--live-metrics metrics)
136
137 (concat
138 (propertize &quot;═══ NSKK Analytics Dashboard ═══\n&quot; 'face 'bold)
139 (format &quot;\n%s\n&quot;
140 (propertize &quot;Real-time Metrics&quot; 'face 'underline))
141 (format &quot; Conversions: %s | Keystrokes: %s | Accuracy: %s | Speed: %s\n&quot;
142 (propertize (format &quot;%d&quot; (nskk-analytics-metrics-conversions metrics))
143 'face 'success)
144 (propertize (format &quot;%d&quot; (nskk-analytics-metrics-keystrokes metrics))
145 'face 'success)
146 (propertize (format &quot;%.1f%%&quot;
147 (* (nskk-analytics-metrics-accuracy metrics) 100))
148 'face (if (&gt; (nskk-analytics-metrics-accuracy metrics) 0.9)
149 'success
150 'warning))
151 (propertize (format &quot;%.3fs&quot; (nskk-analytics-metrics-speed metrics))
152 'face (if (&lt; (nskk-analytics-metrics-speed metrics) 0.1)
153 'success
154 'warning)))
155 (format &quot;\n%s\n&quot;
156 (propertize &quot;Quick Stats&quot; 'face 'underline))
157 (format &quot; Usage Intensity: %s events/day | Efficiency Score: %s/100\n&quot;
158 (propertize (format &quot;%.1f&quot;
159 (or (alist-get 'usage-intensity patterns) 0.0))
160 'face 'info)
161 (propertize (format &quot;%d&quot;
162 (or (alist-get 'efficiency-score patterns) 0))
163 'face (let ((score (or (alist-get 'efficiency-score patterns) 0)))
164 (cond
165 ((&gt;= score 80) 'success)
166 ((&gt;= score 60) 'warning)
167 (t 'error)))))
168 (format &quot;\nView: %s | Last Update: %s\n&quot;
169 (propertize (symbol-name nskk-analytics-dashboard--current-view)
170 'face 'italic)
171 (format-time-string &quot;%H:%M:%S&quot;))
172 (propertize &quot;────────────────────────────────────────\n&quot; 'face 'shadow))))

176(defun nskk-analytics-dashboard-show-metrics ()
177 &quot;詳細メトリクスを表示する。&quot;
178 (interactive)
179 (let ((metrics (nskk-analytics-collect-metrics)))
180 (with-current-buffer (get-buffer-create &quot;*NSKK Metrics*&quot;)
181 (erase-buffer)

183 (goto-char (point-min))
184 (display-buffer (current-buffer)))))

304(defun nskk-analytics-dashboard-toggle-refresh ()
305 &quot;自動更新を切り替える。&quot;
306 (interactive)
307 (if nskk-analytics-dashboard--refresh-timer
308 (progn
309 (cancel-timer nskk-analytics-dashboard--refresh-timer)
310 (setq nskk-analytics-dashboard--refresh-timer nil)
311 (message &quot;Auto-refresh disabled&quot;))
312 (when (&gt; nskk-analytics-dashboard-refresh-interval 0)
313 (setq nskk-analytics-dashboard--refresh-timer
314 (run-at-time nskk-analytics-dashboard-refresh-interval
315 nskk-analytics-dashboard-refresh-interval
316 #'nskk-analytics-dashboard-refresh))
317 (message &quot;Auto-refresh enabled (interval: %ds)&quot;
318 nskk-analytics-dashboard-refresh-interval))))
319
320(defun nskk-analytics-dashboard-refresh ()
321 &quot;ダッシュボードを更新する。&quot;
322 (interactive)
323 (setq nskk-analytics-dashboard--live-metrics nil)
324 ;; Transientが開いている場合は再描画
325 (when (and (boundp 'transient-current-command)
326 transient-current-command)
327 (transient-setup 'nskk-analytics-dashboard)))

344(defun nskk-analytics-dashboard--create-sparkline (data &optional width)
345 &quot;簡易スパークラインを作成する。
346
347DATA: データポイントのリスト
348WIDTH: 幅（デフォルト: 20）
349戻り値: スパークライン文字列&quot;
350 (let* ((w (or width 20))
351 (max-val (apply #'max data))
352 (min-val (apply #'min data))
353 (range (- max-val min-val))
354 (chars '(&quot;▁&quot; &quot;▂&quot; &quot;▃&quot; &quot;▄&quot; &quot;▅&quot; &quot;▆&quot; &quot;▇&quot; &quot;█&quot;)))
355 (mapconcat
356 (lambda (val)
357 (let ((index (if (zerop range)
358 0
359 (min 7 (floor (* 8 (/ (- val min-val) (float range))))))))
360 (nth index chars)))
361 data
362 &quot;&quot;)))

366(defun nskk-analytics-dashboard--create-gauge (value max-value &optional label)
367 &quot;ゲージウィジェットを作成する。
368
369VALUE: 現在値
370MAX-VALUE: 最大値
371LABEL: ラベル（オプション）
372戻り値: フォーマット済み文字列&quot;
373 (let* ((percentage (/ (float value) max-value))
374 (filled (floor (* 20 percentage)))
375 (empty (- 20 filled)))
376 (format &quot;%s [%s%s] %.0f%%&quot;
377 (or label &quot;&quot;)
378 (make-string filled ?█)
379 (make-string empty ?░)
380 (* percentage 100))))
