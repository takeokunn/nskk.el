121(defun nskk-data--load-dictionaries ()
122 &quot;辞書ファイルをロードする。&quot;
123 (setq nskk-data--dictionaries nil)
124 (dolist (path nskk-data-dictionary-paths)
125 (when (file-readable-p path)
126 (condition-case err
127 (let ((dict (nskk-data--load-dictionary-file path)))
128 (push dict nskk-data--dictionaries)
129 (nskk-data--log &quot;Loaded dictionary: %s&quot; path))
130 (error
131 (nskk-data--log &quot;Failed to load dictionary %s: %s&quot; path err))))))

152(defun nskk-data-search (query &optional options)
153 &quot;辞書を検索する。
154QUERYは検索クエリ、OPTIONSは検索オプション。&quot;
155 ;; キャッシュチェック
156 (let ((cache-key (cons query options)))
157 (or (gethash cache-key nskk-data--cache)
158 (let ((results (nskk-data--search-uncached query options)))
159 (puthash cache-key results nskk-data--cache)
160 results))))
161
162(defun nskk-data--search-uncached (query options)
163 &quot;辞書を検索する（キャッシュなし）。
164QUERYは検索クエリ、OPTIONSは検索オプション。&quot;
165 (let ((results nil))
166 (dolist (dict nskk-data--dictionaries)
167 (let ((entries (gethash query dict)))
168 (when entries
169 (setq results (append results entries)))))
170 ;; 学習データでソート
171 (nskk-data--sort-by-learning results query)))
172
173(defun nskk-data--sort-by-learning (results query)
174 &quot;学習データに基づいて結果をソートする。
175RESULTSは検索結果リスト、QUERYは検索クエリ。&quot;
176 (let ((learning-scores (gethash query nskk-data--learning-data)))
177 (if learning-scores
178 (sort results
179 (lambda (a b)
180 (&gt; (or (gethash a learning-scores) 0)
181 (or (gethash b learning-scores) 0))))
182 results)))

186(defun nskk-data--load-learning-data ()
187 &quot;学習データをロードする。&quot;
188 (when (file-readable-p nskk-data-learning-file)
189 (condition-case err
190 (with-temp-buffer
191 (insert-file-contents nskk-data-learning-file)
192 (setq nskk-data--learning-data (read (current-buffer)))
193 (nskk-data--log &quot;Loaded learning data from %s&quot;
194 nskk-data-learning-file))
195 (error
196 (nskk-data--log &quot;Failed to load learning data: %s&quot; err)
197 (setq nskk-data--learning-data (make-hash-table :test 'equal))))))
198
199(defun nskk-data-save-learning-data ()
200 &quot;学習データを保存する。&quot;
201 (interactive)
202 (let ((dir (file-name-directory nskk-data-learning-file)))
203 (unless (file-directory-p dir)
204 (make-directory dir t)))
205 (condition-case err
206 (with-temp-file nskk-data-learning-file
207 (prin1 nskk-data--learning-data (current-buffer))
208 (setq nskk-data--dirty-flag nil)
209 (nskk-data--log &quot;Saved learning data to %s&quot;
210 nskk-data-learning-file)
211 (message &quot;Learning data saved&quot;))
212 (error
213 (nskk-data--log &quot;Failed to save learning data: %s&quot; err)
214 (message &quot;Failed to save learning data: %s&quot; err))))
215
216(defun nskk-data-learn (query candidate &optional context)
217 &quot;学習データを記録する。
218QUERYは検索クエリ、CANDIDATEは選択された候補、
219CONTEXTは学習コンテキスト。&quot;
220 (let ((scores (or (gethash query nskk-data--learning-data)
221 (make-hash-table :test 'equal))))
222 (puthash candidate
223 (1+ (or (gethash candidate scores) 0))
224 scores)
225 (puthash query scores nskk-data--learning-data)
226 (setq nskk-data--dirty-flag t)))

248(defun nskk-data--start-auto-save ()
249 &quot;自動保存を開始する。&quot;
250 (when nskk-data--auto-save-timer
251 (cancel-timer nskk-data--auto-save-timer))
252 (setq nskk-data--auto-save-timer

254 nskk-data-auto-save-interval
255 #'nskk-data--auto-save-handler)))

258 &quot;自動保存を停止する。&quot;
259 (when nskk-data--auto-save-timer
260 (cancel-timer nskk-data--auto-save-timer)
261 (setq nskk-data--auto-save-timer nil)))
262

264 &quot;自動保存ハンドラー。&quot;

315(defun nskk-data-get-statistics ()
316 &quot;Data Access Layerの統計情報を取得する。&quot;
317 (interactive)
318 (let ((dict-count (length nskk-data--dictionaries))
319 (learning-entries (hash-table-count nskk-data--learning-data))
320 (cache-entries (hash-table-count nskk-data--cache)))
321 (message &quot;Data Layer Statistics:\n Dictionaries: %d\n Learning entries: %d\n Cache entries: %d\n Dirty: %s&quot;
322 dict-count learning-entries cache-entries
323 (if nskk-data--dirty-flag &quot;yes&quot; &quot;no&quot;))
324 (list :dictionaries dict-count
325 :learning-entries learning-entries
326 :cache-entries cache-entries
327 :dirty nskk-data--dirty-flag)))

354(defun nskk-data-health-check ()
355 &quot;Data Access Layerのヘルスチェックを実行する。&quot;
356 (interactive)
357 (let ((issues '()))
358 ;; 辞書状態チェック
359 (when (null nskk-data--dictionaries)
360 (push &quot;No dictionaries loaded&quot; issues))
361 ;; 学習データチェック
362 (unless nskk-data--learning-data
363 (push &quot;Learning data not initialized&quot; issues))
364 ;; ファイルアクセスチェック
365 (let ((dir (file-name-directory nskk-data-learning-file)))
366 (unless (file-writable-p dir)
367 (push (format &quot;Learning data directory not writable: %s&quot; dir) issues)))
368 ;; 結果表示
369 (if issues
370 (message &quot;Data Layer issues: %s&quot; (string-join issues &quot;, &quot;))
371 (message &quot;Data Layer: All systems operational&quot;))))
