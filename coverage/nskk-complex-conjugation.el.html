27;; このファイルはNSKKの複雑活用処理エンジンを実装します。
28;;
29;; 対応する複合活用:
30;; - 使役形（書かせる、見させる）
31;; - 受身形（書かれる、見られる）

33;; - 使役受身形（書かせられる）

35;; - 謙譲形（お書きする）
36;;
37;; 特殊活用:
38;; - 不規則動詞の処理
39;; - エッジケース（ら抜き言葉等）
40;;
41;; 使用例:
42;; (nskk-complex-conjugate &quot;書&quot; 'godan 'causative) ;; =&gt; &quot;書かせ&quot;
43;; (nskk-complex-conjugate &quot;見&quot; 'kami-ichidan 'passive) ;; =&gt; &quot;見られ&quot;
44;; (nskk-complex-conjugate &quot;書&quot; 'godan 'potential) ;; =&gt; &quot;書け&quot;

219;;;###autoload
220(defun nskk-complex-conjugate (stem verb-type complex-type)
221 &quot;動詞STEMを複合活用する。
222
223引数:
224 STEM - 動詞の語幹（文字列）
225 VERB-TYPE - 動詞の活用型（symbol）
226 'godan, 'kami-ichidan, 'shimo-ichidan, 'sa-hen, 'ka-hen
227 COMPLEX-TYPE - 複合活用の種類（symbol）
228 'causative, 'passive, 'potential, 'causative-passive,
229 'honorific, 'humble
230
231返り値:
232 複合活用後の文字列、またはnil（無効な入力の場合）
233
234使用例:
235 (nskk-complex-conjugate \&quot;書\&quot; 'godan 'causative) ;; =&gt; \&quot;書かせ\&quot;
236 (nskk-complex-conjugate \&quot;見\&quot; 'kami-ichidan 'passive) ;; =&gt; \&quot;見られ\&quot;
237 (nskk-complex-conjugate \&quot;書\&quot; 'godan 'potential) ;; =&gt; \&quot;書け\&quot;
238 (nskk-complex-conjugate \&quot;食べ\&quot; 'shimo-ichidan 'potential) ;; =&gt; \&quot;食べれ\&quot; (ら抜き時)
239 (nskk-complex-conjugate \&quot;書\&quot; 'godan 'causative-passive) ;; =&gt; \&quot;書かせられ\&quot;
240 (nskk-complex-conjugate \&quot;見\&quot; 'kami-ichidan 'humble) ;; =&gt; \&quot;お見し\&quot;&quot;
241 (unless (stringp stem)
242 (signal 'wrong-type-argument (list 'stringp stem)))
243 (unless (memq verb-type nskk-verb-types)
244 (signal 'wrong-type-argument (list 'nskk-verb-types verb-type)))
245 (unless (memq complex-type nskk-complex-conjugation-types)
246 (signal 'wrong-type-argument (list 'nskk-complex-conjugation-types complex-type)))
247
248 (pcase complex-type
249 ('causative
250 (pcase verb-type
251 ('godan (nskk-complex--causative-godan stem))
252 ('kami-ichidan (nskk-complex--causative-ichidan stem))
253 ('shimo-ichidan (nskk-complex--causative-ichidan stem))
254 ('sa-hen (nskk-complex--causative-sa-hen stem))
255 ('ka-hen (nskk-complex--causative-ka-hen stem))))
256
257 ('passive
258 (pcase verb-type
259 ('godan (nskk-complex--passive-godan stem))
260 ('kami-ichidan (nskk-complex--passive-ichidan stem))
261 ('shimo-ichidan (nskk-complex--passive-ichidan stem))
262 ('sa-hen (nskk-complex--passive-sa-hen stem))
263 ('ka-hen (nskk-complex--passive-ka-hen stem))))
264
265 ('potential
266 (pcase verb-type
267 ('godan (nskk-complex--potential-godan stem))
268 ('kami-ichidan (nskk-complex--potential-ichidan stem))
269 ('shimo-ichidan (nskk-complex--potential-ichidan stem))
270 ('sa-hen (nskk-complex--potential-sa-hen stem))
271 ('ka-hen (nskk-complex--potential-ka-hen stem))))
272
273 ('causative-passive
274 (pcase verb-type
275 ('godan (nskk-complex--causative-passive-godan stem))
276 ('kami-ichidan (nskk-complex--causative-passive-ichidan stem))
277 ('shimo-ichidan (nskk-complex--causative-passive-ichidan stem))
278 ('sa-hen (nskk-complex--causative-passive-sa-hen stem))
279 ('ka-hen (nskk-complex--causative-passive-ka-hen stem))))
280
281 ('honorific
282 (pcase verb-type
283 ('godan (nskk-complex--honorific-godan stem))
284 ('kami-ichidan (nskk-complex--honorific-ichidan stem))
285 ('shimo-ichidan (nskk-complex--honorific-ichidan stem))
286 ('sa-hen (nskk-complex--honorific-sa-hen stem))
287 ('ka-hen (nskk-complex--honorific-ka-hen stem))))
288
289 ('humble
290 (nskk-complex--humble stem verb-type))
291
292 (_ nil)))

295(defun nskk-complex-get-all-forms (stem verb-type)
296 &quot;動詞STEMの全複合活用形を返す。
297
298引数:
299 STEM - 動詞の語幹（文字列）
300 VERB-TYPE - 動詞の活用型（symbol）
301
302返り値:
303 複合活用種類と活用後の文字列のalist
304
305使用例:
306 (nskk-complex-get-all-forms \&quot;書\&quot; 'godan)
307 ;; =&gt; ((causative . \&quot;書かせ\&quot;) (passive . \&quot;書かれ\&quot;) ...)&quot;
308 (mapcar (lambda (type)
309 (cons type (nskk-complex-conjugate stem verb-type type)))
310 nskk-complex-conjugation-types))
311
312;;; 特殊活用の処理
313
314(defvar nskk-complex--irregular-verbs
315 '((&quot;行&quot; . ((potential . &quot;行ける&quot;))) ; 行く→行ける（特殊）
316 (&quot;問&quot; . ((potential . &quot;問える&quot;)))) ; 問う→問える
317 &quot;不規則活用する動詞のリスト。&quot;)
318
319(defun nskk-complex-is-irregular-p (stem)
320 &quot;不規則活用する動詞かどうかを判定する。&quot;
321 (assoc stem nskk-complex--irregular-verbs))
322
323(defun nskk-complex-conjugate-irregular (stem complex-type)
324 &quot;不規則動詞の複合活用を返す。&quot;
325 (let* ((irregular (assoc stem nskk-complex--irregular-verbs))
326 (forms (cdr irregular)))
327 (when forms
328 (cdr (assq complex-type forms)))))
329

332(defun nskk-complex-print-all-forms (stem verb-type)
333 &quot;動詞STEMの全複合活用形を表形式で表示する（デバッグ用）。&quot;
334 (interactive &quot;s語幹: \nS活用型 (godan/kami-ichidan/shimo-ichidan/sa-hen/ka-hen): &quot;)
335 (let ((forms (nskk-complex-get-all-forms stem (intern verb-type))))
336 (with-output-to-temp-buffer &quot;*NSKK Complex Conjugation*&quot;
337 (princ (format &quot;動詞「%s」の複合活用（%s活用）\n&quot; stem verb-type))
338 (princ &quot;====================\n\n&quot;)
339 (dolist (entry forms)
340 (princ (format &quot;%-20s: %s\n&quot;
341 (symbol-name (car entry))
342 (cdr entry)))))))
