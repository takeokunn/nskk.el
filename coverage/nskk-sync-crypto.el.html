27;; このファイルはNSKK同期システムの暗号化・セキュリティ機能を実装します。
28;;
29;; 特徴:
30;; - AES-256-GCM暗号化（データの機密性と完全性）
31;; - TLS 1.3統合（通信の暗号化）
32;; - PBKDF2鍵導出（パスワードベース）
33;; - HMAC-SHA256メッセージ認証
34;; - 安全な鍵管理（auth-source/keychain統合）
35;; - OWASP準拠のセキュリティ実装
36;;
37;; セキュリティ仕様:
38;;
39;; - 暗号化: AES-256-GCM
40;; - 鍵導出: PBKDF2-HMAC-SHA256 (100,000回イテレーション)
41;; - 認証: HMAC-SHA256
42;; - TLS: TLS 1.3以上
43;; - 乱数: /dev/urandom または安全な乱数生成器
44;;
45;; 使用例:
46;;
47;; (require 'nskk-sync-crypto)
48;;
49;; ;; 鍵の生成
50;; (let ((password &quot;my-secret-password&quot;)
51;; (salt (nskk-sync-crypto-generate-salt)))
52;; (let ((key (nskk-sync-crypto-derive-key password salt)))
53;;
54;; ;; データの暗号化
55;; (let* ((plaintext &quot;secret data&quot;)
56;; (encrypted (nskk-sync-crypto-encrypt plaintext key)))
57;;
58;; ;; データの復号
59;; (let ((decrypted (nskk-sync-crypto-decrypt encrypted key)))
60;; (message &quot;Decrypted: %s&quot; decrypted)))))

189(defun nskk-sync-crypto--encrypt-with-openssl (plaintext key)
190 &quot;OpenSSLを使用してPLAINTEXTを暗号化する。&quot;
191 (unless (executable-find &quot;openssl&quot;)
192 (error &quot;OpenSSL not found&quot;))
193
194 (let* ((iv (nskk-sync-crypto-generate-iv))
195 (iv-hex (nskk-sync-crypto--bytes-to-hex iv))
196 (key-hex (nskk-sync-crypto--bytes-to-hex key))
197 (temp-input (make-temp-file &quot;nskk-encrypt-input&quot;))
198 (temp-output (make-temp-file &quot;nskk-encrypt-output&quot;)))
199
200 (unwind-protect
201 (progn
202 ;; 平文を一時ファイルに書き込み
203 (with-temp-file temp-input
204 (insert plaintext))
205
206 ;; OpenSSLで暗号化
207 (let ((exit-code
208 (call-process &quot;openssl&quot; nil nil nil
209 &quot;enc&quot; &quot;-aes-256-gcm&quot;
210 &quot;-K&quot; key-hex
211 &quot;-iv&quot; iv-hex
212 &quot;-in&quot; temp-input
213 &quot;-out&quot; temp-output)))
214 (unless (zerop exit-code)
215 (error &quot;OpenSSL encryption failed: %d&quot; exit-code)))
216
217 ;; 暗号文を読み込み
218 (let ((ciphertext
219 (with-temp-buffer
220 (set-buffer-multibyte nil)
221 (insert-file-contents-literally temp-output)
222 (buffer-string))))
223
224 (nskk-sync-crypto-encrypted--create
225 :ciphertext ciphertext
226 :iv iv
227 :tag &quot;&quot; ;; GCMタグは暗号文に含まれる
228 :algorithm 'aes-256-gcm
229 :version 1)))
230
231 ;; 一時ファイル削除
232 (when (file-exists-p temp-input)
233 (delete-file temp-input))
234 (when (file-exists-p temp-output)
235 (delete-file temp-output)))))

301(defun nskk-sync-crypto--xor-encrypt (plaintext key iv)
302 &quot;簡易XOR暗号（教育目的のみ）。&quot;
303 (let* ((key-bytes (append key iv nil))
304 (key-len (length key-bytes))
305 (result nil))
306 (dotimes (i (length plaintext))
307 (let ((plain-byte (aref plaintext i))
308 (key-byte (nth (mod i key-len) key-bytes)))

310 (apply #'unibyte-string (nreverse result))))
