93;;;###autoload
94(defun nskk-async-learning-initialize ()
95 &quot;非同期学習システムを初期化する。&quot;
96 (interactive)
97 (unless nskk-async-learning--state
98 (when (nskk-thread-pool-available-p)
99 (let ((state (nskk-async-learning-state--create
100 :queue nil
101 :queue-mutex (make-mutex)
102 :pool (nskk-thread-pool-create 2) ; 学習専用プール
103 :frequency-table (make-hash-table :test 'equal)
104 :context-table (make-hash-table :test 'equal))))
105
106 ;; バッチ処理タイマー設定
107 (setf (nskk-async-learning-state-timer state)
108 (run-with-timer nskk-async-learning-batch-interval
109 nskk-async-learning-batch-interval
110 #'nskk-async-learning--process-batch))
111
112 ;; 自動保存タイマー設定
113 (when nskk-async-learning-auto-save
114 (setf (nskk-async-learning-state-save-timer state)
115 (run-with-timer nskk-async-learning-auto-save-interval
116 nskk-async-learning-auto-save-interval
117 #'nskk-async-learning--auto-save)))
118
119 (setq nskk-async-learning--state state)
120 (message &quot;Async learning system initialized&quot;)))))
121
122;;;###autoload
123(defun nskk-async-learning-shutdown ()
124 &quot;非同期学習システムをシャットダウンする。&quot;
125 (interactive)
126 (when nskk-async-learning--state
127 ;; タイマーキャンセル
128 (when (nskk-async-learning-state-timer nskk-async-learning--state)
129 (cancel-timer (nskk-async-learning-state-timer nskk-async-learning--state)))
130
131 (when (nskk-async-learning-state-save-timer nskk-async-learning--state)
132 (cancel-timer (nskk-async-learning-state-save-timer nskk-async-learning--state)))
133
134 ;; 残りのキューを処理
135 (nskk-async-learning--flush-queue)
136
137 ;; スレッドプールシャットダウン
138 (when (nskk-async-learning-state-pool nskk-async-learning--state)
139 (nskk-thread-pool-shutdown (nskk-async-learning-state-pool nskk-async-learning--state)))
140
141 (setq nskk-async-learning--state nil)
142 (message &quot;Async learning system shutdown&quot;)))

146(defun nskk-async-learning--enqueue (record)
147 &quot;学習レコード RECORD をキューに追加する。&quot;
148 (when nskk-async-learning--state
149 (with-mutex (nskk-async-learning-state-queue-mutex nskk-async-learning--state)
150 (let ((queue (nskk-async-learning-state-queue nskk-async-learning--state)))
151 (when (&gt;= (length queue) nskk-async-learning-max-queue-size)
152 (message &quot;Learning queue full, dropping oldest record&quot;)
153 (setq queue (cdr queue)))
154
155 (setf (nskk-async-learning-state-queue nskk-async-learning--state)
156 (append queue (list record)))))))
157
158(defun nskk-async-learning--dequeue-batch ()
159 &quot;バッチサイズ分のレコードをキューから取り出す。&quot;
160 (when nskk-async-learning--state
161 (with-mutex (nskk-async-learning-state-queue-mutex nskk-async-learning--state)
162 (let* ((queue (nskk-async-learning-state-queue nskk-async-learning--state))
163 (batch-size (min nskk-async-learning-batch-size (length queue)))
164 (batch (cl-subseq queue 0 batch-size)))
165
166 (setf (nskk-async-learning-state-queue nskk-async-learning--state)
167 (cl-subseq queue batch-size))
168
169 batch))))

171;;; 学習処理

173(defun nskk-async-learning--update-frequency (key candidate)
174 &quot;KEY と CANDIDATE の使用頻度を更新する。&quot;
175 (when nskk-async-learning--state
176 (let* ((table (nskk-async-learning-state-frequency-table nskk-async-learning--state))
177 (freq-key (format &quot;%s:%s&quot; key candidate))
178 (current-freq (gethash freq-key table 0)))
179 (puthash freq-key (1+ current-freq) table))))
180
181(defun nskk-async-learning--update-context (key candidate context)
182 &quot;KEY と CANDIDATE の文脈情報を更新する。&quot;
183 (when (and nskk-async-learning--state context)
184 (let* ((table (nskk-async-learning-state-context-table nskk-async-learning--state))
185 (ctx-key (format &quot;%s:%s&quot; key candidate))
186 (contexts (gethash ctx-key table nil)))
187 (push context contexts)
188 ;; 最新100件のみ保持
189 (when (&gt; (length contexts) 100)
190 (setq contexts (cl-subseq contexts 0 100)))
191 (puthash ctx-key contexts table))))

193(defun nskk-async-learning--process-record (record)
194 &quot;学習レコード RECORD を処理する。&quot;
195 (let ((key (nskk-learning-record-key record))
196 (candidate (nskk-learning-record-candidate record))
197 (context (nskk-learning-record-context record)))
198
199 ;; 頻度更新
200 (nskk-async-learning--update-frequency key candidate)
201
202 ;; 文脈更新
203 (nskk-async-learning--update-context key candidate context)))
204
205(defun nskk-async-learning--process-batch ()
206 &quot;バッチ処理を実行する。&quot;
207 (when nskk-async-learning--state
208 (let ((batch (nskk-async-learning--dequeue-batch)))
209 (when batch
210 ;; バックグラウンドで処理
211 (nskk-thread-submit
212 (nskk-async-learning-state-pool nskk-async-learning--state)
213 (lambda ()
214 (dolist (record batch)
215 (nskk-async-learning--process-record record)))
216 (lambda (_)
217 (message &quot;Processed %d learning records&quot; (length batch))))))))
218
219(defun nskk-async-learning--flush-queue ()
220 &quot;キュー内の全レコードを処理する。&quot;
221 (when nskk-async-learning--state
222 (while (&gt; (length (nskk-async-learning-state-queue nskk-async-learning--state)) 0)
223 (nskk-async-learning--process-batch)
224 (sleep-for 0.1))))

303(defun nskk-async-learning-wait-for-queue-empty (timeout)
304 &quot;キューが空になるまで TIMEOUT 秒待機する。
305成功した場合 t、タイムアウトした場合 nil を返す。&quot;
306 (when nskk-async-learning--state
307 (let ((start-time (current-time)))
308 (while (and (&gt; (length (nskk-async-learning-state-queue nskk-async-learning--state)) 0)
309 (&lt; (float-time (time-subtract (current-time) start-time)) timeout))
310 (sleep-for 0.1))
311 (= (length (nskk-async-learning-state-queue nskk-async-learning--state)) 0))))
