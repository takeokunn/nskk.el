109;;;###autoload
110(defun nskk-async-candidates-show-async (candidates &optional callback)
111 &quot;候補を非同期で段階的に表示する。
112
113引数:
114 CANDIDATES - 候補リスト
115 CALLBACK - 完了時のコールバック（省略可）
116 引数: (displayed-count) - 表示した候補数
117
118処理フロー:
119 1. 前回の表示をキャンセル
120 2. コンテキスト作成
121 3. タイマー設定
122 4. バッチ処理開始
123
124戻り値:
125 なし（非同期処理）&quot;
126 (when nskk-async-candidates-verbose
127 (message &quot;Starting async candidate display: %d candidates&quot; (length candidates)))
128
129 ;; 前回の表示をキャンセル
130 (nskk-async-candidates-cancel)
131
132 (if (or (not nskk-async-candidates-enable-incremental)
133 (&lt;= (length candidates) nskk-async-candidates-batch-size))
134 ;; 段階的表示が無効、または候補数が少ない場合は一括表示
135 (progn
136 (nskk-async-candidates--display-batch candidates 0 (length candidates))
137 (when callback
138 (funcall callback (length candidates))))
139
140 ;; 段階的表示
141 (let ((context (nskk-async-candidates-context--create
142 :candidates candidates
143 :displayed-count 0
144 :callback callback
145 :cancelled nil
146 :start-time (float-time))))
147
148 (setq nskk-async-candidates--current-context context)
149
150 ;; 最初のバッチを即座に表示（UIブロッキング0ms）
151 (nskk-async-candidates--schedule-next-batch context))))
152
153(defun nskk-async-candidates--schedule-next-batch (context)
154 &quot;次のバッチ表示をスケジュールする。
155
156引数:
157 CONTEXT - nskk-async-candidates-context構造体&quot;
158 (let* ((candidates (nskk-async-candidates-context-candidates context))
159 (displayed (nskk-async-candidates-context-displayed-count context))
160 (total (length candidates))
161 (remaining (- total displayed)))
162
163 (when (and (&gt; remaining 0)
164 (not (nskk-async-candidates-context-cancelled context)))
165
166 ;; 次のバッチサイズを決定
167 (let ((batch-size (min nskk-async-candidates-batch-size remaining)))
168
169 ;; タイマーで次のバッチを表示
170 (setf (nskk-async-candidates-context-timer context)
171 (run-at-time nskk-async-candidates-delay nil
172 (lambda ()
173 (nskk-async-candidates--process-next-batch context batch-size))))))))
174
175(defun nskk-async-candidates--process-next-batch (context batch-size)
176 &quot;次のバッチを処理する。
177
178引数:
179 CONTEXT - nskk-async-candidates-context構造体
180 BATCH-SIZE - バッチサイズ&quot;
181 (when (not (nskk-async-candidates-context-cancelled context))
182 (let* ((candidates (nskk-async-candidates-context-candidates context))
183 (displayed (nskk-async-candidates-context-displayed-count context))
184 (total (length candidates)))
185
186 (when nskk-async-candidates-verbose
187 (message &quot;Processing batch: %d-%d / %d&quot;
188 displayed
189 (+ displayed batch-size)
190 total))
191
192 ;; バッチ表示
193 (nskk-async-candidates--display-batch
194 candidates
195 displayed
196 (min (+ displayed batch-size) total))
197
198 ;; カウント更新
199 (setf (nskk-async-candidates-context-displayed-count context)
200 (+ displayed batch-size))
201
202 ;; 次のバッチをスケジュール、または完了処理
203 (if (&gt;= (+ displayed batch-size) total)
204 (nskk-async-candidates--on-complete context)
205 (nskk-async-candidates--schedule-next-batch context)))))
206
207(defun nskk-async-candidates--display-batch (candidates start end)
208 &quot;候補のバッチを表示する。
209
210引数:
211 CANDIDATES - 候補リスト（全体）
212 START - 開始インデックス（0始まり）
213 END - 終了インデックス（この位置は含まない）
214
215この関数は実際のUI更新を担当します。
216実装は nskk-candidate-window のAPIを使用します。&quot;
217 (let ((batch (cl-subseq candidates start end)))
218
219 (when nskk-async-candidates-verbose
220 (message &quot;Displaying batch: %d-%d (%d candidates)&quot;
221 start end (length batch)))
222
223 ;; nskk-candidate-window のAPIを使用
224 ;; 初回（start = 0）の場合は新規表示、それ以外は更新
225 (if (zerop start)
226 (when (fboundp 'nskk-show-candidates)
227 (nskk-show-candidates candidates))
228 (when (fboundp 'nskk-update-candidates)
229 (nskk-update-candidates candidates 0)))))
230
231(defun nskk-async-candidates--on-complete (context)
232 &quot;表示完了時の処理。
233
234引数:
235 CONTEXT - nskk-async-candidates-context構造体&quot;
236 (when nskk-async-candidates-verbose
237 (let* ((elapsed (- (float-time)
238 (nskk-async-candidates-context-start-time context)))
239 (count (nskk-async-candidates-context-displayed-count context)))
240 (message &quot;Async candidate display completed: %d candidates in %.3fs&quot;
241 count elapsed)))
242
243 ;; コールバック呼び出し
244 (when (nskk-async-candidates-context-callback context)
245 (funcall (nskk-async-candidates-context-callback context)
246 (nskk-async-candidates-context-displayed-count context)))
247
248 ;; コンテキストクリア
249 (setq nskk-async-candidates--current-context nil))

253;;;###autoload
254(defun nskk-async-candidates-cancel ()
255 &quot;実行中の非同期候補表示をキャンセルする。&quot;
256 (interactive)
257 (when nskk-async-candidates--current-context
258 (let ((context nskk-async-candidates--current-context))
259
260 (when nskk-async-candidates-verbose
261 (message &quot;Cancelling async candidate display&quot;))
262
263 ;; キャンセルフラグ設定
264 (setf (nskk-async-candidates-context-cancelled context) t)
265
266 ;; タイマーキャンセル
267 (when (nskk-async-candidates-context-timer context)
268 (cancel-timer (nskk-async-candidates-context-timer context)))
269
270 ;; コンテキストクリア
271 (setq nskk-async-candidates--current-context nil))))

275(defun nskk-async-candidates-running-p ()
276 &quot;非同期候補表示が実行中かどうかを返す。
277
278戻り値:
279 実行中の場合t、そうでない場合nil&quot;
280 (and nskk-async-candidates--current-context
281 (not (nskk-async-candidates-context-cancelled
282 nskk-async-candidates--current-context))
283 t))

285(defun nskk-async-candidates-progress ()
286 &quot;現在の表示進捗を取得する。
287
288戻り値:
289 plist形式の進捗情報、または nil（実行中でない場合）
290 (:displayed N :total M :percentage P)
291 displayed - 表示済み候補数
292 total - 全候補数
293 percentage - 進捗率（0.0-1.0）&quot;
294 (when nskk-async-candidates--current-context
295 (let* ((context nskk-async-candidates--current-context)
296 (displayed (nskk-async-candidates-context-displayed-count context))
297 (total (length (nskk-async-candidates-context-candidates context)))
298 (percentage (if (zerop total) 0.0 (/ (float displayed) total))))
299
300 (list :displayed displayed
301 :total total
302 :percentage percentage))))

306(defun nskk-async-candidates-stats ()
307 &quot;現在の非同期候補表示の統計情報を取得する。
308
309戻り値:
310 plist形式の統計情報
311 (:running BOOL :progress PLIST :elapsed FLOAT)
312 running - 実行中かどうか
313 progress - 進捗情報（nskk-async-candidates-progressの戻り値）
314 elapsed - 経過時間（秒）&quot;
315 (let ((running (nskk-async-candidates-running-p))
316 (progress (nskk-async-candidates-progress))
317 (elapsed (when nskk-async-candidates--current-context
318 (- (float-time)
319 (nskk-async-candidates-context-start-time
320 nskk-async-candidates--current-context)))))
321
322 (list :running running
323 :progress progress
324 :elapsed elapsed)))

326;;;###autoload
327(defun nskk-async-candidates-show-stats ()
328 &quot;現在の統計情報をメッセージとして表示する。&quot;
329 (interactive)
330 (let ((stats (nskk-async-candidates-stats)))
331 (if (plist-get stats :running)
332 (let* ((progress (plist-get stats :progress))
333 (displayed (plist-get progress :displayed))
334 (total (plist-get progress :total))
335 (percentage (* 100 (plist-get progress :percentage)))
336 (elapsed (plist-get stats :elapsed)))
337 (message &quot;Async candidates: %d/%d (%.1f%%) in %.3fs&quot;
338 displayed total percentage elapsed))
339 (message &quot;No async candidate display is running&quot;))))
