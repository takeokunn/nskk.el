112(defun nskk-analytics-generate-html-report (&optional data output-file)
113 &quot;HTMLレポートを生成する。
114
115DATA: レポートデータ（nilの場合は自動収集）
116OUTPUT-FILE: 出力ファイルパス（nilの場合は自動生成）
117戻り値: 生成されたファイルパス&quot;
118 (interactive)
119 (let* ((report-data (or data (nskk-analytics-report--collect-data)))
120 (output (or output-file
121 (nskk-analytics-report--generate-filename &quot;html&quot;)))
122 (html (nskk-analytics-report--build-html report-data)))
123
124 ;; ディレクトリ作成
125 (unless (file-directory-p nskk-analytics-report-directory)
126 (make-directory nskk-analytics-report-directory t))
127
128 ;; HTML書き込み
129 (with-temp-file output
130 (insert html))
131
132 (setq nskk-analytics-report--last-report report-data)
133
134 (message &quot;HTML report generated: %s&quot; output)
135
136 ;; 自動で開く
137 (when (and nskk-analytics-report-auto-open
138 (called-interactively-p 'interactive))
139 (browse-url-of-file output))
140
141 output))

143(defun nskk-analytics-report--collect-data ()
144 &quot;レポートデータを収集する。&quot;
145 (let ((metrics (nskk-analytics-collect-metrics))
146 (patterns (nskk-analytics-detect-patterns))
147 (trends (nskk-analytics-analyze-trends))
148 (optimizations (nskk-analytics-suggest-optimizations)))
149
150 (nskk-analytics-report-create
151 :title &quot;NSKK Analytics Report&quot;
152 :period `(:start ,(nskk-analytics-report--get-period-start)
153 :end ,(current-time))
154 :metrics metrics
155 :charts (when nskk-analytics-report-include-graphs
156 (nskk-analytics-report--generate-charts metrics patterns))
157 :insights (nskk-analytics-report--generate-insights
158 metrics patterns trends optimizations))))

421(defun nskk-analytics-report--generate-bar-chart (title data)
422 &quot;棒グラフのSVGを生成する。
423
424TITLE: グラフタイトル
425DATA: データ配列
426戻り値: SVG文字列&quot;
427 (when (&gt; (length data) 0)
428 (let* ((max-val (apply #'max data))
429 (width 800)
430 (height 400)
431 (bar-width (/ width (float (length data))))
432 (scale (if (&gt; max-val 0) (/ height (float max-val)) 1)))
433 (concat
434 (format &quot;&lt;svg width=\&quot;%d\&quot; height=\&quot;%d\&quot; xmlns=\&quot;http://www.w3.org/2000/svg\&quot;&gt;\n&quot; width height)
435 (format &quot; &lt;text x=\&quot;%d\&quot; y=\&quot;20\&quot; font-size=\&quot;16\&quot; font-weight=\&quot;bold\&quot;&gt;%s&lt;/text&gt;\n&quot;
436 (/ width 2) title)
437 (let ((x 0)
438 (result &quot;&quot;))
439 (dotimes (i (length data))
440 (let* ((current-val (nth i data))
441 (bar-height (* current-val scale)))
442 (setq result
443 (concat result
444 (format &quot; &lt;rect x=\&quot;%d\&quot; y=\&quot;%d\&quot; width=\&quot;%d\&quot; height=\&quot;%d\&quot; fill=\&quot;#3498db\&quot;/&gt;\n&quot;
445 x (- height bar-height) (- bar-width 2) bar-height)
446 (format &quot; &lt;text x=\&quot;%d\&quot; y=\&quot;%d\&quot; font-size=\&quot;10\&quot; text-anchor=\&quot;middle\&quot;&gt;%d&lt;/text&gt;\n&quot;
447 (+ x (/ bar-width 2)) (- height 5) i)))
448 (setq x (+ x bar-width))))
449 result)
450 &quot;&lt;/svg&gt;&quot;))))

452(defun nskk-analytics-report--generate-pie-chart (title data)
453 &quot;円グラフのSVGを生成する。
454
455TITLE: グラフタイトル
456DATA: データのalist
457戻り値: SVG文字列&quot;
458 (let* ((total (apply #'+ (mapcar #'cdr data)))
459 (cx 200)
460 (cy 200)
461 (radius 150)
462 (start-angle 0))
463 (concat
464 &quot;&lt;svg width=\&quot;400\&quot; height=\&quot;400\&quot; xmlns=\&quot;http://www.w3.org/2000/svg\&quot;&gt;\n&quot;
465 (format &quot; &lt;text x=\&quot;200\&quot; y=\&quot;30\&quot; font-size=\&quot;16\&quot; font-weight=\&quot;bold\&quot; text-anchor=\&quot;middle\&quot;&gt;%s&lt;/text&gt;\n&quot;
466 title)
467 (mapconcat
468 (lambda (item)
469 (let* ((value (cdr item))
470 (percentage (/ value total))
471 (angle (* percentage 360))
472 (end-angle (+ start-angle angle))
473 (x1 (+ cx (* radius (cos (* start-angle (/ float-pi 180))))))
474 (y1 (+ cy (* radius (sin (* start-angle (/ float-pi 180))))))
475 (x2 (+ cx (* radius (cos (* end-angle (/ float-pi 180))))))
476 (y2 (+ cy (* radius (sin (* end-angle (/ float-pi 180))))))
477 (large-arc (if (&gt; angle 180) 1 0)))
478 (prog1
479 (format &quot; &lt;path d=\&quot;M %d,%d L %d,%d A %d,%d 0 %d,1 %d,%d Z\&quot; fill=\&quot;%s\&quot;/&gt;\n&quot;
480 cx cy x1 y1 radius radius large-arc x2 y2
481 (if (string= (car item) &quot;Correct&quot;) &quot;#2ecc71&quot; &quot;#e74c3c&quot;))
482 (setq start-angle end-angle))))
483 data
484 &quot;&quot;)
485 &quot;&lt;/svg&gt;&quot;)))

489(defun nskk-analytics-report--generate-insights (metrics patterns trends optimizations)
490 &quot;自動で洞察を生成する。
491
492METRICS: メトリクス構造体
493PATTERNS: パターンのalist
494TRENDS: トレンドのalist
495OPTIMIZATIONS: 最適化提案のリスト
496戻り値: 洞察文字列のリスト&quot;
497 (let ((insights nil))
498
499 ;; 精度に関する洞察
500 (let ((accuracy (nskk-analytics-metrics-accuracy metrics)))
501 (cond
502 ((&gt; accuracy 0.95)
503 (push &quot;Excellent conversion accuracy! Keep up the good work.&quot; insights))
504 ((&gt; accuracy 0.8)
505 (push &quot;Good accuracy, but there's room for improvement. Review error patterns.&quot; insights))
506 (t
507 (push &quot;Accuracy is low. Consider reviewing your dictionary or input method settings.&quot; insights))))
508
509 ;; 速度に関する洞察
510 (let ((speed (nskk-analytics-metrics-speed metrics)))
511 (cond
512 ((&lt; speed 0.05)
513 (push &quot;Lightning-fast conversions! Performance is optimal.&quot; insights))
514 ((&lt; speed 0.1)
515 (push &quot;Good conversion speed. Consider cache optimization for further improvement.&quot; insights))
516 (t
517 (push &quot;Conversion speed could be improved. Check for optimization suggestions.&quot; insights))))
518
519 ;; トレンドに関する洞察
520 (when trends
521 (let ((acc-trend (alist-get 'accuracy trends)))
522 (pcase acc-trend
523 ('increasing
524 (push &quot;Your accuracy is improving over time. Great progress!&quot; insights))
