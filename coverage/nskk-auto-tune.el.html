130 &quot;自動チューニングを有効にする。&quot;
131 (interactive)

136 (nskk-auto-tune--initialize-params)

142 #'nskk-auto-tune-run))
143
144 (setq nskk-auto-tune-enabled t)

146 (message &quot;NSKK auto-tuning enabled (interval: %.0fs)&quot; nskk-auto-tune-interval))

160;;;###autoload
161(defun nskk-auto-tune-run ()
162 &quot;チューニングを1回実行する。&quot;
163 (interactive)
164 (message &quot;NSKK auto-tuning: Starting tuning cycle...&quot;)
165
166 ;; プロファイリング実行
167 (let ((performance-metrics (nskk-auto-tune--measure-performance)))
168
169 ;; スコア計算
170 (let ((score (nskk-auto-tune--calculate-score performance-metrics)))
171
172 ;; 履歴に記録
173 (push (list (current-time)
174 :params (copy-sequence nskk-auto-tune--current-params)
175 :metrics performance-metrics
176 :score score)
177 nskk-auto-tune--history)
178
179 ;; 最高スコア更新チェック
180 (when (or (null nskk-auto-tune--best-score)
181 (&lt; score nskk-auto-tune--best-score))
182 (setq nskk-auto-tune--best-score score
183 nskk-auto-tune--best-params (copy-sequence nskk-auto-tune--current-params))
184 (message &quot;NSKK auto-tuning: New best score: %.3f&quot; score))
185
186 ;; 学習データ更新
187 (when nskk-auto-tune-learning-enabled
188 (nskk-auto-tune--update-learning-data nskk-auto-tune--current-params score))
189
190 ;; 次回のパラメータ決定
191 (nskk-auto-tune--adjust-parameters performance-metrics score)
192
193 (message &quot;NSKK auto-tuning: Cycle completed (score: %.3f)&quot; score))))

197(defun nskk-auto-tune--initialize-params ()
198 &quot;パラメータを初期化する。&quot;
199 (setq nskk-auto-tune--current-params
200 (list :cache-size (if (boundp 'nskk-cache-default-capacity)
201 nskk-cache-default-capacity
202 1000)
203 :gc-cons-threshold gc-cons-threshold
204 :profiler-sampling-interval (if (boundp 'nskk-profiler-sampling-interval)
205 nskk-profiler-sampling-interval
206 1000000))))

208(defun nskk-auto-tune--adjust-parameters (metrics score)
209 &quot;パフォーマンスメトリクスに基づいてパラメータを調整する。
210
211METRICS: パフォーマンスメトリクス
212SCORE: 現在のスコア&quot;
213 (let ((new-params (copy-sequence nskk-auto-tune--current-params)))
214
215 ;; GC頻度が高い場合、gc-cons-thresholdを増やす
216 (when (&gt; (plist-get metrics :gc-frequency) 5.0)
217 (let* ((current-gc-threshold (plist-get new-params :gc-cons-threshold))
218 (new-gc-threshold (min (* current-gc-threshold 2)
219 nskk-auto-tune-gc-cons-threshold-max)))
220 (plist-put new-params :gc-cons-threshold new-gc-threshold)
221 (message &quot;NSKK auto-tuning: Increasing gc-cons-threshold to %d&quot; new-gc-threshold)))
222
223 ;; GC頻度が低すぎる場合、gc-cons-thresholdを減らす
224 (when (&lt; (plist-get metrics :gc-frequency) 0.1)
225 (let* ((current-gc-threshold (plist-get new-params :gc-cons-threshold))
226 (new-gc-threshold (max (/ current-gc-threshold 2)
227 nskk-auto-tune-gc-cons-threshold-min)))
228 (plist-put new-params :gc-cons-threshold new-gc-threshold)
229 (message &quot;NSKK auto-tuning: Decreasing gc-cons-threshold to %d&quot; new-gc-threshold)))
230
231 ;; キャッシュヒット率が低い場合、キャッシュサイズを増やす
232 (when (and (plist-get metrics :cache-hit-rate)
233 (&lt; (plist-get metrics :cache-hit-rate) 0.7))
234 (let* ((current-cache-size (plist-get new-params :cache-size))
235 (new-cache-size (min (floor (* current-cache-size 1.5))
236 nskk-auto-tune-cache-size-max)))
237 (plist-put new-params :cache-size new-cache-size)
238 (message &quot;NSKK auto-tuning: Increasing cache size to %d&quot; new-cache-size)))
239
240 ;; 学習ベースの調整
241 (when nskk-auto-tune-learning-enabled
242 (setq new-params (nskk-auto-tune--apply-learning new-params score)))
243
244 ;; パラメータ適用
245 (nskk-auto-tune--apply-parameters new-params)
246 (setq nskk-auto-tune--current-params new-params)))

248(defun nskk-auto-tune--apply-parameters (params)
249 &quot;パラメータを実際に適用する。
250
251PARAMS: 適用するパラメータのplist&quot;
252 ;; gc-cons-threshold適用
253 (let ((gc-threshold (plist-get params :gc-cons-threshold)))
254 (when gc-threshold
255 (setq gc-cons-threshold gc-threshold)))
256
257 ;; キャッシュサイズ適用（nskk-cacheが存在する場合）
258 (when (boundp 'nskk-cache-default-capacity)
259 (let ((cache-size (plist-get params :cache-size)))
260 (when cache-size
261 (setq nskk-cache-default-capacity cache-size))))
262
263 ;; プロファイラーサンプリング間隔適用
264 (when (boundp 'nskk-profiler-sampling-interval)
265 (let ((sampling-interval (plist-get params :profiler-sampling-interval)))
266 (when sampling-interval
267 (setq nskk-profiler-sampling-interval sampling-interval)))))

271(defun nskk-auto-tune--measure-performance ()
272 &quot;現在のパフォーマンスを測定する。
273
274返り値: plist
275 :avg-conversion-time - 平均変換時間（秒）
276 :gc-frequency - GC頻度（回/秒）
277 :memory-usage - メモリ使用量（cons cells）
278 :cache-hit-rate - キャッシュヒット率&quot;
279 (let ((metrics nil))
280
281 ;; 変換時間測定（nskk-converterが存在する場合）
282 (when (fboundp 'nskk-convert-romaji)
283 (let ((total-time 0.0)
284 (iterations 100))
285 (dotimes (_ iterations)
286 (let ((start-time (current-time)))
287 (ignore-errors (nskk-convert-romaji &quot;konnnichiha&quot;))
288 (setq total-time (+ total-time
289 (float-time (time-subtract (current-time) start-time))))))
290 (setq metrics (plist-put metrics :avg-conversion-time (/ total-time iterations)))))
291
292 ;; GC統計
293 (setq metrics (plist-put metrics :gc-frequency
294 (/ (float gcs-done)
295 (max 1.0 (float-time (time-since before-init-time))))))
296
297 ;; メモリ使用量
298 (let ((mem-counts (memory-use-counts)))
299 (setq metrics (plist-put metrics :memory-usage (nth 0 mem-counts))))
300
301 ;; キャッシュヒット率（キャッシュが存在する場合）
302 (when (boundp 'nskk-cache-default-capacity)
303 ;; 実際のキャッシュインスタンスがあれば、その統計を取得
304 ;; ここでは簡易的にデフォルト値を返す
305 (setq metrics (plist-put metrics :cache-hit-rate 0.5)))
306
307 metrics))
308
309(defun nskk-auto-tune--calculate-score (metrics)
310 &quot;パフォーマンスメトリクスからスコアを計算する。
311
312METRICS: パフォーマンスメトリクス
313返り値: スコア（小さいほど良い）&quot;
314 (let ((score 0.0))
315
316 ;; 変換時間（重み: 10.0）
317 (when (plist-get metrics :avg-conversion-time)
318 (setq score (+ score (* 10.0 (plist-get metrics :avg-conversion-time)))))
319
320 ;; GC頻度（重み: 1.0、理想は1-2回/秒）
321 (when (plist-get metrics :gc-frequency)
322 (let ((gc-freq (plist-get metrics :gc-frequency)))
323 (setq score (+ score (* 1.0 (abs (- gc-freq 1.5)))))))
324
325 ;; メモリ使用量（重み: 0.000001、100万cons cellsで1.0）
326 (when (plist-get metrics :memory-usage)
327 (setq score (+ score (* 0.000001 (plist-get metrics :memory-usage)))))
328
329 ;; キャッシュヒット率（重み: -5.0、高いほど良い）
330 (when (plist-get metrics :cache-hit-rate)
331 (setq score (+ score (* -5.0 (plist-get metrics :cache-hit-rate)))))
332
333 score))

337(defun nskk-auto-tune--update-learning-data (params score)
338 &quot;学習データを更新する。
339
340PARAMS: パラメータ
341SCORE: スコア&quot;
342 (push (cons (copy-sequence params) score) nskk-auto-tune--learning-data)
343
344 ;; 履歴サイズ制限（最新100件のみ保持）
345 (when (&gt; (length nskk-auto-tune--learning-data) 100)
346 (setq nskk-auto-tune--learning-data
347 (cl-subseq nskk-auto-tune--learning-data 0 100))))
348
349(defun nskk-auto-tune--apply-learning (params current-score)
350 &quot;学習データに基づいてパラメータを調整する。
351
352PARAMS: 現在のパラメータ
353CURRENT-SCORE: 現在のスコア
354返り値: 調整後のパラメータ&quot;
355 (if (&lt; (length nskk-auto-tune--learning-data) 10)
356 ;; データ不足の場合は現在のパラメータをそのまま返す
357 params
358
359 ;; 良好なスコアを持つパラメータの平均を計算
360 (let* ((sorted-data (sort (copy-sequence nskk-auto-tune--learning-data)
361 (lambda (a b) (&lt; (cdr a) (cdr b)))))
362 (top-10-percent (cl-subseq sorted-data 0
363 (max 1 (/ (length sorted-data) 10))))
364 (avg-cache-size 0)
365 (avg-gc-threshold 0)
366 (count (length top-10-percent)))
367
368 ;; 平均計算
369 (dolist (entry top-10-percent)
370 (let ((p (car entry)))
371 (setq avg-cache-size (+ avg-cache-size (plist-get p :cache-size)))
372 (setq avg-gc-threshold (+ avg-gc-threshold (plist-get p :gc-cons-threshold)))))
373
374 (setq avg-cache-size (/ avg-cache-size count))
375 (setq avg-gc-threshold (/ avg-gc-threshold count))
376
377 ;; 現在の値と学習した平均値を加重平均
378 (let ((new-params (copy-sequence params))
379 (learning-weight (if nskk-auto-tune-aggressive 0.7 0.3)))
380 (plist-put new-params :cache-size
381 (floor (+ (* learning-weight avg-cache-size)
382 (* (- 1.0 learning-weight) (plist-get params :cache-size)))))
383 (plist-put new-params :gc-cons-threshold
384 (floor (+ (* learning-weight avg-gc-threshold)
385 (* (- 1.0 learning-weight) (plist-get params :gc-cons-threshold)))))
386 new-params))))

404(defun nskk-auto-tune-ab-test-run ()
405 &quot;A/Bテストを実行する。&quot;
406 (interactive)
407 (unless nskk-auto-tune--ab-test-configs
408 (error &quot;No A/B test configs set. Use `nskk-auto-tune-ab-test-setup' first&quot;))
409
410 (message &quot;NSKK auto-tuning: Starting A/B test...&quot;)
411
412 (dolist (config nskk-auto-tune--ab-test-configs)
413 (let ((name (plist-get config :name)))
414 (message &quot;Testing config: %s&quot; name)
415
416 ;; パラメータ適用
417 (nskk-auto-tune--apply-parameters config)
418
419 ;; パフォーマンス測定
420 (let ((metrics (nskk-auto-tune--measure-performance)))
421 (let ((score (nskk-auto-tune--calculate-score metrics)))
422 (push (list :name name
423 :config config
424 :metrics metrics
425 :score score)
426 nskk-auto-tune--ab-test-results)))))
427
428 ;; 結果をソート（スコアが小さい順）
429 (setq nskk-auto-tune--ab-test-results
430 (sort nskk-auto-tune--ab-test-results
431 (lambda (a b) (&lt; (plist-get a :score)
432 (plist-get b :score)))))
433
434 (message &quot;NSKK auto-tuning: A/B test completed&quot;)
435 (nskk-auto-tune-ab-test-report))
