29;; 特徴:
30;; - SKK形式辞書の完全なパース
31;; - エンコーディング自動検出（UTF-8/EUC-JP）
32;; - 注釈（アノテーション）の抽出

34;; - エラーハンドリング（厳格モード/寛容モード）

37;; SKK辞書形式:
38;;
39;; ;; -*- coding: utf-8 -*-
40;; ;; okuri-ari entries.
41;; わたr /渡/航/
42;; わすr /忘/
43;; ;; okuri-nasi entries.
44;; かんじ /漢字/幹事/
45;; あい /愛;love/哀;sorrow/藍;indigo/
46;;
47;; 使用例:
48;;
49;; (require 'nskk-dict-parser)
50;;
51;; ;; 辞書ファイルをパース
52;; (let ((dict (nskk-parse-dictionary &quot;~/.skk/jisyo&quot;)))
53;; (message &quot;Header: %S&quot; (nskk-dict-header dict))
54;; (message &quot;Okuri-ari entries: %d&quot; (length (nskk-dict-okuri-ari dict)))
55;; (message &quot;Okuri-nasi entries: %d&quot; (length (nskk-dict-okuri-nasi dict))))
56;;
57;; ;; エントリを解析
58;; (nskk-parse-entry &quot;かんじ /漢字/幹事/&quot;)

142(defun nskk-dict-parser-detect-encoding (file-path)
143 &quot;FILE-PATH のエンコーディングを検出する。
144UTF-8またはEUC-JPを返す。&quot;
145 ;; ステップ1: coding宣言をチェック
146 (let ((coding-from-header (nskk-dict-parser--detect-from-header file-path)))
147 (if coding-from-header
148 coding-from-header
149 ;; ステップ2: バイトパターン解析
150 (nskk-dict-parser--detect-from-bytes file-path))))
151
152(defun nskk-dict-parser--detect-from-header (file-path)
153 &quot;ファイルヘッダーからエンコーディングを検出する。&quot;
154 (with-temp-buffer
155 (insert-file-contents file-path nil 0 1000)
156 (goto-char (point-min))
157 (when (re-search-forward &quot;-\\*-.*coding:\\s-*\\([-a-z0-9]+\\)&quot; nil t)
158 (let ((coding-str (match-string 1)))
159 (cond
160 ((string-match-p &quot;utf-?8&quot; coding-str) 'utf-8)
161 ((string-match-p &quot;euc-?jp&quot; coding-str) 'euc-jp)
162 (t nil))))))
163
164(defun nskk-dict-parser--detect-from-bytes (file-path)
165 &quot;バイトパターンからエンコーディングを検出する。&quot;
166 (with-temp-buffer
167 (set-buffer-multibyte nil)
168 (insert-file-contents-literally file-path nil 0 8192)
169 (let ((utf8-score 0)
170 (eucjp-score 0))
171 (goto-char (point-min))
172 (while (not (eobp))
173 (let ((byte (char-after)))
174 ;; UTF-8パターン検出
175 (cond
176 ;; UTF-8 2バイトシーケンス
177 ((and (&gt;= byte #xC0) (&lt;= byte #xDF))
178 (when (nskk-dict-parser--valid-utf8-continuation-p (+ (point) 1))
179 (setq utf8-score (+ utf8-score 10))))
180 ;; UTF-8 3バイトシーケンス
181 ((and (&gt;= byte #xE0) (&lt;= byte #xEF))
182 (when (and (nskk-dict-parser--valid-utf8-continuation-p (+ (point) 1))
183 (nskk-dict-parser--valid-utf8-continuation-p (+ (point) 2)))
184 (setq utf8-score (+ utf8-score 10))))
185 ;; EUC-JPパターン検出
186 ((and (&gt;= byte #xA1) (&lt;= byte #xFE))
187 (when (and (&lt; (+ (point) 1) (point-max))
188 (let ((next-byte (char-after (+ (point) 1))))
189 (and (&gt;= next-byte #xA1) (&lt;= next-byte #xFE))))
190 (setq eucjp-score (+ eucjp-score 10)))))
191 (forward-char 1)))
192 ;; スコアに基づく判定
193 (cond
194 ((&gt; utf8-score eucjp-score) 'utf-8)
195 ((&gt; eucjp-score utf8-score) 'euc-jp)
196 (t 'utf-8))))) ; デフォルトはUTF-8
197
198(defun nskk-dict-parser--valid-utf8-continuation-p (pos)
199 &quot;POSの位置がUTF-8継続バイトか判定する。&quot;
200 (when (&lt; pos (point-max))
201 (let ((byte (char-after pos)))
202 (and (&gt;= byte #x80) (&lt;= byte #xBF)))))

206;;;###autoload
207(defun nskk-parse-dictionary (file-path)
208 &quot;FILE-PATH のSKK辞書をパースする。
209戻り値: `nskk-dict' 構造体&quot;
210 (unless (file-exists-p file-path)
211 (signal 'nskk-dict-io-file-not-found (list file-path)))
212
213 (let* ((encoding (nskk-dict-parser-detect-encoding file-path))
214 (dict (nskk-dict--create :file-path file-path
215 :encoding encoding))
216 (current-section 'header))
217 (when nskk-dict-parser-verbose
218 (message &quot;Parsing %s (encoding: %s)&quot; file-path encoding))
219
220 (with-temp-buffer
221 (let ((coding-system-for-read encoding))
222 (insert-file-contents file-path))
223
224 (goto-char (point-min))
225 (let ((line-number 0))
226 (while (not (eobp))
227 (setq line-number (1+ line-number))
228 (let ((line (buffer-substring-no-properties
229 (line-beginning-position)
230 (line-end-position))))
231 (condition-case err
232 (cond
233 ;; 空行
234 ((string-match-p &quot;^\\s-*$&quot; line)
235 nil)
236 ;; コメント行
237 ((string-match &quot;^;;\\s-*\\(.*\\)&quot; line)
238 (let ((comment (match-string 1 line)))
239 (cond
240 ;; 送り仮名ありマーカー
241 ((string-match-p &quot;okuri-ari\\s-+entries\\.&quot; comment)
242 (setq current-section 'okuri-ari))
243 ;; 送り仮名なしマーカー
244 ((string-match-p &quot;okuri-nasi\\s-+entries\\.&quot; comment)
245 (setq current-section 'okuri-nasi))
246 ;; 通常のコメント（ヘッダーに追加）
247 ((eq current-section 'header)
248 (push line (nskk-dict-header dict))))))
249 ;; エントリ
250 (t
251 (let ((entry (nskk-parse-entry line)))
252 (pcase current-section
253 ('okuri-ari
254 (push entry (nskk-dict-okuri-ari dict)))
255 ('okuri-nasi
256 (push entry (nskk-dict-okuri-nasi dict)))
257 ('header
258 (nskk-dict-parser--log-error
259 dict line-number line
260 &quot;Unexpected entry in header section&quot;
261 'warning))))))
262 (error
263 (nskk-dict-parser--handle-error
264 dict line-number line
265 (error-message-string err)))))
266 (forward-line 1))))
267
268 ;; リストを逆順に（ファイル順を保持）
269 (setf (nskk-dict-header dict) (nreverse (nskk-dict-header dict)))
270 (setf (nskk-dict-okuri-ari dict) (nreverse (nskk-dict-okuri-ari dict)))
271 (setf (nskk-dict-okuri-nasi dict) (nreverse (nskk-dict-okuri-nasi dict)))
272
273 (when nskk-dict-parser-verbose
274 (message &quot;Parsed: %d okuri-ari, %d okuri-nasi entries&quot;
275 (length (nskk-dict-okuri-ari dict))
276 (length (nskk-dict-okuri-nasi dict))))
277
278 dict))

281(defun nskk-parse-entry (line)
282 &quot;LINE をSKKエントリとして解析する。
283戻り値: `nskk-dict-entry' 構造体&quot;
284 ;; エントリ形式: &quot;見出し語 /候補1/候補2/...&quot;
285 (unless (string-match &quot;^\\([^ \t]+\\)\\s-+\\(/.*/$\\)&quot; line)
286 (signal 'nskk-dict-parse-invalid-format (list line)))
287
288 (let* ((midashi (match-string 1 line))
289 (candidate-part (match-string 2 line))
290 (candidates (nskk-parse-candidates candidate-part)))
291
292 ;; 候補数チェック
293 (when (&gt; (length candidates) nskk-dict-parser-max-candidates)
294 (warn &quot;Entry has too many candidates (%d &gt; %d): %s&quot;
295 (length candidates) nskk-dict-parser-max-candidates midashi))
296
297 (nskk-dict-entry--create :midashi midashi
298 :candidates candidates)))
299
300(defun nskk-parse-candidates (candidate-part)
301 &quot;CANDIDATE-PART から候補リストを解析する。
302候補リスト形式: \&quot;/候補1/候補2;注釈2/.../\&quot;
303戻り値: ((word . annotation) ...) のリスト&quot;
304 (unless (and (string-prefix-p &quot;/&quot; candidate-part)
305 (string-suffix-p &quot;/&quot; candidate-part))
306 (signal 'nskk-dict-parse-invalid-format
307 (list (format &quot;Candidate list must be enclosed in slashes: %s&quot; candidate-part))))
308
309 ;; 先頭と末尾の&quot;/&quot;を除去
310 (let* ((content (substring candidate-part 1 -1))
311 (raw-candidates (split-string content &quot;/&quot; t))
312 (candidates nil))
313
314 (dolist (raw-cand raw-candidates)
315 (if (string-match &quot;^\\([^;]+\\);\\(.+\\)$&quot; raw-cand)
316 ;; 注釈あり
317 (push (cons (match-string 1 raw-cand)
318 (match-string 2 raw-cand))
319 candidates)
320 ;; 注釈なし
321 (push (cons raw-cand nil) candidates)))
322
323 (nreverse candidates)))

327(defun nskk-dict-parser-validate-entry (entry)
328 &quot;ENTRY の妥当性を検証する。
329問題があれば警告を出力し、エラーメッセージのリストを返す。&quot;
330 (let ((errors nil))
331 ;; 見出し語の検証
332 (when (string-empty-p (nskk-dict-entry-midashi entry))
333 (push &quot;Empty midashi&quot; errors))
334
335 ;; 候補の検証
336 (when (null (nskk-dict-entry-candidates entry))
337 (push &quot;No candidates&quot; errors))
338
339 (dolist (cand (nskk-dict-entry-candidates entry))
340 (when (string-empty-p (car cand))
341 (push &quot;Empty candidate word&quot; errors)))
342
343 errors))

345(defun nskk-dict-parser-validate-okuri-ari-entry (entry)
346 &quot;送り仮名ありエントリ ENTRY の妥当性を検証する。&quot;
347 (let ((errors (nskk-dict-parser-validate-entry entry)))
348 ;; 送り仮名マーカー（ローマ字接尾辞）の検証
349 (unless (nskk-dict-parser--has-romaji-suffix-p
350 (nskk-dict-entry-midashi entry))
351 (push &quot;Okuri-ari entry must have romaji suffix&quot; errors))
352 errors))

354(defun nskk-dict-parser--has-romaji-suffix-p (midashi)
355 &quot;MIDASHI が送り仮名マーカー（ローマ字接尾辞）を持つか判定する。&quot;
356 (and (&gt; (length midashi) 0)
357 (let ((last-char (aref midashi (1- (length midashi)))))
358 (and (&gt;= last-char ?a) (&lt;= last-char ?z)))))

362(defun nskk-dict-parser--log-error (dict line-number line message severity)
363 &quot;エラーを記録する。&quot;
364 (let ((error-obj (nskk-parse-error--create
365 :line-number line-number
366 :line line
367 :message message
368 :severity severity)))
369 (push error-obj (nskk-dict-errors dict))
370 (when nskk-dict-parser-verbose
371 (message &quot;[%s] Line %d: %s&quot; severity line-number message))))

373(defun nskk-dict-parser--handle-error (dict line-number line message)
374 &quot;エラーを処理戦略に従って処理する。&quot;
375 (pcase nskk-dict-parser-error-strategy
376 ('strict
377 (signal 'nskk-dict-parse-error
378 (list (format &quot;Parse error at line %d: %s (Line: %s)&quot;
379 line-number message line))))
380 ('lenient
381 (nskk-dict-parser--log-error dict line-number line message 'error))
382 ('ignore
383 nil)))

387(defun nskk-dict-parser-entry-to-string (entry)
388 &quot;ENTRY を文字列形式に変換する。&quot;
389 (concat (nskk-dict-entry-midashi entry)
390 &quot; /&quot;
391 (mapconcat (lambda (cand)
392 (if (cdr cand)
393 (format &quot;%s;%s&quot; (car cand) (cdr cand))
394 (car cand)))
395 (nskk-dict-entry-candidates entry)
396 &quot;/&quot;)
397 &quot;/&quot;))
