291;;;###autoload
292(defun nskk-server-error-with-retry (request-fn success-callback error-callback
293 &optional max-retries)
294 &quot;リトライロジック付きでサーバーリクエストを実行する。
295
296引数:
297 REQUEST-FN - リクエスト関数（success-callback, error-callback -&gt; void）
298 SUCCESS-CALLBACK - 成功コールバック
299 ERROR-CALLBACK - エラーコールバック（最終的に失敗した場合）
300 MAX-RETRIES - 最大リトライ回数（省略時はnskk-server-error-max-retries）
301
302処理フロー:
303 1. リクエスト実行
304 2. 失敗時、指数バックオフでリトライ
305 3. 最大リトライ回数到達で最終エラーコールバック呼び出し&quot;
306 (let ((max-retries (or max-retries nskk-server-error-max-retries))
307 (retry-count 0))
308 (cl-labels ((retry ()
309 (funcall request-fn
310 ;; 成功コールバック
311 success-callback
312 ;; エラーコールバック（リトライ付き）
313 (lambda (error-msg)
314 (if (&lt; retry-count max-retries)
315 (let ((delay (* nskk-server-error-retry-delay
316 (expt nskk-server-error-retry-backoff
317 retry-count))))
318 (cl-incf retry-count)
319 (when nskk-server-error-verbose
320 (message &quot;Retrying (%d/%d) after %.2fs: %s&quot;
321 retry-count max-retries delay error-msg))
322 (run-at-time delay nil #'retry))
323 ;; 最大リトライ回数到達
324 (when nskk-server-error-verbose
325 (message &quot;Max retries reached: %s&quot; error-msg))
326 (funcall error-callback error-msg))))))
327 (retry))))

337引数:

339 PORT - サーバーポート

345

351 (unless (nskk-server-error--circuit-breaker-allow-request-p host port)

373 (message &quot;Falling back to local dict: %s&quot; error-msg))
374 (nskk-server-error--fallback-to-local-dict
