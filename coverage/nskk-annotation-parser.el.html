27;; このファイルはSKK注釈（アノテーション）のパーサーを実装します。
28;;
29;; 特徴:
30;; - SKK標準注釈形式（;で始まる）のパース
31;; - マルチメディア注釈（画像・音声パス）
32;; - 構造化注釈（JSON/plist形式）
33;; - 複数注釈形式の統合処理
34;; - エスケープ処理
35;;
36;; 注釈形式の例:
37;;
38;; 基本形式:
39;; あい /愛;love/哀;sorrow/
40;;
41;; マルチメディア:
42;; さくら /桜;img:/path/to/sakura.png/
43;; おと /音;snd:/path/to/sound.wav/
44;;
45;; 構造化（JSON風）:
46;; たんご /単語;{&quot;読み&quot;:&quot;たんご&quot;,&quot;品詞&quot;:&quot;名詞&quot;}/
47;;
48;; 複合:
49;; れい /例;example;img:/path/to/image.png/
50;;
51;; 使用例:
52;;
53;; (require 'nskk-annotation-parser)
54;;
55;; ;; 基本的な注釈のパース
56;; (nskk-parse-annotation &quot;love&quot;)
57;; ;; =&gt; (:text &quot;love&quot; :type simple)
58;;
59;; ;; マルチメディア注釈のパース
60;; (nskk-parse-annotation &quot;img:/path/to/image.png&quot;)
61;; ;; =&gt; (:text &quot;img:/path/to/image.png&quot; :type multimedia :media-type image :path &quot;/path/to/image.png&quot;)
62;;
63;; ;; 構造化注釈のパース
64;; (nskk-parse-annotation &quot;{\&quot;読み\&quot;:\&quot;たんご\&quot;}&quot;)
65;; ;; =&gt; (:text &quot;{\&quot;読み\&quot;:\&quot;たんご\&quot;}&quot; :type structured :data ((&quot;読み&quot; . &quot;たんご&quot;)))

126(defun nskk-parse-annotation (text)
127 &quot;TEXT を注釈としてパースする。
128戻り値: `nskk-annotation' 構造体&quot;
129 (when (and text (not (string-empty-p text)))
130 ;; 長さチェック
131 (when (&gt; (length text) nskk-annotation-parser-max-length)
132 (setq text (substring text 0 nskk-annotation-parser-max-length)))
133
134 ;; パース実行（優先順位順にチェック）
135 (cond
136 ;; セミコロンを含まないマルチメディア注釈
137 ((and nskk-annotation-parser-enable-multimedia
138 (nskk-annotation-parser--multimedia-p text)
139 (not (string-match-p &quot;;&quot; text)))
140 (nskk-annotation-parser--parse-multimedia text))
141
142 ;; セミコロンを含まない構造化注釈
143 ((and nskk-annotation-parser-enable-structured
144 (nskk-annotation-parser--structured-p text)
145 (not (string-match-p &quot;;&quot; text)))
146 (nskk-annotation-parser--parse-structured text))
147
148 ;; セミコロンを含む複合注釈
149 ;; ただし、マルチメディアまたは構造化が無効の場合でも、
150 ;; 各要素は有効な機能のみでパースされる
151 ((string-match-p &quot;;&quot; text)
152 (nskk-annotation-parser--parse-compound text))
153
154 ;; 上記に該当しない場合は単純注釈
155 (t
156 (nskk-annotation-parser--parse-simple text)))))

170 &quot;TEXT がマルチメディア注釈かどうかを判定する。&quot;
171 (cl-some (lambda (protocol)

173 nskk-annotation-parser-multimedia-protocols))

175(defun nskk-annotation-parser--parse-multimedia (text)
176 &quot;TEXT をマルチメディア注釈としてパースする。
177形式: \&quot;img:/path/to/file\&quot; または \&quot;http://example.com/file\&quot;&quot;
178 (let* ((media-type nil)
179 (path nil))
180 (cond
181 ;; 画像
182 ((string-prefix-p &quot;img:&quot; text)
183 (setq media-type 'image)
184 (setq path (substring text 4)))
185
186 ;; 音声
187 ((string-prefix-p &quot;snd:&quot; text)
188 (setq media-type 'sound)
189 (setq path (substring text 4)))
190
191 ;; 動画
192 ((string-prefix-p &quot;video:&quot; text)
193 (setq media-type 'video)
194 (setq path (substring text 6)))
195
196 ;; ファイル
197 ((string-prefix-p &quot;file:&quot; text)
198 (setq media-type 'file)
199 (setq path (substring text 5)))
200
201 ;; URL
202 ((or (string-prefix-p &quot;http:&quot; text)
203 (string-prefix-p &quot;https:&quot; text))
204 (setq media-type 'url)
205 (setq path text))
206
207 ;; その他（デフォルト）
208 (t
209 (setq media-type 'unknown)
210 (setq path text)))
211
212 (nskk-annotation--create
213 :text text
214 :type 'multimedia
215 :media-type media-type
216 :path path
217 :description (format &quot;[%s] %s&quot; media-type path))))

221(defun nskk-annotation-parser--structured-p (text)
222 &quot;TEXT が構造化注釈かどうかを判定する。&quot;
223 (or (and (string-prefix-p &quot;{&quot; text) (string-suffix-p &quot;}&quot; text))
224 (and (string-prefix-p &quot;[&quot; text) (string-suffix-p &quot;]&quot; text))
225 (string-prefix-p &quot;(&quot; text)))
226
227(defun nskk-annotation-parser--parse-structured (text)
228 &quot;TEXT を構造化注釈としてパースする。
229JSON形式またはplist形式をサポート。&quot;
230 (let ((data nil)
231 (description nil))
232 (condition-case err
233 (cond
234 ;; JSON形式
235 ((or (string-prefix-p &quot;{&quot; text) (string-prefix-p &quot;[&quot; text))
236 (setq data (nskk-annotation-parser--parse-json text))
237 (setq description (nskk-annotation-parser--format-structured data)))
238
239 ;; plist形式（簡易）
240 ((string-prefix-p &quot;(&quot; text)
241 (setq data (nskk-annotation-parser--parse-plist text))
242 (setq description (nskk-annotation-parser--format-structured data)))
243
244 ;; パース不可
245 (t
246 (setq description text)))
247 (error
248 ;; パースエラー時は単純注釈として扱う
249 (setq description text)))
250
251 (nskk-annotation--create
252 :text text
253 :type 'structured
254 :data data
255 :description (or description text))))

257(defun nskk-annotation-parser--parse-json (text)
258 &quot;TEXT をJSON形式としてパースする。&quot;
259 (condition-case nil
260 (let ((json-object-type 'alist)
261 (json-array-type 'list)
262 (json-key-type 'string))
263 (json-read-from-string text))
264 (error nil)))
265
266(defun nskk-annotation-parser--parse-plist (text)
267 &quot;TEXT をplist形式としてパースする（簡易実装）。&quot;
268 (condition-case nil
269 (let* ((trimmed (string-trim text &quot;(&quot; &quot;)&quot;))
270 (pairs (split-string trimmed))
271 (result nil))
272 (while pairs
273 (let ((key (pop pairs))
274 (val (pop pairs)))
275 (when (and key val)
276 (push (cons key val) result))))
277 (nreverse result))
278 (error nil)))

280(defun nskk-annotation-parser--format-structured (data)
281 &quot;構造化データ DATA を読みやすい文字列に整形する。&quot;
282 (cond
283 ((null data) &quot;&quot;)
284 ((listp data)
285 (mapconcat (lambda (pair)
286 (format &quot;%s: %s&quot;
287 (if (stringp (car pair)) (car pair) (format &quot;%s&quot; (car pair)))
288 (if (stringp (cdr pair)) (cdr pair) (format &quot;%s&quot; (cdr pair)))))
289 data &quot;, &quot;))
290 (t (format &quot;%s&quot; data))))

294(defun nskk-annotation-parser--parse-compound (text)
295 &quot;TEXT を複合注釈としてパースする。
296セミコロンで区切られた複数の注釈要素を処理。&quot;
297 (let* ((parts (split-string text &quot;;&quot; t &quot;\\s-+&quot;))
298 (components (mapcar #'nskk-parse-annotation parts))
299 (descriptions (mapcar (lambda (comp)
300 (or (nskk-annotation-description comp)
301 (nskk-annotation-text comp)))
302 components)))
303 (nskk-annotation--create
304 :text text
305 :type 'compound
306 :components components
307 :description (string-join descriptions &quot;; &quot;))))

311(defun nskk-annotation-parser-escape (text)
312 &quot;TEXT 内の特殊文字をエスケープする。&quot;
313 (when text
314 (let ((result text))
315 ;; セミコロンのエスケープ
316 (setq result (replace-regexp-in-string &quot;;&quot; &quot;\\\\;&quot; result))
317 ;; スラッシュのエスケープ
318 (setq result (replace-regexp-in-string &quot;/&quot; &quot;\\\\/&quot; result))
319 result)))
320
321(defun nskk-annotation-parser-unescape (text)
322 &quot;TEXT 内のエスケープ文字を元に戻す。&quot;
323 (when text
324 (let ((result text))
325 ;; エスケープ解除
326 (setq result (replace-regexp-in-string &quot;\\\\;&quot; &quot;;&quot; result))
327 (setq result (replace-regexp-in-string &quot;\\\\/&quot; &quot;/&quot; result))
328 result)))

378(defun nskk-annotation-parser-statistics (annotations)
379 &quot;ANNOTATIONS のリストから統計情報を生成する。&quot;
380 (let ((total (length annotations))
381 (simple 0)
382 (multimedia 0)
383 (structured 0)
384 (compound 0))
385 (dolist (ann annotations)
386 (pcase (nskk-annotation-type ann)
387 ('simple (cl-incf simple))
388 ('multimedia (cl-incf multimedia))
389 ('structured (cl-incf structured))
390 ('compound (cl-incf compound))))
391 (list :total total
392 :simple simple
393 :multimedia multimedia
394 :structured structured
395 :compound compound)))
