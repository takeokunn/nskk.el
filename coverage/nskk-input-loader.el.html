92;;; 入力方式のロード
93
94(defun nskk-input-loader-load (method-name &optional force)
95 &quot;METHOD-NAMEの入力方式をロードする。
96FORCEが非nilの場合、既にロード済みでも再ロードする。&quot;
97 (when (or force
98 (not (nskk-input-loader-loaded-p method-name)))
99 (let ((start-time (when nskk-input-loader-measure-time
100 (current-time))))
101
102 ;; 依存関係を先にロード
103 (nskk-input-loader--load-dependencies method-name)
104
105 ;; ライブラリをロード
106 (let ((lib-name (nskk-input-loader--get-library-name method-name)))
107 (when lib-name
108 (require lib-name)
109
110 ;; 登録関数を呼び出し
111 (when-let ((register-fn (nskk-input-loader--get-register-function method-name)))
112 (funcall register-fn))
113
114 ;; ロード済みとしてマーク
115 (cl-pushnew method-name nskk-input-loader-loaded-methods)
116
117 ;; ロード時間を記録
118 (when nskk-input-loader-measure-time
119 (let ((elapsed-time (float-time (time-subtract (current-time) start-time))))
120 (setf (alist-get method-name nskk-input-loader-load-times) elapsed-time)
121 (message &quot;Loaded %s in %.3f ms&quot; method-name (* elapsed-time 1000))))
122
123 t)))))
124
125(defun nskk-input-loader-loaded-p (method-name)
126 &quot;METHOD-NAMEが既にロード済みかどうかを返す。&quot;
127 (memq method-name nskk-input-loader-loaded-methods))
128
129(defun nskk-input-loader-unload (method-name)
130 &quot;METHOD-NAMEの入力方式をアンロードする。&quot;
131 (when (nskk-input-loader-loaded-p method-name)
132 (let ((lib-name (nskk-input-loader--get-library-name method-name)))
133 (when lib-name
134 (unload-feature lib-name t)
135 (setq nskk-input-loader-loaded-methods
136 (delq method-name nskk-input-loader-loaded-methods))
137 (message &quot;Unloaded %s&quot; method-name)))))
138
139;;; 依存関係の解決

141(defun nskk-input-loader--load-dependencies (method-name)
142 &quot;METHOD-NAMEの依存関係をロードする。&quot;
143 (when-let ((deps (alist-get method-name nskk-input-loader-dependencies)))
144 (dolist (dep deps)
145 (unless (nskk-input-loader-loaded-p dep)
146 (nskk-input-loader-load dep)))))

154(defun nskk-input-loader--get-register-function (method-name)
155 &quot;METHOD-NAMEの登録関数を取得する。&quot;
156 (let ((fn-name (intern (format &quot;nskk-input-%s-register&quot; method-name))))
157 (when (fboundp fn-name)
158 fn-name)))

162;;;###autoload
163(defun nskk-input-loader-setup-autoloads ()
164 &quot;全ての入力方式のautoload設定を行う。&quot;
165 (interactive)
166 (dolist (entry nskk-input-loader-autoload-alist)
167 (let* ((method-name (car entry))
168 (lib-name (cdr entry))
169 (lookup-fn (intern (format &quot;nskk-input-%s-lookup&quot; method-name)))
170 (register-fn (intern (format &quot;nskk-input-%s-register&quot; method-name))))
171
172 ;; 検索関数のautoload
173 (autoload lookup-fn (symbol-name lib-name)
174 (format &quot;Lookup function for %s input method.&quot; method-name)
175 nil nil)
176
177 ;; 登録関数のautoload
178 (autoload register-fn (symbol-name lib-name)
179 (format &quot;Register function for %s input method.&quot; method-name)
180 t nil))))

244(defun nskk-input-loader-benchmark ()
245 &quot;全ての入力方式のロード時間をベンチマークする。&quot;
246 (interactive)
247 (let ((results nil))
248 ;; 全てアンロード
249 (nskk-input-loader-unload-all)
250
251 ;; 各入力方式をロードして測定
252 (dolist (entry nskk-input-loader-autoload-alist)
253 (let ((method (car entry)))
254 (garbage-collect) ; GCの影響を最小化
255 (nskk-input-loader-load method)
256 (push (cons method (alist-get method nskk-input-loader-load-times))
257 results)))
258
259 ;; 結果を表示
260 (with-output-to-temp-buffer &quot;*NSKK Loader Benchmark*&quot;
261 (princ &quot;NSKK Input Method Load Time Benchmark\n&quot;)
262 (princ &quot;=====================================\n\n&quot;)
263 (dolist (result (nreverse results))
264 (princ (format &quot;%-10s : %.3f ms\n&quot;
265 (car result)
266 (* (cdr result) 1000))))
267 (princ (format &quot;\nTotal: %.3f ms\n&quot;
268 (* (cl-reduce #'+ (mapcar #'cdr results)) 1000))))))
