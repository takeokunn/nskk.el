83(defun nskk-parallel-search--partition-hash (entries num-partitions)
84 &quot;ENTRIES をハッシュベースで NUM-PARTITIONS に分割する。&quot;
85 (let ((partitions (make-vector num-partitions nil)))
86 (dolist (entry entries)
87 (let* ((key (car entry))
88 (hash (sxhash key))
89 (partition-idx (mod hash num-partitions)))
90 (push entry (aref partitions partition-idx))))
91 (cl-coerce partitions 'list)))
92
93(defun nskk-parallel-search--partition-range (entries num-partitions)
94 &quot;ENTRIES を範囲ベースで NUM-PARTITIONS に分割する。&quot;
95 (let* ((sorted-entries (sort (copy-sequence entries)
96 (lambda (a b)
97 (string&lt; (car a) (car b)))))
98 (total (length sorted-entries))
99 (partition-size (/ total num-partitions))
100 (partitions nil))
101 (dotimes (i num-partitions)
102 (let* ((start (* i partition-size))
103 (end (if (= i (1- num-partitions))
104 total
105 (* (1+ i) partition-size)))
106 (partition (cl-subseq sorted-entries start end)))
107 (push partition partitions)))
108 (nreverse partitions)))

110(defun nskk-parallel-search--partition (entries strategy num-partitions)
111 &quot;ENTRIES を STRATEGY に従って NUM-PARTITIONS に分割する。&quot;
112 (pcase strategy
113 ('hash (nskk-parallel-search--partition-hash entries num-partitions))
114 ('range (nskk-parallel-search--partition-range entries num-partitions))
115 (_ (error &quot;Unknown partition strategy: %s&quot; strategy))))

119(defun nskk-parallel-search--search-partition (partition query search-type)
120 &quot;PARTITION 内で QUERY を SEARCH-TYPE で検索する。&quot;
121 (let ((results nil))
122 (pcase search-type
123 ('exact
124 ;; 完全一致検索
125 (let ((entry (assoc query partition)))
126 (when entry
127 (push entry results))))
128
129 ('prefix
130 ;; 前方一致検索
131 (dolist (entry partition)
132 (when (string-prefix-p query (car entry))
133 (push entry results))))
134
135 ('partial
136 ;; 部分一致検索
137 (dolist (entry partition)
138 (when (string-match-p (regexp-quote query) (car entry))
139 (push entry results))))
140
141 ('fuzzy
142 ;; ファジー検索
143 (dolist (entry partition)
144 (let ((distance (nskk-parallel-search--levenshtein-distance
145 query (car entry))))
146 (when (&lt;= distance nskk-search-fuzzy-threshold)
147 (push (cons entry distance) results)))))
148
149 (_ (error &quot;Unknown search type: %s&quot; search-type)))
150
151 results))

155(defun nskk-parallel-search--levenshtein-distance (s1 s2)
156 &quot;S1 と S2 の Levenshtein 距離を計算する。&quot;
157 (let* ((len1 (length s1))
158 (len2 (length s2))
159 (matrix (make-vector (1+ len1) nil)))
160
161 ;; 行列初期化
162 (dotimes (i (1+ len1))
163 (aset matrix i (make-vector (1+ len2) 0))
164 (aset (aref matrix i) 0 i))
165 (dotimes (j (1+ len2))
166 (aset (aref matrix 0) j j))
167
168 ;; DP計算
169 (dotimes (i len1)
170 (dotimes (j len2)
171 (let ((cost (if (= (aref s1 i) (aref s2 j)) 0 1)))
172 (aset (aref matrix (1+ i)) (1+ j)
173 (min (1+ (aref (aref matrix i) (1+ j)))
174 (1+ (aref (aref matrix (1+ i)) j))
175 (+ (aref (aref matrix i) j) cost))))))
176
177 (aref (aref matrix len1) len2)))

179(defun nskk-parallel-search--merge-results (results search-type)
180 &quot;複数の検索結果 RESULTS をマージする。&quot;
181 (let ((merged nil))
182 (dolist (result results)
183 (setq merged (append merged result)))
184
185 ;; 重複除去
186 (setq merged (delete-dups merged))
187
188 ;; ソート
189 (pcase search-type
190 ('fuzzy
191 ;; ファジー検索: 距離でソート
192 (sort merged (lambda (a b) (&lt; (cdr a) (cdr b)))))
193 (_
194 ;; その他: 辞書順
195 (sort merged (lambda (a b) (string&lt; (car a) (car b))))))))

200(defun nskk-parallel-search (index query search-type &optional max-results)
201 &quot;INDEX 内で QUERY を並列検索する。
202SEARCH-TYPE は検索タイプ（'exact, 'prefix, 'partial, 'fuzzy）。
203MAX-RESULTS は最大結果数（オプション）。&quot;
204 ;; 初期化確認
205 (unless nskk-parallel-search--pool
206 (nskk-parallel-search-initialize))
207
208 ;; 並列検索が利用不可能な場合は通常検索にフォールバック
209 (unless (and nskk-parallel-search--pool
210 (nskk-thread-pool-available-p))
211 (return-from nskk-parallel-search
212 (nskk-search index query search-type nil max-results)))
213
214 ;; エントリ数チェック
215 (let* ((entries (nskk-index-get-all-entries index))
216 (entry-count (length entries)))
217
218 ;; エントリ数が少ない場合は通常検索
219 (when (&lt; entry-count nskk-parallel-search-min-entries)
220 (return-from nskk-parallel-search
221 (nskk-search index query search-type nil max-results)))
222
223 ;; 並列検索実行
224 (let* ((num-workers (nskk-thread-pool-size nskk-parallel-search--pool))
225 (partitions (nskk-parallel-search--partition
226 entries
227 nskk-parallel-search-partition-strategy
228 num-workers))
229 (results-mutex (make-mutex))
230 (all-results nil)
231 (completed-count 0))
232
233 ;; 各パーティションを並列検索
234 (dolist (partition partitions)
235 (nskk-thread-submit
236 nskk-parallel-search--pool
237 (lambda ()
238 (nskk-parallel-search--search-partition
239 partition query search-type))
240 (lambda (partition-results)
241 (with-mutex results-mutex
242 (push partition-results all-results)
243 (setq completed-count (1+ completed-count))))))
244
245 ;; 全検索完了待機
246 (let ((start-time (current-time)))
247 (while (and (&lt; completed-count num-workers)
248 (&lt; (float-time (time-subtract (current-time) start-time))
249 nskk-parallel-search-timeout))
250 (sleep-for 0.01)))
251
252 ;; タイムアウトチェック
253 (when (&lt; completed-count num-workers)
254 (message &quot;Parallel search timeout, using partial results&quot;))
255
256 ;; 結果マージ
257 (let ((merged (nskk-parallel-search--merge-results all-results search-type)))
258 ;; 最大結果数制限
259 (if max-results
260 (cl-subseq merged 0 (min (length merged) max-results))
261 merged)))))

264(defun nskk-parallel-search-benchmark (index query search-type iterations)
265 &quot;並列検索と通常検索のベンチマーク。
266INDEX 内で QUERY を SEARCH-TYPE で ITERATIONS 回検索し、
267平均実行時間と高速化率を報告する。&quot;
268 (interactive)
269
270 ;; 通常検索ベンチマーク
271 (let ((sequential-times nil)
272 (parallel-times nil))
273
274 (message &quot;Running sequential search benchmark...&quot;)
275 (dotimes (_ iterations)
276 (let ((start (current-time)))
277 (nskk-search index query search-type)
278 (push (float-time (time-subtract (current-time) start))
279 sequential-times)))
280
281 ;; 並列検索ベンチマーク
282 (message &quot;Running parallel search benchmark...&quot;)
283 (dotimes (_ iterations)
284 (let ((start (current-time)))
285 (nskk-parallel-search index query search-type)
286 (push (float-time (time-subtract (current-time) start))
287 parallel-times)))
288
289 ;; 結果分析
290 (let* ((seq-avg (/ (apply #'+ sequential-times) iterations))
291 (par-avg (/ (apply #'+ parallel-times) iterations))
292 (speedup (/ seq-avg par-avg)))
293
294 (message &quot;Benchmark Results:&quot;)
295 (message &quot; Sequential: %.3f ms&quot; (* seq-avg 1000))
296 (message &quot; Parallel: %.3f ms&quot; (* par-avg 1000))
297 (message &quot; Speedup: %.2fx&quot; speedup)
298
299 (list :sequential seq-avg
300 :parallel par-avg
301 :speedup speedup))))
