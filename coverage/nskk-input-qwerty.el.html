14;; QWERTY-JIS入力方式の実装。
15;;
16;; これは標準的なローマ字入力方式で、QWERTYキーボード配列を
17;; 使用した最も一般的な日本語入力方法です。
18;; 既存の `nskk-romaji-tables` をそのまま使用します。
19;;
20;; 使用例:
21;; (require 'nskk-input-qwerty)
22;; (nskk-register-input-method 'qwerty)
23
24;;; Code:
25
26(require 'nskk-romaji-tables)
27
28;;; QWERTY入力方式は標準ローマ字テーブルをそのまま使用
29
30(defvar nskk-input-qwerty-hash-table nil
31 &quot;QWERTY入力方式用のハッシュテーブル。
32実際には `nskk-romaji-hash-table` への参照。&quot;)
33
34(defun nskk-input-qwerty-init-hash-table ()
35 &quot;QWERTY入力方式用のハッシュテーブルを初期化する。
36標準ローマ字テーブルを再利用する。&quot;
37 (unless nskk-romaji-hash-table
38 (nskk-romaji-init-hash-table))
39 (setq nskk-input-qwerty-hash-table nskk-romaji-hash-table))
40
41;;; 検索関数
42
43(defun nskk-input-qwerty-lookup (romaji)
44 &quot;QWERTY入力方式でROMAJIをひらがなに変換する。
45
46使用例:
47 (nskk-input-qwerty-lookup \&quot;ka\&quot;) ;; =&gt; \&quot;か\&quot;
48 (nskk-input-qwerty-lookup \&quot;kya\&quot;) ;; =&gt; \&quot;きゃ\&quot;&quot;
49 (unless nskk-input-qwerty-hash-table
50 (nskk-input-qwerty-init-hash-table))
51 (gethash romaji nskk-input-qwerty-hash-table))
52
53(defun nskk-input-qwerty-get-candidates (prefix)
54 &quot;PREFIX で始まる全てのQWERTYローマ字候補を取得する。&quot;
55 (nskk-romaji-get-candidates prefix))
56
57;;; 入力方式登録
58
59(defun nskk-input-qwerty-register ()
60 &quot;QWERTY入力方式をNSKKに登録する。&quot;
61 (nskk-input-qwerty-init-hash-table))
62
63;;; 統計情報
64
65(defun nskk-input-qwerty-stats ()
66 &quot;QWERTY入力方式の統計情報を返す。&quot;
67 (nskk-romaji-table-stats))

71(defsubst nskk-input-qwerty-lookup-fast (romaji)
72 &quot;QWERTY入力方式でROMAJIを高速変換する。&quot;
73 (gethash romaji nskk-input-qwerty-hash-table))
