175;;;###autoload
176(defun nskk-ai-pattern-learn (midashi candidate context)
177 &quot;変換パターンを学習する。
178MIDASHI: 見出し語
179CANDIDATE: 変換候補
180CONTEXT: 文脈情報（単語リスト）&quot;
181 (when (and (stringp midashi) (stringp candidate))
182 ;; パターンエントリを作成
183 (let* ((features (nskk-ai-pattern--extract-features midashi candidate context))
184 (entry (nskk-ai-pattern-entry--create
185 :midashi midashi
186 :candidate candidate
187 :context context
188 :timestamp (current-time)
189 :features features)))
190
191 ;; 履歴に追加
192 (push entry nskk-ai-pattern-history)
193
194 ;; ユーザーパターンを更新
195 (nskk-ai-pattern--update-user-patterns midashi candidate context)
196
197 ;; 時系列データを更新
198 (nskk-ai-pattern--update-time-series midashi candidate)
199
200 ;; 定期的にクラスタリングを実行（100エントリごと）
201 (when (and nskk-ai-pattern-enable-clustering
202 (zerop (mod (length nskk-ai-pattern-history) 100)))
203 (nskk-ai-pattern-cluster-patterns))
204
205 ;; 異常検出
206 (when nskk-ai-pattern-enable-anomaly-detection
207 (nskk-ai-pattern--detect-anomaly entry))
208
209 ;; 履歴サイズ制限（最新10000件）
210 (when (&gt; (length nskk-ai-pattern-history) 10000)
211 (setq nskk-ai-pattern-history
212 (seq-take nskk-ai-pattern-history 10000))))))

214(defun nskk-ai-pattern--extract-features (midashi candidate context)
215 &quot;パターンの特徴ベクトルを抽出する。
216MIDASHI: 見出し語
217CANDIDATE: 変換候補
218CONTEXT: 文脈情報
219戻り値: 特徴ベクトル（vector）&quot;
220 (let ((features (make-vector 10 0.0)))
221 ;; 特徴1: 見出し語の長さ
222 (aset features 0 (float (length midashi)))
223 ;; 特徴2: 候補の長さ
224 (aset features 1 (float (length candidate)))
225 ;; 特徴3: 文脈の単語数
226 (aset features 2 (float (length context)))
227 ;; 特徴4: 見出し語のひらがな率
228 (aset features 3 (nskk-ai-pattern--hiragana-ratio midashi))
229 ;; 特徴5: 候補の漢字率
230 (aset features 4 (nskk-ai-pattern--kanji-ratio candidate))
231 ;; 特徴6: 長さ比率
232 (aset features 5 (if (&gt; (length midashi) 0)
233 (/ (float (length candidate)) (length midashi))
234 0.0))
235 ;; 特徴7-9: 文脈の最後3単語の長さ平均
236 (let ((context-avg 0.0))
237 (when (&gt; (length context) 0)
238 (setq context-avg
239 (/ (apply #'+ (mapcar #'length (seq-take (reverse context) 3)))
240 (float (min 3 (length context))))))
241 (aset features 6 context-avg))
242 ;; 特徴8: 時刻（0-23の時間帯）
243 (aset features 7 (float (nth 2 (decode-time))))
244 ;; 特徴9: 曜日（0-6）
245 (aset features 8 (float (nth 6 (decode-time))))
246 features))

248(defsubst nskk-ai-pattern--hiragana-ratio (str)
249 &quot;文字列 STR のひらがな比率を計算する。&quot;
250 (if (zerop (length str))
251 0.0
252 (let ((hiragana-count 0))
253 (dotimes (i (length str))
254 (let ((char (aref str i)))
255 (when (and (&gt;= char #x3040) (&lt;= char #x309F))
256 (cl-incf hiragana-count))))
257 (/ (float hiragana-count) (length str)))))
258
259(defsubst nskk-ai-pattern--kanji-ratio (str)
260 &quot;文字列 STR の漢字比率を計算する。&quot;
261 (if (zerop (length str))
262 0.0
263 (let ((kanji-count 0))
264 (dotimes (i (length str))
265 (let ((char (aref str i)))
266 (when (or (and (&gt;= char #x4E00) (&lt;= char #x9FFF))
267 (and (&gt;= char #x3400) (&lt;= char #x4DBF)))
268 (cl-incf kanji-count))))
269 (/ (float kanji-count) (length str)))))

316(defun nskk-ai-pattern--update-time-series (midashi candidate)
317 &quot;時系列データを更新する。&quot;
318 (let* ((key (cons midashi candidate))
319 (ts (gethash key nskk-ai-pattern-time-series-data)))
320 (unless ts
321 (setq ts (nskk-ai-time-series--create
322 :window-size nskk-ai-pattern-time-window))
323 (puthash key ts nskk-ai-pattern-time-series-data))
324
325 ;; タイムスタンプと値を追加
326 (push (current-time) (nskk-ai-time-series-timestamps ts))
327 (push 1 (nskk-ai-time-series-values ts))
328
329 ;; 古いデータを削除（ウィンドウサイズ外）
330 (let* ((now (current-time))
331 (cutoff-time (time-subtract now (seconds-to-time nskk-ai-pattern-time-window)))
332 (filtered-timestamps nil)
333 (filtered-values nil))
334 (cl-loop for timestamp in (nskk-ai-time-series-timestamps ts)
335 for value in (nskk-ai-time-series-values ts)
336 when (time-less-p cutoff-time timestamp)
337 do (push timestamp filtered-timestamps)
338 and do (push value filtered-values))
339 (setf (nskk-ai-time-series-timestamps ts) filtered-timestamps)
340 (setf (nskk-ai-time-series-values ts) filtered-values))))

362(defun nskk-ai-pattern-cluster-patterns ()
363 &quot;パターン履歴をクラスタリングする（k-means）。
364戻り値: クラスタのリスト&quot;
365 (interactive)
366 (when (&lt; (length nskk-ai-pattern-history) nskk-ai-pattern-cluster-count)
367 (error &quot;Not enough patterns for clustering&quot;))
368
369 (let ((k nskk-ai-pattern-cluster-count)
370 (max-iter nskk-ai-pattern-max-iterations)
371 (data-points (mapcar #'nskk-ai-pattern-entry-features
372 nskk-ai-pattern-history)))
373
374 ;; 初期クラスタ中心をランダムに選択
375 (let ((clusters (nskk-ai-pattern--initialize-clusters k data-points)))
376
377 ;; k-meansアルゴリズム
378 (dotimes (_ max-iter)
379 ;; 各データポイントを最も近いクラスタに割り当て
380 (nskk-ai-pattern--assign-to-clusters clusters data-points)
381 ;; クラスタ中心を再計算
382 (nskk-ai-pattern--update-centroids clusters))
383
384 ;; クラスタIDをパターンエントリに設定
385 (dotimes (i (length nskk-ai-pattern-history))
386 (let* ((entry (nth i nskk-ai-pattern-history))
387 (features (nskk-ai-pattern-entry-features entry))
388 (cluster-id (nskk-ai-pattern--find-nearest-cluster features clusters)))
389 (setf (nskk-ai-pattern-entry-cluster-id entry) cluster-id)))
390
391 (setq nskk-ai-pattern-clusters clusters)
392 clusters)))

394(defun nskk-ai-pattern--initialize-clusters (k data-points)
395 &quot;クラスタを初期化する（k-means++）。&quot;
396 (let ((clusters nil)
397 (n (length data-points)))
398 (dotimes (i k)
399 (let* ((random-idx (random n))
400 (centroid (copy-sequence (nth random-idx data-points)))
401 (cluster (nskk-ai-cluster--create
402 :id i
403 :centroid centroid)))
404 (push cluster clusters)))
405 (nreverse clusters)))

407(defun nskk-ai-pattern--assign-to-clusters (clusters data-points)
408 &quot;各データポイントを最も近いクラスタに割り当てる。&quot;
409 (dolist (cluster clusters)
410 (setf (nskk-ai-cluster-members cluster) nil)
411 (setf (nskk-ai-cluster-size cluster) 0))
412
413 (dolist (point data-points)
414 (let* ((nearest-cluster (nskk-ai-pattern--find-nearest-cluster point clusters))
415 (cluster (nth nearest-cluster clusters)))
416 (push point (nskk-ai-cluster-members cluster))
417 (cl-incf (nskk-ai-cluster-size cluster)))))

419(defun nskk-ai-pattern--update-centroids (clusters)
420 &quot;クラスタの重心を更新する。&quot;
421 (dolist (cluster clusters)
422 (let ((members (nskk-ai-cluster-members cluster)))
423 (when (&gt; (length members) 0)
424 (let* ((dim (length (car members)))
425 (new-centroid (make-vector dim 0.0)))
426 ;; 各次元の平均を計算
427 (dotimes (d dim)
428 (let ((sum 0.0))
429 (dolist (member members)
430 (setq sum (+ sum (aref member d))))
431 (aset new-centroid d (/ sum (float (length members))))))
432 (setf (nskk-ai-cluster-centroid cluster) new-centroid))))))

434(defun nskk-ai-pattern--find-nearest-cluster (point clusters)
435 &quot;POINT に最も近いクラスタのIDを返す。&quot;
436 (let ((min-distance most-positive-fixnum)
437 (nearest-id 0))
438 (dotimes (i (length clusters))
439 (let* ((cluster (nth i clusters))
440 (distance (nskk-ai-pattern--euclidean-distance
441 point
442 (nskk-ai-cluster-centroid cluster))))
443 (when (&lt; distance min-distance)
444 (setq min-distance distance)
445 (setq nearest-id i))))
446 nearest-id))

448(defsubst nskk-ai-pattern--euclidean-distance (vec1 vec2)
449 &quot;2つのベクトル VEC1 と VEC2 のユークリッド距離を計算する。&quot;
450 (let ((sum 0.0)
451 (len (min (length vec1) (length vec2))))
452 (dotimes (i len)
453 (let ((diff (- (aref vec1 i) (aref vec2 i))))
454 (setq sum (+ sum (* diff diff)))))
455 (sqrt sum)))

459(defun nskk-ai-pattern--detect-anomaly (entry)
460 &quot;パターンエントリ ENTRY が異常かどうかを検出する。&quot;
461 (let ((features (nskk-ai-pattern-entry-features entry)))
462 ;; 各特徴の統計を計算
463 (dotimes (dim (length features))
464 (let* ((values (mapcar (lambda (e)
465 (aref (nskk-ai-pattern-entry-features e) dim))
466 nskk-ai-pattern-history))
467 (mean (nskk-ai-pattern--mean values))
468 (stddev (nskk-ai-pattern--stddev values mean))
469 (value (aref features dim))
470 (z-score (if (&gt; stddev 0) (/ (abs (- value mean)) stddev) 0.0)))
471
472 ;; 閾値を超えた場合、異常として記録
473 (when (&gt; z-score nskk-ai-pattern-anomaly-threshold)
474 (let ((anomaly (nskk-ai-anomaly--create
475 :entry entry
476 :score z-score
477 :reason (format &quot;Feature %d: z-score=%.2f&quot; dim z-score)
478 :timestamp (current-time))))
479 (push anomaly nskk-ai-pattern-anomalies)
480 ;; 異常リストのサイズ制限（最新100件）
481 (when (&gt; (length nskk-ai-pattern-anomalies) 100)
482 (setq nskk-ai-pattern-anomalies
483 (seq-take nskk-ai-pattern-anomalies 100)))))))))
484
485(defsubst nskk-ai-pattern--mean (values)
486 &quot;VALUES の平均を計算する。&quot;
487 (if (&gt; (length values) 0)
488 (/ (apply #'+ values) (float (length values)))
489 0.0))
490
491(defsubst nskk-ai-pattern--stddev (values mean)
492 &quot;VALUES の標準偏差を計算する。MEAN は平均値。&quot;
493 (if (&gt; (length values) 1)
494 (let ((sum-sq 0.0))
495 (dolist (val values)
496 (let ((diff (- val mean)))
497 (setq sum-sq (+ sum-sq (* diff diff)))))
498 (sqrt (/ sum-sq (float (1- (length values))))))
499 0.0))

510(defun nskk-ai-pattern-predict-candidate (midashi context)
511 &quot;見出し語 MIDASHI と文脈 CONTEXT から候補を予測する。
512戻り値: ((candidate . confidence) ...) のリスト&quot;
513 (let ((predictions (make-hash-table :test 'equal))
514 (target-features (nskk-ai-pattern--extract-features midashi &quot;&quot; context)))
515
516 ;; 履歴から類似パターンを検索
517 (dolist (entry nskk-ai-pattern-history)
518 (when (equal (nskk-ai-pattern-entry-midashi entry) midashi)
519 (let* ((distance (nskk-ai-pattern--euclidean-distance
520 target-features
521 (nskk-ai-pattern-entry-features entry)))
522 (confidence (exp (- distance))) ; 距離を信頼度に変換
523 (candidate (nskk-ai-pattern-entry-candidate entry))
524 (current-conf (gethash candidate predictions 0.0)))
525 (puthash candidate (max confidence current-conf) predictions))))
526
527 ;; 信頼度順にソート
528 (let ((result nil))
529 (maphash (lambda (cand conf) (push (cons cand conf) result)) predictions)
530 (sort result (lambda (a b) (&gt; (cdr a) (cdr b)))))))
