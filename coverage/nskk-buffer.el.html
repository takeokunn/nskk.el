90(cl-defstruct (nskk-buffer-history-entry
91 (:constructor nskk-buffer-history-entry-create)
92 (:copier nil))
93 &quot;バッファ操作履歴のエントリ。
94
95スロット:
96 operation - 操作種別（'insert, 'delete, 'clear 等）
97 data - 操作に関連するデータ
98 state - 操作時の状態スナップショット&quot;
99 (operation nil :type symbol :read-only t)
100 (data nil :read-only t)
101 (state nil :read-only t))
102
103(defun nskk-buffer--save-history (operation data)
104 &quot;履歴スタックに OPERATION と DATA を保存する。&quot;
105 (unless nskk-buffer--in-undo-redo
106 (when (&lt; (length nskk-buffer--undo-stack) nskk-buffer-undo-limit)
107 (let ((entry (nskk-buffer-history-entry-create
108 :operation operation
109 :data data
110 :state (when nskk-current-state
111 (nskk-state-copy nskk-current-state)))))
112 (push entry nskk-buffer--undo-stack)
113 ;; リドゥスタックをクリア（新しい操作が行われたため）
114 (setq nskk-buffer--redo-stack nil)))))
115
116(defun nskk-buffer--clear-history ()
117 &quot;アンドゥ・リドゥ履歴をクリアする。&quot;
118 (setq nskk-buffer--undo-stack nil)
119 (setq nskk-buffer--redo-stack nil))

123(defun nskk-buffer--ensure-overlay ()
124 &quot;オーバーレイが存在することを確認し、なければ作成する。&quot;
125 (unless (and nskk-buffer--overlay
126 (overlay-buffer nskk-buffer--overlay))
127 (setq nskk-buffer--overlay (make-overlay (point) (point)))
128 (overlay-put nskk-buffer--overlay 'face nskk-buffer-pending-face)
129 (overlay-put nskk-buffer--overlay 'nskk-overlay t)))
130
131(defun nskk-buffer--delete-overlay ()
132 &quot;オーバーレイを削除する。&quot;
133 (when (and nskk-buffer--overlay
134 (overlay-buffer nskk-buffer--overlay))
135 (delete-overlay nskk-buffer--overlay)
136 (setq nskk-buffer--overlay nil)))

138(defun nskk-buffer--update-overlay (start end face)
139 &quot;オーバーレイの位置と表示を更新する。&quot;
140 (nskk-buffer--ensure-overlay)
141 (move-overlay nskk-buffer--overlay start end)
142 (overlay-put nskk-buffer--overlay 'face face))

146(defun nskk-buffer-insert (string)
147 &quot;STRING を現在位置に挿入する。
148
149引数:
150 STRING - 挿入する文字列
151
152副作用:
153 - バッファへの文字列挿入
154 - マーカー位置の更新
155 - アンドゥ履歴への記録&quot;
156 (unless nskk-current-state
157 (nskk-state-init))
158
159 (let ((start (point)))
160 ;; バッファに挿入
161 (insert string)
162
163 ;; マーカー更新
164 (unless (nskk-state-marker-start nskk-current-state)
165 (nskk-state-set-markers nskk-current-state start (point)))
166 (when (nskk-state-marker-end nskk-current-state)
167 (set-marker (nskk-state-marker-end nskk-current-state) (point)))
168
169 ;; オーバーレイ更新
170 (when nskk-buffer-show-pending-input
171 (nskk-buffer--update-overlay
172 start (point)
173 (if (nskk-state-in-conversion-p nskk-current-state)
174 nskk-buffer-conversion-face
175 nskk-buffer-pending-face)))
176
177 ;; 履歴保存
178 (nskk-buffer--save-history 'insert string)))

180(defun nskk-buffer-delete-region (start end)
181 &quot;START から END までの領域を削除する。
182
183引数:
184 START - 削除開始位置
185 END - 削除終了位置
186
187副作用:
188 - バッファからのテキスト削除
189 - アンドゥ履歴への記録&quot;
190 (let ((deleted-text (buffer-substring-no-properties start end)))
191 (delete-region start end)
192
193 ;; 履歴保存
194 (nskk-buffer--save-history 'delete
195 (list :text deleted-text
196 :start start
197 :end end))))

199(defun nskk-buffer-delete-backward-char (&optional n)
200 &quot;カーソルの前の N 文字を削除する。
201
202引数:
203 N - 削除する文字数（デフォルト: 1）
204
205返り値:
206 削除に成功した場合は t、失敗した場合は nil&quot;
207 (interactive &quot;p&quot;)
208 (setq n (or n 1))
209
210 (when (and nskk-current-state
211 (nskk-state-marker-start nskk-current-state))
212 (let ((start (marker-position (nskk-state-marker-start nskk-current-state)))
213 (current (point)))
214 (when (&gt;= (- current start) n)
215 (nskk-buffer-delete-region (- current n) current)
216
217 ;; オーバーレイ更新
218 (when nskk-buffer-show-pending-input
219 (if (= (point) start)
220 (nskk-buffer--delete-overlay)
221 (nskk-buffer--update-overlay
222 start (point)
223 (if (nskk-state-in-conversion-p nskk-current-state)
224 nskk-buffer-conversion-face
225 nskk-buffer-pending-face))))
226 t))))

249(defun nskk-buffer-clear ()
250 &quot;未確定入力を全てクリアする。
251
252副作用:
253 - バッファから未確定テキストを削除
254 - マーカーのクリア
255 - オーバーレイの削除
256 - 状態のクリア&quot;
257 (interactive)
258
259 (when (and nskk-current-state
260 (nskk-state-marker-start nskk-current-state)
261 (nskk-state-marker-end nskk-current-state))
262 (let ((start (marker-position (nskk-state-marker-start nskk-current-state)))

264
265 ;; バッファから削除

268

271 (nskk-state-clear-markers nskk-current-state)
272
273 ;; オーバーレイ削除
274 (nskk-buffer--delete-overlay)
275
276 ;; 履歴保存
277 (nskk-buffer--save-history 'clear nil))))

281(defun nskk-buffer-commit (text)
282 &quot;TEXT を確定してバッファに挿入する。
283
284引数:
285 TEXT - 確定するテキスト
286
287副作用:
288 - 未確定入力の削除
289 - 確定テキストの挿入
290 - 状態のクリア&quot;
291 (when (and nskk-current-state
292 (nskk-state-marker-start nskk-current-state))
293 (let ((start (marker-position (nskk-state-marker-start nskk-current-state))))

307(defun nskk-buffer-undo ()
308 &quot;最後のバッファ操作をアンドゥする。
309
310返り値:
311 アンドゥに成功した場合は t、履歴がない場合は nil&quot;
312 (interactive)
313
314 (if-let ((entry (pop nskk-buffer--undo-stack)))
315 (progn
316 ;; リドゥ用に保存
317 (push entry nskk-buffer--redo-stack)
318
319 ;; 操作に応じてアンドゥ処理
320 (let ((operation (nskk-buffer-history-entry-operation entry))
321 (data (nskk-buffer-history-entry-data entry))
322 (saved-state (nskk-buffer-history-entry-state entry)))
323
324 (pcase operation
325 ('insert
326 ;; 挿入のアンドゥ: 削除
327 (when (and nskk-current-state
328 (nskk-state-marker-end nskk-current-state))
329 (let ((end (marker-position (nskk-state-marker-end nskk-current-state)))
330 (len (length data)))
331 (delete-region (- end len) end))))
332
333 ('delete
334 ;; 削除のアンドゥ: 挿入
335 (let ((text (plist-get data :text))
336 (start (plist-get data :start)))
337 (goto-char start)
338 (insert text)))
339
340 ('clear
341 ;; クリアのアンドゥ: 状態復元
342 (when saved-state
343 (setq nskk-current-state (nskk-state-copy saved-state))))
344
345 ('commit
346 ;; 確定のアンドゥ: テキスト削除
347 (let ((len (length data)))
348 (delete-region (- (point) len) (point))))))
349
350 t)
351 nil))

353(defun nskk-buffer-redo ()
354 &quot;アンドゥした操作をリドゥする。
355
356返り値:
357 リドゥに成功した場合は t、履歴がない場合は nil&quot;
358 (interactive)
359
360 (if-let ((entry (pop nskk-buffer--redo-stack)))
361 (let ((nskk-buffer--in-undo-redo t))
362 ;; アンドゥ用に保存
363 (push entry nskk-buffer--undo-stack)
364
365 ;; 操作に応じてリドゥ処理
366 (let ((operation (nskk-buffer-history-entry-operation entry))
367 (data (nskk-buffer-history-entry-data entry)))
368
369 (pcase operation
370 ('insert
371 ;; 挿入のリドゥ
372 (nskk-buffer-insert data))
373
374 ('delete
375 ;; 削除のリドゥ
376 (let ((start (plist-get data :start))
377 (end (plist-get data :end)))
378 (nskk-buffer-delete-region start end)))
379
380 ('clear
381 ;; クリアのリドゥ
382 (nskk-buffer-clear))
383
384 ('commit
385 ;; 確定のリドゥ
386 (nskk-buffer-commit data))))
387
388 t)
389 nil))

393(defun nskk-buffer-has-pending-input-p ()
394 &quot;未確定入力が存在するかどうかを判定する。&quot;
395 (and nskk-current-state
396 (nskk-state-marker-start nskk-current-state)
397 (nskk-state-marker-end nskk-current-state)
398 (&lt; (marker-position (nskk-state-marker-start nskk-current-state))
399 (marker-position (nskk-state-marker-end nskk-current-state)))))

401(defun nskk-buffer-pending-text ()
402 &quot;未確定入力のテキストを取得する。
403未確定入力がない場合は nil を返す。&quot;
404 (when (nskk-buffer-has-pending-input-p)
405 (buffer-substring-no-properties
406 (marker-position (nskk-state-marker-start nskk-current-state))
407 (marker-position (nskk-state-marker-end nskk-current-state)))))

409(defun nskk-buffer-pending-length ()
410 &quot;未確定入力の文字数を取得する。&quot;
411 (if-let ((text (nskk-buffer-pending-text)))
412 (length text)
413 0))

426(defun nskk-buffer-stats ()
427 &quot;バッファ管理の統計情報を返す。
428
429返り値は以下の要素を含むplist:
430 :has-pending - 未確定入力の有無
431 :pending-length - 未確定入力の文字数
432 :undo-stack-size - アンドゥ履歴の数
433 :redo-stack-size - リドゥ履歴の数
434 :overlay-active - オーバーレイの有無&quot;
435 (list :has-pending (nskk-buffer-has-pending-input-p)
436 :pending-length (nskk-buffer-pending-length)
437 :undo-stack-size (length nskk-buffer--undo-stack)
438 :redo-stack-size (length nskk-buffer--redo-stack)
439 :overlay-active (and nskk-buffer--overlay
440 (overlay-buffer nskk-buffer--overlay))))

442(defun nskk-buffer-describe ()
443 &quot;バッファの現在状態を表示する。&quot;
444 (interactive)
445 (let ((stats (nskk-buffer-stats)))
446 (message &quot;NSKK Buffer:
447 Pending input: %s
448 Pending length: %d
449 Undo stack: %d
450 Redo stack: %d
451 Overlay: %s&quot;
452 (if (plist-get stats :has-pending)
453 (format &quot;\&quot;%s\&quot;&quot; (nskk-buffer-pending-text))
454 &quot;none&quot;)
455 (plist-get stats :pending-length)
456 (plist-get stats :undo-stack-size)
457 (plist-get stats :redo-stack-size)
458 (if (plist-get stats :overlay-active) &quot;active&quot; &quot;inactive&quot;))))
