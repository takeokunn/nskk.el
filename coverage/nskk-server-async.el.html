29;; 特徴:
30;; - make-network-processを使用した非同期通信

222(defun nskk-server-async-search (host port key okuri-type
223 success-callback error-callback
224 &optional encoding timeout)
225 &quot;SKKサーバーに非同期検索リクエストを送信する。
226
227引数:
228 HOST - サーバーホスト名
229 PORT - サーバーポート番号
230 KEY - 検索キー
231 OKURI-TYPE - 送り仮名タイプ（'okuri-ari/'okuri-nasi）
232 SUCCESS-CALLBACK - 成功時のコールバック（nskk-server-response -&gt; void）
233 ERROR-CALLBACK - エラー時のコールバック（error-message -&gt; void）
234 ENCODING - エンコーディング（省略時はnskk-server-protocol-encoding）
235 TIMEOUT - タイムアウト時間（秒、省略時はnskk-server-async-timeout）
236
237処理フロー:
238 1. リクエスト文字列生成
239 2. コネクション取得（プールまたは新規作成）
240 3. リクエストキューに追加
241 4. タイムアウトタイマー設定
242 5. 送信
243
244戻り値:
245 なし（非同期処理）&quot;
246 (let* ((encoding (or encoding nskk-server-protocol-encoding))
247 (timeout (or timeout nskk-server-async-timeout))
248 (request-str (nskk-server-protocol-make-request key okuri-type encoding))
249 (conn (or (nskk-server-async--get-connection host port)
250 (nskk-server-async--create-connection host port encoding))))
251
252 (when nskk-server-async-verbose
253 (message &quot;Async search: %s@%s:%d&quot; key host port))
254
255 ;; リクエストコンテキスト作成
256 (let* ((timeout-time (+ (float-time) timeout))
257 (context (nskk-server-request-context--create
258 :request request-str
259 :callback success-callback
260 :error-callback error-callback
261 :timeout timeout-time)))
262
263 ;; タイムアウトタイマー設定
264 (setf (nskk-server-request-context-timer context)
265 (run-at-time timeout nil
266 (lambda ()
267 (nskk-server-async--handle-timeout conn context))))
268
269 ;; リクエストキューに追加
270 (push context (nskk-server-connection-pending conn))
271
272 ;; 接続済みの場合は即座に送信
273 (when (eq (nskk-server-connection-state conn) 'connected)
274 (nskk-server-async--send-next-request conn)))))

276(defun nskk-server-async--create-connection (host port encoding)
277 &quot;新しいサーバー接続を作成する。
278
279引数:
280 HOST - ホスト名
281 PORT - ポート番号
282 ENCODING - エンコーディング
283
284戻り値:
285 nskk-server-connection構造体&quot;
286 (when nskk-server-async-verbose
287 (message &quot;Creating connection to %s:%d&quot; host port))
288
289 (let ((conn (nskk-server-connection--create
290 :host host
291 :port port
292 :state 'connecting
293 :encoding encoding
294 :created-time (float-time)
