117(defun nskk-debug-profile-start ()
118 &quot;プロファイリングを開始する。&quot;
119 (interactive)
120 (when nskk-debug--profiler-running
121 (error &quot;プロファイラーは既に実行中です&quot;))
122 (setq nskk-debug--profiler-running t
123 nskk-debug--profiler-start-time (current-time)
124 nskk-debug--function-call-counts (make-hash-table :test 'eq)
125 nskk-debug--function-total-times (make-hash-table :test 'eq))
126 (profiler-start 'cpu+mem)
127 (message &quot;プロファイリングを開始しました&quot;))

148(defun nskk-debug--show-custom-profile-report ()
149 &quot;カスタムプロファイルレポートを表示する。&quot;
150 (let ((buffer (get-buffer-create nskk-debug-profile-buffer-name)))
151 (with-current-buffer buffer
152 (erase-buffer)
153 (insert &quot;=== NSKK プロファイルレポート ===\n\n&quot;)
154 (insert (format &quot;実行時間: %.3f 秒\n\n&quot;
155 (float-time (time-subtract (current-time)
156 nskk-debug--profiler-start-time))))
157 (insert &quot;=== 関数呼び出し統計 ===\n\n&quot;)
158 (insert (format &quot;%-40s %10s %15s\n&quot; &quot;関数名&quot; &quot;呼び出し回数&quot; &quot;総実行時間(ms)&quot;))
159 (insert (make-string 70 ?-) &quot;\n&quot;)
160
161 ;; 統計データを表示
162 (let ((stats '()))
163 (maphash
164 (lambda (func count)
165 (let ((total-time (gethash func nskk-debug--function-total-times 0)))
166 (push (list func count total-time) stats)))
167 nskk-debug--function-call-counts)
168
169 ;; 実行時間でソート
170 (setq stats (sort stats (lambda (a b) (&gt; (nth 2 a) (nth 2 b)))))
171
172 ;; 上位20件を表示
173 (dolist (stat (seq-take stats 20))
174 (insert (format &quot;%-40s %10d %15.3f\n&quot;
175 (nth 0 stat)
176 (nth 1 stat)
177 (* (nth 2 stat) 1000)))))
178
179 (goto-char (point-min))
180 (special-mode))
181 (display-buffer buffer)))

199(defun nskk-debug-log (level message &rest args)
200 &quot;LEVEL と MESSAGE でログを記録する。&quot;
201 (let ((entry (list :time (current-time)
202 :level level
203 :message (apply #'format message args))))
204 (push entry nskk-debug--log-entries)
205 ;; エントリ数制限
206 (when (&gt; (length nskk-debug--log-entries) nskk-debug-max-log-entries)
207 (setq nskk-debug--log-entries
208 (seq-take nskk-debug--log-entries nskk-debug-max-log-entries)))))

210(defun nskk-debug-logs ()
211 &quot;ログビューアーを表示する。&quot;
212 (interactive)
213 (let ((buffer (get-buffer-create nskk-debug-log-buffer-name)))
214 (with-current-buffer buffer
215 (erase-buffer)
216 (insert &quot;=== NSKK デバッグログ ===\n\n&quot;)
217 (insert (format &quot;フィルター: %s\n&quot; nskk-debug--log-level-filter))
218 (insert (make-string 80 ?-) &quot;\n\n&quot;)
219
220 (dolist (entry (reverse nskk-debug--log-entries))
221 (when (or (eq nskk-debug--log-level-filter 'all)
222 (eq (plist-get entry :level) nskk-debug--log-level-filter))
223 (insert (format &quot;[%s] [%s] %s\n&quot;
224 (format-time-string &quot;%Y-%m-%d %H:%M:%S&quot;
225 (plist-get entry :time))
226 (upcase (symbol-name (plist-get entry :level)))
227 (plist-get entry :message)))))
228
229 (goto-char (point-max))
230 (special-mode))
231 (display-buffer buffer)))

343(defun nskk-debug--update-metrics ()
344 &quot;メトリクスデータを更新する。&quot;
345 ;; 実際のメトリクス収集（プレースホルダー）
346 (puthash 'conversion-time (+ 0.5 (random 2.0)) nskk-debug--metrics)
347 (puthash 'search-time (+ 1.0 (random 5.0)) nskk-debug--metrics)
348 (puthash 'key-response (+ 0.1 (random 0.5)) nskk-debug--metrics)
349 (puthash 'total-memory (* 1024 1024 (+ 10 (random 10))) nskk-debug--metrics)
350 (puthash 'cache-memory (* 1024 1024 (+ 5 (random 5))) nskk-debug--metrics)
351 (puthash 'gc-count gcs-done nskk-debug--metrics)
352 (puthash 'conversion-count (+ 100 (random 1000)) nskk-debug--metrics)
353 (puthash 'search-count (+ 500 (random 5000)) nskk-debug--metrics)
354 (puthash 'cache-hit-rate (+ 0.7 (/ (random 30) 100.0)) nskk-debug--metrics))

401(defun nskk-debug-export-report ()
402 &quot;デバッグレポートをファイルに出力する。&quot;
403 (interactive)
404 (let* ((timestamp (format-time-string &quot;%Y%m%d-%H%M%S&quot;))
405 (filename (expand-file-name
406 (format &quot;nskk-debug-report-%s.txt&quot; timestamp)
407 user-emacs-directory)))
408 (with-temp-file filename
409 (insert &quot;=== NSKK デバッグレポート ===\n\n&quot;)
410 (insert (format &quot;生成日時: %s\n\n&quot;
411 (format-time-string &quot;%Y-%m-%d %H:%M:%S&quot; (current-time))))
412
413 ;; プロファイル情報
414 (insert &quot;=== プロファイル情報 ===\n&quot;)
415 (insert (format &quot;プロファイラー状態: %s\n&quot;
416 (if nskk-debug--profiler-running &quot;実行中&quot; &quot;停止&quot;)))
417 (insert &quot;\n&quot;)
418
419 ;; メトリクス情報
420 (insert &quot;=== メトリクス ===\n&quot;)
421 (maphash
422 (lambda (key value)
423 (insert (format &quot;%s: %s\n&quot; key value)))
424 nskk-debug--metrics)
425 (insert &quot;\n&quot;)
426
427 ;; ログ情報
428 (insert &quot;=== ログ（最新100件） ===\n&quot;)
429 (dolist (entry (seq-take (reverse nskk-debug--log-entries) 100))
430 (insert (format &quot;[%s] [%s] %s\n&quot;
431 (format-time-string &quot;%Y-%m-%d %H:%M:%S&quot;
432 (plist-get entry :time))
433 (upcase (symbol-name (plist-get entry :level)))
434 (plist-get entry :message)))))
435
436 (message &quot;デバッグレポートを %s に出力しました&quot; filename)))
