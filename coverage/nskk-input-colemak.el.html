65;;; Colemak用のローマ字テーブル生成
66
67(defun nskk-input-colemak-convert-table ()
68 &quot;Colemak配列用のローマ字テーブルを生成する。
69QWERTYベースのローマ字テーブルをColemak配列に変換。&quot;
70 (let ((colemak-table nil))
71 (dolist (entry nskk-romaji-table)
72 (let ((romaji (car entry))
73 (kana (cdr entry))
74 (converted &quot;&quot;))
75 ;; ローマ字の各文字をColemakキーに変換
76 (dotimes (i (length romaji))
77 (let* ((char (substring romaji i (1+ i)))
78 (colemak-char (or (car (rassoc char nskk-input-colemak-key-map))
79 char)))
80 (setq converted (concat converted colemak-char))))
81 (push (cons converted kana) colemak-table)))
82 (nreverse colemak-table)))
83
84(defconst nskk-input-colemak-table
85 (nskk-input-colemak-convert-table)
86 &quot;Colemak配列用の変換テーブル。&quot;)

93(defun nskk-input-colemak-init-hash-table ()
94 &quot;Colemak入力方式用のハッシュテーブルを初期化する。&quot;
95 (setq nskk-input-colemak-hash-table
96 (make-hash-table :test 'equal :size 500))
97 (dolist (entry nskk-input-colemak-table)
98 (puthash (car entry) (cdr entry) nskk-input-colemak-hash-table)))

102(defun nskk-input-colemak-lookup (colemak-romaji)
103 &quot;Colemak入力方式でCOLEMAK-ROMAJIをひらがなに変換する。
104
105使用例:
106 (nskk-input-colemak-lookup \&quot;ta\&quot;) ;; Colemak配列での入力&quot;
107 (unless nskk-input-colemak-hash-table
108 (nskk-input-colemak-init-hash-table))
109 (gethash colemak-romaji nskk-input-colemak-hash-table))

111(defun nskk-input-colemak-get-candidates (prefix)
112 &quot;PREFIX で始まる全てのColemakローマ字候補を取得する。&quot;
113 (let ((candidates nil))
114 (dolist (entry nskk-input-colemak-table)
115 (when (string-prefix-p prefix (car entry))
116 (push (car entry) candidates)))
117 (nreverse candidates)))
