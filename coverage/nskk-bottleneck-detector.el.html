119(defun nskk-bottleneck-detect-start ()
120 &quot;ボトルネック検出を開始する。&quot;
121 (interactive)
122 (when nskk-bottleneck--active
123 (error &quot;Bottleneck detector is already running&quot;))
124
125 ;; プロファイラー起動
126 (unless nskk-profiler--active
127 (nskk-profile-start))
128
129 ;; 初期化
130 (setq nskk-bottleneck--active t
131 nskk-bottleneck--start-time (current-time)
132 nskk-bottleneck--hotpaths nil
133 nskk-bottleneck--slow-functions nil
134 nskk-bottleneck--memory-issues nil
135 nskk-bottleneck--gc-issues nil)
136
137 (message &quot;NSKK bottleneck detector started&quot;))

140(defun nskk-bottleneck-detect-stop ()
141 &quot;ボトルネック検出を停止し、分析を実行する。&quot;
142 (interactive)

144 (error &quot;Bottleneck detector is not running&quot;))
145

150 ;; 分析実行
151 (nskk-bottleneck--analyze)
152
153 (setq nskk-bottleneck--active nil)
154
155 (message &quot;NSKK bottleneck detector stopped. Found %d issues.&quot;
156 (+ (length nskk-bottleneck--hotpaths)
157 (length nskk-bottleneck--slow-functions)
158 (length nskk-bottleneck--memory-issues)
159 (length nskk-bottleneck--gc-issues))))

163(defun nskk-bottleneck--analyze ()
164 &quot;プロファイリングデータを分析してボトルネックを検出する。&quot;
165 ;; 関数レベルのボトルネック検出
166 (nskk-bottleneck--analyze-functions)
167
168 ;; メモリボトルネック検出
169 (nskk-bottleneck--analyze-memory)
170

172 (nskk-bottleneck--analyze-gc)

178(defun nskk-bottleneck--analyze-functions ()
179 &quot;関数レベルのボトルネックを分析する。&quot;
180 (setq nskk-bottleneck--hotpaths nil
181 nskk-bottleneck--slow-functions nil)
182
183 (dolist (entry nskk-profiler--function-stats)
184 (let* ((name (car entry))
185 (count (nth 1 entry))
186 (total (nth 2 entry))
187 (min-time (nth 3 entry))
188 (max-time (nth 4 entry))
189 (avg (/ total count)))
190
191 ;; ホットパス検出（呼び出し回数が多い）
192 (when (&gt;= count nskk-bottleneck-call-count-threshold)
193 (push (list name
194 :call-count count
195 :total-time total
196 :avg-time avg)
197 nskk-bottleneck--hotpaths))
198
199 ;; 遅延関数検出（平均実行時間が長い）
200 (when (&gt;= avg nskk-bottleneck-latency-threshold)
201 (push (list name
202 :max-time max-time
203 :avg-time avg
204 :call-count count)
205 nskk-bottleneck--slow-functions))))
206
207 ;; ソート
208 (setq nskk-bottleneck--hotpaths
209 (sort nskk-bottleneck--hotpaths
210 (lambda (a b) (&gt; (plist-get (cdr a) :call-count)
211 (plist-get (cdr b) :call-count)))))
212
213 (setq nskk-bottleneck--slow-functions
214 (sort nskk-bottleneck--slow-functions
215 (lambda (a b) (&gt; (plist-get (cdr a) :avg-time)
216 (plist-get (cdr b) :avg-time))))))

218(defun nskk-bottleneck--analyze-memory ()
219 &quot;メモリ使用量を分析してボトルネックを検出する。&quot;
220 (setq nskk-bottleneck--memory-issues nil)
221
222 (when (&gt;= (length nskk-profiler--memory-snapshots) 2)
223 (let* ((snapshots (reverse nskk-profiler--memory-snapshots))
224 (initial (cdr (car snapshots)))
225 (final (cdr (car (last snapshots))))
226 (cons-delta (- (or (plist-get final :cons-cells) 0)
227 (or (plist-get initial :cons-cells) 0))))
228
229 ;; Cons cells増加チェック
230 (when (&gt;= cons-delta nskk-bottleneck-memory-delta-threshold)
231 (push (list :excessive-cons-allocation
232 :description (format &quot;Excessive cons cell allocation: %d cells&quot;
233 cons-delta)
234 :severity 'high)
235 nskk-bottleneck--memory-issues))
236
237 ;; その他のメモリ問題チェック
238 (dolist (key '(:floats :vectors :strings))
239 (let* ((delta (- (or (plist-get final key) 0)
240 (or (plist-get initial key) 0)))
241 (threshold (* nskk-bottleneck-memory-delta-threshold 0.1)))
242 (when (&gt;= delta threshold)
243 (push (list (intern (format &quot;excessive-%s-allocation&quot; (substring (symbol-name key) 1)))
244 :description (format &quot;High %s allocation: %d&quot;
245 (substring (symbol-name key) 1)
246 delta)
247 :severity 'medium)
248 nskk-bottleneck--memory-issues)))))))

250(defun nskk-bottleneck--analyze-gc ()
251 &quot;GCパフォーマンスを分析してボトルネックを検出する。&quot;
252 (setq nskk-bottleneck--gc-issues nil)
253
254 (when (&gt; (length nskk-profiler--gc-events) 0)
255 (let* ((elapsed (if nskk-bottleneck--start-time
256 (float-time (time-subtract (current-time)
257 nskk-bottleneck--start-time))
258 1.0))
259 (gc-count (length nskk-profiler--gc-events))
260 (gc-frequency (/ gc-count elapsed)))
261
262 ;; GC頻度チェック
263 (when (&gt;= gc-frequency nskk-bottleneck-gc-frequency-threshold)
264 (push (list :high-gc-frequency
265 :description (format &quot;High GC frequency: %.2f GCs/sec&quot; gc-frequency)
266 :severity 'high)
267 nskk-bottleneck--gc-issues))
268
269 ;; GC時間チェック
270 (when (&gt; gc-count 0)
271 (let* ((events (reverse nskk-profiler--gc-events))
272 (first-gc (plist-get (cadr (car events)) :gc-elapsed))
273 (last-gc (plist-get (cadr (car (last events))) :gc-elapsed))
274 (gc-time-delta (- last-gc first-gc))
275 (gc-ratio (/ gc-time-delta elapsed)))
276 (when (&gt;= gc-ratio 0.1) ; GC時間が全体の10%以上
277 (push (list :excessive-gc-time
278 :description (format &quot;Excessive GC time: %.2f%% of total&quot;
279 (* gc-ratio 100))
280 :severity 'high)
281 nskk-bottleneck--gc-issues)))))))

285(defun nskk-bottleneck--generate-alerts ()
286 &quot;検出されたボトルネックに対してアラートを生成する。&quot;
287 (let ((total-issues (+ (length nskk-bottleneck--hotpaths)
288 (length nskk-bottleneck--slow-functions)
289 (length nskk-bottleneck--memory-issues)
290 (length nskk-bottleneck--gc-issues))))
291 (when (&gt; total-issues 0)
292 (message &quot;⚠ NSKK Bottleneck Alert: %d issue(s) detected&quot; total-issues)
293
294 ;; 重要度の高い問題を個別に通知
295 (dolist (issue nskk-bottleneck--memory-issues)
296 (when (eq (plist-get (cdr issue) :severity) 'high)
297 (message &quot; • Memory: %s&quot; (plist-get (cdr issue) :description))))
298
299 (dolist (issue nskk-bottleneck--gc-issues)
300 (when (eq (plist-get (cdr issue) :severity) 'high)
301 (message &quot; • GC: %s&quot; (plist-get (cdr issue) :description))))
302
303 (when (&gt; (length nskk-bottleneck--slow-functions) 0)
304 (let ((slowest (car nskk-bottleneck--slow-functions)))
305 (message &quot; • Slowest function: %s (%.3fs avg)&quot;
306 (car slowest)
307 (plist-get (cdr slowest) :avg-time)))))))

311(defun nskk-bottleneck-report ()
312 &quot;ボトルネック検出レポートを表示する。&quot;
313 (interactive)
314 (with-output-to-temp-buffer &quot;*NSKK Bottleneck Report*&quot;
315 (princ &quot;=======================================================\n&quot;)
316 (princ &quot; NSKK Bottleneck Detection Report\n&quot;)
317 (princ &quot;=======================================================\n\n&quot;)
318
319 ;; サマリー
320 (princ (format &quot;Total Issues Found: %d\n&quot;
321 (+ (length nskk-bottleneck--hotpaths)
322 (length nskk-bottleneck--slow-functions)
323 (length nskk-bottleneck--memory-issues)
324 (length nskk-bottleneck--gc-issues))))
325 (princ &quot;\n&quot;)
326
327 ;; ホットパス
328 (when nskk-bottleneck--hotpaths
329 (princ &quot;--- Hot Paths (Frequently Called) ---\n\n&quot;)
330 (princ (format &quot;%-40s %12s %15s %15s\n&quot;
331 &quot;Function&quot; &quot;Calls&quot; &quot;Total(s)&quot; &quot;Avg(s)&quot;))
332 (princ (make-string 82 ?-))
333 (princ &quot;\n&quot;)
334 (dolist (hotpath (cl-subseq nskk-bottleneck--hotpaths 0
335 (min 10 (length nskk-bottleneck--hotpaths))))
336 (let ((name (car hotpath))
337 (info (cdr hotpath)))
338 (princ (format &quot;%-40s %12d %15.6f %15.6f\n&quot;
339 name
340 (plist-get info :call-count)
341 (plist-get info :total-time)
342 (plist-get info :avg-time)))))
343 (princ &quot;\n&quot;))
344
345 ;; 遅い関数
346 (when nskk-bottleneck--slow-functions
347 (princ &quot;--- Slow Functions (High Latency) ---\n\n&quot;)
348 (princ (format &quot;%-40s %15s %15s %12s\n&quot;
349 &quot;Function&quot; &quot;Max(s)&quot; &quot;Avg(s)&quot; &quot;Calls&quot;))
350 (princ (make-string 82 ?-))
351 (princ &quot;\n&quot;)
352 (dolist (slow (cl-subseq nskk-bottleneck--slow-functions 0
353 (min 10 (length nskk-bottleneck--slow-functions))))
354 (let ((name (car slow))
355 (info (cdr slow)))
356 (princ (format &quot;%-40s %15.6f %15.6f %12d\n&quot;
357 name
358 (plist-get info :max-time)
359 (plist-get info :avg-time)
360 (plist-get info :call-count)))))
361 (princ &quot;\n&quot;))
362
363 ;; メモリ問題
364 (when nskk-bottleneck--memory-issues
365 (princ &quot;--- Memory Issues ---\n\n&quot;)
366 (dolist (issue nskk-bottleneck--memory-issues)
367 (let ((severity (plist-get (cdr issue) :severity))
368 (desc (plist-get (cdr issue) :description)))
369 (princ (format &quot;[%s] %s\n&quot;
370 (upcase (symbol-name severity))
371 desc))))
372 (princ &quot;\n&quot;))
373
374 ;; GC問題
375 (when nskk-bottleneck--gc-issues
376 (princ &quot;--- Garbage Collection Issues ---\n\n&quot;)
377 (dolist (issue nskk-bottleneck--gc-issues)
378 (let ((severity (plist-get (cdr issue) :severity))
379 (desc (plist-get (cdr issue) :description)))
380 (princ (format &quot;[%s] %s\n&quot;
381 (upcase (symbol-name severity))
382 desc))))
383 (princ &quot;\n&quot;))
384
385 ;; 推奨事項
386 (princ &quot;--- Recommendations ---\n\n&quot;)
387 (nskk-bottleneck--generate-recommendations)
388
389 (princ &quot;\n=======================================================\n&quot;)))

391(defun nskk-bottleneck--generate-recommendations ()
392 &quot;ボトルネックに対する推奨事項を生成する。&quot;
393 (let ((recommendations nil))
394
395 ;; ホットパス最適化
396 (when nskk-bottleneck--hotpaths
397 (push &quot;• Optimize hot paths: Consider caching or memoization for frequently called functions&quot;
398 recommendations))
399
400 ;; 遅延関数最適化
401 (when nskk-bottleneck--slow-functions
402 (push &quot;• Reduce latency: Profile slow functions with `profiler-report' for detailed analysis&quot;
403 recommendations))
404
405 ;; メモリ最適化
406 (when nskk-bottleneck--memory-issues
407 (push &quot;• Reduce memory allocation: Consider using `cl-loop' instead of `mapcar' for large datasets&quot;
408 recommendations)
409 (push &quot;• Increase gc-cons-threshold temporarily during heavy operations&quot;
410 recommendations))
411
412 ;; GC最適化
413 (when nskk-bottleneck--gc-issues
414 (push &quot;• Reduce GC pressure: Minimize temporary object creation&quot;
415 recommendations)
416 (push &quot;• Consider using object pooling for frequently allocated objects&quot;
417 recommendations))
418
419 (if recommendations
420 (dolist (rec (reverse recommendations))
421 (princ (format &quot;%s\n&quot; rec)))
422 (princ &quot;No specific recommendations. Performance looks good!\n&quot;))))

460 &quot;ボトルネック検出器の状態を返す。
