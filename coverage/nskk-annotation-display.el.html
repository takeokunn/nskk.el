27;; このファイルはSKK注釈の表示システムを実装します。
28;;
29;; 特徴:
30;; - ポップアップ表示（tooltip）
31;; - インライン表示（候補横）
32;; - エコーエリア表示
33;; - オーバーレイ表示
34;; - 画像・音声対応準備
35;; - カスタマイズ可能な表示スタイル
36;;
37;; 表示方式:
38;;
39;; 1. tooltip方式（デフォルト）:
40;; マウスポインタ付近にツールチップ表示
41;;
42;; 2. inline方式:
43;; 候補文字列の直後に注釈を表示
44;; 例: 「愛 (love)」
45;;
46;; 3. echo-area方式:
47;; エコーエリアに注釈を表示
48;;
49;; 4. overlay方式:
50;; バッファ内の候補上にオーバーレイ表示
51;;
52;; 5. popup方式:
53;; 専用ウィンドウでポップアップ表示
54;;
55;; 使用例:
56;;
57;; (require 'nskk-annotation-display)
58;;
59;; ;; 注釈を表示
60;; (nskk-show-annotation annotation)
61;;
62;; ;; 表示スタイルを変更
63;; (setq nskk-annotation-display-style 'inline)
64;;
65;; ;; 注釈を非表示
66;; (nskk-hide-annotation)

160(defun nskk-show-annotation (annotation &optional position)
161 &quot;ANNOTATION を表示する。
162POSITION が指定された場合、その位置に表示する（デフォルト: ポイント位置）。&quot;
163 (when (and annotation
164 (not (eq nskk-annotation-display-style 'none)))
165 ;; 既存の注釈を非表示
166 (nskk-hide-annotation)
167
168 ;; 表示遅延がある場合はタイマーを設定
169 (if (and nskk-annotation-display-delay
170 (&gt; nskk-annotation-display-delay 0))
171 (setq nskk-annotation-display--timer
172 (run-with-timer
173 nskk-annotation-display-delay nil
174 #'nskk-annotation-display--show-internal
175 annotation position))
176 ;; 遅延なしで即座に表示
177 (nskk-annotation-display--show-internal annotation position))))
178
179(defun nskk-annotation-display--show-internal (annotation position)
180 &quot;ANNOTATION を実際に表示する内部関数。&quot;
181 (setq nskk-annotation-display--current-annotation annotation)
182
183 ;; 表示スタイルに応じて表示
184 (pcase nskk-annotation-display-style
185 ('tooltip
186 (nskk-annotation-display--show-tooltip annotation position))
187 ('inline
188 (nskk-annotation-display--show-inline annotation position))
189 ('echo-area
190 (nskk-annotation-display--show-echo-area annotation))
191 ('overlay
192 (nskk-annotation-display--show-overlay annotation position))
193 ('popup
194 (nskk-annotation-display--show-popup annotation position)))
195
196 ;; 自動非表示タイマーを設定
197 (when nskk-annotation-display-duration
198 (setq nskk-annotation-display--hide-timer
199 (run-with-timer nskk-annotation-display-duration nil
200 #'nskk-hide-annotation))))

202;;;###autoload
203(defun nskk-hide-annotation ()
204 &quot;表示中の注釈を非表示にする。&quot;
205 (interactive)
206
207 ;; タイマーをキャンセル
208 (when nskk-annotation-display--timer
209 (cancel-timer nskk-annotation-display--timer)
210 (setq nskk-annotation-display--timer nil))
211
212 (when nskk-annotation-display--hide-timer
213 (cancel-timer nskk-annotation-display--hide-timer)
214 (setq nskk-annotation-display--hide-timer nil))
215
216 ;; オーバーレイを削除
217 (when nskk-annotation-display--overlay
218 (delete-overlay nskk-annotation-display--overlay)
219 (setq nskk-annotation-display--overlay nil))
220
221 ;; 状態をクリア
222 (setq nskk-annotation-display--current-annotation nil))

233(defun nskk-annotation-display--show-inline (annotation position)
234 &quot;ANNOTATION をインライン表示する。&quot;
235 (let* ((text (nskk-annotation-display--format-text annotation))
236 (pos (or position (point)))
237 (formatted (format nskk-annotation-display-inline-format text)))
238 (when (and text (not (string-empty-p text)))
239 (setq nskk-annotation-display--overlay
240 (make-overlay pos pos))
241 (overlay-put nskk-annotation-display--overlay
242 'after-string
243 (propertize formatted 'face nskk-annotation-display-face)))))

245(defun nskk-annotation-display--show-echo-area (annotation)
246 &quot;ANNOTATION をエコーエリアに表示する。&quot;
247 (let ((text (nskk-annotation-display--format-text annotation)))
248 (when text
249 (message &quot;%s&quot; text))))

251(defun nskk-annotation-display--show-overlay (annotation position)
252 &quot;ANNOTATION をオーバーレイで表示する。&quot;
253 (let* ((text (nskk-annotation-display--format-text annotation))
254 (pos (or position (point)))
255 (end-pos (min (+ pos 1) (point-max))))
256 (when text
257 (setq nskk-annotation-display--overlay
258 (make-overlay pos end-pos))
259 (overlay-put nskk-annotation-display--overlay
260 'before-string
261 (concat (propertize text 'face nskk-annotation-display-face) &quot;\n&quot;)))))

267 (nskk-annotation-display--show-tooltip annotation position))

271(defun nskk-annotation-display--format-text (annotation)
272 &quot;ANNOTATION を表示用テキストに整形する。&quot;
273 (when annotation
274 (let ((text (nskk-annotation-parser-get-description annotation)))
275 ;; 画像注釈の場合は画像を含める（オプション）
276 (when (and nskk-annotation-display-enable-images
277 (nskk-annotation-parser-has-media-p annotation)
278 (eq (nskk-annotation-media-type annotation) 'image))
279 (let ((image-text (nskk-annotation-display--format-image annotation)))
280 (when image-text
281 (setq text (concat text &quot;\n&quot; image-text)))))
282
283 ;; 最大幅で折り返し
284 (setq text (nskk-annotation-display--wrap-text text
285 nskk-annotation-display-max-width))
286
287 ;; 最大高さで切り詰め
288 (let ((lines (split-string text &quot;\n&quot;)))
289 (when (&gt; (length lines) nskk-annotation-display-max-height)
290 (setq lines (seq-take lines nskk-annotation-display-max-height))
291 (setq text (string-join lines &quot;\n&quot;))))
292
293 text)))
294
295(defun nskk-annotation-display--wrap-text (text max-width)
296 &quot;TEXT を MAX-WIDTH で折り返す。&quot;
297 (if (&lt;= (length text) max-width)
298 text
299 (let ((lines nil)
300 (current-line &quot;&quot;)
301 (words (split-string text)))
302 (dolist (word words)
303 (if (&gt; (+ (length current-line) (length word) 1) max-width)
304 (progn
305 (push current-line lines)
306 (setq current-line word))
307 (setq current-line
308 (if (string-empty-p current-line)
309 word
310 (concat current-line &quot; &quot; word)))))
311 (when (not (string-empty-p current-line))
312 (push current-line lines))
313 (string-join (nreverse lines) &quot;\n&quot;))))

315(defun nskk-annotation-display--format-image (annotation)
316 &quot;ANNOTATION から画像を表示用にフォーマットする。&quot;
317 (when-let ((path (nskk-annotation-parser-get-media-path annotation)))
318 (condition-case err
319 (when (and (file-exists-p path)
320 (image-type-available-p (image-type-from-file-name path)))
321 ;; 画像を読み込んでサイズ調整
322 (let* ((image (create-image path nil nil
323 :max-width nskk-annotation-display-image-max-width
324 :max-height nskk-annotation-display-image-max-height))
325 (image-string (propertize &quot; &quot; 'display image)))
326 image-string))
327 (error
328 (format &quot;[Image error: %s]&quot; (error-message-string err))))))

332;;;###autoload
333(defun nskk-annotation-display-with-candidate (candidate annotation)
334 &quot;CANDIDATE と ANNOTATION を組み合わせて表示用文字列を生成する。&quot;
335 (if (and annotation
336 (eq nskk-annotation-display-style 'inline))
337 (let ((ann-text (nskk-annotation-display--format-text annotation)))
338 (concat candidate
339 (when (and ann-text (not (string-empty-p ann-text)))
340 (format nskk-annotation-display-inline-format ann-text))))
341 candidate))

343;;;###autoload
344(defun nskk-annotation-display-candidates-with-annotations (candidates)
345 &quot;CANDIDATES（(word . annotation)のリスト）を表示用文字列のリストに変換する。&quot;
346 (mapcar (lambda (cand)
347 (nskk-annotation-display-with-candidate
348 (car cand)
349 (when (cdr cand)
350 (nskk-parse-annotation (cdr cand)))))
351 candidates))
