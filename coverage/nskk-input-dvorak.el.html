69;;; Dvorak用のローマ字テーブル生成
70
71(defun nskk-input-dvorak-convert-table ()
72 &quot;Dvorak配列用のローマ字テーブルを生成する。
73QWERTYベースのローマ字テーブルをDvorak配列に変換。&quot;
74 (let ((dvorak-table nil))
75 (dolist (entry nskk-romaji-table)
76 (let ((romaji (car entry))
77 (kana (cdr entry))
78 (converted &quot;&quot;))
79 ;; ローマ字の各文字をDvorakキーに変換
80 (dotimes (i (length romaji))
81 (let* ((char (substring romaji i (1+ i)))
82 (dvorak-char (or (car (rassoc char nskk-input-dvorak-key-map))
83 char)))
84 (setq converted (concat converted dvorak-char))))
85 (push (cons converted kana) dvorak-table)))
86 (nreverse dvorak-table)))
87
88(defconst nskk-input-dvorak-table
89 (nskk-input-dvorak-convert-table)
90 &quot;Dvorak配列用の変換テーブル。&quot;)

97(defun nskk-input-dvorak-init-hash-table ()
98 &quot;Dvorak入力方式用のハッシュテーブルを初期化する。&quot;
99 (setq nskk-input-dvorak-hash-table
100 (make-hash-table :test 'equal :size 500))
101 (dolist (entry nskk-input-dvorak-table)
102 (puthash (car entry) (cdr entry) nskk-input-dvorak-hash-table)))

106(defun nskk-input-dvorak-lookup (dvorak-romaji)
107 &quot;Dvorak入力方式でDVORAK-ROMAJIをひらがなに変換する。
108
109使用例:
110 (nskk-input-dvorak-lookup \&quot;ta\&quot;) ;; Dvorak配列での入力&quot;
111 (unless nskk-input-dvorak-hash-table

115(defun nskk-input-dvorak-get-candidates (prefix)
116 &quot;PREFIX で始まる全てのDvorakローマ字候補を取得する。&quot;
117 (let ((candidates nil))
118 (dolist (entry nskk-input-dvorak-table)
119 (when (string-prefix-p prefix (car entry))
120 (push (car entry) candidates)))
121 (nreverse candidates)))
