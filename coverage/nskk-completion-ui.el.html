157;;;###autoload
158(defun nskk-completion-ui-show (candidates &optional method)
159 &quot;補完候補を表示する。
160
161引数:
162 CANDIDATES - 補完候補（nskk-completion-engine-result構造体のリスト）
163 METHOD - 表示方法（省略時はnskk-completion-ui-display-method）
164
165処理:
166 指定された表示方法で補完候補を表示&quot;
167 (if (null candidates)
168 (progn
169 (nskk-completion-ui-hide)
170 nil)
171 (let ((display-method (or method nskk-completion-ui-display-method)))
172 (setq nskk-completion-ui--current-candidates candidates)
173 (setq nskk-completion-ui--current-index 0)
174
175 (pcase display-method
176 ('inline
177 (nskk-completion-ui--show-inline candidates))
178 ('popup
179 (nskk-completion-ui--show-popup candidates))
180 ('minibuffer
181 (nskk-completion-ui--show-minibuffer candidates))
182 ('tooltip
183 (nskk-completion-ui--show-tooltip candidates))
184 (_
185 (nskk-completion-ui--show-inline candidates))))))
186
187;;;###autoload
188(defun nskk-completion-ui-hide ()
189 &quot;補完候補表示を隠す。&quot;
190 (interactive)
191 (nskk-completion-ui--hide-inline)
192 (nskk-completion-ui--hide-popup)
193 (setq nskk-completion-ui--current-candidates nil)
194 (setq nskk-completion-ui--current-index 0)
195 (when nskk-completion-ui--timer
196 (cancel-timer nskk-completion-ui--timer)
197 (setq nskk-completion-ui--timer nil)))

200(defun nskk-completion-ui-show-delayed (candidates &optional method delay)
201 &quot;補完候補を遅延表示する。
202
203引数:
204 CANDIDATES - 補完候補
205 METHOD - 表示方法
206 DELAY - 遅延時間（秒、省略時はnskk-completion-ui-delay）&quot;
207 (when nskk-completion-ui--timer
208 (cancel-timer nskk-completion-ui--timer))
209
210 (setq nskk-completion-ui--timer
211 (run-with-timer (or delay nskk-completion-ui-delay)
212 nil
213 #'nskk-completion-ui-show
214 candidates
215 method)))

294(defun nskk-completion-ui--format-candidates (candidates display-method)
295 &quot;補完候補をフォーマットする（内部関数）。
296
297引数:
298 CANDIDATES - 補完候補
299 DISPLAY-METHOD - 表示方法
300
301戻り値:
302 フォーマット済み文字列&quot;
303 (let ((display-candidates (seq-take candidates nskk-completion-ui-max-display-items))
304 (lines nil))
305
306 (pcase display-method
307 ('inline
308 ;; インライン: 1行で表示
309 (mapconcat
310 (lambda (cand)
311 (nskk-completion-ui--format-single-candidate cand 'inline))
312 display-candidates
313 &quot; &quot;))
314
315 ('popup
316 ;; ポップアップ: 複数行
317 (mapconcat
318 (lambda (cand)
319 (nskk-completion-ui--format-single-candidate cand 'popup))
320 display-candidates
321 &quot;\n&quot;))
322
323 ('minibuffer
324 ;; ミニバッファ: 1行（コンパクト）
325 (concat &quot;補完候補: &quot;
326 (mapconcat
327 (lambda (cand)
328 (nskk-completion-ui--format-single-candidate cand 'minibuffer))
329 display-candidates
330 &quot; &quot;)))
331
332 ('tooltip
333 ;; ツールチップ: 複数行
334 (mapconcat
335 (lambda (cand)
336 (nskk-completion-ui--format-single-candidate cand 'tooltip))
337 display-candidates
338 &quot;\n&quot;))
339
340 (_
341 &quot;&quot;))))

343(defun nskk-completion-ui--format-single-candidate (candidate display-method)
344 &quot;単一の補完候補をフォーマットする（内部関数）。
345
346引数:
347 CANDIDATE - 補完候補
348 DISPLAY-METHOD - 表示方法
349
350戻り値:
351 フォーマット済み文字列&quot;
352 (let ((midashi (nskk-completion-engine-result-midashi candidate))
353 (rank (nskk-completion-engine-result-rank candidate))
354 (score (nskk-completion-engine-result-total-score candidate))
355 (parts nil))
356
357 ;; ランク
358 (when nskk-completion-ui-show-rank
359 (push (propertize (format &quot;%d&quot; rank)
360 'face 'nskk-completion-ui-rank-face)
361 parts)
362 (push (propertize &quot;.&quot; 'face 'nskk-completion-ui-separator-face)
363 parts))
364
365 ;; 見出し語
366 (push midashi parts)
367
368 ;; スコア
369 (when nskk-completion-ui-show-score
370 (push (propertize (format &quot; (%.3f)&quot; score)
371 'face 'nskk-completion-ui-score-face)
372 parts))
373
374 (apply #'concat (nreverse parts))))

411(defun nskk-completion-ui--update-display ()
412 &quot;表示を更新する（内部関数）。&quot;
413 (when nskk-completion-ui--current-candidates
414 (let ((saved-index nskk-completion-ui--current-index)
415 (candidates nskk-completion-ui--current-candidates)
416 (display-method nskk-completion-ui-display-method))
417 (pcase display-method
418 ('inline
419 (nskk-completion-ui--show-inline candidates))
420 ('popup
421 (nskk-completion-ui--show-popup candidates))
422 ('minibuffer

474;;;###autoload
475(defun nskk-completion-ui-complete (index query &rest args)
476 &quot;統合補完を実行してUIに表示する。
477
478引数:
479 INDEX - 辞書インデックス
480 QUERY - 検索クエリ
481
482キーワード引数:
483 nskk-completion-engine-searchと同じ
484
485処理:
486 1. 補完検索を実行
487 2. 結果をUIに表示
488
489戻り値:
490 補完候補のリスト&quot;
491 (let* ((results (apply #'nskk-completion-engine-search index query args))
492 (method (or (plist-get args :display-method)
493 nskk-completion-ui-display-method)))
494
495 (if nskk-completion-ui-auto-show
496 (if (&gt; nskk-completion-ui-delay 0)
497 (nskk-completion-ui-show-delayed results method)
498 (nskk-completion-ui-show results method))
499 ;; 自動表示しない場合は結果のみ返す
500 nil)
501
502 results))
