180;;;###autoload
181(defun nskk-ai-learning-online-update (features label)
182 &quot;オンライン学習で単一サンプルから学習する。
183FEATURES: 特徴ベクトル
184LABEL: 正解ラベル（候補テキスト）&quot;
185 (when (and nskk-ai-learning-enable-online
186 (vectorp features))
187
188 ;; サンプルを作成
189 (let ((sample (nskk-ai-learning-sample--create
190 :features features
191 :label label
192 :timestamp (current-time))))
193
194 ;; 訓練データに追加
195 (push sample nskk-ai-learning-training-data)
196
197 ;; バッチサイズに達したら学習
198 (when (&gt;= (length nskk-ai-learning-training-data)
199 nskk-ai-learning-batch-size)
200 (nskk-ai-learning--train-batch)
201 (setq nskk-ai-learning-training-data nil))
202
203 ;; 統計更新
204 (cl-incf (plist-get nskk-ai-learning-statistics :total-updates))
205 (plist-put nskk-ai-learning-statistics :last-update-time (current-time))
206
207 ;; 定期的にモデルを圧縮
208 (when (and nskk-ai-learning-enable-compression
209 (zerop (mod (nskk-ai-learning-model-training-steps
210 nskk-ai-learning-model)
211 nskk-ai-learning-auto-save-interval)))
212 (nskk-ai-learning-compress-model)))))
213
214(defun nskk-ai-learning--train-batch ()
215 &quot;ミニバッチで学習を実行する。&quot;
216 (let* ((model nskk-ai-learning-model)
217 (weights (nskk-ai-learning-model-weights model))
218 (velocity (nskk-ai-learning-model-velocity model))
219 (batch nskk-ai-learning-training-data)
220 (batch-size (length batch))
221 (total-loss 0.0))
222
223 (when (&gt; batch-size 0)
224 ;; 各サンプルで勾配を計算
225 (let ((gradient (make-vector (length weights) 0.0)))
226
227 (dolist (sample batch)
228 (let* ((features (nskk-ai-learning-sample-features sample))
229 (prediction (nskk-ai-learning--predict features))
230 (error (nskk-ai-learning--calc-error prediction sample))
231 (sample-weight (nskk-ai-learning-sample-weight sample)))
232
233 ;; 損失を累積
234 (setq total-loss (+ total-loss (* error error)))
235
236 ;; 勾配を計算（簡易版: error * feature）
237 (dotimes (i (min (length features) (length gradient)))
238 (aset gradient i
239 (+ (aref gradient i)
240 (* error (aref features i) sample-weight))))))
241
242 ;; 勾配の平均
243 (dotimes (i (length gradient))
244 (aset gradient i (/ (aref gradient i) batch-size)))
245
246 ;; モーメンタム付きSGD
247 (dotimes (i (length weights))
248 ;; 速度更新
249 (aset velocity i
250 (+ (* nskk-ai-learning-momentum (aref velocity i))
251 (* nskk-ai-learning-learning-rate (aref gradient i))))
252 ;; 重み更新
253 (aset weights i (- (aref weights i) (aref velocity i))))
254
255 ;; 統計更新
256 (let ((avg-loss (/ total-loss batch-size)))
257 (plist-put nskk-ai-learning-statistics :avg-loss avg-loss)
258 (push avg-loss (nskk-ai-learning-model-loss-history model))
259
260 ;; 損失履歴のサイズ制限
261 (when (&gt; (length (nskk-ai-learning-model-loss-history model)) 1000)
262 (setf (nskk-ai-learning-model-loss-history model)
263 (seq-take (nskk-ai-learning-model-loss-history model) 1000))))
264
265 ;; 学習ステップ数を増やす
266 (cl-incf (nskk-ai-learning-model-training-steps model))))))

268(defsubst nskk-ai-learning--predict (features)
269 &quot;特徴ベクトル FEATURES から予測値を計算する。&quot;
270 (let* ((model nskk-ai-learning-model)
271 (weights (nskk-ai-learning-model-weights model))
272 (bias (nskk-ai-learning-model-bias model))
273 (score bias))
274 ;; 内積を計算
275 (dotimes (i (min (length features) (length weights)))
276 (setq score (+ score (* (aref features i) (aref weights i)))))
277 ;; シグモイド関数で正規化
278 (/ 1.0 (+ 1.0 (exp (- score))))))
279
280(defsubst nskk-ai-learning--calc-error (prediction sample)

282簡易版: 常に予測値を0-1の範囲で評価し、正解は1とする。&quot;
283 (- prediction 1.0)) ; 正解ラベルを1.0と仮定

362;;;###autoload
363(defun nskk-ai-learning-compress-model ()
364 &quot;現在のモデルを圧縮する。
365戻り値: 圧縮されたモデル&quot;
366 (interactive)
367 (when nskk-ai-learning-enable-compression
368 (let* ((model nskk-ai-learning-model)
369 (weights (nskk-ai-learning-model-weights model))
370 (compressed-size (floor (* (length weights)
371 nskk-ai-learning-compression-ratio)))
372 (compressed-weights (make-vector compressed-size 0.0)))
373
374 ;; 重要度の高い重みのみ保持（簡易版: 絶対値の大きい順）
375 (let* ((weight-list nil)
376 (i 0))
377 ;; 重みをリストに変換
378 (dotimes (j (length weights))
379 (push (cons j (aref weights j)) weight-list))
380
381 ;; 絶対値でソート
382 (setq weight-list
383 (sort weight-list
384 (lambda (a b) (&gt; (abs (cdr a)) (abs (cdr b))))))
385
386 ;; 上位のみ保持
387 (dolist (entry (seq-take weight-list compressed-size))
388 (aset compressed-weights i (cdr entry))
389 (cl-incf i)))
390
391 ;; 圧縮モデルを保存
392 (let ((compressed-model
393 (nskk-ai-learning-model--create
394 :weights compressed-weights
395 :bias (nskk-ai-learning-model-bias model)
396 :version (1+ (nskk-ai-learning-model-version model)))))
397
398 (puthash (nskk-ai-learning-model-version model)
399 compressed-model
400 nskk-ai-learning-compressed-models)
401
402 ;; 統計更新
403 (plist-put nskk-ai-learning-statistics :model-size
404 (* compressed-size 8)) ; float = 8 bytes
405
406 compressed-model))))

429;;;###autoload
430(defun nskk-ai-learning-adjust-learning-rate ()
431 &quot;学習率を適応的に調整する。
432損失関数の履歴に基づいて自動調整。&quot;
433 (when nskk-ai-learning-adaptive-lr-enabled
434 (let ((loss-history (nskk-ai-learning-model-loss-history
435 nskk-ai-learning-model)))
436 (when (&gt; (length loss-history) 10)
437 ;; 最近の損失の傾向を分析
438 (let* ((recent-loss (seq-take loss-history 10))
439 (avg-recent (/ (apply #'+ recent-loss) 10.0))
440 (older-loss (seq-take (nthcdr 10 loss-history) 10))
441 (avg-older (if (&gt; (length older-loss) 0)
442 (/ (apply #'+ older-loss) (float (length older-loss)))
443 avg-recent)))
444
445 ;; 損失が改善していない場合、学習率を減衰
446 (when (&gt;= avg-recent avg-older)
447 (setq nskk-ai-learning-learning-rate
448 (* nskk-ai-learning-learning-rate
449 nskk-ai-learning-lr-decay-rate))))))))

455 &quot;テストサンプル TEST-SAMPLES でモデルを評価する。
456戻り値: plist形式の評価結果&quot;

464 (error (nskk-ai-learning--calc-error prediction sample)))

467 (setq total-loss (+ total-loss (* error error)))
468

470 (when (&gt; prediction 0.5)
471 (cl-incf correct))))
472
473 (list :accuracy (if (&gt; total 0) (/ (float correct) total) 0.0)
474 :avg-loss (if (&gt; total 0) (/ total-loss total) 0.0)
