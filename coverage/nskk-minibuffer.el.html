145(defun nskk-minibuffer-init ()
146 &quot;ミニバッファUIを初期化する。&quot;

150(defun nskk-minibuffer-cleanup ()
151 &quot;ミニバッファUIのリソースをクリーンアップする。&quot;
152 (when nskk-minibuffer--state
153 ;; オーバーレイの削除
154 (when (nskk-minibuffer-state-overlay nskk-minibuffer--state)
155 (delete-overlay (nskk-minibuffer-state-overlay nskk-minibuffer--state)))
156 (when (nskk-minibuffer-state-inline-overlay nskk-minibuffer--state)
157 (delete-overlay (nskk-minibuffer-state-inline-overlay nskk-minibuffer--state)))
158 (setq nskk-minibuffer--state nil)))

162(defun nskk-minibuffer--ensure-overlay ()
163 &quot;テキスト装飾用のオーバーレイを確保する。&quot;
164 (nskk-minibuffer-init)
165 (unless (and (nskk-minibuffer-state-overlay nskk-minibuffer--state)
166 (overlay-buffer (nskk-minibuffer-state-overlay nskk-minibuffer--state)))
167 (let ((ov (make-overlay (point-min) (point-min))))
168 (overlay-put ov 'nskk-minibuffer t)
169 (setf (nskk-minibuffer-state-overlay nskk-minibuffer--state) ov))))
170

172 &quot;インライン候補用のオーバーレイを確保する。&quot;
173 (nskk-minibuffer-init)
174 (unless (and (nskk-minibuffer-state-inline-overlay nskk-minibuffer--state)
175 (overlay-buffer (nskk-minibuffer-state-inline-overlay nskk-minibuffer--state)))
176 (let ((ov (make-overlay (point-min) (point-min))))
177 (overlay-put ov 'nskk-minibuffer-inline t)
178 (setf (nskk-minibuffer-state-inline-overlay nskk-minibuffer--state) ov))))

208(defun nskk-minibuffer-show (text &optional face mode)
209 &quot;ミニバッファに TEXT を表示する。
210
211引数:
212 TEXT - 表示するテキスト
213 FACE - 使用するフェイス（オプション）
214 MODE - 現在のモード（オプション）
215
216副作用:
217 - ミニバッファへのテキスト挿入
218 - オーバーレイの適用
219 - 状態の更新&quot;
220 (nskk-minibuffer-init)
221
222 (let* ((actual-mode (or mode 'hiragana))
223 (actual-face (or face (nskk-minibuffer--select-face actual-mode)))
224 (prompt (nskk-minibuffer--make-prompt actual-mode)))
225
226 ;; 状態更新
227 (setf (nskk-minibuffer-state-text nskk-minibuffer--state) text)
228 (setf (nskk-minibuffer-state-mode nskk-minibuffer--state) actual-mode)
229 (setf (nskk-minibuffer-state-prompt nskk-minibuffer--state) (or prompt &quot;&quot;))
230 (setf (nskk-minibuffer-state-visible nskk-minibuffer--state) t)
231
232 ;; overlayを使用する場合
233 (when nskk-minibuffer-use-overlay
234 (nskk-minibuffer--ensure-overlay)
235 (let ((ov (nskk-minibuffer-state-overlay nskk-minibuffer--state)))
236 ;; オーバーレイ位置の更新は呼び出し側が行う
237 (overlay-put ov 'face actual-face)
238 (when prompt
239 (overlay-put ov 'before-string prompt))))))

243
244副作用:

246 - 状態のクリア&quot;
247 (when nskk-minibuffer--state

256
257 ;; 状態クリア

259 (setf (nskk-minibuffer-state-inline-candidate nskk-minibuffer--state) nil)
260 (setf (nskk-minibuffer-state-visible nskk-minibuffer--state) nil)))

281(defun nskk-minibuffer-show-inline-candidate (candidate)
282 &quot;インライン候補 CANDIDATE を表示する。
283
284引数:
285 CANDIDATE - 表示する候補文字列
286
287副作用:
288 - インライン候補の表示
289 - after-string プロパティの設定
290
291インライン候補はカーソル位置の後にグレー表示される。
292TABキーで確定可能。&quot;
293 (when (and nskk-minibuffer-enable-inline
294 nskk-minibuffer--state
295 (nskk-minibuffer-state-visible nskk-minibuffer--state)
296 candidate
297 (not (string-empty-p candidate)))
298
299 (nskk-minibuffer--ensure-inline-overlay)
300
301 (let ((ov (nskk-minibuffer-state-inline-overlay nskk-minibuffer--state))
302 (inline-text (propertize candidate 'face nskk-minibuffer-inline-face)))
303
304 ;; after-stringを使用してインライン候補を表示
305 (overlay-put ov 'after-string inline-text)
306 (setf (nskk-minibuffer-state-inline-candidate nskk-minibuffer--state) candidate))))
