71(defmacro nskk-measure-time (&rest body)
72 &quot;BODY の実行時間を測定し、結果をマイクロ秒単位で返す。
73
74返り値: (実行時間(μs) . BODYの返り値)
75
76例:
77 (nskk-measure-time (nskk-convert-romaji \&quot;ka\&quot;))
78 ;; =&gt; (15.2 . #s(nskk-converter-result ...))&quot;
79 (declare (indent 0) (debug t))
80 (let ((start-time (make-symbol &quot;start-time&quot;))
81 (end-time (make-symbol &quot;end-time&quot;))
82 (result (make-symbol &quot;result&quot;)))
83 `(let ((,start-time (current-time))
84 (,result (progn ,@body))
85 (,end-time (current-time)))
86 (cons (* (float-time (time-subtract ,end-time ,start-time)) 1000000.0)
87 ,result))))

89(defmacro nskk-with-profiling (name &rest body)
90 &quot;BODY の実行をプロファイリングし、NAME として記録する。
91
92`nskk-optimize-enable-profiling' が非nilの場合のみ有効。
93
94例:
95 (nskk-with-profiling \&quot;romaji-conversion\&quot;
96 (nskk-convert-romaji \&quot;konnnichiha\&quot;))&quot;
97 (declare (indent 1) (debug t))
98 `(if nskk-optimize-enable-profiling
99 (let ((start-time (current-time))
100 (result (progn ,@body))
101 (end-time (current-time)))
102 (nskk-optimize--record-profile
103 ,name
104 (time-subtract end-time start-time))
105 result)
106 (progn ,@body)))

114(defun nskk-optimize--record-profile (name elapsed-time)
115 &quot;プロファイリングデータを記録する。
116NAME: プロファイル名
117ELAPSED-TIME: 経過時間（time-subtract の返り値）&quot;
118 (let* ((elapsed-us (* (float-time elapsed-time) 1000000.0))
119 (entry (assoc name nskk-optimize--profile-data))
120 (count (if entry (nth 1 entry) 0))
121 (total (if entry (nth 2 entry) 0.0))
122 (min-time (if entry (min (nth 3 entry) elapsed-us) elapsed-us))
123 (max-time (if entry (max (nth 4 entry) elapsed-us) elapsed-us)))
124 (setf (alist-get name nskk-optimize--profile-data nil nil #'string=)
125 (list (1+ count) (+ total elapsed-us) min-time max-time))))

132(defun nskk-optimize-show-profile ()
133 &quot;プロファイリングデータを表示する。&quot;
134 (interactive)
135 (if (null nskk-optimize--profile-data)
136 (message &quot;No profiling data available&quot;)
137 (with-output-to-temp-buffer &quot;*NSKK Profile*&quot;
138 (princ (format &quot;%-30s %10s %15s %15s %15s %15s\n&quot;
139 &quot;Name&quot; &quot;Count&quot; &quot;Total(μs)&quot; &quot;Avg(μs)&quot; &quot;Min(μs)&quot; &quot;Max(μs)&quot;))
140 (princ (make-string 110 ?-))
141 (princ &quot;\n&quot;)
142 (dolist (entry (sort (copy-sequence nskk-optimize--profile-data)
143 (lambda (a b) (&gt; (nth 2 (cdr a)) (nth 2 (cdr b))))))
144 (let* ((name (car entry))
145 (count (nth 1 entry))
146 (total (nth 2 entry))
147 (min-time (nth 3 entry))
148 (max-time (nth 4 entry))
149 (avg (/ total count)))
150 (princ (format &quot;%-30s %10d %15.2f %15.2f %15.2f %15.2f\n&quot;
151 name count total avg min-time max-time)))))))

155(defun nskk-benchmark-romaji-conversion (input iterations)
156 &quot;ローマ字変換のベンチマークを実行する。
157
158INPUT: 変換する入力文字列
159ITERATIONS: 反復回数
160
161返り値: plist (:total-time :avg-time :iterations :throughput)
162 :total-time - 合計実行時間（秒）
163 :avg-time - 平均実行時間（マイクロ秒）
164 :iterations - 反復回数
165 :throughput - スループット（回/秒）&quot;
166 (require 'nskk-converter)
167 (garbage-collect) ; GC の影響を最小化
168 (let ((start-time (current-time))
169 (result nil))
170 (dotimes (_ iterations)
171 (setq result (nskk-convert-romaji input)))
172 (let* ((end-time (current-time))
173 (elapsed (float-time (time-subtract end-time start-time)))
174 (avg-us (* (/ elapsed iterations) 1000000.0))
175 (throughput (/ iterations elapsed)))
176 (list :total-time elapsed
177 :avg-time avg-us
178 :iterations iterations
179 :throughput throughput
180 :input input
181 :result result))))

300(defun nskk-measure-memory-delta (func)
301 &quot;FUNC の実行前後のメモリ使用量の差分を測定する。
302
303FUNC: 引数なしの関数
304
305返り値: plist
306 :before - 実行前のメモリ使用量
307 :after - 実行後のメモリ使用量
308 :delta - 差分&quot;
309 (garbage-collect)
310 (let ((before (nskk-benchmark-memory-usage)))
311 (funcall func)
312 (garbage-collect)
313 (let* ((after (nskk-benchmark-memory-usage))
314 (delta (list :cons-cells
315 (- (plist-get after :cons-cells-allocated)
316 (plist-get before :cons-cells-allocated))
317 :floats
318 (- (plist-get after :floats-allocated)
319 (plist-get before :floats-allocated))
320 :strings
321 (- (plist-get after :strings-allocated)
322 (plist-get before :strings-allocated)))))
323 (list :before before :after after :delta delta))))

327(defmacro nskk-optimize-loop (var limit &rest body)
328 &quot;最適化されたループマクロ。
329VARが0からLIMIT-1まで反復し、BODYを実行する。
330
331通常のdotimesより高速だが、副作用に注意。
332
333例:
334 (nskk-optimize-loop i 1000
335 (do-something i))&quot;
336 (declare (indent 2) (debug (symbolp form body)))
337 `(let ((,var 0)
338 (limit ,limit))
339 (while (&lt; ,var limit)
340 ,@body
341 (setq ,var (1+ ,var)))))

364(defun nskk-optimize-check-regression ()
365 &quot;パフォーマンス回帰をチェックする。
366
367ベースラインと比較して、10%以上の性能低下がある場合は警告。
368
369返り値: plist
370 :baseline - ベースライン平均時間
371 :current - 現在の平均時間
372 :ratio - 比率（current/baseline）
373 :regression - 回帰の有無（10%以上遅い場合t）&quot;
374 (interactive)
375 (unless nskk-optimize--performance-baseline
376 (error &quot;No baseline set. Run `nskk-optimize-set-baseline' first&quot;))
377
378 (let* ((current (nskk-benchmark-romaji-conversion &quot;konnnichiha&quot; 1000))
379 (baseline-time (plist-get nskk-optimize--performance-baseline :avg-time))
380 (current-time (plist-get current :avg-time))
381 (ratio (/ current-time baseline-time))
382 (regression (&gt; ratio 1.1)))
383
384 (let ((result (list :baseline baseline-time
385 :current current-time
386 :ratio ratio
387 :regression regression)))
388 (when (called-interactively-p 'any)
389 (message &quot;Baseline: %.2fμs, Current: %.2fμs, Ratio: %.2f%s&quot;
390 baseline-time current-time ratio
391 (if regression &quot; [REGRESSION DETECTED]&quot; &quot;&quot;)))
392 result)))
