137;;; ハッシュテーブル
138
139(defvar nskk-input-kana-hash-table nil
140 &quot;かな入力方式用のハッシュテーブル。&quot;)
141
142(defun nskk-input-kana-init-hash-table ()
143 &quot;かな入力方式用のハッシュテーブルを初期化する。&quot;
144 (setq nskk-input-kana-hash-table
145 (make-hash-table :test 'equal :size 150))
146 (dolist (entry nskk-input-kana-table)
147 (puthash (car entry) (cdr entry) nskk-input-kana-hash-table)))
148
149;;; 検索関数
150
151(defun nskk-input-kana-lookup (key)
152 &quot;かな入力方式でKEYをかなに変換する。
153
154使用例:
155 (nskk-input-kana-lookup \&quot;k\&quot;) ;; =&gt; \&quot;の\&quot;
156 (nskk-input-kana-lookup \&quot;a\&quot;) ;; =&gt; \&quot;ち\&quot;&quot;
157 (unless nskk-input-kana-hash-table
158 (nskk-input-kana-init-hash-table))
159 (gethash key nskk-input-kana-hash-table))
160
161(defun nskk-input-kana-get-candidates (prefix)
162 &quot;PREFIX で始まる全てのかな入力候補を取得する。&quot;
163 (let ((candidates nil))
164 (dolist (entry nskk-input-kana-table)
165 (when (string-prefix-p prefix (car entry))
166 (push (car entry) candidates)))
167 (nreverse candidates)))

171(defun nskk-input-kana-register ()
172 &quot;かな入力方式をNSKKに登録する。&quot;
173 (nskk-input-kana-init-hash-table))
