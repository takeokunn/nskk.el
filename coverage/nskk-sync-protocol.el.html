195(defun nskk-sync-connect (session)
196 &quot;SESSION を使用してサーバーに接続する。&quot;
197 (when (eq (nskk-sync-session-state session) 'connected)
198 (warn &quot;Session already connected&quot;)
199 (cl-return-from nskk-sync-connect session))
200
201 (setf (nskk-sync-session-state session) 'connecting)
202
203 (condition-case err
204 (let* ((url (nskk-sync-session-server-url session))
205 (connection (nskk-sync--establish-connection url session)))
206 (setf (nskk-sync-session-connection session) connection)
207 (setf (nskk-sync-session-state session) 'connected)
208 (setf (nskk-sync-session-reconnect-count session) 0)
209
210 ;; Pingタイマー開始
211 (nskk-sync--start-ping-timer session)
212
213 ;; キューにあるメッセージを送信
214 (nskk-sync--flush-message-queue session)
215
216 ;; 接続成功コールバック
217 (nskk-sync--invoke-callback session 'on-connect)
218
219 session)
220 (error
221 (setf (nskk-sync-session-state session) 'error)
222 (nskk-sync--invoke-callback session 'on-error err)
223 (nskk-sync--maybe-reconnect session)
224 (signal (car err) (cdr err)))))
