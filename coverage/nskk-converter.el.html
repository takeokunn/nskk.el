63 &quot;促音として扱う子音文字。

68(defcustom nskk-converter-n-processing-mode 'smart
69 &quot;撥音「ん」の処理モード。
70- `smart': 文脈依存（n+子音で自動確定、n+母音で継続）
71- `explicit': nn, n', xn のみで確定
72- `aggressive': n単独でも即座に確定&quot;
73 :type '(choice (const :tag &quot;スマート処理&quot; smart)
74 (const :tag &quot;明示的のみ&quot; explicit)
75 (const :tag &quot;積極的確定&quot; aggressive))
76 :group 'nskk-converter)

91(defsubst nskk-converter--is-vowel (char)
92 &quot;CHAR が母音（a, i, u, e, o）かどうかを判定する。&quot;
93 (and char (memq char '(?a ?i ?u ?e ?o))))

95(defsubst nskk-converter--is-sokuon-char (char)
96 &quot;CHAR が促音化可能な子音かどうかを判定する。&quot;
97 (and char
98 nskk-converter-use-sokuon
99 (string-match-p (regexp-quote (char-to-string char))
100 nskk-converter-sokuon-chars)))

102(defsubst nskk-converter--can-be-sokuon (str)
103 &quot;STR が促音パターン（同じ子音の連続）かどうかを判定する。
104例: \&quot;kk\&quot;, \&quot;tt\&quot;, \&quot;pp\&quot; など&quot;
105 (and (&gt;= (length str) 2)
106 (= (aref str 0) (aref str 1))
107 (nskk-converter--is-sokuon-char (aref str 0))))

109(defun nskk-converter--process-sokuon (input)
110 &quot;INPUT から促音パターンを処理する。
111促音が検出された場合は (\&quot;っ\&quot; . 残り) を返す。
112検出されない場合は nil を返す。
113
114例:
115 (nskk-converter--process-sokuon \&quot;kka\&quot;) ;; =&gt; (\&quot;っ\&quot; . \&quot;ka\&quot;)
116 (nskk-converter--process-sokuon \&quot;ka\&quot;) ;; =&gt; nil
117 (nskk-converter--process-sokuon \&quot;kk\&quot;) ;; =&gt; nil (未確定)&quot;
118 (when (and nskk-converter-use-sokuon
119 ;; 3文字以上必要（kk + 続く文字）
120 (&gt;= (length input) 3))
121 (let ((first-char (aref input 0))
122 (second-char (aref input 1)))
123 (when (and (= first-char second-char)
124 (nskk-converter--is-sokuon-char first-char)
125 ;; &quot;nn&quot; は促音ではなく撥音
126 (not (= first-char ?n)))
127 (cons &quot;っ&quot; (substring input 1))))))

129(defun nskk-converter--process-n (input)
130 &quot;INPUT の先頭が 'n' の場合の特殊処理を行う。
131撥音「ん」として確定できる場合は (\&quot;ん\&quot; . 残り) を返す。
132確定できない場合は nil を返す。
133
134処理ルール:
135- nn, n', xn → 即座に「ん」として確定
136- n + 子音 → スマートモードでは「ん」として確定
137- n + 母音 → 確定せず（「な」等の可能性）
138- n単独 → モード依存&quot;
139 (when (and (&gt;= (length input) 1)
140 (= (aref input 0) ?n))
141 (cond
142 ;; 2文字以上の場合
143 ((&gt;= (length input) 2)
144 (let ((second-char (aref input 1)))
145 (cond
146 ;; nn, n', xn → 確定
147 ((or (= second-char ?n)
148 (= second-char ?')
149 (and (= (aref input 0) ?x) (= second-char ?n)))
150 (cons &quot;ん&quot; (substring input 2)))
151
152 ;; n + 母音 → 確定しない（na, ni, nu, ne, no の可能性）
153 ((nskk-converter--is-vowel second-char)
154 nil)
155
156 ;; n + 子音 → smartモードでは確定
157 ((eq nskk-converter-n-processing-mode 'smart)
158 (cons &quot;ん&quot; (substring input 1)))
159
160 ;; それ以外 → 確定しない
161 (t nil))))
162
163 ;; n単独の場合
164 ((= (length input) 1)
165 (when (eq nskk-converter-n-processing-mode 'aggressive)
166 (cons &quot;ん&quot; &quot;&quot;)))
167
168 (t nil))))

170(defun nskk-converter--longest-match (input)
171 &quot;INPUT に対して最長一致するローマ字変換を試みる。
172成功した場合は (変換後文字列 . 残り) を返す。
173失敗した場合は nil を返す。
174
175アルゴリズム:
1761. INPUTの長さから降順に検索
1772. 最初にマッチした（最長の）パターンを採用
1783. マッチしない場合は nil
179
180例:
181 (nskk-converter--longest-match \&quot;kya\&quot;) ;; =&gt; (\&quot;きゃ\&quot; . \&quot;\&quot;)
182 (nskk-converter--longest-match \&quot;ka\&quot;) ;; =&gt; (\&quot;か\&quot; . \&quot;\&quot;)
183 (nskk-converter--longest-match \&quot;xyz\&quot;) ;; =&gt; nil&quot;
184 (nskk-converter--init-max-length)
185 (let* ((max-len (min (length input) nskk-converter--max-romaji-length))
186 (result nil))
187 (cl-loop for len from max-len downto 1
188 until result
189 do (let* ((prefix (substring input 0 len))
190 (kana (nskk-romaji-lookup prefix)))
191 (when kana
192 (setq result (cons kana (substring input len))))))
193 result))

195(defun nskk-converter--has-prefix-match (input)
196 &quot;INPUT が何らかのローマ字パターンのプレフィックスかどうかを判定する。
197プレフィックスマッチがある場合は t、ない場合は nil を返す。
198
199これは未確定入力の判定に使用される。
200
201例:
202 (nskk-converter--has-prefix-match \&quot;k\&quot;) ;; =&gt; t (\&quot;ka\&quot;, \&quot;ki\&quot; 等の候補)
203 (nskk-converter--has-prefix-match \&quot;ky\&quot;) ;; =&gt; t (\&quot;kya\&quot;, \&quot;kyu\&quot; 等の候補)
204 (nskk-converter--has-prefix-match \&quot;xyz\&quot;) ;; =&gt; nil (候補なし)&quot;
205 (let ((candidates (nskk-romaji-get-candidates input)))
206 (and candidates t)))

223(defun nskk-convert-romaji (input)
224 &quot;INPUT をローマ字→かなに変換する。
225
226返り値は `nskk-converter-result' 構造体:
227 - converted: 確定した変換結果
228 - pending: 未確定の入力
229 - consumed: 消費した文字数
230
231変換ルール:
2321. 促音処理（kka → っか）
2332. 撥音処理（nn → ん、n+子音 → ん+子音）
2343. 最長一致変換
2354. プレフィックスマッチング（未確定判定）
236
237例:
238 (nskk-convert-romaji \&quot;kya\&quot;)
239 ;; =&gt; #s(nskk-converter-result \&quot;きゃ\&quot; \&quot;\&quot; 3)
240
241 (nskk-convert-romaji \&quot;k\&quot;)
242 ;; =&gt; #s(nskk-converter-result \&quot;\&quot; \&quot;k\&quot; 0)
243
244 (nskk-convert-romaji \&quot;kka\&quot;)
245 ;; =&gt; #s(nskk-converter-result \&quot;っか\&quot; \&quot;\&quot; 3)
246
247 (nskk-convert-romaji \&quot;nn\&quot;)
248 ;; =&gt; #s(nskk-converter-result \&quot;ん\&quot; \&quot;\&quot; 2)&quot;
249 (if (string-empty-p input)
250 (nskk-converter-result-create :converted &quot;&quot;
251 :pending &quot;&quot;
252 :consumed 0)
253
254 ;; 促音処理を試行
255 (if-let ((sokuon-result (nskk-converter--process-sokuon input)))
256 (let* ((sokuon-kana (car sokuon-result))
257 (rest-input (cdr sokuon-result))
258 ;; 残りを再帰的に変換
259 (rest-result (nskk-convert-romaji rest-input)))
260 (nskk-converter-result-create
261 :converted (concat sokuon-kana
262 (nskk-converter-result-converted rest-result))
263 :pending (nskk-converter-result-pending rest-result)
264 :consumed (+ 1 (nskk-converter-result-consumed rest-result))))
265
266 ;; 撥音処理を試行
267 (if-let ((n-result (nskk-converter--process-n input)))
268 (let* ((n-kana (car n-result))
269 (rest-input (cdr n-result))
270 (rest-result (nskk-convert-romaji rest-input)))
271 (nskk-converter-result-create
272 :converted (concat n-kana
273 (nskk-converter-result-converted rest-result))
274 :pending (nskk-converter-result-pending rest-result)
275 :consumed (+ (- (length input) (length rest-input))
276 (nskk-converter-result-consumed rest-result))))
277
278 ;; 最長一致変換を試行
279 (if-let ((match-result (nskk-converter--longest-match input)))
280 (let* ((kana (car match-result))
281 (rest-input (cdr match-result))
282 (consumed-len (- (length input) (length rest-input)))
283 (rest-result (nskk-convert-romaji rest-input)))
284 (nskk-converter-result-create
285 :converted (concat kana
286 (nskk-converter-result-converted rest-result))
287 :pending (nskk-converter-result-pending rest-result)
288 :consumed (+ consumed-len
289 (nskk-converter-result-consumed rest-result))))
290
291 ;; プレフィックスマッチがあれば未確定として保持
292 (if (nskk-converter--has-prefix-match input)
293 (nskk-converter-result-create
294 :converted &quot;&quot;
295 :pending input
296 :consumed 0)
297
298 ;; どのパターンにもマッチしない → 最初の1文字をそのまま出力
299 (let ((rest-result (nskk-convert-romaji (substring input 1))))
300 (nskk-converter-result-create
301 :converted (concat (substring input 0 1)
302 (nskk-converter-result-converted rest-result))
303 :pending (nskk-converter-result-pending rest-result)
304 :consumed (1+ (nskk-converter-result-consumed rest-result))))))))))
