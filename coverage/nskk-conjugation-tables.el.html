130(defun nskk-conjugation--cache-init ()
131 &quot;キャッシュを初期化する。&quot;
132 (when nskk-conjugation-enable-cache
133 (setq nskk-conjugation--cache (make-hash-table :test 'equal))
134 (setq nskk-conjugation--cache-hits 0)
135 (setq nskk-conjugation--cache-misses 0)))

137(defun nskk-conjugation--cache-key (stem type form)
138 &quot;キャッシュキーを生成する。&quot;
139 (list stem type form))
140
141(defun nskk-conjugation--cache-get (key)
142 &quot;キャッシュから値を取得する。&quot;
143 (when nskk-conjugation-enable-cache
144 (let ((entry (gethash key nskk-conjugation--cache)))
145 (if entry
146 (progn
147 (cl-incf nskk-conjugation--cache-hits)
148 (car entry))
149 (cl-incf nskk-conjugation--cache-misses)
150 nil))))
151
152(defun nskk-conjugation--cache-put (key value)
153 &quot;キャッシュに値を保存する。&quot;
154 (when nskk-conjugation-enable-cache
155 ;; キャッシュサイズ制限
156 (when (&gt;= (hash-table-count nskk-conjugation--cache)
157 nskk-conjugation-cache-size)
158 (nskk-conjugation--cache-evict))
159
160 (puthash key (cons value (current-time)) nskk-conjugation--cache)))
161
162(defun nskk-conjugation--cache-evict ()
163 &quot;キャッシュから古いエントリを削除する（LRU方式）。&quot;
164 (let ((oldest-key nil)
165 (oldest-time nil))
166 ;; 最も古いエントリを探す
167 (maphash (lambda (key entry)
168 (let ((time (cdr entry)))
169 (when (or (null oldest-time)
170 (time-less-p time oldest-time))
171 (setq oldest-key key)
172 (setq oldest-time time))))
173 nskk-conjugation--cache)
174 ;; 削除
175 (when oldest-key
176 (remhash oldest-key nskk-conjugation--cache))))

180;;;###autoload
181(defun nskk-conjugation-table-lookup (stem type form)
182 &quot;最適化されたテーブルから活用形を高速検索する。
183
184引数:
185 STEM - 語幹（文字列）
186 TYPE - 活用型（symbol）
187 FORM - 活用形（symbol）
188
189返り値:
190 活用後の文字列、またはnil
191
192使用例:
193 (nskk-conjugation-table-lookup \&quot;書\&quot; 'godan 'renyou) ;; =&gt; \&quot;き\&quot;
194 (nskk-conjugation-table-lookup \&quot;高\&quot; 'i-adjective 'renyou-ku) ;; =&gt; \&quot;高く\&quot;&quot;
195 (if (and nskk-conjugation--last-value
196 (equal stem nskk-conjugation--last-stem)
197 (eq type nskk-conjugation--last-type)
198 (eq form nskk-conjugation--last-form))
199 (progn
200 (cl-incf nskk-conjugation--cache-hits)
201 nskk-conjugation--last-value)
202 (let* ((cache-key (nskk-conjugation--cache-key stem type form))
203 (cached (nskk-conjugation--cache-get cache-key)))
204 (if cached
205 cached
206 ;; キャッシュミス時は通常の活用処理
207 (let ((result (cond
208 ;; 動詞活用
209 ((memq type nskk-verb-types)
210 (nskk-conjugate-verb stem type form))
211 ;; 形容詞活用
212 ((memq type nskk-adjective-types)
213 (nskk-conjugate-adjective stem type form))
214 ;; 複合活用
215 ((memq form nskk-complex-conjugation-types)
216 (nskk-complex-conjugate stem type form))
217 (t nil))))
218 (when result
219 (setq nskk-conjugation--last-stem stem
220 nskk-conjugation--last-type type
221 nskk-conjugation--last-form form
222 nskk-conjugation--last-value result)
223 (nskk-conjugation--cache-put cache-key result))
224 result)))))

227(defun nskk-conjugation-table-optimize ()
228 &quot;活用テーブルを最適化する。
229キャッシュの初期化と高速検索テーブルの構築を行う。&quot;
230 (interactive)
231 (message &quot;活用テーブルを最適化中...&quot;)
232 (let ((start-time (current-time)))
233 ;; キャッシュ初期化
234 (nskk-conjugation--cache-init)
235
236 ;; 高速検索テーブル初期化
237 (nskk-conjugation--init-fast-lookup)
238
239 ;; 頻出パターンの事前計算とキャッシング
240 (nskk-conjugation--precompute-common-patterns)
241
242 ;; ウォームアップ時のヒット/ミスは統計から除外する
243 (setq nskk-conjugation--cache-hits 0)
244 (setq nskk-conjugation--cache-misses 0)
245
246 (let* ((end-time (current-time))
247 (elapsed-ms (* 1000.0 (float-time (time-subtract end-time start-time)))))
248 (message &quot;活用テーブル最適化完了 (%.2f ms)&quot; elapsed-ms)))
249 nil)

251(defun nskk-conjugation--precompute-common-patterns ()
252 &quot;頻出する活用パターンを事前計算してキャッシュする。&quot;
253 (let ((common-verbs '(&quot;書&quot; &quot;読&quot; &quot;見&quot; &quot;食べ&quot; &quot;し&quot; &quot;来&quot;))
254 (common-adjectives '(&quot;高&quot; &quot;静か&quot;)))
255 ;; 頻出動詞の全活用形を事前計算
256 (dolist (verb common-verbs)
257 (let ((type (cond
258 ((equal verb &quot;し&quot;) 'sa-hen)
259 ((equal verb &quot;来&quot;) 'ka-hen)
260 ((member verb '(&quot;見&quot; &quot;食べ&quot;)) 'kami-ichidan)
261 (t 'godan))))
262 (dolist (form nskk-verb-forms)
263 (nskk-conjugation-table-lookup verb type form))))
264
265 ;; 頻出形容詞の全活用形を事前計算
266 (dolist (adj common-adjectives)
267 (let ((type (if (equal adj &quot;静か&quot;) 'na-adjective 'i-adjective)))
268 (dolist (form nskk-adjective-forms)
269 (nskk-conjugation-table-lookup adj type form))))))

308(defun nskk-conjugation-memory-usage ()
309 &quot;活用テーブルのメモリ使用量を推定する（バイト）。&quot;
310 (interactive)

327(defun nskk-conjugation-benchmark ()
328 &quot;活用処理のベンチマークを実行する。&quot;
329 (interactive)
330 (let ((iterations 10000)
331 (test-cases '((&quot;書&quot; godan renyou)
332 (&quot;見&quot; kami-ichidan renyou)
333 (&quot;高&quot; i-adjective renyou-ku)
334 (&quot;静か&quot; na-adjective renyou-ni))))
335 (message &quot;活用処理ベンチマーク開始 (%d回 x %d種類)...&quot;
336 iterations (length test-cases))
337
338 (dolist (case test-cases)
339 (let* ((stem (nth 0 case))
340 (type (nth 1 case))
341 (form (nth 2 case))
342 (start-time (current-time)))
343
344 ;; キャッシュなしでベンチマーク
345 (let ((nskk-conjugation-enable-cache nil))
346 (dotimes (_ iterations)
347 (nskk-conjugation-table-lookup stem type form))
348 (let* ((end-time (current-time))
349 (elapsed-ms (* 1000.0 (float-time (time-subtract end-time start-time))))
350 (per-call-ms (/ elapsed-ms iterations)))
351 (message &quot; %s/%s/%s (キャッシュなし): %.3f ms/call&quot;
352 stem type form per-call-ms)))
353
354 ;; キャッシュありでベンチマーク
355 (setq start-time (current-time))
356 (let ((nskk-conjugation-enable-cache t))
357 (nskk-conjugation--cache-init)
358 (dotimes (_ iterations)
359 (nskk-conjugation-table-lookup stem type form))
360 (let* ((end-time (current-time))
361 (elapsed-ms (* 1000.0 (float-time (time-subtract end-time start-time))))
362 (per-call-ms (/ elapsed-ms iterations)))
363 (message &quot; %s/%s/%s (キャッシュあり): %.3f ms/call&quot;
364 stem type form per-call-ms)))))
365
366 (nskk-conjugation-cache-stats)))
