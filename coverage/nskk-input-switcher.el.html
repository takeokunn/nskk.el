114(defun nskk-input-switcher-switch (method-name)
115 &quot;入力方式をMETHOD-NAMEに切り替える。
116
117METHOD-NAMEは利用可能な入力方式のシンボル。
118状態の保存と復元を自動的に行う。&quot;
119 (unless (memq method-name nskk-input-switcher-available-methods)
120 (error &quot;Unknown input method: %s&quot; method-name))
121
122 ;; 現在の状態を保存
123 (when (and nskk-input-switcher-save-state
124 nskk-input-switcher-current-method)
125 (nskk-input-switcher--save-state nskk-input-switcher-current-method))
126
127 ;; 履歴に追加
128 (push nskk-input-switcher-current-method nskk-input-switcher-history)
129 (when (&gt; (length nskk-input-switcher-history)
130 nskk-input-switcher-max-history)
131 (setq nskk-input-switcher-history
132 (cl-subseq nskk-input-switcher-history 0
133 nskk-input-switcher-max-history)))
134
135 ;; 入力方式を切り替え
136 (setq nskk-input-switcher-current-method method-name)
137
138 ;; 新しい入力方式を初期化
139 (nskk-input-switcher--init-method method-name)
140
141 ;; 状態を復元
142 (when nskk-input-switcher-restore-state
143 (nskk-input-switcher--restore-state method-name))
144
145 ;; モードラインを更新
146 (when nskk-input-switcher-show-in-modeline
147 (nskk-input-switcher--update-modeline))
148
149 ;; メッセージ表示
150 (message &quot;Input method: %s&quot;
151 (nskk-input-switcher--get-method-name method-name))
152
153 method-name)

155(defun nskk-input-switcher-select ()
156 &quot;インタラクティブに入力方式を選択する。&quot;
157 (interactive)
158 (let* ((choices (mapcar (lambda (m)
159 (cons (nskk-input-switcher--get-method-name m) m))
160 nskk-input-switcher-available-methods))
161 (choice (completing-read &quot;Select input method: &quot; choices nil t))
162 (method (cdr (assoc choice choices))))
163 (when method
164 (nskk-input-switcher-switch method))))
165
166(defun nskk-input-switcher-cycle ()
167 &quot;次の入力方式に循環的に切り替える。&quot;
168 (interactive)
169 (let* ((current-pos (cl-position nskk-input-switcher-current-method
170 nskk-input-switcher-available-methods))
171 (next-pos (mod (1+ current-pos)
172 (length nskk-input-switcher-available-methods)))
173 (next-method (nth next-pos nskk-input-switcher-available-methods)))
174 (nskk-input-switcher-switch next-method)))
175
176(defun nskk-input-switcher-previous ()
177 &quot;履歴から前の入力方式に戻る。&quot;
178 (interactive)
179 (if nskk-input-switcher-history
180 (let ((prev-method (pop nskk-input-switcher-history)))
181 (nskk-input-switcher-switch prev-method))
182 (message &quot;No previous input method in history&quot;)))

186(defun nskk-input-switcher--save-state (method-name)
187 &quot;METHOD-NAMEの現在の状態を保存する。&quot;
188 (let ((state (nskk-input-switcher--get-current-state method-name)))
189 (setf (alist-get method-name nskk-input-switcher-state-alist) state)))
190
191(defun nskk-input-switcher--restore-state (method-name)
192 &quot;METHOD-NAMEの以前の状態を復元する。&quot;
193 (when-let ((state (alist-get method-name nskk-input-switcher-state-alist)))
194 (nskk-input-switcher--apply-state method-name state)))

199 ;; 現在は基本的な状態のみ

204将来的には各入力方式固有の状態を復元可能。&quot;
205 ;; 現在は何もしない
206 nil)

210(defun nskk-input-switcher--init-method (method-name)
211 &quot;METHOD-NAMEの入力方式を初期化する。&quot;
212 (let ((init-fn (nskk-input-switcher--get-init-function method-name)))
213 (when (functionp init-fn)
214 (funcall init-fn))))

216(defun nskk-input-switcher--get-init-function (method-name)
217 &quot;METHOD-NAMEの初期化関数を取得する。&quot;
218 (cond
219 ((eq method-name 'qwerty)
220 (require 'nskk-input-qwerty)
221 #'nskk-input-qwerty-register)
222 ((eq method-name 'azik)
223 (require 'nskk-input-azik)
224 #'nskk-input-azik-register)
225 ((eq method-name 'act)
226 (require 'nskk-input-act)
227 #'nskk-input-act-register)
228 ((eq method-name 'tutcode)
229 (require 'nskk-input-tutcode)
230 #'nskk-input-tutcode-register)
231 ((eq method-name 'nicola)
232 (require 'nskk-input-nicola)
233 #'nskk-input-nicola-register)
234 ((eq method-name 'kana)
235 (require 'nskk-input-kana)
236 #'nskk-input-kana-register)
237 ((eq method-name 'dvorak)
238 (require 'nskk-input-dvorak)
239 #'nskk-input-dvorak-register)
240 ((eq method-name 'colemak)
241 (require 'nskk-input-colemak)
242 #'nskk-input-colemak-register)
243 ((eq method-name 'hybrid)
244 (require 'nskk-input-hybrid)

246 (t nil)))
