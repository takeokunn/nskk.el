123(defvar nskk-input-act-hash-table nil
124 &quot;ACT入力方式用のハッシュテーブル。&quot;)

126(defun nskk-input-act-init-hash-table ()
127 &quot;ACT入力方式用のハッシュテーブルを初期化する。&quot;
128 (setq nskk-input-act-hash-table
129 (make-hash-table :test 'equal :size 600))
130 (dolist (entry nskk-input-act-table)
131 (puthash (car entry) (cdr entry) nskk-input-act-hash-table)))
132
133;;; 検索関数
134
135(defun nskk-input-act-lookup (romaji)
136 &quot;ACT入力方式でROMAJIをひらがなに変換する。
137
138使用例:
139 (nskk-input-act-lookup \&quot;kh\&quot;) ;; =&gt; \&quot;きゃ\&quot;
140 (nskk-input-act-lookup \&quot;ka\&quot;) ;; =&gt; \&quot;か\&quot;&quot;
141 (unless nskk-input-act-hash-table
142 (nskk-input-act-init-hash-table))
143 (gethash romaji nskk-input-act-hash-table))
144
145(defun nskk-input-act-get-candidates (prefix)
146 &quot;PREFIX で始まる全てのACTローマ字候補を取得する。&quot;
147 (let ((candidates nil))
148 (dolist (entry nskk-input-act-table)
149 (when (string-prefix-p prefix (car entry))
150 (push (car entry) candidates)))
151 (nreverse candidates)))
