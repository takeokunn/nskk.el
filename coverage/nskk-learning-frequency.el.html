27;; このファイルは頻度学習エンジンを実装します。
28;;
29;; 特徴:
30;; - 使用頻度追跡（LRU/LFUアルゴリズム）
31;; - 個人辞書への自動反映
32;; - 頻度減衰機能（時間経過による重み調整）
33;; - メモリ効率的な実装
34;;
35;; アルゴリズム:
36;; - LRU (Least Recently Used): 最近使用された順に優先度付け
37;; - LFU (Least Frequently Used): 使用頻度に基づいて優先度付け
38;; - ハイブリッドモード: LRUとLFUの組み合わせ
39;;
40;; 頻度減衰:
41;; - 時間経過とともに頻度カウントを減衰
42;; - 最近の使用パターンを重視
43;;
44;; 使用例:
45;; (nskk-update-frequency &quot;かんじ&quot; &quot;漢字&quot;)
46;; (nskk-get-frequency-score &quot;かんじ&quot; &quot;漢字&quot;)

135;;;###autoload
136(defun nskk-update-frequency (midashi candidate)
137 &quot;見出し語 MIDASHI と候補 CANDIDATE の使用頻度を更新する。&quot;
138 (unless (and (stringp midashi) (stringp candidate))
139 (error &quot;Invalid arguments: midashi and candidate must be strings&quot;))
140
141 (let* ((key (cons midashi candidate))
142 (entry (gethash key nskk-frequency-table)))
143
144 (if entry
145 ;; 既存エントリを更新
146 (progn
147 (cl-incf (nskk-frequency-entry-count entry))
148 (setf (nskk-frequency-entry-last-used entry) (current-time))
149 (nskk-frequency--update-score entry))
150
151 ;; 新規エントリを作成
152 (let ((new-entry (nskk-frequency-entry--create
153 :midashi midashi
154 :candidate candidate
155 :count 1
156 :last-used (current-time)
157 :created (current-time))))
158 (nskk-frequency--update-score new-entry)
159 (puthash key new-entry nskk-frequency-table)))
160
161 ;; 定期的な減衰処理
162 (when nskk-frequency-decay-enabled
163 (nskk-frequency--maybe-decay))
164
165 ;; エントリ数制限チェック
166 (nskk-frequency--check-max-entries)))
167
168;;;###autoload
169(defun nskk-get-frequency-score (midashi candidate)
170 &quot;見出し語 MIDASHI と候補 CANDIDATE の頻度スコアを取得する。
171エントリが存在しない場合は 0.0 を返す。&quot;
172 (let* ((key (cons midashi candidate))
173 (entry (gethash key nskk-frequency-table)))
174 (if entry
175 (nskk-frequency-entry-score entry)
176 0.0)))
177
178;;;###autoload
179(defun nskk-get-sorted-candidates (midashi candidates)
180 &quot;見出し語 MIDASHI の候補リスト CANDIDATES を頻度順にソートする。
181CANDIDATES は文字列のリスト。
182戻り値: 頻度順にソートされた候補リスト&quot;
183 (sort (copy-sequence candidates)
184 (lambda (a b)
185 (&gt; (nskk-get-frequency-score midashi a)
186 (nskk-get-frequency-score midashi b)))))

188;;; スコア計算
189
190(defun nskk-frequency--update-score (entry)
191 &quot;ENTRY のスコアを計算して更新する。&quot;
192 (let ((score (pcase nskk-frequency-algorithm
193 ('lru (nskk-frequency--calc-lru-score entry))
194 ('lfu (nskk-frequency--calc-lfu-score entry))
195 ('hybrid (nskk-frequency--calc-hybrid-score entry))
196 (_ (error &quot;Unknown frequency algorithm: %s&quot; nskk-frequency-algorithm)))))
197 (setf (nskk-frequency-entry-score entry) score)))
198
199(defun nskk-frequency--calc-lru-score (entry)
200 &quot;LRUスコアを計算する。
201最近使用されたものほど高スコア。&quot;
202 (let* ((last-used (nskk-frequency-entry-last-used entry))
203 (elapsed (float-time (time-since last-used))))
204 ;; 経過時間が短いほど高スコア（指数関数的に減衰）
205 (exp (- (/ elapsed 86400.0)))))
206
207(defun nskk-frequency--calc-lfu-score (entry)
208 &quot;LFUスコアを計算する。
209使用回数が多いものほど高スコア。&quot;
210 (float (nskk-frequency-entry-count entry)))
211
212(defun nskk-frequency--calc-hybrid-score (entry)
213 &quot;ハイブリッドスコア（LRU + LFU）を計算する。&quot;
214 (+ (* nskk-frequency-lru-weight (nskk-frequency--calc-lru-score entry))
215 (* nskk-frequency-lfu-weight (nskk-frequency--calc-lfu-score entry))))
216
217;;; 頻度減衰
218
219(defun nskk-frequency--maybe-decay ()
220 &quot;必要に応じて頻度減衰を実行する。&quot;
221 (when (&gt; (float-time (time-since nskk-frequency-last-decay-time))
222 nskk-frequency-decay-interval)
223 (nskk-frequency--apply-decay)
224 (setq nskk-frequency-last-decay-time (current-time))))
225
226(defun nskk-frequency--apply-decay ()
227 &quot;全エントリに頻度減衰を適用する。&quot;
228 (maphash
229 (lambda (_key entry)
230 (let ((new-count (* (nskk-frequency-entry-count entry)
231 nskk-frequency-decay-rate)))
232 ;; カウントが1未満になったら削除対象
233 (if (&lt; new-count 1.0)
234 (remhash (cons (nskk-frequency-entry-midashi entry)
235 (nskk-frequency-entry-candidate entry))
236 nskk-frequency-table)
237 (setf (nskk-frequency-entry-count entry) (floor new-count))
238 (nskk-frequency--update-score entry))))
239 nskk-frequency-table))
240
241;;; エントリ数制限
242
243(defun nskk-frequency--check-max-entries ()
244 &quot;エントリ数が上限を超えた場合、低スコアのエントリを削除する。&quot;
245 (when (&gt; (hash-table-count nskk-frequency-table)
246 nskk-frequency-max-entries)
247 (nskk-frequency--prune-entries)))
248
249(defun nskk-frequency--prune-entries ()
250 &quot;低スコアのエントリを削除してエントリ数を制限内に収める。&quot;
251 (let ((entries nil))
252 ;; 全エントリをリストに変換
253 (maphash
254 (lambda (key entry)
255 (push (cons key (nskk-frequency-entry-score entry)) entries))
256 nskk-frequency-table)
257
258 ;; スコアでソート（降順）
259 (setq entries (sort entries (lambda (a b) (&gt; (cdr a) (cdr b)))))
260
261 ;; 上位 nskk-frequency-max-entries * 0.9 個を残す
262 (let ((keep-count (floor (* nskk-frequency-max-entries 0.9)))
263 (new-table (make-hash-table :test 'equal)))
264 (dolist (entry (seq-take entries keep-count))
265 (puthash (car entry)
266 (gethash (car entry) nskk-frequency-table)
267 new-table))
268 (setq nskk-frequency-table new-table))))

272;;;###autoload
273(defun nskk-frequency-statistics ()
274 &quot;頻度学習の統計情報を返す。
275戻り値: plist形式の統計情報&quot;
276 (let ((total-entries (hash-table-count nskk-frequency-table))
277 (total-count 0)
278 (avg-count 0.0)
279 (max-count 0)
280 (min-count most-positive-fixnum))
281
282 (maphash
283 (lambda (_key entry)
284 (let ((count (nskk-frequency-entry-count entry)))
285 (setq total-count (+ total-count count))
286 (setq max-count (max max-count count))
287 (setq min-count (min min-count count))))
288 nskk-frequency-table)
289
290 (when (&gt; total-entries 0)
291 (setq avg-count (/ (float total-count) total-entries)))
292
293 (list :total-entries total-entries
294 :total-count total-count
295 :average-count avg-count
296 :max-count max-count
297 :min-count (if (= min-count most-positive-fixnum) 0 min-count)
298 :algorithm nskk-frequency-algorithm
299 :decay-enabled nskk-frequency-decay-enabled)))

338(defun nskk-frequency-get-top-entries (n)
339 &quot;上位 N 件の頻度エントリを取得する。
340戻り値: ((midashi candidate score) ...) のリスト&quot;
341 (let ((entries nil))
342 (maphash
343 (lambda (key entry)
344 (push (list (car key)
345 (cdr key)
346 (nskk-frequency-entry-score entry)
347 (nskk-frequency-entry-count entry))
348 entries))
349 nskk-frequency-table)
350
351 (seq-take (sort entries (lambda (a b) (&gt; (nth 2 a) (nth 2 b)))) n)))

354 &quot;上位 N 件（デフォルト10件）の頻度エントリを表示する。&quot;
