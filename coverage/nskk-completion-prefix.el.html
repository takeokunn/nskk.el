27;; このファイルはトライ木を活用した高速前方一致補完を実装します。
28;;
29;; 特徴:
30;; - トライ木による O(k + n) の高速検索 (k=prefix長, n=結果数)
31;; - インクリメンタル補完サポート
32;; - 送り仮名タイプの柔軟な指定
33;; - キャッシュ統合準備
34;;
35;; パフォーマンス目標:
36;; - 検索応答: &lt; 1ms
37;; - 100件候補取得: &lt; 5ms
38;; - メモリ効率: 既存トライ木を活用
39;;
40;; 使用例:
41;;
42;; (require 'nskk-completion-prefix)
43;;
44;; ;; 基本的な前方一致補完
45;; (nskk-completion-prefix-search index &quot;かん&quot;)
46;; ;; =&gt; ((&quot;かん&quot; . entry1) (&quot;かんじ&quot; . entry2) (&quot;かんたん&quot; . entry3))
47;;
48;; ;; 制限付き検索
49;; (nskk-completion-prefix-search index &quot;か&quot; :limit 10)
50;; ;; =&gt; 最大10件
51;;
52;; ;; 送り仮名タイプ指定
53;; (nskk-completion-prefix-search index &quot;かんが&quot; :okuri-type 'okuri-ari)

109;;;###autoload
110(defun nskk-completion-prefix-search (index prefix &rest args)
111 &quot;前方一致補完検索を実行する。
112
113引数:
114 INDEX - nskk-dict-index構造体
115 PREFIX - 検索プレフィックス
116
117キーワード引数:
118 :limit - 最大結果数（デフォルト: nskk-completion-prefix-default-limit）
119 :okuri-type - 送り仮名タイプ（'okuri-ari/'okuri-nasi/nil）
120 :sort - ソート方法（デフォルト: nskk-completion-prefix-sort-method）
121 :exact - 完全一致を含めるか（デフォルト: t）
122
123戻り値:
124 nskk-completion-prefix-result構造体のリスト
125
126パフォーマンス: O(k + n) (k=prefix長, n=結果数)&quot;
127 (unless (stringp prefix)
128 (error &quot;Prefix must be a string: %s&quot; prefix))
129
130 ;; プレフィックス長チェック
131 (when (&lt; (length prefix) nskk-completion-prefix-min-length)
132 (return nil))
133
134 (let* ((limit (or (plist-get args :limit) nskk-completion-prefix-default-limit))
135 (okuri-type (plist-get args :okuri-type))
136 (sort-method (or (plist-get args :sort) nskk-completion-prefix-sort-method))
137 (include-exact (if (plist-member args :exact)
138 (plist-get args :exact)
139 t))
140 (results nil))
141
142 ;; トライ木から前方一致検索
143 (setq results (nskk-completion-prefix--search-tries
144 index prefix limit okuri-type include-exact))
145
146 ;; ソート
147 (setq results (nskk-completion-prefix--sort-results results sort-method))
148
149 ;; limit適用（ソート後）
150 (when (and limit (&gt; (length results) limit))
151 (setq results (seq-take results limit)))
152
153 results))

155(defun nskk-completion-prefix--search-tries (index prefix limit okuri-type include-exact)
156 &quot;トライ木から前方一致検索を実行する（内部関数）。
157
158引数:
159 INDEX - nskk-dict-index構造体
160 PREFIX - 検索プレフィックス
161 LIMIT - 最大結果数
162 OKURI-TYPE - 送り仮名タイプ
163 INCLUDE-EXACT - 完全一致を含めるか
164
165戻り値:
166 nskk-completion-prefix-result構造体のリスト&quot;
167 (let ((results nil)
168 (exact-match nil)
169 (tries (cond
170 ((eq okuri-type 'okuri-ari)
171 (list (cons (nskk-dict-index-trie-ari index) 'okuri-ari)))
172 ((eq okuri-type 'okuri-nasi)
173 (list (cons (nskk-dict-index-trie-nasi index) 'okuri-nasi)))
174 (t
175 ;; 両方から検索（送り仮名なしを優先）
176 (list (cons (nskk-dict-index-trie-nasi index) 'okuri-nasi)
177 (cons (nskk-dict-index-trie-ari index) 'okuri-ari))))))
178
179 ;; 完全一致を先にチェック
180 (when include-exact
181 (cl-block nil
182 (dolist (trie-pair tries)
183 (let* ((trie (car trie-pair))
184 (lookup-result (nskk-trie-lookup trie prefix)))
185 (when (cdr lookup-result)
186 (setq exact-match
187 (nskk-completion-prefix-result--create
188 :midashi prefix
189 :entry (car lookup-result)
190 :score (nskk-dict-entry-frequency (car lookup-result))
191 :match-type 'exact))
192 ;; 最初の完全一致のみ
193 (cl-return))))))
194
195 ;; 前方一致検索
196 (dolist (trie-pair tries)
197 (let ((trie (car trie-pair)))
198 (when trie
199 (let ((trie-results (nskk-trie-prefix-search trie prefix limit)))
200 (dolist (result trie-results)
201 (let ((midashi (car result))
202 (entry (cdr result)))
203 ;; 完全一致は除外（既に追加済み）
204 (unless (and include-exact (equal midashi prefix))
205 (push (nskk-completion-prefix-result--create
206 :midashi midashi
207 :entry entry
208 :score (nskk-dict-entry-frequency entry)
209 :match-type 'prefix)
210 results))))))))
211
212 ;; 完全一致を先頭に追加
213 (when exact-match
214 (push exact-match results))
215
216 ;; 重複除去（同じ見出し語が複数のトライ木に存在する場合）
217 (nskk-completion-prefix--remove-duplicates results)))

219(defun nskk-completion-prefix--remove-duplicates (results)
220 &quot;補完結果から重複を除去する（内部関数）。
221
222引数:
223 RESULTS - nskk-completion-prefix-result構造体のリスト
224
225戻り値:
226 重複除去後のリスト（最初に出現したものを保持）&quot;
227 (let ((seen (make-hash-table :test 'equal))
228 (unique nil))
229 (dolist (result results)
230 (let ((midashi (nskk-completion-prefix-result-midashi result)))
231 (unless (gethash midashi seen)
232 (puthash midashi t seen)
233 (push result unique))))
234 (nreverse unique)))

259(defun nskk-completion-prefix--sort-by-frequency (results)
260 &quot;補完結果を使用頻度でソートする。
261
262引数:
263 RESULTS - nskk-completion-prefix-result構造体のリスト
264
265処理:
266 1. 完全一致を最優先
267 2. 頻度の高い順
268 3. 同頻度の場合は文字列長の短い順
269
270戻り値:
271 ソート済みリスト&quot;
272 (sort results
273 (lambda (a b)
274 (let ((a-exact (eq (nskk-completion-prefix-result-match-type a) 'exact))
275 (b-exact (eq (nskk-completion-prefix-result-match-type b) 'exact))
276 (a-score (nskk-completion-prefix-result-score a))
277 (b-score (nskk-completion-prefix-result-score b)))
278 (cond
279 ;; 完全一致を優先
280 ((and a-exact (not b-exact)) t)
281 ((and (not a-exact) b-exact) nil)
282 ;; 頻度で比較
283 ((&gt; a-score b-score) t)
284 ((&lt; a-score b-score) nil)
285 ;; 同頻度の場合は文字列長の短い順
286 (t (&lt; (length (nskk-completion-prefix-result-midashi a))
287 (length (nskk-completion-prefix-result-midashi b)))))))))

322引数:
323 INDEX - nskk-dict-index構造体
324 PREFIX-LIST - プレフィックスのリスト（例: '(\&quot;か\&quot; \&quot;かん\&quot; \&quot;かんじ\&quot;)）

329

333
