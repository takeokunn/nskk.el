186(defmacro nskk-native-compile-with-profiling (name &rest body)

273(defun nskk-native-compile-file (file)
274 &quot;指定されたファイルをネイティブコンパイルする。
275
276引数:
277 FILE - コンパイルするファイルパス
278
279返り値:
280 コンパイル成功時はt、失敗時はnil&quot;
281 (interactive &quot;fFile to native compile: &quot;)
282 (unless (nskk-native-compile-available-p)
283 (error &quot;Native compilation is not available&quot;))
284
285 (condition-case err
286 (progn
287 (when (fboundp 'native-compile)
288 (native-compile file)
289 (message &quot;Native compiled: %s&quot; file)
290 t))
291 (error
292 (message &quot;Native compilation failed for %s: %s&quot; file err)
293 nil)))

295(defun nskk-native-compile-package ()
296 &quot;NSKKパッケージ全体をネイティブコンパイルする。&quot;
297 (interactive)
298 (unless (nskk-native-compile-available-p)
299 (error &quot;Native compilation is not available&quot;))
300
301 (let* ((nskk-dir (file-name-directory (locate-library &quot;nskk&quot;)))
302 (files (directory-files nskk-dir t &quot;^nskk-.*\\.el$&quot;))
303 (success-count 0)
304 (fail-count 0))
305
306 (message &quot;Starting native compilation of %d files...&quot; (length files))
307
308 (dolist (file files)
309 (if (nskk-native-compile-file file)
310 (cl-incf success-count)
311 (cl-incf fail-count)))
312
313 (message &quot;Native compilation completed: %d succeeded, %d failed&quot;
314 success-count fail-count)
315
316 (list :success success-count :failed fail-count)))
