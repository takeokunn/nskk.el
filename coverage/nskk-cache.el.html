27;; このファイルはNSKKのキャッシュ機構を実装します。
28;;
29;; サポートするキャッシュアルゴリズム:
30;; - LRU (Least Recently Used): 最近使用されていないものを削除
31;; - LFU (Least Frequently Used): 使用頻度が低いものを削除
32;;
33;; 特徴:
34;; - O(1) get/put操作（ハッシュテーブル + 双方向リンクリスト）
35;; - 容量管理（エントリ数ベース）
36;; - 自動エビクション
37;; - キャッシュヒット率測定
38;; - パターンマッチ無効化
39;;
40;; パフォーマンス目標:
41;; - get操作: O(1), &lt; 0.1ms
42;; - put操作: O(1), &lt; 0.1ms
43;; - キャッシュヒット時の検索: &lt; 10ms
44;;
45;; 使用例:
46;;
47;; (require 'nskk-cache)
48;;
49;; ;; LRUキャッシュ作成
50;; (setq cache (nskk-cache-create 'lru 1000))
51;;
52;; ;; データ追加
53;; (nskk-cache-put cache &quot;key1&quot; &quot;value1&quot;)
54;;
55;; ;; データ取得
56;; (nskk-cache-get cache &quot;key1&quot;)
57;; ;; =&gt; &quot;value1&quot;
58;;
59;; ;; 統計情報
60;; (nskk-cache-stats cache)
61;; ;; =&gt; (:hits 10 :misses 5 :hit-rate 0.666 ...)
62

167 &quot;LRUキャッシュを作成する。
168
169引数:
170 CAPACITY - 最大容量（エントリ数）
171
172戻り値:
173 nskk-cache-lru構造体&quot;
174 (let* ((head (nskk-cache-lru-node--create))
175 (tail (nskk-cache-lru-node--create)))
176 ;; ダミーヘッドとテールを接続
177 (setf (nskk-cache-lru-node-next head) tail)
178 (setf (nskk-cache-lru-node-prev tail) head)

181 :size 0
182 :hash (make-hash-table :test 'equal :size capacity)
183 :head head
184 :tail tail
185 :hits 0
186 :misses 0)))

188(defsubst nskk-cache-lru--remove-node (node)
189 &quot;ノードをリストから削除する（内部関数）。
190
191引数:
192 NODE - 削除するノード&quot;
193 (let ((prev-node (nskk-cache-lru-node-prev node))
194 (next-node (nskk-cache-lru-node-next node)))
195 (setf (nskk-cache-lru-node-next prev-node) next-node)
196 (setf (nskk-cache-lru-node-prev next-node) prev-node)))

199 &quot;ノードをヘッドの直後に追加する（内部関数）。

202 CACHE - LRUキャッシュ
203 NODE - 追加するノード&quot;
204 (let ((head (nskk-cache-lru-head cache))
205 (next-node (nskk-cache-lru-node-next (nskk-cache-lru-head cache))))
206 (setf (nskk-cache-lru-node-next node) next-node)
207 (setf (nskk-cache-lru-node-prev node) head)
208 (setf (nskk-cache-lru-node-next head) node)
209 (setf (nskk-cache-lru-node-prev next-node) node)))

234(defun nskk-cache-lru-get (cache key)
235 &quot;LRUキャッシュから値を取得する。
236
237引数:
238 CACHE - LRUキャッシュ
239 KEY - キー
240
241戻り値:
242 値（存在しない場合はnil）&quot;
243 (let ((node (gethash key (nskk-cache-lru-hash cache))))
244 (if node
245 (progn
246 ;; キャッシュヒット：ノードをヘッドに移動
247 (nskk-cache-lru--move-to-head cache node)
248 (cl-incf (nskk-cache-lru-hits cache))
249 (nskk-cache-lru-node-value node))
250 ;; キャッシュミス
251 (cl-incf (nskk-cache-lru-misses cache))
252 nil)))

254;;;###autoload
255(defun nskk-cache-lru-put (cache key value)
256 &quot;LRUキャッシュに値を追加する。
257
258引数:
259 CACHE - LRUキャッシュ
260 KEY - キー
261 VALUE - 値&quot;
262 (let ((node (gethash key (nskk-cache-lru-hash cache))))
263 (if node
264 ;; 既存エントリ：値を更新してヘッドに移動
265 (progn
266 (setf (nskk-cache-lru-node-value node) value)
267 (nskk-cache-lru--move-to-head cache node))
268 ;; 新規エントリ
269 (let ((new-node (nskk-cache-lru-node--create :key key :value value)))
270 (puthash key new-node (nskk-cache-lru-hash cache))
271 (nskk-cache-lru--add-to-head cache new-node)
272 (cl-incf (nskk-cache-lru-size cache))
273 ;; 容量超過チェック
274 (when (&gt; (nskk-cache-lru-size cache) (nskk-cache-lru-capacity cache))
275 (let ((tail-node (nskk-cache-lru--remove-tail cache)))
276 (remhash (nskk-cache-lru-node-key tail-node) (nskk-cache-lru-hash cache))
277 (cl-decf (nskk-cache-lru-size cache))))))))

322戻り値:
323 nskk-cache-lfu構造体&quot;

333(defsubst nskk-cache-lfu--update-freq (cache entry old-freq)
334 &quot;エントリの頻度を更新する（内部関数）。
335
336引数:
337 CACHE - LFUキャッシュ
338 ENTRY - 更新するエントリ
339 OLD-FREQ - 古い頻度&quot;
340 (let ((freq-table (nskk-cache-lfu-freq cache))
341 (key (nskk-cache-lfu-entry-key entry))
342 (new-freq (nskk-cache-lfu-entry-frequency entry)))
343 ;; 古い頻度のリストから削除
344 (when old-freq
345 (let ((keys (gethash old-freq freq-table)))
346 (setq keys (delq key keys))
347 (if keys
348 (puthash old-freq keys freq-table)
349 (remhash old-freq freq-table)
350 ;; 最小頻度が削除された場合、更新
351 (when (= old-freq (nskk-cache-lfu-min-freq cache))
352 (setf (nskk-cache-lfu-min-freq cache) new-freq)))))
353 ;; 新しい頻度のリストに追加
354 (let ((keys (gethash new-freq freq-table)))
355 (puthash new-freq (cons key keys) freq-table))))

380;;;###autoload
381(defun nskk-cache-lfu-put (cache key value)
382 &quot;LFUキャッシュに値を追加する。
383
384引数:
385 CACHE - LFUキャッシュ
386 KEY - キー
387 VALUE - 値&quot;
388 (let ((entry (gethash key (nskk-cache-lfu-hash cache))))
389 (if entry
390 ;; 既存エントリ：値を更新して頻度を増やす
391 (let ((old-freq (nskk-cache-lfu-entry-frequency entry)))
392 (setf (nskk-cache-lfu-entry-value entry) value)
393 (cl-incf (nskk-cache-lfu-entry-frequency entry))
394 (nskk-cache-lfu--update-freq cache entry old-freq))
395 ;; 新規エントリ
396 (when (&gt;= (nskk-cache-lfu-size cache) (nskk-cache-lfu-capacity cache))
397 ;; 容量超過：最小頻度のエントリを削除
398 (let* ((min-freq (nskk-cache-lfu-min-freq cache))
399 (keys (gethash min-freq (nskk-cache-lfu-freq cache)))
400 (evict-key (car keys)))
401 (when evict-key
402 (remhash evict-key (nskk-cache-lfu-hash cache))
403 (setq keys (cdr keys))
404 (if keys
405 (puthash min-freq keys (nskk-cache-lfu-freq cache))
406 (remhash min-freq (nskk-cache-lfu-freq cache)))
407 (cl-decf (nskk-cache-lfu-size cache)))))
408 ;; 新規エントリを追加
409 (let ((new-entry (nskk-cache-lfu-entry--create :key key :value value :frequency 1)))
410 (puthash key new-entry (nskk-cache-lfu-hash cache))
411 (nskk-cache-lfu--update-freq cache new-entry nil)
412 (setf (nskk-cache-lfu-min-freq cache) 1)
413 (cl-incf (nskk-cache-lfu-size cache))))))

533 &quot;パターンに一致するキーを無効化する。

555;;;###autoload
556(defun nskk-cache-stats (cache)
557 &quot;キャッシュの統計情報を取得する。
558
559引数:
560 CACHE - キャッシュ
561
562戻り値:
563 統計情報のplist
564 (:type TYPE :capacity CAPACITY :size SIZE :hits HITS :misses MISSES :hit-rate RATE)&quot;
565 (let* ((type (cond ((nskk-cache-lru-p cache) 'lru)
566 ((nskk-cache-lfu-p cache) 'lfu)
567 (t 'unknown)))
568 (capacity (cond ((nskk-cache-lru-p cache) (nskk-cache-lru-capacity cache))
569 ((nskk-cache-lfu-p cache) (nskk-cache-lfu-capacity cache))
570 (t 0)))
571 (size (cond ((nskk-cache-lru-p cache) (nskk-cache-lru-size cache))
572 ((nskk-cache-lfu-p cache) (nskk-cache-lfu-size cache))
573 (t 0)))
574 (hits (cond ((nskk-cache-lru-p cache) (nskk-cache-lru-hits cache))
575 ((nskk-cache-lfu-p cache) (nskk-cache-lfu-hits cache))
576 (t 0)))
577 (misses (cond ((nskk-cache-lru-p cache) (nskk-cache-lru-misses cache))
578 ((nskk-cache-lfu-p cache) (nskk-cache-lfu-misses cache))
579 (t 0)))
580 (total (+ hits misses))
581 (hit-rate (if (&gt; total 0) (/ (float hits) total) 0.0)))
582 (list :type type
583 :capacity capacity
584 :size size
585 :hits hits
586 :misses misses
587 :hit-rate hit-rate)))
