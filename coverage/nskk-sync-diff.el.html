27;; このファイルはNSKK辞書の差分同期エンジンを実装します。
28;;
29;; 特徴:
30;; - Myers差分アルゴリズム（効率的な差分計算）
31;; - 増分同期（変更された部分のみ転送）
32;; - zlib圧縮による帯域最適化
33;; - チャンクベース転送（大量データ対応）
34;; - デルタ圧縮（類似データの効率的転送）
35;; - ハッシュベース変更検出
36;;
37;; アルゴリズム:
38;;
39;; 1. ハッシュ計算（SHA-256）
40;; 2. 変更検出（ハッシュ比較）
41;; 3. Myers差分計算
42;; 4. デルタ圧縮
43;; 5. zlib圧縮
44;; 6. チャンク分割（必要に応じて）
45;;
46;; 使用例:
47;;
48;; (require 'nskk-sync-diff)
49;;
50;; ;; 辞書の差分を計算
51;; (let* ((local-dict (nskk-dict-load &quot;local.dict&quot;))
52;; (remote-dict (nskk-dict-load &quot;remote.dict&quot;))
53;; (diff (nskk-sync-diff-calculate local-dict remote-dict)))
54;;
55;; ;; 差分のサイズを確認
56;; (message &quot;Added: %d, Modified: %d, Deleted: %d&quot;
57;; (length (nskk-sync-diff-added diff))
58;; (length (nskk-sync-diff-modified diff))
59;; (length (nskk-sync-diff-deleted diff)))
60;;
61;; ;; 差分を適用
62;; (nskk-sync-diff-apply remote-dict diff))

154;;;###autoload
155(defun nskk-sync-diff-calculate (local-dict remote-dict)
156 &quot;LOCAL-DICT と REMOTE-DICT の差分を計算する。
157
158LOCAL-DICT: ローカル辞書
159REMOTE-DICT: リモート辞書
160
161戻り値: `nskk-sync-diff' オブジェクト&quot;
162 (let* ((local-entries (nskk-sync-diff--dict-to-hash local-dict))
163 (remote-entries (nskk-sync-diff--dict-to-hash remote-dict))
164 (added nil)
165 (modified nil)
166 (deleted nil))
167
168 ;; 追加・変更検出
169 (maphash
170 (lambda (midashi local-entry)
171 (let ((remote-entry (gethash midashi remote-entries)))
172 (cond
173 ;; リモートに存在しない → 追加
174 ((null remote-entry)
175 (push (nskk-sync-diff-entry--create
176 :midashi midashi
177 :new-value local-entry
178 :timestamp (floor (float-time)))
179 added))
180 ;; 内容が異なる → 変更
181 ((not (nskk-sync-diff--entries-equal-p local-entry remote-entry))
182 (push (nskk-sync-diff-entry--create
183 :midashi midashi
184 :old-value remote-entry
185 :new-value local-entry
186 :timestamp (floor (float-time)))
187 modified)))))
188 local-entries)
189
190 ;; 削除検出
191 (maphash
192 (lambda (midashi remote-entry)
193 (unless (gethash midashi local-entries)
194 (push (nskk-sync-diff-entry--create
195 :midashi midashi
196 :old-value remote-entry
197 :timestamp (floor (float-time)))
198 deleted)))
199 remote-entries)
200
201 ;; 差分オブジェクト作成
202 (nskk-sync-diff--create
203 :added (nreverse added)
204 :modified (nreverse modified)
205 :deleted (nreverse deleted)
206 :metadata (list :timestamp (floor (float-time))
207 :local-size (hash-table-count local-entries)
208 :remote-size (hash-table-count remote-entries)))))

211(defun nskk-sync-diff-apply (dict diff)
212 &quot;DICT に DIFF を適用する。
213
214DICT: 辞書（変更される）
215DIFF: 差分オブジェクト
216
217戻り値: 更新された辞書&quot;
218 ;; 削除
219 (dolist (entry (nskk-sync-diff-deleted diff))
220 (nskk-dict-remove-entry dict (nskk-sync-diff-entry-midashi entry)))
221
222 ;; 追加
223 (dolist (entry (nskk-sync-diff-added diff))
224 (nskk-dict-add-entry dict
225 (nskk-sync-diff-entry-midashi entry)
226 (nskk-sync-diff-entry-new-value entry)))
227
228 ;; 変更
229 (dolist (entry (nskk-sync-diff-modified diff))
230 (nskk-dict-update-entry dict
231 (nskk-sync-diff-entry-midashi entry)
232 (nskk-sync-diff-entry-new-value entry)))
233
234 dict)

238(defun nskk-sync-diff--myers-diff (seq-a seq-b)
239 &quot;Myers差分アルゴリズムでSEQ-AとSEQ-Bの差分を計算する。
240
241SEQ-A, SEQ-B: 要素のリスト
242
243戻り値: 差分操作のリスト（'add, 'delete, 'keep）&quot;
244 (let* ((n (length seq-a))
245 (m (length seq-b))
246 (max-d (+ n m))
247 (v (make-vector (+ (* 2 max-d) 1) 0))
248 (trace nil))
249
250 ;; 最短編集距離を探索
251 (catch 'found
252 (cl-loop for d from 0 to max-d do
253 (cl-loop for k from (- d) to d by 2 do
254 (let* ((x (if (or (= k (- d))
255 (and (/= k d)
256 (&lt; (aref v (+ k 1 max-d))
257 (aref v (- k 1 max-d)))))
258 (aref v (+ k 1 max-d))
259 (+ (aref v (- k 1 max-d)) 1)))
260 (y (- x k)))
261
262 ;; 対角線を進む
263 (while (and (&lt; x n) (&lt; y m)
264 (equal (nth x seq-a) (nth y seq-b)))
265 (cl-incf x)
266 (cl-incf y))
267
268 (aset v (+ k max-d) x)
269
270 ;; ゴール到達
271 (when (and (&gt;= x n) (&gt;= y m))
272 (throw 'found d))))))
273
274 ;; バックトラック（簡略版）
275 (nskk-sync-diff--myers-backtrack seq-a seq-b)))
276
277(defun nskk-sync-diff--myers-backtrack (seq-a seq-b)
278 &quot;Myers差分のバックトラック（簡略実装）。&quot;
279 (let ((result nil)
280 (i 0)
281 (j 0))
282 (while (or (&lt; i (length seq-a))
283 (&lt; j (length seq-b)))
284 (cond
285 ;; 両方終了
286 ((and (&gt;= i (length seq-a))
287 (&gt;= j (length seq-b)))
288 nil)
289 ;; Aのみ残っている → 削除
290 ((&gt;= j (length seq-b))
291 (push (list 'delete (nth i seq-a)) result)
292 (cl-incf i))
293 ;; Bのみ残っている → 追加
294 ((&gt;= i (length seq-a))
295 (push (list 'add (nth j seq-b)) result)
296 (cl-incf j))
297 ;; 一致 → 保持
298 ((equal (nth i seq-a) (nth j seq-b))
299 (push (list 'keep (nth i seq-a)) result)
300 (cl-incf i)
301 (cl-incf j))
302 ;; 不一致 → 変更
303 (t
304 (push (list 'delete (nth i seq-a)) result)
305 (push (list 'add (nth j seq-b)) result)
306 (cl-incf i)
307 (cl-incf j))))
308 (nreverse result)))
