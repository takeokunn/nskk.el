172(defun nskk-ai-candidates-generate (midashi &optional base-candidates)
173 &quot;見出し語 MIDASHI に対してスマート候補を生成する。
174BASE-CANDIDATES は辞書からの基本候補リスト。
175戻り値: ランキングされた候補リスト（文字列のリスト）&quot;
176 (when (stringp midashi)
177 (cl-incf (plist-get nskk-ai-candidates-statistics :total-queries))
178
179 ;; キャッシュチェック
180 (let* ((context (when (and (featurep 'nskk-ai-context)
181 nskk-ai-candidates-enable-context-adaptation)
182 (nskk-ai-context-extract-context)))
183 (context-hash (sxhash-equal context))
184 (cache-key (cons midashi context-hash))
185 (cached (gethash cache-key nskk-ai-candidates-cache)))
186
187 (if cached
188 (progn
189 (cl-incf (plist-get nskk-ai-candidates-statistics :cache-hits))
190 cached)
191
192 ;; キャッシュミス - 候補を生成
193 (cl-incf (plist-get nskk-ai-candidates-statistics :cache-misses))
194
195 (let ((candidates (nskk-ai-candidates--generate-internal midashi base-candidates context)))
196 ;; キャッシュに保存
197 (puthash cache-key candidates nskk-ai-candidates-cache)
198
199 ;; キャッシュサイズ制限
200 (when (&gt; (hash-table-count nskk-ai-candidates-cache)
201 nskk-ai-candidates-cache-size)
202 (nskk-ai-candidates--prune-cache))
203
204 candidates)))))
205
206(defun nskk-ai-candidates--generate-internal (midashi base-candidates context)
207 &quot;内部的な候補生成処理。
208MIDASHI: 見出し語
209BASE-CANDIDATES: 辞書からの基本候補
210CONTEXT: 文脈テキスト
211戻り値: 候補リスト（文字列）&quot;
212 (let ((ai-candidates (make-hash-table :test 'equal))
213 (context-words (when context
214 (and (featurep 'nskk-ai-context)
215 (nskk-ai-context-parse-sentence context)))))
216
217 ;; 1. 辞書候補をAI候補に変換
218 (dolist (cand base-candidates)
219 (let ((ai-cand (nskk-ai-candidates--create-candidate
220 midashi cand :dict context-words)))
221 (puthash (nskk-ai-candidate-text ai-cand) ai-cand ai-candidates)))
222
223 ;; 2. パターンベース候補を追加
224 (when (and (featurep 'nskk-ai-pattern)
225 nskk-ai-candidates-enable-hybrid-mode)
226 (let ((pattern-candidates (nskk-ai-pattern-predict-candidate midashi context-words)))
227 (dolist (pred pattern-candidates)
228 (let* ((cand-text (car pred))
229 (confidence (cdr pred))
230 (existing (gethash cand-text ai-candidates)))
231 (if existing
232 ;; 既存候補のスコアを更新
233 (setf (nskk-ai-candidate-confidence existing)
234 (max (nskk-ai-candidate-confidence existing) confidence))
235 ;; 新規候補を追加
236 (let ((ai-cand (nskk-ai-candidates--create-candidate
237 midashi cand-text :pattern context-words)))
238 (setf (nskk-ai-candidate-confidence ai-cand) confidence)
239 (puthash cand-text ai-cand ai-candidates)))))))
240
241 ;; 3. 候補をスコアリング
242 (let ((scored-candidates nil))
243 (maphash
244 (lambda (_key ai-cand)
245 (nskk-ai-candidates--score-candidate ai-cand midashi context-words)
246 (when (&gt;= (nskk-ai-candidate-confidence ai-cand)
247 nskk-ai-candidates-min-confidence)
248 (push ai-cand scored-candidates)))
249 ai-candidates)
250
251 ;; 4. スコア順にソート
252 (setq scored-candidates
253 (sort scored-candidates
254 (lambda (a b)
255 (&gt; (nskk-ai-candidate-score a)
256 (nskk-ai-candidate-score b)))))
257
258 ;; 5. 上位N件を返す（文字列のリスト）
259 (mapcar #'nskk-ai-candidate-text
260 (seq-take scored-candidates nskk-ai-candidates-max-candidates)))))
261
262(defun nskk-ai-candidates--create-candidate (midashi text source context-words)
263 &quot;AI候補を作成する。
264MIDASHI: 見出し語
265TEXT: 候補テキスト
266SOURCE: ソース（:dict/:pattern/:ai）
267CONTEXT-WORDS: 文脈の単語リスト
268戻り値: nskk-ai-candidate構造体&quot;
269 (nskk-ai-candidate--create
270 :text text
271 :source source
272 :confidence 0.5 ; 初期値
273 :metadata (list :midashi midashi
274 :context-length (length context-words))))

276;;; スコアリング
277
278(defun nskk-ai-candidates--score-candidate (candidate midashi context-words)
279 &quot;候補 CANDIDATE をスコアリングする。
280MIDASHI: 見出し語
281CONTEXT-WORDS: 文脈の単語リスト&quot;
282 (let* ((text (nskk-ai-candidate-text candidate))
283 (freq-score (nskk-ai-candidates--calc-frequency-score midashi text))
284 (context-score (nskk-ai-candidates--calc-context-score midashi text context-words))
285 (pattern-score (nskk-ai-candidates--calc-pattern-score midashi text context-words))
286 (weights (nskk-ai-ranking-model-weights nskk-ai-candidates-ranking-model))
287 (total-score 0.0))
288
289 ;; 重み付き合算
290 (setq total-score
291 (+ (* (aref weights 0) freq-score)
292 (* (aref weights 1) context-score)
293 (* (aref weights 2) pattern-score)))
294
295 ;; スコアを設定
296 (setf (nskk-ai-candidate-score candidate) total-score)
297 (setf (nskk-ai-candidate-scores candidate)
298 (list :frequency freq-score
299 :context context-score
300 :pattern pattern-score
301 :total total-score))
302
303 ;; 信頼度を更新（スコアベース）
304 (setf (nskk-ai-candidate-confidence candidate)
305 (min 1.0 (max 0.0 total-score)))))
306
307(defsubst nskk-ai-candidates--calc-frequency-score (midashi candidate)
308 &quot;頻度スコアを計算する。&quot;
309 (if (featurep 'nskk-learning-frequency)
310 (let ((score (nskk-get-frequency-score midashi candidate)))
311 ;; スコアを0-1に正規化（仮定: 最大100）
312 (min 1.0 (/ score 100.0)))
313 0.5)) ; デフォルト
314
315(defsubst nskk-ai-candidates--calc-context-score (midashi candidate context-words)
316 &quot;文脈スコアを計算する。&quot;
317 (if (and (featurep 'nskk-ai-context)
318 nskk-ai-candidates-enable-context-adaptation
319 context-words)
320 (nskk-ai-context-score-candidate midashi candidate)
321 ;; 文脈学習のフォールバック
322 (if (featurep 'nskk-learning-context)
323 (let ((score (nskk-context-score midashi candidate)))
324 (min 1.0 (max 0.0 score)))
325 0.5))) ; デフォルト
326
327(defsubst nskk-ai-candidates--calc-pattern-score (midashi candidate context-words)
328 &quot;パターンスコアを計算する。&quot;
329 (if (and (featurep 'nskk-ai-pattern)
330 nskk-ai-candidates-enable-hybrid-mode)
331 (let ((predictions (nskk-ai-pattern-predict-candidate midashi context-words)))
332 (or (cdr (assoc candidate predictions))
333 0.1)) ; 予測に含まれない場合は低スコア
334 0.5)) ; デフォルト

339(defun nskk-ai-candidates-merge-sources (dict-candidates ai-candidates)
340 &quot;辞書候補 DICT-CANDIDATES とAI候補 AI-CANDIDATES を統合する。
341戻り値: 統合された候補リスト&quot;
342 (when nskk-ai-candidates-enable-hybrid-mode
343 (let ((merged (make-hash-table :test 'equal)))
344 ;; 辞書候補を追加
345 (dolist (cand dict-candidates)
346 (puthash cand (list :source :dict :score 1.0) merged))
347
348 ;; AI候補を追加/マージ
349 (dolist (cand ai-candidates)
350 (let ((existing (gethash cand merged)))
351 (if existing
352 ;; 既存候補のスコアを更新
353 (plist-put existing :score
354 (max (plist-get existing :score) 0.8))
355 ;; 新規候補を追加
356 (puthash cand (list :source :ai :score 0.8) merged))))
357
358 ;; スコア順にソート
359 (let ((result nil))
360 (maphash
361 (lambda (cand info)
362 (push (cons cand (plist-get info :score)) result))
363 merged)
364 (mapcar #'car
365 (sort result (lambda (a b) (&gt; (cdr a) (cdr b)))))))))

369;;;###autoload
370(defun nskk-ai-candidates-learn-selection (midashi selected-candidate all-candidates)
371 &quot;ユーザーが選択した候補から学習する。
372MIDASHI: 見出し語
373SELECTED-CANDIDATE: 選択された候補
374ALL-CANDIDATES: 全候補リスト&quot;
375 (when (and nskk-ai-candidates-enable-learning
376 (stringp midashi)
377 (stringp selected-candidate))
378
379 ;; フィードバックを記録
380 (let* ((context (when (featurep 'nskk-ai-context)
381 (nskk-ai-context-extract-context)))
382 (context-words (when context
383 (nskk-ai-context-parse-sentence context)))
384 (rank (cl-position selected-candidate all-candidates :test #'equal))
385 (feedback (list :midashi midashi
386 :selected selected-candidate
387 :rank (or rank -1)
388 :context context-words
389 :timestamp (current-time))))
390
391 (push feedback nskk-ai-candidates-feedback-history)
392
393 ;; 精度統計を更新
394 (when rank
395 (when (= rank 0)
396 (cl-incf (plist-get nskk-ai-candidates-statistics :top1-accuracy)))
397 (when (&lt;= rank 2)
398 (cl-incf (plist-get nskk-ai-candidates-statistics :top3-accuracy))))
399
400 ;; 各学習システムに通知
401 (when (featurep 'nskk-learning-frequency)
402 (nskk-update-frequency midashi selected-candidate))
403
404 (when (and (featurep 'nskk-learning-context) context-words)
405 (let ((prev (car (last context-words))))
406 (when prev
407 (nskk-learn-context prev selected-candidate))))
408
409 (when (and (featurep 'nskk-ai-pattern) context-words)
410 (nskk-ai-pattern-learn midashi selected-candidate context-words))
411
412 (when (featurep 'nskk-ai-context)
413 (nskk-ai-context-learn-text (concat context selected-candidate)))
414
415 ;; キャッシュをクリア（学習により古いランキングは無効）
416 (nskk-ai-candidates--clear-cache-for-midashi midashi)
417
418 ;; 定期的にモデルを更新（100回ごと）
419 (when (zerop (mod (length nskk-ai-candidates-feedback-history) 100))
420 (nskk-ai-candidates--update-ranking-model)))))
421
422(defun nskk-ai-candidates--update-ranking-model ()
423 &quot;ランキングモデルを更新する（簡易版）。
424フィードバック履歴から最適な重みを学習。&quot;
425 (when (&gt; (length nskk-ai-candidates-feedback-history) 10)
426 ;; 簡易的な重み調整（正答率ベース）
427 (let* ((recent-feedback (seq-take nskk-ai-candidates-feedback-history 100))
428 (success-count 0)
429 (total-count (length recent-feedback)))
430
431 (dolist (fb recent-feedback)
432 (when (&lt;= (plist-get fb :rank) 2)
433 (cl-incf success-count)))
434
435 (let ((accuracy (/ (float success-count) total-count)))
436 (setf (nskk-ai-ranking-model-accuracy nskk-ai-candidates-ranking-model)
437 accuracy)
438
439 ;; 精度が低い場合、重みを調整（ヒューリスティック）
440 (when (&lt; accuracy 0.7)
441 (let ((weights (nskk-ai-ranking-model-weights nskk-ai-candidates-ranking-model)))
442 ;; 文脈の重みを増やす
443 (aset weights 1 (min 1.0 (* (aref weights 1) 1.1)))
444 ;; 正規化
445 (let ((sum (+ (aref weights 0) (aref weights 1) (aref weights 2))))
446 (dotimes (i 3)
447 (aset weights i (/ (aref weights i) sum))))))))))

451(defun nskk-ai-candidates--prune-cache ()
452 &quot;キャッシュから古いエントリを削除する。&quot;
453 (let ((keys nil))
454 (maphash (lambda (k _v) (push k keys)) nskk-ai-candidates-cache)
455 ;; ランダムに半分削除（簡易的なLRU代替）
456 (dolist (key (seq-take (shuffle keys)
457 (/ (length keys) 2)))
458 (remhash key nskk-ai-candidates-cache))))

460(defun nskk-ai-candidates--clear-cache-for-midashi (midashi)
461 &quot;見出し語 MIDASHI に関連するキャッシュをクリアする。&quot;
462 (let ((keys-to-remove nil))
463 (maphash
464 (lambda (key _value)
465 (when (equal (car key) midashi)
466 (push key keys-to-remove)))
467 nskk-ai-candidates-cache)
468 (dolist (key keys-to-remove)
469 (remhash key nskk-ai-candidates-cache))))
