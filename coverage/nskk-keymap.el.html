238(defun nskk-define-key (keymap key command &optional force)
239 &quot;KEYMAPにKEYとCOMANDのバインディングを定義する。
240
241引数:
242 KEYMAP - キーマップ
243 KEY - キー定義（文字列またはベクター）
244 COMMAND - コマンド（シンボルまたは関数）
245 FORCE - 非nilの場合、既存のバインディングを上書き
246
247戻り値: t（成功）、nil（既存バインディングあり&forceがnil）&quot;
248 (let* ((key-sequence (if (stringp key) (kbd key) key))
249 (existing (lookup-key keymap key-sequence)))
250 (when (or force (not existing) (numberp existing))
251 (define-key keymap key-sequence command)
252 t)))
253
254(defun nskk-undefine-key (keymap key)
255 &quot;KEYMAPからKEYのバインディングを削除する。
256
257引数:
258 KEYMAP - キーマップ
259 KEY - キー定義（文字列またはベクター）&quot;
260 (let ((key-sequence (if (stringp key) (kbd key) key)))
261 (define-key keymap key-sequence nil)))
262
263(defun nskk-define-keys (keymap bindings &optional force)
264 &quot;KEYMAPに複数のキーバインディングを一括定義する。
265
266引数:
267 KEYMAP - キーマップ
268 BINDINGS - ((key . command) ...) 形式のリスト
269 FORCE - 非nilの場合、既存のバインディングを上書き
270
271戻り値: 成功したバインディング数&quot;
272 (let ((count 0))
273 (dolist (binding bindings)
274 (when (nskk-define-key keymap (car binding) (cdr binding) force)
275 (setq count (1+ count))))
276 count))

383(defun nskk-get-current-keymap ()
384 &quot;現在のNSKK入力モードに対応するキーマップを返す。
385モードが初期化されていない場合はnskk-mode-mapを返す。&quot;
386 (if (bound-and-true-p nskk-current-state)
387 (let ((mode (nskk-state-mode nskk-current-state)))
388 (cond
389 ((eq mode 'hiragana) nskk-hiragana-mode-map)
390 ((eq mode 'katakana) nskk-katakana-mode-map)
391 ((eq mode 'latin) nskk-ascii-mode-map)
392 ((eq mode 'zenkaku-latin) nskk-zenkaku-latin-mode-map)
393 ((eq mode 'abbrev) nskk-abbrev-mode-map)
394 ((eq mode 'conversion)
395 (if (nskk-state-has-candidates-p nskk-current-state)
396 nskk-kouho-mode-map
397 nskk-henkan-mode-map))
398 (t nskk-mode-map)))
399 nskk-mode-map))

406
407戻り値:
408 対応するキーマップ、または不明な場合はnil&quot;
409 (cond
410 ((eq mode 'hiragana) nskk-hiragana-mode-map)
411 ((eq mode 'katakana) nskk-katakana-mode-map)
412 ((eq mode 'latin) nskk-ascii-mode-map)
413 ((eq mode 'zenkaku-latin) nskk-zenkaku-latin-mode-map)
414 ((eq mode 'abbrev) nskk-abbrev-mode-map)
415 ((eq mode 'henkan) nskk-henkan-mode-map)
416 ((eq mode 'kouho) nskk-kouho-mode-map)
417 (t nil)))

421(defun nskk-customize-key (mode key command)
422 &quot;指定されたモードのキーバインドをカスタマイズする。
423
424引数:
425 MODE - モードシンボル
426 KEY - キー定義（文字列）
427 COMMAND - バインドするコマンド
428
429例:
430 (nskk-customize-key 'hiragana \&quot;C-j\&quot; 'my-custom-command)&quot;
431 (interactive
432 (list
433 (intern (completing-read &quot;Mode: &quot;
434 '(hiragana katakana latin zenkaku-latin
435 abbrev henkan kouho)))
436 (read-string &quot;Key: &quot;)
437 (read-command &quot;Command: &quot;)))
438 (when-let ((keymap (nskk-get-keymap-for-mode mode)))
439 (nskk-define-key keymap key command t)
440 (message &quot;Customized key binding: %s in %s mode -&gt; %s&quot;
441 key mode command)))

445(defun nskk-describe-keymap (&optional mode)
446 &quot;指定されたMODEのキーマップ情報を表示する。
447MODEが省略された場合は現在のモードのキーマップを表示。&quot;
448 (interactive)
449 (let* ((target-mode (or mode
450 (when (bound-and-true-p nskk-current-state)
451 (nskk-state-mode nskk-current-state))
452 'hiragana))
453 (keymap (nskk-get-keymap-for-mode target-mode)))
