27;; このファイルはNSKKの形容詞活用エンジンを実装します。
28;;
29;; 対応する活用:
30;; - イ形容詞活用（高い→高く、高かっ、高けれ）
31;; - ナ形容詞活用（静か→静かに、静かだ、静かで）
32;; - 形容動詞活用（きれい→きれいに、きれいだ）
33;;
34;; 活用形:
35;; - 未然形（なかろ）
36;; - 連用形（く/に、かっ/だっ）
37;; - 終止形（い/だ）
38;; - 連体形（い/な）
39;; - 仮定形（けれ/なら）
40;;
41;; 使用例:
42;; (nskk-conjugate-adjective &quot;高&quot; 'i-adjective 'renyou-ku) ;; =&gt; &quot;高く&quot;
43;; (nskk-conjugate-adjective &quot;静か&quot; 'na-adjective 'renyou-ni) ;; =&gt; &quot;静かに&quot;

106(defun nskk-adjective--remove-i (stem)
107 &quot;イ形容詞の語幹から語尾「い」を除去する。
108STEMが「い」で終わっている場合、それを除去した部分を返す。&quot;
109 (if (and stem
110 (&gt; (length stem) 0)
111 (string-suffix-p &quot;い&quot; stem))
112 (substring stem 0 -1)
113 stem))
114
115(defun nskk-adjective--conjugate-i-adjective (stem form)
116 &quot;イ形容詞の活用を返す。
117STEMは語幹（例: \&quot;高\&quot;、\&quot;高い\&quot;のいずれも可）、FORMは活用形。&quot;
118 (let* ((base (nskk-adjective--remove-i stem))
119 (ending (cdr (assq form nskk-i-adjective-endings))))
120 (when ending
121 (concat base ending))))
122
123(defun nskk-adjective--conjugate-na-adjective (stem form)
124 &quot;ナ形容詞の活用を返す。
125STEMは語幹（例: \&quot;静か\&quot;）、FORMは活用形。&quot;
126 (let ((ending (cdr (assq form nskk-na-adjective-endings))))
127 (when ending
128 (concat stem ending))))

132;;;###autoload
133(defun nskk-conjugate-adjective (stem type form)
134 &quot;形容詞STEMをTYPE（活用型）とFORM（活用形）に基づいて活用する。
135
136引数:
137 STEM - 形容詞の語幹（文字列）
138 TYPE - 活用型（symbol）
139 'i-adjective, 'na-adjective
140 FORM - 活用形（symbol）
141 イ形容詞: 'mizen, 'renyou-ku, 'renyou-katta, 'shushi, 'rentai, 'katei
142 ナ形容詞: 'mizen, 'renyou-ni, 'renyou-datta, 'shushi-da, 'rentai-na, 'katei-nara
143
144返り値:
145 活用後の文字列、またはnil（無効な入力の場合）
146
147使用例:
148 (nskk-conjugate-adjective \&quot;高\&quot; 'i-adjective 'renyou-ku) ;; =&gt; \&quot;高く\&quot;
149 (nskk-conjugate-adjective \&quot;高い\&quot; 'i-adjective 'renyou-ku) ;; =&gt; \&quot;高く\&quot;
150 (nskk-conjugate-adjective \&quot;高\&quot; 'i-adjective 'renyou-katta) ;; =&gt; \&quot;高かっ\&quot;
151 (nskk-conjugate-adjective \&quot;高\&quot; 'i-adjective 'katei) ;; =&gt; \&quot;高けれ\&quot;
152 (nskk-conjugate-adjective \&quot;静か\&quot; 'na-adjective 'renyou-ni) ;; =&gt; \&quot;静かに\&quot;
153 (nskk-conjugate-adjective \&quot;静か\&quot; 'na-adjective 'shushi-da) ;; =&gt; \&quot;静かだ\&quot;
154 (nskk-conjugate-adjective \&quot;静か\&quot; 'na-adjective 'rentai-na) ;; =&gt; \&quot;静かな\&quot;&quot;
155 (unless (stringp stem)
156 (signal 'wrong-type-argument (list 'stringp stem)))
157 (unless (memq type nskk-adjective-types)
158 (signal 'wrong-type-argument (list 'nskk-adjective-types type)))
159 (unless (memq form nskk-adjective-forms)
160 (signal 'wrong-type-argument (list 'nskk-adjective-forms form)))
161
162 (pcase type
163 ('i-adjective (nskk-adjective--conjugate-i-adjective stem form))
164 ('na-adjective (nskk-adjective--conjugate-na-adjective stem form))
165 (_ nil)))

167;;;###autoload
168(defun nskk-adjective-get-all-forms (stem type)
169 &quot;形容詞STEMの全活用形を返す。
170
171引数:
172 STEM - 形容詞の語幹（文字列）
173 TYPE - 活用型（symbol）
174
175返り値:
176 活用形と活用後の文字列のalist
177
178使用例:
179 (nskk-adjective-get-all-forms \&quot;高\&quot; 'i-adjective)
180 ;; =&gt; ((mizen . \&quot;高かろ\&quot;) (renyou-ku . \&quot;高く\&quot;) ...)&quot;
181 (let ((forms (if (eq type 'i-adjective)
182 '(mizen renyou-ku renyou-katta shushi rentai katei)
183 '(mizen renyou-ni renyou-datta shushi-da rentai-na katei-nara))))
184 (mapcar (lambda (form)
185 (cons form (nskk-conjugate-adjective stem type form)))
186 forms)))

190(defun nskk-adjective-is-special-p (stem)
191 &quot;特殊な活用をする形容詞かどうかを判定する。
192現在は「いい」「よい」などの特殊形容詞をサポート。&quot;
193 (member stem '(&quot;いい&quot; &quot;良い&quot; &quot;よい&quot;)))
194
195(defun nskk-adjective-conjugate-special (stem form)
196 &quot;特殊形容詞の活用を返す。
197「いい」「よい」などの不規則活用に対応。&quot;
198 (cond
199 ;; 「いい」「よい」の活用
200 ((or (equal stem &quot;いい&quot;) (equal stem &quot;良い&quot;) (equal stem &quot;よい&quot;))
201 (pcase form
202 ('mizen &quot;よかろ&quot;)
203 ('renyou-ku &quot;よく&quot;)
204 ('renyou-katta &quot;よかっ&quot;)
205 ('shushi (if (equal stem &quot;いい&quot;) &quot;いい&quot; &quot;よい&quot;))
206 ('rentai (if (equal stem &quot;いい&quot;) &quot;いい&quot; &quot;よい&quot;))
207 ('katei &quot;よけれ&quot;)
208 (_ nil)))
209 (t nil)))

213(defun nskk-adjective-get-base-form (conjugated type)
214 &quot;活用形CONJUGATEDから基本形を推測する（逆変換）。
215完全な実装ではなく、ヒューリスティックな推測を行う。&quot;
216 (cond
217 ((eq type 'i-adjective)
218 ;; イ形容詞の場合
219 (cond
220 ((string-suffix-p &quot;く&quot; conjugated)
221 (concat (substring conjugated 0 -1) &quot;い&quot;))
222 ((string-suffix-p &quot;かっ&quot; conjugated)
223 (concat (substring conjugated 0 -2) &quot;い&quot;))
224 ((string-suffix-p &quot;けれ&quot; conjugated)
225 (concat (substring conjugated 0 -2) &quot;い&quot;))
226 ((string-suffix-p &quot;かろ&quot; conjugated)
227 (concat (substring conjugated 0 -2) &quot;い&quot;))
228 ((string-suffix-p &quot;い&quot; conjugated)
229 conjugated)
230 (t nil)))
231 ((eq type 'na-adjective)
232 ;; ナ形容詞の場合
233 (cond
234 ((string-suffix-p &quot;に&quot; conjugated)
235 (substring conjugated 0 -1))
236 ((string-suffix-p &quot;だっ&quot; conjugated)
237 (substring conjugated 0 -2))
238 ((string-suffix-p &quot;だ&quot; conjugated)
239 (substring conjugated 0 -1))
240 ((string-suffix-p &quot;な&quot; conjugated)
241 (substring conjugated 0 -1))
242 ((string-suffix-p &quot;なら&quot; conjugated)
243 (substring conjugated 0 -2))
244 (t nil)))
245 (t nil)))

259(defun nskk-adjective-print-all-forms (stem type)
260 &quot;形容詞STEMの全活用形を表形式で表示する（デバッグ用）。&quot;
261 (interactive &quot;s語幹: \nS活用型 (i-adjective/na-adjective): &quot;)

263 (with-output-to-temp-buffer &quot;*NSKK Adjective Conjugation*&quot;
264 (princ (format &quot;形容詞「%s」の活用（%s）\n&quot; stem type))
265 (princ &quot;====================\n\n&quot;)
266 (dolist (entry forms)
267 (princ (format &quot;%-15s: %s\n&quot;
268 (symbol-name (car entry))
269 (cdr entry)))))))
