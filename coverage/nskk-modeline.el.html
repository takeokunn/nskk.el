269(defun nskk-modeline-expand-format (format-string mode state dict-status)
270 &quot;FORMAT-STRING 内のプレースホルダーを展開する。
271
272引数:
273 FORMAT-STRING - フォーマット文字列
274 MODE - 入力モード
275 STATE - 入力状態
276 DICT-STATUS - 辞書状態
277
278返り値:
279 展開された文字列&quot;
280 (let ((mode-indicator (nskk-state-mode-indicator mode))
281 (state-indicator (nskk-modeline-get-state-indicator state))
282 (dict-indicator (nskk-modeline-get-dict-status-indicator dict-status))
283 (result format-string))
284 ;; まず %% を一時的な文字列に置換
285 (setq result (replace-regexp-in-string &quot;%%&quot; &quot;\x00PERCENT\x00&quot; result t t))
286 ;; プレースホルダーを順番に置換
287 (setq result (replace-regexp-in-string &quot;%m&quot; mode-indicator result t t))
288 (setq result (replace-regexp-in-string &quot;%s&quot; state-indicator result t t))
289 (setq result (replace-regexp-in-string &quot;%d&quot; dict-indicator result t t))
290 ;; 一時文字列を % に戻す
291 (setq result (replace-regexp-in-string &quot;\x00PERCENT\x00&quot; &quot;%&quot; result t t))
292 result))

296(defun nskk-modeline-format (mode state &optional dict-status)
297 &quot;モードライン表示用の文字列を生成する。
298
299引数:
300 MODE - 入力モード（'hiragana, 'katakana, 'latin など）
301 STATE - 入力状態（'normal, 'converting, 'selecting）
302 DICT-STATUS - 辞書状態（nil, 'loading, 'error）
303
304返り値:
305 装飾されたモードライン文字列&quot;
306 (let* ((mode-indicator (nskk-state-mode-indicator mode))
307 (state-indicator (nskk-modeline-get-state-indicator state))
308 (dict-indicator (nskk-modeline-get-dict-status-indicator dict-status))
309
310 ;; 各要素を装飾
311 (mode-text (nskk-modeline-propertize
312 mode-indicator
313 (nskk-modeline-get-mode-face mode)))
314 (state-text (if (string-empty-p state-indicator)
315 &quot;&quot;
316 (nskk-modeline-propertize
317 state-indicator
318 (nskk-modeline-get-state-indicator-face state))))
319 (dict-text (if (string-empty-p dict-indicator)
320 &quot;&quot;
321 (nskk-modeline-propertize
322 dict-indicator
323 (nskk-modeline-get-dict-status-face dict-status))))
324
325 ;; フォーマット文字列を処理
326 (format-str nskk-modeline-format)
327 (result format-str))
328
329 ;; プレースホルダーを装飾されたテキストで置換
330 ;; まず %% を一時的な文字列に置換
331 (setq result (replace-regexp-in-string &quot;%%&quot; &quot;\x00PERCENT\x00&quot; result t t))
332 ;; プレースホルダーを順番に置換（装飾されたテキストを使用）
333 (setq result (replace-regexp-in-string &quot;%m&quot; mode-text result t t))
334 (setq result (replace-regexp-in-string &quot;%s&quot; state-text result t t))
335 (setq result (replace-regexp-in-string &quot;%d&quot; dict-text result t t))
336 ;; 一時文字列を % に戻す
337 (setq result (replace-regexp-in-string &quot;\x00PERCENT\x00&quot; &quot;%&quot; result t t))
338 result))

355(defun nskk-modeline-state-from-nskk-state (nskk-state)
356 &quot;NSKK-STATE から nskk-modeline-state を生成する。
357
358引数:
359 NSKK-STATE - nskk-state 構造体
360
361返り値:
362 nskk-modeline-state 構造体&quot;
363 (let* ((mode (nskk-state-mode nskk-state))
364 (has-candidates (nskk-state-has-candidates-p nskk-state))
365 (has-conversion-buffer (not (string-empty-p
366 (nskk-state-conversion-buffer nskk-state))))
367 ;; 状態を判定
368 (state (cond
369 ((and has-candidates (eq mode 'conversion))
370 'selecting)
371 (has-conversion-buffer
372 'converting)
373 (t
374 'normal))))
375 (nskk-modeline-state-create
376 :mode mode
377 :state state)))

384(defun nskk-modeline-update (&optional mode state dict-status)
385 &quot;モードライン表示を更新する。
386
387引数:
388 MODE - 入力モード（省略時は現在の状態から取得）
389 STATE - 入力状態（省略時は現在の状態から判定）
390 DICT-STATUS - 辞書状態（省略時は nil）
391
392副作用:
393 `nskk-modeline-string' を更新し、モードラインを再描画する。&quot;
394 (interactive)
395 (let* ((nskk-state (or nskk-current-state
396 (nskk-state-create)))
397 (actual-mode (or mode (nskk-state-mode nskk-state)))
398 (ml-state (if (and (null mode) (null state))
399 ;; 引数が省略された場合は状態から自動判定
400 (nskk-modeline-state-from-nskk-state nskk-state)
401 ;; 引数が指定された場合はそれを使用
402 (nskk-modeline-state-create
403 :mode actual-mode
404 :state (or state 'normal)
405 :dict-status dict-status))))
406
407 (setq nskk-modeline-string
408 (nskk-modeline-format-from-state ml-state))
409
410 ;; モードラインを強制再描画
411 (force-mode-line-update)))

415(defun nskk-modeline-install ()
416 &quot;NSKKモードライン表示を mode-line-format に組み込む。&quot;
417 (unless (memq 'nskk-modeline-string mode-line-format)
418 ;; mode-line-position の前に挿入
419 (let ((pos (cl-position nskk-modeline-position mode-line-format)))
420 (if pos
421 (setq mode-line-format
422 (append (cl-subseq mode-line-format 0 pos)
423 '(nskk-modeline-string)
424 (cl-subseq mode-line-format pos)))
425 ;; 見つからない場合は末尾に追加
426 (setq mode-line-format
427 (append mode-line-format '(nskk-modeline-string)))))))
428
429(defun nskk-modeline-uninstall ()
430 &quot;NSKKモードライン表示を mode-line-format から削除する。&quot;
431 (setq mode-line-format
432 (delq 'nskk-modeline-string mode-line-format))
433 (setq nskk-modeline-string &quot;&quot;))

453(defun nskk-modeline-preview-all-modes ()
454 &quot;全てのモードのモードライン表示をプレビューする。&quot;
455 (interactive)
456 (let ((modes '(hiragana katakana latin zenkaku-latin abbrev conversion))
457 (states '(normal converting selecting))
458 (preview-buffer (get-buffer-create &quot;*NSKK Modeline Preview*&quot;)))
459 (with-current-buffer preview-buffer
460 (erase-buffer)
461 (insert &quot;NSKK Modeline Preview\n&quot;)
462 (insert &quot;=====================\n\n&quot;)
463
464 ;; 各モードと状態の組み合わせを表示
465 (dolist (mode modes)
466 (insert (format &quot;Mode: %s\n&quot; mode))
467 (dolist (state states)
468 (let ((ml-string (nskk-modeline-format mode state nil)))
469 (insert (format &quot; %s: %s\n&quot; state ml-string))))
470 (insert &quot;\n&quot;))
471
472 ;; 辞書状態付き
473 (insert &quot;Dictionary Status:\n&quot;)
474 (insert (format &quot; Loading: %s\n&quot;
475 (nskk-modeline-format 'hiragana 'normal 'loading)))
476 (insert (format &quot; Error: %s\n&quot;
477 (nskk-modeline-format 'hiragana 'normal 'error)))
478 (insert &quot;\n&quot;)
479
480 (goto-char (point-min))
481 (special-mode))
482 (display-buffer preview-buffer)))
