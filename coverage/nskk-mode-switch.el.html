131(defun nskk-mode-switch (to-mode &optional force no-hooks)
132 &quot;現在のNSKK状態を TO-MODE に切り替える。
133
134引数:
135 TO-MODE - 切り替え先のモード（`nskk-state-modes' のいずれか）
136 FORCE - 非nilの場合、遷移可能性チェックをスキップ
137 NO-HOOKS - 非nilの場合、フックの実行をスキップ
138
139返り値:
140 切り替えに成功した場合は t、失敗した場合は nil
141
142副作用:
143 - `nskk-current-state' のモードを変更
144 - 未確定入力のクリア（設定による）
145 - フックの実行
146 - モードライン更新
147 - メッセージ表示&quot;
148 (unless nskk-current-state
149 (nskk-state-init))
150
151 ;; 再帰的な切り替えを防止
152 (when nskk-mode-switch--in-progress
153 (error &quot;Mode switch already in progress&quot;))
154
155 (let ((from-mode (nskk-state-mode nskk-current-state))
156 (nskk-mode-switch--in-progress t))
157
158 (unwind-protect
159 (progn
160 ;; 同じモードへの切り替えはスキップ
161 (if (eq from-mode to-mode)
162 t
163 ;; before フック実行
164 (unless no-hooks
165 (run-hook-with-args 'nskk-mode-switch-before-hook
166 from-mode to-mode))
167
168 ;; 状態遷移を実行
169 (if (nskk-state-transition nskk-current-state to-mode force)
170 (progn
171 ;; 入力バッファクリア
172 (nskk-mode-switch--clear-input-if-needed nskk-current-state)
173
174 ;; after フック実行
175 (unless no-hooks
176 (run-hook-with-args 'nskk-mode-switch-after-hook
177 from-mode to-mode)
178 ;; モード別フック実行
179 (run-hooks (nskk-mode-switch--get-mode-hook to-mode)))
180
181 ;; UI更新
182 (nskk-mode-switch--update-modeline)
183 (nskk-mode-switch--show-message to-mode)
184 t)
185 nil)))
186
187 ;; cleanup
188 (setq nskk-mode-switch--in-progress nil))))

219(defun nskk-mode-toggle-kana ()
220 &quot;ひらがなモードとカタカナモードを切り替える。&quot;
221 (interactive)
222 (unless nskk-current-state
223 (nskk-state-init))
224
225 (let ((current-mode (nskk-state-mode nskk-current-state)))
226 (cond
227 ((eq current-mode 'hiragana)
228 (nskk-mode-switch 'katakana))
229 ((eq current-mode 'katakana)
230 (nskk-mode-switch 'hiragana))
231 (t
232 (nskk-mode-switch 'hiragana)))))
233
234(defun nskk-mode-toggle-latin ()
235 &quot;かなモードと英数モードを切り替える。
236現在のモードに応じて最適なモードに切り替える。&quot;
237 (interactive)
238 (unless nskk-current-state
239 (nskk-state-init))
240
241 (let ((current-mode (nskk-state-mode nskk-current-state)))
242 (cond
243 ((memq current-mode '(hiragana katakana abbrev))
244 (nskk-mode-switch 'latin))
245 ((eq current-mode 'latin)
246 ;; 前回のかなモードに戻る、なければひらがな
247 (let ((prev-mode (nskk-state-previous-mode nskk-current-state)))
248 (if (memq prev-mode '(hiragana katakana))
249 (nskk-mode-switch prev-mode)
250 (nskk-mode-switch 'hiragana))))
251 (t
252 (nskk-mode-switch 'latin)))))

271(defun nskk-mode-cycle-forward ()
272 &quot;モードを順方向に循環切り替えする。
273ひらがな → カタカナ → 全角英数 → 半角英数 → ひらがな&quot;
274 (interactive)
275 (unless nskk-current-state
276 (nskk-state-init))
277
278 (let ((current-mode (nskk-state-mode nskk-current-state)))
279 (cond
280 ((eq current-mode 'hiragana) (nskk-mode-switch 'katakana))
281 ((eq current-mode 'katakana) (nskk-mode-switch 'zenkaku-latin))
282 ((eq current-mode 'zenkaku-latin) (nskk-mode-switch 'latin))
283 ((eq current-mode 'latin) (nskk-mode-switch 'hiragana))
284 (t (nskk-mode-switch 'hiragana)))))
