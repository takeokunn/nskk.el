251(defun nskk-server-protocol-parse-response (response-str &optional encoding)
252 &quot;サーバーレスポンスをパースする。
253
254引数:
255 RESPONSE-STR - レスポンス文字列（バイトシーケンス）
256 ENCODING - エンコーディング（省略時はnskk-server-protocol-encoding）
257
258戻り値:
259 nskk-server-response構造体
260
261レスポンス形式:
262 - '1/&lt;candidate1&gt;/&lt;candidate2&gt;/.../' : 見つかった
263 - '4' : 見つからない
264 - '&lt;info-text&gt;' : サーバー情報&quot;
265 (unless (and (stringp response-str) (&gt; (length response-str) 0))
266 (signal 'nskk-server-protocol-invalid-response
267 (list &quot;Response must be a non-empty string&quot;)))
268
269 (let* ((encoding (or encoding nskk-server-protocol-encoding))
270 ;; 改行を除去
271 (response (string-trim-right response-str &quot;[\r\n]+&quot;))
272 ;; デコード
273 (decoded (decode-coding-string response encoding)))
274
275 (when nskk-server-protocol-verbose
276 (message &quot;Parsing response: %s&quot; decoded))
277
278 (cond
279 ;; 見つからない
280 ((string= decoded &quot;4&quot;)
281 (nskk-server-response--create :status 'not-found))
282
283 ;; 検索結果あり
284 ((string-prefix-p &quot;1/&quot; decoded)
285 (let ((candidates (nskk-server-protocol--parse-candidates decoded)))
286 (nskk-server-response--create
287 :status 'found
288 :candidates candidates)))
289
290 ;; サーバー情報
291 (t
292 (nskk-server-response--create
293 :status 'info
294 :info decoded)))))

296(defun nskk-server-protocol--parse-candidates (response-str)
297 &quot;候補リストをパースする（内部関数）。
298
299引数:
300 RESPONSE-STR - レスポンス文字列（\&quot;1/...\&quot; 形式）
301
302戻り値:
303 候補リスト（(word . annotation)のリスト）
304
305形式:
306 1/候補1/候補2;注釈2/.../&quot;
307 (unless (string-prefix-p &quot;1/&quot; response-str)
308 (signal 'nskk-server-protocol-parse-error
309 (list (format &quot;Invalid response format: %s&quot; response-str))))
310
311 ;; &quot;1/&quot; を除去して候補部分を取得
312 (let* ((candidate-part (substring response-str 2))
313 (candidates nil))
314
315 ;; 末尾の &quot;/&quot; を除去
316 (when (string-suffix-p &quot;/&quot; candidate-part)
317 (setq candidate-part (substring candidate-part 0 -1)))
318
319 ;; &quot;/&quot; で分割
320 (when (&gt; (length candidate-part) 0)
321 (let ((raw-candidates (split-string candidate-part &quot;/&quot; t)))
322 (dolist (raw-cand raw-candidates)
323 ;; 注釈をパース（候補;注釈 形式）
324 (if (string-match &quot;^\\([^;]+\\);\\(.+\\)$&quot; raw-cand)
325 (push (cons (match-string 1 raw-cand)
326 (match-string 2 raw-cand))
327 candidates)
328 (push (cons raw-cand nil) candidates)))))
329
330 (nreverse candidates)))

369 &quot;リクエストの妥当性を検証する。
370
371引数:
372 REQUEST - nskk-server-request構造体
373
374戻り値:
375 エラーメッセージのリスト（空リストの場合は妥当）&quot;
376 (let ((errors nil)

390 (when (memq command

392 nskk-server-protocol-command-search-okuri-ari))
393 (unless (and (stringp key) (&gt; (length key) 0))

396 errors))
