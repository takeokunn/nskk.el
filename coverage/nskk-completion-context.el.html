141(defun nskk-completion-context-search (index query &rest args)
142 &quot;文脈を考慮した補完検索を実行する。
143
144引数:
145 INDEX - nskk-dict-index構造体
146 QUERY - 検索クエリ
147
148キーワード引数:
149 :context - 文脈（直前の単語のリスト、新しい順）
150 :model - 文脈モデル（nskk-completion-context-model構造体）
151 :limit - 最大結果数
152 :okuri-type - 送り仮名タイプ
153 :model-type - モデルタイプ（'bigram/'trigram）
154
155戻り値:
156 nskk-completion-context-result構造体のリスト（スコア順）&quot;
157 (unless (stringp query)
158 (error &quot;Query must be a string: %s&quot; query))
159
160 (let* ((context (plist-get args :context))
161 (ctx-model (plist-get args :model))
162 (limit (or (plist-get args :limit)
163 nskk-completion-context-default-limit))
164 (okuri-type (plist-get args :okuri-type))
165 (model-type (or (plist-get args :model-type)
166 nskk-completion-context-model-type))
167 (results nil)
168 (tables (cond
169 ((eq okuri-type 'okuri-ari)
170 (list (nskk-dict-index-okuri-ari-table index)))
171 ((eq okuri-type 'okuri-nasi)
172 (list (nskk-dict-index-okuri-nasi-table index)))
173 (t
174 (list (nskk-dict-index-okuri-nasi-table index)
175 (nskk-dict-index-okuri-ari-table index))))))
176
177 ;; クエリに前方一致する候補を収集
178 (dolist (table tables)
179 (maphash
180 (lambda (midashi entry)
181 (when (string-prefix-p query midashi)
182 (let* ((context-score (if (and ctx-model context)
183 (nskk-completion-context--calculate-score
184 ctx-model context midashi model-type)
185 0.5))
186 (freq (nskk-dict-entry-frequency entry))
187 (total-score (nskk-completion-context--combine-scores
188 context-score freq)))
189 (push (nskk-completion-context-result--create
190 :midashi midashi
191 :entry entry
192 :context-score context-score
193 :frequency freq
194 :total-score total-score)
195 results))))
196 table))
197
198 ;; 総合スコアでソート
199 (setq results (sort results
200 (lambda (a b)
201 (&gt; (nskk-completion-context-result-total-score a)
202 (nskk-completion-context-result-total-score b)))))
203
204 ;; limit適用
205 (when (and limit (&gt; (length results) limit))
206 (setq results (seq-take results limit)))
207
208 results))

212(defun nskk-completion-context--calculate-score (model context word model-type)
213 &quot;文脈スコアを計算する（内部関数）。
214
215引数:
216 MODEL - 文脈モデル
217 CONTEXT - 文脈（直前の単語のリスト）
218 WORD - 候補単語
219 MODEL-TYPE - モデルタイプ（'bigram/'trigram）
220
221戻り値:
222 文脈スコア（0.0-1.0の確率）&quot;
223 (pcase model-type
224 ('bigram
225 (nskk-completion-context--bigram-score model context word))
226 ('trigram
227 (nskk-completion-context--trigram-score model context word))
228 (_
229 (nskk-completion-context--bigram-score model context word))))

231(defun nskk-completion-context--bigram-score (model context word)
232 &quot;バイグラムスコアを計算する（内部関数）。
233
234引数:
235 MODEL - 文脈モデル
236 CONTEXT - 文脈（直前の単語のリスト）
237 WORD - 候補単語
238
239処理:
240 P(word|prev) = count(prev, word) / count(prev)
241 スムージングを適用
242
243戻り値:
244 バイグラムスコア&quot;
245 (if (null context)
246 0.5 ; 文脈なしの場合は中立スコア
247 (let* ((prev-word (car context))
248 (bigram-key (cons prev-word word))
249 (bigram-count (gethash bigram-key (nskk-completion-context-model-bigram-table model) 0))
250 (prev-count (gethash prev-word (nskk-completion-context-model-unigram-table model) 0)))
251 (if (zerop prev-count)
252 nskk-completion-context-smoothing-factor
253 ;; スムージング適用
254 (/ (+ bigram-count nskk-completion-context-smoothing-factor)
255 (+ prev-count 1.0))))))
256
257(defun nskk-completion-context--trigram-score (model context word)
258 &quot;トライグラムスコアを計算する（内部関数）。
259
260引数:
261 MODEL - 文脈モデル
262 CONTEXT - 文脈（直前の単語のリスト）
263 WORD - 候補単語
264
265処理:
266 P(word|prev2,prev1) = count(prev2, prev1, word) / count(prev2, prev1)
267 バイグラムスコアとのバックオフ
268
269戻り値:
270 トライグラムスコア&quot;
271 (if (&lt; (length context) 2)
272 ;; 文脈が不十分な場合はバイグラムにフォールバック
273 (nskk-completion-context--bigram-score model context word)
274 (let* ((prev1 (nth 0 context))
275 (prev2 (nth 1 context))
276 (trigram-key (cons prev2 (cons prev1 word)))
277 (bigram-key (cons prev1 word))
278 (trigram-count (gethash trigram-key (nskk-completion-context-model-trigram-table model) 0))
279 (bigram-count-context (gethash (cons prev2 prev1)
280 (nskk-completion-context-model-bigram-table model) 0)))
281 (if (zerop bigram-count-context)
282 ;; トライグラムデータがない場合はバイグラムにバックオフ
283 (nskk-completion-context--bigram-score model (list prev1) word)
284 ;; 補間法: トライグラム * 0.7 + バイグラム * 0.3
285 (let ((trigram-prob (/ (+ trigram-count nskk-completion-context-smoothing-factor)
286 (+ bigram-count-context 1.0)))
287 (bigram-prob (nskk-completion-context--bigram-score model (list prev1) word)))
288 (+ (* 0.7 trigram-prob) (* 0.3 bigram-prob)))))))

290(defun nskk-completion-context--combine-scores (context-score frequency)
291 &quot;文脈スコアと頻度を組み合わせて総合スコアを計算する（内部関数）。
292
293引数:
294 CONTEXT-SCORE - 文脈スコア（0.0-1.0）
295 FREQUENCY - 頻度
296
297処理:
298 総合スコア = context-score * 0.7 + frequency-score * 0.3
299
300戻り値:
301 総合スコア&quot;
302 (let ((freq-score (/ (float frequency) (+ frequency 10.0))))
303 (+ (* context-score 0.7) (* freq-score 0.3))))

308(defun nskk-completion-context-learn (model word-sequence)
309 &quot;文脈モデルを更新する（学習）。
310
311引数:
312 MODEL - 文脈モデル
313 WORD-SEQUENCE - 単語シーケンス（リスト）
314
315処理:
316 1. ユニグラム、バイグラム、トライグラムを抽出
317 2. 各カウントをインクリメント
318
319戻り値:
320 更新された文脈モデル&quot;
321 (unless (listp word-sequence)
322 (error &quot;Word sequence must be a list: %s&quot; word-sequence))
323
324 (let ((seq-len (length word-sequence)))
325 ;; ユニグラム
326 (dolist (word word-sequence)
327 (let ((count (gethash word (nskk-completion-context-model-unigram-table model) 0)))
328 (puthash word (1+ count) (nskk-completion-context-model-unigram-table model))))
329
330 ;; バイグラム
331 (when (&gt;= seq-len 2)
332 (dotimes (i (1- seq-len))
333 (let* ((word1 (nth i word-sequence))
334 (word2 (nth (1+ i) word-sequence))
335 (key (cons word1 word2))
336 (count (gethash key (nskk-completion-context-model-bigram-table model) 0)))
337 (puthash key (1+ count) (nskk-completion-context-model-bigram-table model)))))
338
339 ;; トライグラム
340 (when (&gt;= seq-len 3)
341 (dotimes (i (- seq-len 2))
342 (let* ((word1 (nth i word-sequence))
343 (word2 (nth (1+ i) word-sequence))
344 (word3 (nth (+ i 2) word-sequence))
345 (key (cons word1 (cons word2 word3)))
346 (count (gethash key (nskk-completion-context-model-trigram-table model) 0)))
347 (puthash key (1+ count) (nskk-completion-context-model-trigram-table model)))))
348
349 ;; 総カウント更新
350 (cl-incf (nskk-completion-context-model-total-count model))
351
352 model))

375(defun nskk-completion-context--tokenize (text)
376 &quot;テキストを単語に分割する（内部関数）。
377
378引数:
379 TEXT - 分析対象テキスト
380
381処理:
382 簡易的な分かち書き（スペース、句読点で分割）
383
384戻り値:
385 単語のリスト&quot;
386 (let ((words nil)
387 (current-word &quot;&quot;))
388 (dotimes (i (length text))
389 (let ((char (aref text i)))
390 (cond
391 ;; 区切り文字
392 ((or (= char ?\s) (= char ?\t) (= char ?\n)
393 (= char ?、) (= char ?。) (= char ?！) (= char ?？))
394 (when (&gt; (length current-word) 0)
395 (push current-word words)
396 (setq current-word &quot;&quot;)))
397 ;; 通常の文字
398 (t
399 (setq current-word (concat current-word (char-to-string char)))))))
400 ;; 最後の単語
401 (when (&gt; (length current-word) 0)
402 (push current-word words))
403 (nreverse words)))
