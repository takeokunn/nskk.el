65(defun nskk-input-custom-define (name table &rest options)
66 &quot;カスタム入力方式を定義する。
67
68NAME は入力方式の名前（シンボル）。
69TABLE は変換テーブル（連想リスト）。
70OPTIONS はキーワード引数:
71 :base SYMBOL - ベースとなる入力方式
72 :description STRING - 説明文字列
73 :priority NUMBER - 優先度（デフォルト: 100）
74
75使用例:
76 (nskk-input-custom-define
77 'my-azik
78 '((\&quot;dh\&quot; . \&quot;で\&quot;) (\&quot;kk\&quot; . \&quot;っか\&quot;))
79 :base 'azik
80 :description \&quot;My AZIK extension\&quot;
81 :priority 150)&quot;
82 (let* ((base (plist-get options :base))
83 (description (plist-get options :description))
84 (priority (or (plist-get options :priority) 100))
85 (method (nskk-input-custom-method-create
86 :name name
87 :table table
88 :base base
89 :description description
90 :priority priority)))

92 ;; ハッシュテーブルの初期化
93 (nskk-input-custom--init-hash-table method)
94
95 ;; 既存の定義を削除
96 (setq nskk-input-custom-methods
97 (cl-remove-if (lambda (m)
98 (eq (nskk-input-custom-method-name m) name))
99 nskk-input-custom-methods))
100
101 ;; 新しい定義を追加
102 (push method nskk-input-custom-methods)
103
104 ;; 優先度順にソート
105 (setq nskk-input-custom-methods
106 (cl-sort nskk-input-custom-methods #'&gt;
107 :key #'nskk-input-custom-method-priority))
108
109 method))

113(defun nskk-input-custom--init-hash-table (method)
114 &quot;METHODのハッシュテーブルを初期化する。&quot;
115 (let ((hash-table (make-hash-table :test 'equal
116 :size (max 100 (* 2 (length (nskk-input-custom-method-table method)))))))
117 ;; ベース入力方式がある場合、その変換テーブルを含める
118 (when (nskk-input-custom-method-base method)
119 (let ((base-table (nskk-input-custom--get-base-table
120 (nskk-input-custom-method-base method))))
121 (dolist (entry base-table)
122 (puthash (car entry) (cdr entry) hash-table))))
123
124 ;; カスタムテーブルを追加（上書き）
125 (dolist (entry (nskk-input-custom-method-table method))
126 (puthash (car entry) (cdr entry) hash-table))
127
128 (setf (nskk-input-custom-method-hash-table method) hash-table)))

130(defun nskk-input-custom--get-base-table (base-name)
131 &quot;BASE-NAMEの変換テーブルを取得する。&quot;
132 (cond
133 ((eq base-name 'azik)
134 (require 'nskk-input-azik)
135 nskk-input-azik-table)
136 ((eq base-name 'act)
137 (require 'nskk-input-act)
138 nskk-input-act-table)
139 ((eq base-name 'tutcode)
140 (require 'nskk-input-tutcode)
141 nskk-input-tutcode-table)
142 ((eq base-name 'qwerty)
143 (require 'nskk-romaji-tables)
144 nskk-romaji-table)
145 (t
146 (warn &quot;Unknown base input method: %s&quot; base-name)
147 nil)))

165(defun nskk-input-custom-get-candidates (name prefix)
166 &quot;NAMEのカスタム入力方式でPREFIXで始まる候補を取得する。&quot;
167 (let ((method (nskk-input-custom-get-method name)))
168 (when method
169 (let ((candidates nil)
170 (hash-table (nskk-input-custom-method-hash-table method)))
171 (maphash (lambda (key _value)
172 (when (string-prefix-p prefix key)
173 (push key candidates)))
174 hash-table)
175 (nreverse candidates)))))
