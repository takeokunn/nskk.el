128(defun nskk-qa-run-all-tests ()
129 &quot;すべてのテストを実行する。&quot;
130 (interactive)
131 (nskk-qa--log &quot;Running all tests...&quot;)
132 (let ((start-time (float-time))
133 (results '()))
134 ;; 各レイヤーのテストを実行
135 (push (nskk-qa--run-layer-tests 'presentation) results)
136 (push (nskk-qa--run-layer-tests 'extension) results)
137 (push (nskk-qa--run-layer-tests 'application) results)
138 (push (nskk-qa--run-layer-tests 'core) results)
139 (push (nskk-qa--run-layer-tests 'data) results)
140 (push (nskk-qa--run-layer-tests 'infrastructure) results)
141
142 (let* ((end-time (float-time))
143 (duration (- end-time start-time))
144 (total-passed (apply #'+ (mapcar (lambda (r) (plist-get r :passed)) results)))
145 (total-failed (apply #'+ (mapcar (lambda (r) (plist-get r :failed)) results))))
146 (nskk-qa--update-metrics :tests-run (+ total-passed total-failed)
147 :tests-passed total-passed
148 :tests-failed total-failed)
149 (message &quot;Tests completed in %.2fs: %d passed, %d failed&quot;
150 duration total-passed total-failed)
151 (list :duration duration
152 :passed total-passed
153 :failed total-failed
154 :results results))))

198(defun nskk-qa--benchmark-function (function description)
199 &quot;関数のベンチマークを実行する。
200FUNCTIONはベンチマーク対象関数、DESCRIPTIONは説明。&quot;
201 (let ((start-time (float-time))
202 (iterations nskk-qa-benchmark-iterations))
203 (dotimes (_ iterations)
204 (funcall function))
205 (let* ((end-time (float-time))
206 (total-time (- end-time start-time))
207 (avg-time (/ total-time iterations)))
208 (puthash description
209 (list :total-time total-time
210 :avg-time avg-time
211 :iterations iterations)
212 nskk-qa--benchmark-results)
213 (list :description description
214 :total-time total-time
215 :avg-time avg-time
216 :iterations iterations))))

269(defun nskk-qa-verify-performance ()
270 &quot;パフォーマンス目標を検証する。&quot;
271 (interactive)
272 (nskk-qa--log &quot;Verifying performance targets...&quot;)
273 (let ((results '())
274 (all-passed t))
275 ;; ローマ字変換 &lt; 0.1ms
276 (let ((result (nskk-qa--verify-performance-target
277 #'nskk-qa--dummy-conversion
278 0.0001
279 &quot;Romaji conversion&quot;)))
280 (push result results)
281 (unless (plist-get result :passed)
282 (setq all-passed nil)))
283
284 ;; 辞書検索 &lt; 10ms
285 (let ((result (nskk-qa--verify-performance-target
286 #'nskk-qa--dummy-search
287 0.01
288 &quot;Dictionary search&quot;)))
289 (push result results)
290 (unless (plist-get result :passed)
291 (setq all-passed nil)))
292
293 ;; 結果表示
294 (nskk-qa--display-performance-results results all-passed)
295 results))

297(defun nskk-qa--verify-performance-target (function target description)
298 &quot;パフォーマンス目標を検証する。
299FUNCTIONは検証対象関数、TARGETは目標時間（秒）、
300DESCRIPTIONは説明。&quot;
301 (let ((start-time (float-time)))
302 (funcall function)
303 (let* ((end-time (float-time))
304 (duration (- end-time start-time))
305 (passed (&lt;= duration target)))
306 (list :description description
307 :target target
308 :actual duration
309 :passed passed))))

391(defun nskk-qa--update-metrics (&rest metrics)
392 &quot;メトリクスを更新する。
393METRICSは更新するメトリクスのplist。&quot;
394 (let ((plist metrics))
395 (while plist
