27;; このファイルはSKK辞書の効率的な検索・管理のためのデータ構造を実装します。
28;;
29;; 特徴:
30;; - cl-defstructによる型安全なデータ構造
31;; - ハッシュテーブルによる O(1) 平均検索
32;; - 前方一致検索のための最適化インデックス
33;; - メモリ効率を考慮した設計
34;; - Task 1.17 (トライ木) への移行を考慮した拡張性
35;;
36;; パフォーマンス目標:
37;; - エントリ検索: O(1) 平均（ハッシュテーブル）
38;; - 前方一致検索: O(k + m) (k=prefix長, m=結果数)
39;; - メモリ使用: 10万エントリで &lt; 50MB
40;; - 構造変換: 10万エントリで &lt; 500ms
41;;
42;; 使用例:
43;;
44;; (require 'nskk-dict-struct)
45;; (require 'nskk-dict-parser)
46;;
47;; ;; パーサー出力を最適化された構造に変換
48;; (let* ((parsed (nskk-parse-dictionary &quot;~/.skk/jisyo&quot;))
49;; (index (nskk-dict-struct-from-parser parsed)))
50;; ;; 基本検索
51;; (nskk-dict-struct-lookup index &quot;かんじ&quot; 'okuri-nasi)
52;; ;; =&gt; nskk-dict-entry構造体
53;;
54;; ;; 前方一致検索
55;; (nskk-dict-struct-prefix-search index &quot;かん&quot; 10)
56;; ;; =&gt; エントリのリスト
57;; )

280;;;###autoload
281(defun nskk-dict-struct-from-parser (parsed-dict)
282 &quot;パーサー出力（nskk-dict）を最適化されたインデックス構造に変換する。
283
284引数:
285 PARSED-DICT - nskk-parse-dictionary の戻り値（nskk-dict構造体）
286
287戻り値:
288 nskk-dict-index構造体
289
290パフォーマンス目標:
291 10万エントリで &lt; 500ms
292
293変更点（Task 1.18）:
294 - トライ木を構築して高速前方一致検索をサポート
295 - 既存のprefix-indexも後方互換性のため保持&quot;
296 (let* ((start-time (float-time))
297 (okuri-ari-table (make-hash-table :test 'equal :size 10000))
298 (okuri-nasi-table (make-hash-table :test 'equal :size 100000))
299 ;; トライ木の作成（Task 1.18）
300 (trie-ari (nskk-trie-create))
301 (trie-nasi (nskk-trie-create))
302 ;; 既存のprefix-indexも保持（後方互換性）
303 (prefix-ari (nskk-dict-prefix-index--create
304 :char-map (make-hash-table :test 'equal :size 1000)
305 :depth nskk-dict-struct-prefix-index-depth))
306 (prefix-nasi (nskk-dict-prefix-index--create
307 :char-map (make-hash-table :test 'equal :size 10000)
308 :depth nskk-dict-struct-prefix-index-depth))
309 (stats (when nskk-dict-struct-enable-statistics
310 (nskk-dict-statistics--create)))
311 (total-candidates 0))
312
313 ;; 送り仮名ありエントリを変換
314 ;; トライ木とプレフィックスインデックスを同時に構築
315 (let ((ari-entries nil))
316 (dolist (parsed-entry (nskk-dict-okuri-ari parsed-dict))
317 (let* ((midashi (nskk-dict-entry-midashi parsed-entry))
318 (candidates (nskk-dict-struct--convert-candidates
319 (nskk-dict-entry-candidates parsed-entry)))
320 (entry (nskk-dict-entry--create
321 :midashi midashi
322 :candidates candidates
323 :okuri-type 'okuri-ari)))
324 (puthash midashi entry okuri-ari-table)
325 (push (cons midashi entry) ari-entries)
326 ;; トライ木に挿入（Task 1.18）
327 (nskk-trie-insert trie-ari midashi entry)
328 (when stats
329 (setq total-candidates (+ total-candidates (length candidates))))))
330 ;; プレフィックスインデックス一括構築（後方互換性）
331 (dolist (entry-pair ari-entries)
332 (nskk-dict-struct--add-to-prefix-index prefix-ari (car entry-pair) (cdr entry-pair))))
333
334 ;; 送り仮名なしエントリを変換
335 (let ((nasi-entries nil))
336 (dolist (parsed-entry (nskk-dict-okuri-nasi parsed-dict))
337 (let* ((midashi (nskk-dict-entry-midashi parsed-entry))
338 (candidates (nskk-dict-struct--convert-candidates
339 (nskk-dict-entry-candidates parsed-entry)))
340 (entry (nskk-dict-entry--create
341 :midashi midashi
342 :candidates candidates
343 :okuri-type 'okuri-nasi)))
344 (puthash midashi entry okuri-nasi-table)
345 (push (cons midashi entry) nasi-entries)
346 ;; トライ木に挿入（Task 1.18）
347 (nskk-trie-insert trie-nasi midashi entry)
348 (when stats
349 (setq total-candidates (+ total-candidates (length candidates))))))
350 ;; プレフィックスインデックス一括構築（後方互換性）
351 (dolist (entry-pair nasi-entries)
352 (nskk-dict-struct--add-to-prefix-index prefix-nasi (car entry-pair) (cdr entry-pair))))
353
354 ;; メタデータ構築
355 (let* ((metadata (nskk-dict-metadata--create
356 :file-path (nskk-dict-file-path parsed-dict)
357 :encoding (nskk-dict-encoding parsed-dict)
358 :entry-count (+ (hash-table-count okuri-ari-table)
359 (hash-table-count okuri-nasi-table))
360 :last-modified (float-time)))
361 (build-time (- (float-time) start-time)))
362
363 ;; 統計情報を更新
364 (when stats
365 (setf (nskk-dict-statistics-okuri-ari-count stats)
366 (hash-table-count okuri-ari-table))
367 (setf (nskk-dict-statistics-okuri-nasi-count stats)
368 (hash-table-count okuri-nasi-table))
369 (setf (nskk-dict-statistics-total-candidates stats)
370 total-candidates)
371 (setf (nskk-dict-statistics-avg-candidates stats)
372 (if (&gt; (nskk-dict-metadata-entry-count metadata) 0)
373 (/ (float total-candidates)
374 (nskk-dict-metadata-entry-count metadata))
375 0.0))
376 (setf (nskk-dict-statistics-build-time stats) build-time)
377 (setf (nskk-dict-statistics-memory-usage stats)
378 (nskk-dict-struct--estimate-memory-usage
379 okuri-ari-table okuri-nasi-table)))
380
381 ;; インデックス構造を返す（トライ木を追加）
382 (nskk-dict-index--create
383 :okuri-ari-table okuri-ari-table
384 :okuri-nasi-table okuri-nasi-table
385 :trie-ari trie-ari
386 :trie-nasi trie-nasi
387 :prefix-index-ari prefix-ari
388 :prefix-index-nasi prefix-nasi
389 :metadata metadata
390 :statistics stats))))

407(defun nskk-dict-struct--add-to-prefix-index (prefix-index midashi entry)
408 &quot;前方一致インデックスにエントリを追加する。
409
410引数:
411 PREFIX-INDEX - nskk-dict-prefix-index構造体
412 MIDASHI - 見出し語
413 ENTRY - nskk-dict-entry構造体
414
415深度分のすべてのプレフィックスに対してエントリを登録する。
416例: 深度2で「かんじ」なら、「か」と「かん」の両方に登録。&quot;
417 (let* ((depth (nskk-dict-prefix-index-depth prefix-index))
418 (char-map (nskk-dict-prefix-index-char-map prefix-index))
419 (midashi-len (length midashi)))
420 ;; 1文字から深度までのすべてのプレフィックスに登録
421 (dotimes (i (min depth midashi-len))
422 (let* ((prefix-len (1+ i))
423 (prefix (substring midashi 0 prefix-len))
424 (entries (gethash prefix char-map)))
425 (puthash prefix (cons entry entries) char-map)))))

427(defun nskk-dict-struct--estimate-memory-usage (okuri-ari-table okuri-nasi-table)
428 &quot;辞書インデックスの推定メモリ使用量を計算する（バイト）。
429
430引数:
431 OKURI-ARI-TABLE - 送り仮名ありハッシュテーブル
432 OKURI-NASI-TABLE - 送り仮名なしハッシュテーブル
433
434戻り値:
435 推定メモリ使用量（バイト）
436
437注: これは概算値です。実際のメモリ使用量はEmacsの内部実装に依存します。&quot;
438 (let ((total 0))
439 ;; ハッシュテーブルオーバーヘッド（概算）
440 (setq total (+ total
441 (* (hash-table-count okuri-ari-table) 32)
442 (* (hash-table-count okuri-nasi-table) 32)))
443
444 ;; エントリとキーのサイズ
445 (maphash (lambda (key value)
446 (setq total (+ total
447 (* (length key) 4) ; 文字列キー（UTF-8想定）
448 200))) ; エントリ構造体のオーバーヘッド
449 okuri-ari-table)
450 (maphash (lambda (key value)
451 (setq total (+ total
452 (* (length key) 4)
453 200)))
454 okuri-nasi-table)
455
456 total))

462 &quot;インデックスから見出し語を検索する。
463
464引数:
465 INDEX - nskk-dict-index構造体
466 MIDASHI - 見出し語
467 OKURI-TYPE - 'okuri-ari または 'okuri-nasi（省略時は両方検索）

469戻り値:
470 nskk-dict-entry構造体、見つからない場合はnil
471
472パフォーマンス:
473 O(1) 平均（ハッシュテーブル検索）&quot;
474 (cond
475 ((eq okuri-type 'okuri-ari)
476 (gethash midashi (nskk-dict-index-okuri-ari-table index)))
477 ((eq okuri-type 'okuri-nasi)
478 (gethash midashi (nskk-dict-index-okuri-nasi-table index)))
479 (t
480 ;; okuri-typeが指定されていない場合、送り仮名なしを優先して検索

482 (gethash midashi (nskk-dict-index-okuri-ari-table index))))))

484;;;###autoload
485(defun nskk-dict-struct-prefix-search (index prefix &optional limit okuri-type)
486 &quot;前方一致検索を実行する。
487
488引数:
489 INDEX - nskk-dict-index構造体
490 PREFIX - 検索プレフィックス（見出し語の先頭部分）
491 LIMIT - 最大結果数（省略時は無制限）
492 OKURI-TYPE - 'okuri-ari または 'okuri-nasi（省略時は両方検索）
493
494戻り値:
495 nskk-dict-entry構造体のリスト
496
497パフォーマンス:
498 O(k + m) (k=prefix長, m=結果数)&quot;
499 (let ((results nil)
500 (count 0)
501 (tables (cond
502 ((eq okuri-type 'okuri-ari)
503 (list (cons (nskk-dict-index-okuri-ari-table index)
504 (nskk-dict-index-prefix-index-ari index))))
505 ((eq okuri-type 'okuri-nasi)
506 (list (cons (nskk-dict-index-okuri-nasi-table index)
507 (nskk-dict-index-prefix-index-nasi index))))
508 (t
509 (list (cons (nskk-dict-index-okuri-nasi-table index)
510 (nskk-dict-index-prefix-index-nasi index))
511 (cons (nskk-dict-index-okuri-ari-table index)
512 (nskk-dict-index-prefix-index-ari index)))))))
513
514 (catch 'done
515 (dolist (table-pair tables)
516 (let* ((table (car table-pair))
517 (prefix-idx (cdr table-pair))
518 (depth (nskk-dict-prefix-index-depth prefix-idx))
519 (prefix-key (substring prefix 0 (min depth (length prefix)))))
520
521 ;; 前方一致インデックスを使用して候補を絞り込む
522 (let ((candidate-entries (gethash prefix-key
523 (nskk-dict-prefix-index-char-map prefix-idx))))
524 (dolist (entry candidate-entries)
525 (when (and entry
526 (string-prefix-p prefix (nskk-dict-entry-midashi entry)))
527 (push entry results)
528 (setq count (1+ count))
529 (when (and limit (&gt;= count limit))
530 (throw 'done nil))))))))
531
532 (nreverse results)))

604(defun nskk-dict-struct-validate-index (index)
605 &quot;辞書インデックスの整合性を検証する。
606
607引数:
608 INDEX - nskk-dict-index構造体
609
610戻り値:
611 問題がなければt、問題があればエラーメッセージのリスト&quot;
612 (let ((errors nil))
613 ;; 基本構造チェック
614 (unless (nskk-dict-index-p index)
615 (push &quot;Invalid index structure&quot; errors))
616
617 (unless (hash-table-p (nskk-dict-index-okuri-ari-table index))
618 (push &quot;Invalid okuri-ari table&quot; errors))
619
620 (unless (hash-table-p (nskk-dict-index-okuri-nasi-table index))
621 (push &quot;Invalid okuri-nasi table&quot; errors))
622
623 ;; エントリ整合性チェック
624 (maphash (lambda (key entry)
625 (unless (equal key (nskk-dict-entry-midashi entry))
626 (push (format &quot;Mismatched key-midashi: %s != %s&quot;
627 key (nskk-dict-entry-midashi entry))
628 errors)))
629 (nskk-dict-index-okuri-ari-table index))
630
631 (maphash (lambda (key entry)
632 (unless (equal key (nskk-dict-entry-midashi entry))
633 (push (format &quot;Mismatched key-midashi: %s != %s&quot;
634 key (nskk-dict-entry-midashi entry))
635 errors)))
636 (nskk-dict-index-okuri-nasi-table index))
637
638 (if errors
639 (nreverse errors)
640 t)))
