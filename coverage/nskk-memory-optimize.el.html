92(defun nskk-memory-pool-get-string (size)
93 &quot;文字列プールから指定サイズの文字列を取得する。
94
95引数:
96 SIZE - 文字列のサイズ
97
98返り値:
99 文字列（再利用可能）&quot;
100 (if (not nskk-memory-enable-pooling)
101 (make-string size ?\ )
102 (let* ((pool-entry (assoc size nskk-memory--string-pool))
103 (pool (cdr pool-entry)))
104 (if (and pool (car pool))
105 (let ((str (pop (cdr pool-entry))))
106 (fillarray str ?\ )
107 str)
108 (make-string size ?\ )))))
109
110(defun nskk-memory-pool-return-string (str)
111 &quot;文字列をプールに返却する。
112
113引数:
114 STR - 返却する文字列&quot;
115 (when nskk-memory-enable-pooling
116 (let* ((size (length str))
117 (pool-entry (assoc size nskk-memory--string-pool))
118 (pool (cdr pool-entry)))
119 (when (or (null pool)
120 (&lt; (length pool) nskk-memory--string-pool-max-size))
121 (if pool-entry
122 (push str (cdr pool-entry))
123 (push (cons size (list str)) nskk-memory--string-pool))))))

276(defun nskk-memory-usage ()
277 &quot;現在のNSKK関連のメモリ使用量を推定する。
278
279返り値:
280 plist
281 :string-pool-bytes - 文字列プールのメモリ使用量
282 :cons-pool-bytes - consプールのメモリ使用量
283 :vector-pool-bytes - ベクトルプールのメモリ使用量
284 :interned-bytes - インターン文字列のメモリ使用量
285 :total-bytes - 合計メモリ使用量
286 :gc-count - GC実行回数
287 :gc-elapsed - GC累積時間&quot;
288 (let ((string-pool-bytes
289 (cl-loop for (_size . strings) in nskk-memory--string-pool
290 sum (* (length strings)
291 (or (ignore-errors (length (car strings))) 0))))
292 (cons-pool-bytes
293 (* (length nskk-memory--cons-pool) 16)) ; consセル = 16バイト（推定）
294 (vector-pool-bytes
295 (cl-loop for (_size . vectors) in nskk-memory--vector-pool
296 sum (* (length vectors)
297 (or (ignore-errors (* (length (car vectors)) 8)) 0))))
298 (interned-bytes
299 (let ((total 0))
300 (maphash (lambda (_k v)
301 (cl-incf total (length v)))
302 nskk-memory--interned-strings)
303 total)))
304 (list :string-pool-bytes string-pool-bytes
305 :cons-pool-bytes cons-pool-bytes
306 :vector-pool-bytes vector-pool-bytes
307 :interned-bytes interned-bytes
308 :total-bytes (+ string-pool-bytes cons-pool-bytes
309 vector-pool-bytes interned-bytes)
310 :gc-count gcs-done
311 :gc-elapsed gc-elapsed)))

390(defun nskk-memory-check-pressure ()
391 &quot;メモリプレッシャーをチェックする。
392
393返り値:
394 plist
395 :pressure - メモリプレッシャー（high/medium/low）
396 :usage - 現在の使用量
397 :threshold - 閾値&quot;
398 (let* ((usage-info (nskk-memory-usage))
399 (total-bytes (plist-get usage-info :total-bytes))
400 (pressure (cond
401 ((&gt; total-bytes nskk-memory--pressure-threshold) 'high)
402 ((&gt; total-bytes (/ nskk-memory--pressure-threshold 2)) 'medium)
403 (t 'low))))
404 (list :pressure pressure
405 :usage total-bytes
406 :threshold nskk-memory--pressure-threshold)))
