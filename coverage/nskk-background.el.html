163;;;###autoload
164(defun nskk-background-start ()
165 &quot;バックグラウンドタスクを開始する。&quot;
166 (interactive)
167 (when nskk-background-enable
168 (when nskk-background-verbose
169 (message &quot;Starting background tasks&quot;))
170
171 ;; デフォルトタスクを登録
172 (nskk-background--register-default-tasks)
173
174 ;; アイドルタイマー開始
175 (unless nskk-background--idle-timer
176 (setq nskk-background--idle-timer
177 (run-with-idle-timer nskk-background-idle-time t
178 #'nskk-background--on-idle)))
179
180 ;; リソース監視開始
181 (nskk-background--start-resource-monitor)))
182
183;;;###autoload
184(defun nskk-background-stop ()
185 &quot;バックグラウンドタスクを停止する。&quot;
186 (interactive)
187 (when nskk-background-verbose
188 (message &quot;Stopping background tasks&quot;))
189
190 ;; すべてのタスクタイマーをキャンセル
191 (dolist (task nskk-background--tasks)
192 (when (nskk-background-task-timer task)
193 (cancel-timer (nskk-background-task-timer task))))
194
195 ;; アイドルタイマーキャンセル
196 (when nskk-background--idle-timer
197 (cancel-timer nskk-background--idle-timer)
198 (setq nskk-background--idle-timer nil))
199
200 ;; リソース監視停止
201 (nskk-background--stop-resource-monitor)
202
203 ;; タスクリストクリア
204 (setq nskk-background--tasks nil))

291 (not (nskk-background-task-timer task)))

294(defun nskk-background--schedule-task (task)
295 &quot;タスクをスケジュールする。
296
297引数:
298 TASK - nskk-background-task構造体&quot;
299 (let ((interval (nskk-background-task-interval task)))
300
301 (when nskk-background-verbose
302 (message &quot;Scheduling task: %s (interval: %ds)&quot;
303 (nskk-background-task-name task)
304 interval))
305
306 (setf (nskk-background-task-timer task)
307 (run-at-time interval interval
308 (lambda ()
309 (nskk-background--execute-task task))))))
310
311(defun nskk-background--execute-task (task)
312 &quot;タスクを実行する。
313
314引数:
315 TASK - nskk-background-task構造体&quot;
316 (when (and (nskk-background-task-enabled task)
317 (not nskk-background--suspended))
318
319 (when nskk-background-verbose
320 (message &quot;Executing background task: %s&quot;
321 (nskk-background-task-name task)))
322
323 (let ((start-time (float-time)))
324
325 (condition-case err
326 (funcall (nskk-background-task-function task))
327 (error
328 (message &quot;Background task error [%s]: %s&quot;
329 (nskk-background-task-name task)
330 (error-message-string err))))
331
332 ;; 統計更新
333 (let ((elapsed (- (float-time) start-time)))
334 (setf (nskk-background-task-last-run task) (float-time))
335 (cl-incf (nskk-background-task-run-count task))
336 (cl-incf (nskk-background-task-total-time task) elapsed)
337
338 (when nskk-background-verbose
339 (message &quot;Background task completed: %s (%.3fs)&quot;
340 (nskk-background-task-name task)
341 elapsed))))))

393(defun nskk-background--check-resources ()
394 &quot;リソース使用量をチェックし、必要に応じてタスクを一時停止する。&quot;
395 (let ((stats (nskk-background--collect-stats)))
396
397 ;; CPU使用率チェック
398 (when (&gt; (nskk-background-stats-cpu-usage stats)
399 nskk-background-max-cpu-usage)
400 (nskk-background--suspend-tasks
401 (format &quot;High CPU usage: %.1f%%&quot;
402 (nskk-background-stats-cpu-usage stats))))
403
404 ;; メモリ使用量チェック
405 (when (&gt; (nskk-background-stats-memory-usage stats)
406 nskk-background-max-memory-usage)
407 (nskk-background--suspend-tasks
408 (format &quot;High memory usage: %dMB&quot;
409 (nskk-background-stats-memory-usage stats))))))

433(defun nskk-background--collect-stats ()
434 &quot;現在のリソース統計を収集する。
435
436戻り値:
437 nskk-background-stats構造体&quot;
438 (let* (;; メモリ使用量を簡易的に計算（MB単位）
439 ;; memory-use-countsはリストを返すので、最初の要素を使用
440 (memory-counts (memory-use-counts))
441 (memory-usage (/ (if (listp memory-counts)
442 (car memory-counts)
443 memory-counts)
444 1048576))
445 ;; CPU使用率の推定（簡易版）
446 (cpu-usage 0.0) ; 実際にはより詳細な計算が必要
447 ;; GC回数（gcs-doneはリストの場合がある）
448 (gc-count (cond
449 ((numberp gcs-done) gcs-done)
450 ((listp gcs-done) (if (numberp (car gcs-done))
451 (car gcs-done)
452 0))
453 (t 0))))
454
455 (nskk-background-stats--create
456 :cpu-usage cpu-usage
457 :memory-usage memory-usage
458 :gc-count gc-count
459 :timestamp (float-time))))

464(defun nskk-background-task-stats ()
465 &quot;すべてのタスクの統計情報を取得する。
466
467戻り値:
468 タスク統計のリスト（plist形式）

471 (let* ((run-count (nskk-background-task-run-count task))

485(defun nskk-background-resource-stats ()
486 &quot;リソース統計の概要を取得する。
487
488戻り値:
489 plist形式の統計情報
490 (:current STATS :avg-cpu N :avg-memory M :max-cpu N :max-memory M)&quot;

497 :avg-memory 0
498 :max-cpu 0.0

510 :key #'nskk-background-stats-memory-usage)))
