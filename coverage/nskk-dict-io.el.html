209;;;###autoload
210(defun nskk-load-dictionary (path &optional force-reload)
211 &quot;辞書ファイルPATHを読み込み、インデックス構造を返す。
212
213引数:
214 PATH - 辞書ファイルパス
215 FORCE-RELOAD - 非nilの場合、キャッシュを無視して再読み込み
216
217処理フロー:
218 1. ファイル存在確認
219 2. メタデータ確認（チェックサム、最終更新時刻）
220 3. キャッシュ確認（変更なければキャッシュ返却）
221 4. パーサーで読み込み (nskk-parse-dictionary)
222 5. データ構造に変換 (nskk-dict-struct-from-parser)
223 6. キャッシュに保存
224 7. インデックス構造を返す
225
226戻り値:
227 nskk-dict-index 構造体
228
229エラー:
230 ファイルが存在しない場合、エラーをシグナル&quot;
231 (unless (file-exists-p path)
232 (signal 'nskk-dict-io-file-not-found (list path)))
233
234 (unless (file-readable-p path)
235 (signal 'nskk-dict-io-permission-denied (list path)))
236
237 (let* ((cache-key (nskk-dict-io--cache-key path))
238 (cached-entry (gethash cache-key nskk-dict-io--cache))
239 (use-cache (and nskk-dict-io-cache-enabled
240 (not force-reload)
241 cached-entry
242 (nskk-dict-io--cache-valid-p path (cdr cached-entry)))))
243
244 (if use-cache
245 (progn
246 (when nskk-dict-io-verbose
247 (message &quot;Using cached dictionary: %s&quot; path))
248 (car cached-entry))
249
250 ;; キャッシュが無効なので新規読み込み
251 (when nskk-dict-io-verbose
252 (message &quot;Loading dictionary: %s&quot; path))
253
254 (let* ((start-time (float-time))
255 ;; パーサーで読み込み
256 (parsed (nskk-parse-dictionary path))
257 ;; データ構造に変換
258 (index (nskk-dict-struct-from-parser parsed))
259 ;; メタデータ作成
260 (checksum (nskk-dict-io--compute-checksum path))
261 (mtime (nskk-dict-io--get-file-mtime path))
262 (cache-metadata (nskk-dict-io-cache-metadata--create
263 :checksum checksum
264 :mtime mtime
265 :cached-time (float-time)))
266 (load-time (- (float-time) start-time)))
267
268 (when nskk-dict-io-verbose
269 (message &quot;Dictionary loaded in %.3f sec: %s&quot; load-time path))
270
271 ;; キャッシュに保存
272 (when nskk-dict-io-cache-enabled
273 (puthash cache-key (cons index cache-metadata) nskk-dict-io--cache))
274
275 index))))

338;;;###autoload
339(defun nskk-save-dictionary (index output-path &optional encoding)
340 &quot;インデックス構造INDEXを辞書ファイルOUTPUT-PATHに書き込む。
341
342引数:
343 INDEX - nskk-dict-index構造体
344 OUTPUT-PATH - 出力ファイルパス
345 ENCODING - エンコーディング（未指定の場合はnskk-dict-io-default-encodingを使用）
346
347処理フロー:
348 1. バックアップ作成（既存ファイルがあれば）
349 2. 一時ファイルに書き込み
350 3. エンコーディング指定
351 4. SKK辞書形式でシリアライズ
352 5. アトミックリネーム（一時ファイル→本ファイル）
353 6. パーミッション保持
354
355戻り値:
356 OUTPUT-PATH
357
358エラー:
359 書き込み権限がない場合、エラーをシグナル&quot;
360 (let* ((output-path (expand-file-name output-path))
361 (encoding (or encoding nskk-dict-io-default-encoding))
362 (file-exists (file-exists-p output-path)))
363
364 ;; ディレクトリの存在確認
365 (let ((output-dir (file-name-directory output-path)))
366 (unless (file-directory-p output-dir)
367 (signal 'nskk-dict-io-error (list (format &quot;Output directory does not exist: %s&quot; output-dir)))))
368
369 ;; 書き込み権限確認
370 (when (and file-exists (not (file-writable-p output-path)))
371 (signal 'nskk-dict-io-permission-denied (list output-path)))
372
373 ;; バックアップ作成
374 (when (and nskk-dict-io-backup-enabled file-exists)
375 (nskk-backup-dictionary output-path))
376
377 (when nskk-dict-io-verbose
378 (message &quot;Saving dictionary: %s&quot; output-path))
379
380 (let* ((start-time (float-time))
381 ;; 一時ファイル作成（同じディレクトリに作成してアトミックリネームを保証）
382 (temp-file (let ((default-directory (file-name-directory output-path)))
383 (make-temp-file
384 (concat (file-name-nondirectory output-path) &quot;.&quot;)
385 nil
386 &quot;.tmp&quot;)))
387 ;; 既存ファイルのパーミッション保存
388 (original-modes (when file-exists
389 (file-modes output-path))))
390
391 (unwind-protect
392 (progn
393 ;; 一時ファイルに書き込み
394 (with-temp-buffer
395 (let ((coding-system-for-write encoding))
396 ;; ヘッダー書き込み
397 (nskk-dict-io--write-header encoding)
398 (insert &quot;\n&quot;)
399
400 ;; 送り仮名ありエントリ
401 (insert &quot;;; okuri-ari entries.\n&quot;)
402 (let ((okuri-ari-entries (nskk-dict-io--collect-all-entries index 'okuri-ari)))
403 (dolist (entry okuri-ari-entries)
404 (insert (nskk-dict-io--serialize-entry entry))
405 (insert &quot;\n&quot;)))
406 (insert &quot;\n&quot;)
407
408 ;; 送り仮名なしエントリ
409 (insert &quot;;; okuri-nasi entries.\n&quot;)
410 (let ((okuri-nasi-entries (nskk-dict-io--collect-all-entries index 'okuri-nasi)))
411 (dolist (entry okuri-nasi-entries)
412 (insert (nskk-dict-io--serialize-entry entry))
413 (insert &quot;\n&quot;)))
414
415 ;; 一時ファイルに書き込み
416 (write-region (point-min) (point-max) temp-file nil 'silent)))
417
418 ;; アトミックリネーム
419 (rename-file temp-file output-path t)
420
421 ;; パーミッション復元
422 (when original-modes
423 (set-file-modes output-path original-modes))
424
425 (let ((save-time (- (float-time) start-time)))
426 (when nskk-dict-io-verbose
427 (message &quot;Dictionary saved in %.3f sec: %s&quot; save-time output-path))))
428
429 ;; エラー時に一時ファイルを削除
430 (when (file-exists-p temp-file)
431 (delete-file temp-file)))
432
433 ;; キャッシュをクリア（ファイルが更新されたため）
434 (nskk-clear-dictionary-cache output-path)
435
436 output-path)))

440(defun nskk-dict-io--merge-entries (existing new)
441 &quot;エントリをマージする（重複排除、頻度統合）。
442
443引数:
444 EXISTING - 既存のnskk-dict-entry構造体
445 NEW - 新しいnskk-dict-entry構造体
446
447戻り値:
448 マージされたnskk-dict-entry構造体
449
450マージロジック:
451 - 候補は重複排除（単語が同じものは1つに）
452 - 新しい候補を既存候補の前に配置（学習順を反映）
453 - 頻度情報を統合&quot;
454 (let* ((existing-candidates (nskk-dict-entry-candidates existing))
455 (new-candidates (nskk-dict-entry-candidates new))
456 ;; 既存候補を単語→候補のマップに変換
457 (candidate-map (make-hash-table :test 'equal))
458 (merged-candidates nil))
459
460 ;; 既存候補をマップに登録
461 (dolist (cand existing-candidates)
462 (puthash (nskk-dict-candidate-word cand) cand candidate-map))
463
464 ;; 新しい候補を処理（重複があれば上書き）
465 (dolist (cand new-candidates)
466 (puthash (nskk-dict-candidate-word cand) cand candidate-map))
467
468 ;; 新しい候補を優先してマージ（学習順）
469 (dolist (cand new-candidates)
470 (let ((word (nskk-dict-candidate-word cand)))
471 (push (gethash word candidate-map) merged-candidates)))
472
473 ;; 既存候補で新規に含まれないものを追加
474 (dolist (cand existing-candidates)
475 (let ((word (nskk-dict-candidate-word cand)))
476 (unless (cl-find word new-candidates
477 :key #'nskk-dict-candidate-word
478 :test #'equal)
479 (push cand merged-candidates))))
480
481 ;; マージされたエントリを作成
482 (nskk-dict-entry--create
483 :midashi (nskk-dict-entry-midashi existing)
484 :candidates (nreverse merged-candidates)
485 :frequency (+ (nskk-dict-entry-frequency existing)
486 (nskk-dict-entry-frequency new))
487 :last-used (or (nskk-dict-entry-last-used new)
488 (nskk-dict-entry-last-used existing))
489 :okuri-type (nskk-dict-entry-okuri-type existing)
490 :metadata (nskk-dict-entry-metadata existing))))

493(defun nskk-update-dictionary-incremental (index new-entries output-path)
494 &quot;既存辞書INDEXにNEW-ENTRIESをマージして保存。
495
496引数:
497 INDEX - 既存のnskk-dict-index構造体
498 NEW-ENTRIES - 新しいエントリのリスト（nskk-dict-entryのリスト）
499 OUTPUT-PATH - 出力ファイルパス
500
501処理フロー:
502 1. 既存エントリとマージ
503 2. 重複チェック
504 3. 頻度情報の統合
505 4. ソート（見出し語順）
506 5. 書き込み
507
508戻り値:
509 更新されたnskk-dict-index構造体&quot;
510 (when nskk-dict-io-verbose
511 (message &quot;Incremental update: %d new entries&quot; (length new-entries)))
512
513 (let ((start-time (float-time)))
514 ;; 新しいエントリをインデックスにマージ
515 (dolist (new-entry new-entries)
516 (let* ((midashi (nskk-dict-entry-midashi new-entry))
517 (okuri-type (nskk-dict-entry-okuri-type new-entry))
518 (table (if (eq okuri-type 'okuri-ari)
519 (nskk-dict-index-okuri-ari-table index)
520 (nskk-dict-index-okuri-nasi-table index)))
521 (existing-entry (gethash midashi table)))
522
523 (if existing-entry
524 ;; 既存エントリとマージ
525 (let ((merged (nskk-dict-io--merge-entries existing-entry new-entry)))
526 (puthash midashi merged table))
527 ;; 新規エントリとして追加
528 (puthash midashi new-entry table))))
529
530 ;; 辞書を保存
531 (nskk-save-dictionary index output-path)
532
533 (let ((update-time (- (float-time) start-time)))
534 (when nskk-dict-io-verbose
535 (message &quot;Incremental update completed in %.3f sec&quot; update-time)))
536
537 index))
