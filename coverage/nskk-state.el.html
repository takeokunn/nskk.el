83(cl-defstruct (nskk-state
84 (:constructor nskk-state--create)
85 (:copier nskk-state-copy))
86 &quot;NSKK入力状態を表す構造体。
87
88スロット:
89 mode - 現在の入力モード（`nskk-state-modes' のいずれか）
90 submode - サブモード（nil または 'okuri-ari, 'okuri-nasi）
91 input-buffer - 未確定入力文字列
92 conversion-buffer - 変換対象文字列
93 candidates - 変換候補リスト
94 candidate-index - 現在選択中の候補インデックス
95 okuri-char - 送り仮名の最初の文字
96 marker-start - 入力開始位置（マーカー）
97 marker-end - 入力終了位置（マーカー）
98 previous-mode - 前回のモード（モード復帰用）
99 timestamp - 最終更新時刻
100 properties - 拡張プロパティ（plist形式）&quot;
101 (mode 'hiragana
102 :type symbol
103 :documentation &quot;現在の入力モード&quot;)
104 (submode nil
105 :type (or null symbol)
106 :documentation &quot;サブモード（送り仮名の状態）&quot;)
107 (input-buffer &quot;&quot;
108 :type string
109 :documentation &quot;未確定のローマ字入力&quot;)
110 (conversion-buffer &quot;&quot;
111 :type string
112 :documentation &quot;変換対象の文字列&quot;)
113 (candidates nil
114 :type list
115 :documentation &quot;変換候補のリスト&quot;)
116 (candidate-index 0
117 :type integer
118 :documentation &quot;現在選択中の候補インデックス&quot;)
119 (okuri-char nil
120 :type (or null character)
121 :documentation &quot;送り仮名の最初の文字&quot;)
122 (marker-start nil
123 :type (or null marker)
124 :documentation &quot;入力開始位置のマーカー&quot;)
125 (marker-end nil
126 :type (or null marker)
127 :documentation &quot;入力終了位置のマーカー&quot;)
128 (previous-mode nil
129 :type (or null symbol)
130 :documentation &quot;前回のモード&quot;)
131 (timestamp (current-time)
132 :type list
133 :documentation &quot;最終更新時刻&quot;)
134 (properties nil
135 :type list
136 :documentation &quot;拡張プロパティ（plist）&quot;))

140(defun nskk-state-create (&rest args)
141 &quot;新しいNSKK状態を生成する。
142
143引数:
144 ARGS - cl-defstructのキーワード引数
145
146例:
147 (nskk-state-create)
148 (nskk-state-create :mode 'katakana)
149 (nskk-state-create :mode 'hiragana :input-buffer \&quot;ka\&quot;)&quot;
150 (let ((mode (or (plist-get args :mode) 'hiragana)))
151 (unless (memq mode nskk-state-modes)
152 (error &quot;Invalid mode: %s&quot; mode))
153 (apply #'nskk-state--create args)))

158 &quot;MODE が有効なNSKK入力モードかどうかを判定する。&quot;
159 (and (symbolp mode)
160 (memq mode nskk-state-modes)))

176(defun nskk-state-has-candidates-p (state)
177 &quot;STATE が変換候補を持っているかどうかを判定する。&quot;
178 (and (nskk-state-candidates state) t))

180(defun nskk-state-current-candidate (state)
181 &quot;STATE の現在選択中の候補を取得する。
182候補がない場合は nil を返す。&quot;
183 (when-let ((candidates (nskk-state-candidates state))
184 (index (nskk-state-candidate-index state)))
185 (nth index candidates)))

187(defun nskk-state-empty-p (state)
188 &quot;STATE が空（入力も変換もない）かどうかを判定する。&quot;
189 (and (string-empty-p (nskk-state-input-buffer state))
190 (string-empty-p (nskk-state-conversion-buffer state))
191 (null (nskk-state-candidates state))))

195(defun nskk-state-update-timestamp (state)
196 &quot;STATE のタイムスタンプを現在時刻に更新する。&quot;
197 (setf (nskk-state-timestamp state) (current-time))
198 state)

200(defun nskk-state-set-mode (state mode)
201 &quot;STATE のモードを MODE に設定する。
202前回のモードを保存し、タイムスタンプを更新する。&quot;
203 (unless (nskk-state-valid-mode-p mode)
204 (error &quot;Invalid mode: %s&quot; mode))
205 (setf (nskk-state-previous-mode state) (nskk-state-mode state))
206 (setf (nskk-state-mode state) mode)
207 (nskk-state-update-timestamp state))

209(defun nskk-state-restore-previous-mode (state)
210 &quot;STATE を前回のモードに戻す。
211前回のモードがない場合は hiragana モードに戻る。&quot;
212 (let ((prev-mode (or (nskk-state-previous-mode state) 'hiragana)))
213 (nskk-state-set-mode state prev-mode)))

216 &quot;STATE の入力バッファをクリアする。&quot;

221 &quot;STATE の変換関連データをクリアする。&quot;
222 (setf (nskk-state-conversion-buffer state) &quot;&quot;)
223 (setf (nskk-state-candidates state) nil)

228(defun nskk-state-clear-all (state)
229 &quot;STATE の全ての入力・変換データをクリアする。&quot;
230 (nskk-state-clear-input state)

233(defun nskk-state-append-input (state char)
234 &quot;STATE の入力バッファに CHAR を追加する。&quot;
235 (setf (nskk-state-input-buffer state)
236 (concat (nskk-state-input-buffer state)
237 (char-to-string char)))

240(defun nskk-state-set-candidates (state candidates &optional index)
241 &quot;STATE に変換候補 CANDIDATES を設定する。
242INDEX が指定された場合は、その候補をデフォルト選択にする。&quot;
243 (setf (nskk-state-candidates state) candidates)
244 (setf (nskk-state-candidate-index state) (or index 0))
245 (nskk-state-update-timestamp state))
246
247(defun nskk-state-next-candidate (state)
248 &quot;STATE の候補選択を次に進める。
249最後の候補の場合は最初に戻る。&quot;
250 (when-let ((candidates (nskk-state-candidates state)))
251 (let* ((index (nskk-state-candidate-index state))
252 (next-index (mod (1+ index) (length candidates))))
253 (setf (nskk-state-candidate-index state) next-index)
254 (nskk-state-update-timestamp state)
255 next-index)))
256
257(defun nskk-state-previous-candidate (state)
258 &quot;STATE の候補選択を前に戻す。
259最初の候補の場合は最後に戻る。&quot;
260 (when-let ((candidates (nskk-state-candidates state)))
261 (let* ((index (nskk-state-candidate-index state))
262 (prev-index (mod (1- index) (length candidates))))
263 (setf (nskk-state-candidate-index state) prev-index)
264 (nskk-state-update-timestamp state)
265 prev-index)))

267;;; 状態遷移定義
268
269(defconst nskk-state-transitions
270 '((hiragana . (katakana zenkaku-latin latin abbrev conversion))
271 (katakana . (hiragana zenkaku-latin latin abbrev conversion))
272 (zenkaku-latin . (hiragana katakana latin abbrev conversion))
273 (latin . (hiragana katakana zenkaku-latin abbrev conversion))
274 (abbrev . (hiragana katakana zenkaku-latin latin conversion))
275 (conversion . (hiragana katakana zenkaku-latin latin abbrev)))
276 &quot;各モードから遷移可能なモードのリスト。&quot;)
277
278(defun nskk-state-can-transition-p (from-mode to-mode)
279 &quot;FROM-MODE から TO-MODE への遷移が可能かどうかを判定する。&quot;
280 (and (nskk-state-valid-mode-p from-mode)
281 (nskk-state-valid-mode-p to-mode)
282 (memq to-mode (alist-get from-mode nskk-state-transitions))))
283
284(defun nskk-state-transition (state to-mode &optional force)
285 &quot;STATE を TO-MODE に遷移させる。
286FORCE が非nilの場合、遷移可能性チェックをスキップする。
287
288遷移が成功した場合は t を返し、失敗した場合は nil を返す。&quot;
289 (let ((from-mode (nskk-state-mode state)))
290 (cond
291 ;; 同じモードへの遷移は何もしない
292 ((eq from-mode to-mode)
293 t)
294
295 ;; 強制モードまたは遷移可能な場合
296 ((or force (nskk-state-can-transition-p from-mode to-mode))
297 (nskk-state-set-mode state to-mode)
298 t)
299
300 ;; 遷移不可能
301 (t
302 (message &quot;Cannot transition from %s to %s&quot; from-mode to-mode)
303 nil))))

307(defun nskk-state-get-property (state key &optional default)
308 &quot;STATE の拡張プロパティから KEY の値を取得する。

310 (or (plist-get (nskk-state-properties state) key)

313(defun nskk-state-set-property (state key value)
314 &quot;STATE の拡張プロパティに KEY と VALUE を設定する。&quot;
315 (setf (nskk-state-properties state)
316 (plist-put (nskk-state-properties state) key value))
317 (nskk-state-update-timestamp state))
318
319(defun nskk-state-remove-property (state key)
320 &quot;STATE の拡張プロパティから KEY を削除する。&quot;
321 (let ((props (nskk-state-properties state)))
322 (setf (nskk-state-properties state)
323 (cl-loop for (k v) on props by #'cddr
324 unless (eq k key)
325 append (list k v))))
326 (nskk-state-update-timestamp state))

330(defun nskk-state-set-markers (state start end)
331 &quot;STATE にバッファ位置マーカーを設定する。
332START と END は buffer position または marker。&quot;
333 (when (nskk-state-marker-start state)
334 (set-marker (nskk-state-marker-start state) nil))
335 (when (nskk-state-marker-end state)
336 (set-marker (nskk-state-marker-end state) nil))
337
338 (setf (nskk-state-marker-start state)
339 (if (markerp start) start (copy-marker start)))
340 (setf (nskk-state-marker-end state)
341 (if (markerp end) end (copy-marker end)))
342
343 (nskk-state-update-timestamp state))
344
345(defun nskk-state-clear-markers (state)
346 &quot;STATE のマーカーをクリアする。&quot;
347 (when (nskk-state-marker-start state)
348 (set-marker (nskk-state-marker-start state) nil)
349 (setf (nskk-state-marker-start state) nil))
350 (when (nskk-state-marker-end state)
351 (set-marker (nskk-state-marker-end state) nil)
352 (setf (nskk-state-marker-end state) nil))
353 (nskk-state-update-timestamp state))
354
355(defun nskk-state-marker-region (state)
356 &quot;STATE のマーカーで示される領域を (start . end) で返す。
357マーカーが設定されていない場合は nil を返す。&quot;
358 (when (and (nskk-state-marker-start state)
359 (nskk-state-marker-end state))
360 (cons (marker-position (nskk-state-marker-start state))
361 (marker-position (nskk-state-marker-end state)))))

402(defun nskk-state-init ()
403 &quot;現在のバッファのNSKK状態を初期化する。&quot;
404 (unless nskk-current-state
405 (setq nskk-current-state (nskk-state-create))))

407(defun nskk-state-cleanup ()
408 &quot;現在のバッファのNSKK状態をクリーンアップする。&quot;
409 (when nskk-current-state
410 (nskk-state-clear-markers nskk-current-state)
411 (setq nskk-current-state nil)))
