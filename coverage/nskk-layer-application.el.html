145(defun nskk-application-process-input (input)
146 &quot;入力を処理する。
147INPUTは入力文字またはキーイベント。&quot;
148 ;; 拡張ポイント: 入力前処理
149 (nskk-application--run-extension-point :before-input-processing input)
150
151 ;; 現在のモードに応じて処理
152 (let ((result
153 (pcase nskk-application--current-mode
154 ('hiragana (nskk-application--process-hiragana-input input))
155 ('katakana (nskk-application--process-katakana-input input))
156 ('latin (nskk-application--process-latin-input input))
157 ('zenkaku-latin (nskk-application--process-zenkaku-latin-input input))
158 (_ (nskk-application--process-default-input input)))))
159
160 ;; 拡張ポイント: 入力後処理
161 (nskk-application--run-extension-point :after-input-processing result)
162 result))
163

168 (nskk-application--delegate-to-core :convert-romaji input))

194(defun nskk-application-start-conversion (query)
195 &quot;変換を開始する。
196QUERYは変換クエリ文字列。&quot;
197 ;; 拡張ポイント: 変換前処理
198 (nskk-application--run-extension-point :before-conversion query)
199
200 ;; 変換状態を作成
201 (setq nskk-application--conversion-state
202 (make-nskk-application-conversion-state
203 :query query
204 :candidates nil
205 :index 0
206 :context (nskk-application--create-context)
207 :timestamp (float-time)))
208
209 ;; Data Layerから候補を取得
210 (let ((candidates (nskk-application--fetch-candidates query)))
211 (setf (nskk-application-conversion-state-candidates
212 nskk-application--conversion-state)
213 candidates)
214 (setq nskk-application--candidate-list candidates)
215 (setq nskk-application--candidate-index 0)
216
217 ;; 拡張ポイント: 変換後処理
218 (nskk-application--run-extension-point :after-conversion candidates)
219
220 ;; イベント発行
221 (nskk-application--emit-event :conversion-started
222 :query query
223 :candidates candidates)
224 candidates))
225
226(defun nskk-application--fetch-candidates (query)
227 &quot;候補を取得する。
228QUERYは検索クエリ。&quot;
229 ;; Data Layerに問い合わせ
230 (let ((results (nskk-application--delegate-to-data :search query)))
231 ;; 候補数を制限
232 (seq-take results nskk-application-candidate-limit)))
233
234(defun nskk-application-select-candidate (index)
235 &quot;候補を選択する。
236INDEXは候補のインデックス。&quot;
237 (when (and nskk-application--candidate-list
238 (&lt; index (length nskk-application--candidate-list)))
239 (let ((candidate (nth index nskk-application--candidate-list)))
240 ;; 拡張ポイント: 候補選択前処理
241 (nskk-application--run-extension-point :before-candidate-selection candidate)
242
243 ;; 候補を確定
244 (nskk-application-commit-candidate candidate)
245
246 ;; 学習
247 (when nskk-application-enable-learning
248 (nskk-application--learn-selection
249 (nskk-application-conversion-state-query nskk-application--conversion-state)
250 candidate
251 index))
252
253 ;; 拡張ポイント: 候補選択後処理
254 (nskk-application--run-extension-point :after-candidate-selection candidate)
255
256 ;; イベント発行
257 (nskk-application--emit-event :candidate-selected
258 :candidate candidate
259 :index index)
260 candidate)))
261
262(defun nskk-application-commit-candidate (candidate)
263 &quot;候補を確定する。
264CANDIDATEは確定する候補。&quot;
265 ;; 変換状態をクリア
266 (setq nskk-application--conversion-state nil)
267 (setq nskk-application--candidate-list nil)
268 (setq nskk-application--candidate-index 0)
269
270 ;; イベント発行
271 (nskk-application--emit-event :conversion-committed :candidate candidate)
272
273 candidate)

275(defun nskk-application-next-candidate ()
276 &quot;次の候補を選択する。&quot;
277 (interactive)
278 (when nskk-application--candidate-list
279 (setq nskk-application--candidate-index
280 (mod (1+ nskk-application--candidate-index)
281 (length nskk-application--candidate-list)))
282 (nskk-application--emit-event :candidate-changed
283 :index nskk-application--candidate-index)))
284
285(defun nskk-application-previous-candidate ()
286 &quot;前の候補を選択する。&quot;
287 (interactive)
288 (when nskk-application--candidate-list
289 (setq nskk-application--candidate-index
290 (mod (1- nskk-application--candidate-index)
291 (length nskk-application--candidate-list)))
292 (nskk-application--emit-event :candidate-changed
293 :index nskk-application--candidate-index)))

297(defun nskk-application-switch-mode (new-mode)
298 &quot;入力モードを切り替える。
299NEW-MODEは新しいモード。&quot;
300 (let ((old-mode nskk-application--current-mode))
301 ;; 拡張ポイント: モード変更前処理
302 (when (nskk-application--run-extension-point :before-mode-change
303 old-mode new-mode)
304 (setq nskk-application--current-mode new-mode)
305
306 ;; 拡張ポイント: モード変更後処理
307 (nskk-application--run-extension-point :after-mode-change old-mode new-mode)
308
309 ;; イベント発行
310 (nskk-application--emit-event :mode-changed
311 :from old-mode
312 :to new-mode)
313
314 new-mode)))

429(defun nskk-application-health-check ()
430 &quot;Application Layerのヘルスチェックを実行する。&quot;
431 (interactive)
432 (let ((issues '()))
433 ;; セッション状態チェック
434 (unless nskk-application--session-id
435 (push &quot;No active session&quot; issues))
436 ;; モード状態チェック
437 (unless nskk-application--current-mode
438 (push &quot;No current mode set&quot; issues))
439 ;; 結果表示
440 (if issues
441 (message &quot;Application Layer issues: %s&quot; (string-join issues &quot;, &quot;))
442 (message &quot;Application Layer: All systems operational&quot;))))
