161(defun nskk-candidate-window--normalize-candidate (candidate)
162 &quot;候補データを正規化する。
163
164CANDIDATE: 候補（文字列またはplist）
165
166plist形式の場合:
167 (:text \&quot;候補\&quot; :annotation \&quot;注釈\&quot; :priority 0)
168
169文字列を返すか、plistを返す。&quot;
170 (cond
171 ((stringp candidate)
172 ;; 文字列の場合はそのまま
173 candidate)
174 ((and (listp candidate) (plist-get candidate :text))
175 ;; plist形式
176 candidate)
177 (t
178 ;; その他の場合は文字列化
179 (format &quot;%s&quot; candidate))))

199(defun nskk-candidate-window--format-number (index)
200 &quot;候補番号をフォーマットする。
201
202INDEX: インデックス（0始まり）&quot;
203 (let ((num (1+ index)))
204 (pcase nskk-candidate-number-style
205 ('arabic (format &quot;%d&quot; num))
206 ('alphabet (if (&lt;= num 26)
207 (char-to-string (+ ?a (1- num)))
208 (format &quot;%d&quot; num)))
209 (_ (format &quot;%d&quot; num)))))

213(defun nskk-candidate-window--format-candidate (candidate index selected-p)
214 &quot;1つの候補をフォーマットする。
215
216CANDIDATE: 候補データ
217INDEX: インデックス（ページ内での0始まり）
218SELECTED-P: 選択中かどうか&quot;
219 (let* ((text (nskk-candidate-window--get-text candidate))
220 (annotation (nskk-candidate-window--get-annotation candidate))
221 (number (when nskk-candidate-show-index
222 (nskk-candidate-window--format-number index)))
223 (face (if selected-p
224 'nskk-candidate-selected-face
225 'nskk-candidate-face))
226 (parts nil))
227
228 ;; 番号
229 (when number
230 (push (propertize number 'face 'nskk-candidate-number-face) parts)
231 (push &quot;:&quot; parts))
232
233 ;; 候補テキスト
234 (push (propertize text 'face face) parts)
235
236 ;; 注釈
237 (when annotation
238 (push (propertize nskk-candidate-annotation-separator
239 'face 'nskk-candidate-separator-face)
240 parts)
241 (push (propertize annotation 'face 'nskk-candidate-annotation-face)
242 parts))
243
244 (apply #'concat (nreverse parts))))

246(defun nskk-candidate-window--format-page (window)
247 &quot;現在のページの候補を整形する。
248
249WINDOW: 候補ウィンドウ構造体&quot;
250 (let* ((page-size (nskk-candidate-window-page-size window))
251 (current-page (nskk-candidate-window-current-page window))
252 (selected-index (nskk-candidate-window-selected-index window))
253 (candidates (nskk-candidate-window-candidates window))
254 (start-index (* current-page page-size))
255 (end-index (min (+ start-index page-size) (length candidates)))
256 (page-candidates (cl-subseq candidates start-index end-index))
257 (lines nil))
258
259 ;; 各候補を整形
260 (cl-loop for candidate in page-candidates
261 for i from start-index
262 for page-i from 0
263 do (let ((selected-p (= i selected-index)))
264 (push (nskk-candidate-window--format-candidate
265 candidate page-i selected-p)
266 lines)))
267
268 ;; ページ情報を追加
269 (let* ((total-pages (ceiling (length candidates) (float page-size)))
270 (page-info (when (&gt; total-pages 1)
271 (format &quot;[%d/%d]&quot;
272 (1+ current-page)
273 total-pages))))
274 (when page-info
275 (push (propertize page-info 'face 'nskk-candidate-number-face)
276 lines)))
277
278 (mapconcat #'identity (nreverse lines) &quot;\n&quot;)))

282(defun nskk-candidate-window--calculate-position (position)
283 &quot;候補ウィンドウの表示位置を計算する。
284
285POSITION: 基準位置（ポイント位置）
286
287(LINE . COLUMN) の形式で返す。&quot;
288 (save-excursion
289 (goto-char position)
290 (let* ((window (selected-window))
291 (window-start (window-start window))
292 (window-height (window-height window))
293 (current-line (count-lines window-start (point)))
294 (current-column (current-column))
295 (lines-below (- window-height current-line))
296 (margin nskk-candidate-window-margin))
297
298 ;; 下に十分なスペースがあるか確認
299 (cons (if (&gt;= lines-below (+ margin 3))
300 (1+ current-line) ;; 下に表示
301 (max 0 (- current-line margin 1))) ;; 上に表示
302 current-column))))

306(defun nskk-candidate-window--create-overlay (position)
307 &quot;候補表示用のoverlayを作成する。
308
309POSITION: 表示位置（ポイント位置）&quot;
310 (let* ((pos-info (nskk-candidate-window--calculate-position position))
311 (line (car pos-info))
312 (overlay-pos (save-excursion
313 (goto-char (window-start))
314 (forward-line line)
315 (point)))
316 (overlay (make-overlay overlay-pos overlay-pos)))
317
318 ;; overlay設定
319 (overlay-put overlay 'window (selected-window))
320 (overlay-put overlay 'priority 1000)
321 (overlay-put overlay 'evaporate t)
322 overlay))

342(defun nskk-show-candidates (candidates &optional position)
343 &quot;候補ウィンドウを表示する。
344
345CANDIDATES: 候補リスト（文字列またはplistのリスト）
346POSITION: 表示位置（省略時は現在のポイント位置）&quot;
347 (interactive)
348 (let* ((pos (or position (point)))
349 (normalized-candidates (mapcar #'nskk-candidate-window--normalize-candidate
350 candidates))
351 (window (nskk-candidate-window--create
352 :candidates normalized-candidates
353 :selected-index 0
354 :page-size nskk-candidate-page-size
355 :current-page 0
356 :position pos))
357 (overlay (nskk-candidate-window--create-overlay pos))
358 (content (nskk-candidate-window--format-page window)))
359
360 (setf (nskk-candidate-window-overlay window) overlay)
361 (nskk-candidate-window--update-overlay overlay content)
362 (setq nskk-candidate-window--current window)
363 window))

376(defun nskk-update-candidates (candidates selected-index)
377 &quot;候補ウィンドウの内容を更新する。
378
379CANDIDATES: 新しい候補リスト
380SELECTED-INDEX: 選択中の候補インデックス&quot;
381 (interactive)
382 (if (not nskk-candidate-window--current)
383 (nskk-show-candidates candidates)
384 (let* ((window nskk-candidate-window--current)
385 (normalized-candidates (mapcar #'nskk-candidate-window--normalize-candidate
386 candidates))
387 (page-size (nskk-candidate-window-page-size window))
388 (current-page (/ selected-index page-size))
389 (overlay (nskk-candidate-window-overlay window)))
390
391 (setf (nskk-candidate-window-candidates window) normalized-candidates)
392 (setf (nskk-candidate-window-selected-index window) selected-index)
393 (setf (nskk-candidate-window-current-page window) current-page)
394
395 (let ((content (nskk-candidate-window--format-page window)))
396 (nskk-candidate-window--update-overlay overlay content))
397
398 window)))

401(defun nskk-scroll-candidates (direction)
402 &quot;候補ウィンドウをスクロールする。
403
404DIRECTION: スクロール方向
405 'next - 次のページ
406 'previous - 前のページ
407 'first - 最初のページ
408 'last - 最後のページ&quot;
409 (interactive)
410 (unless nskk-candidate-window--current
411 (error &quot;候補ウィンドウが表示されていません&quot;))
412
413 (let* ((window nskk-candidate-window--current)
414 (candidates (nskk-candidate-window-candidates window))
415 (page-size (nskk-candidate-window-page-size window))
416 (current-page (nskk-candidate-window-current-page window))
417 (total-pages (ceiling (length candidates) (float page-size)))
418 (new-page
419 (pcase direction
420 ('next (min (1+ current-page) (1- total-pages)))
421 ('previous (max 0 (1- current-page)))
422 ('first 0)
423 ('last (1- total-pages))
424 (_ current-page)))
425 (new-index (* new-page page-size)))
426
427 (when (/= new-page current-page)
428 (setf (nskk-candidate-window-current-page window) new-page)
429 (setf (nskk-candidate-window-selected-index window) new-index)
430
431 (let* ((overlay (nskk-candidate-window-overlay window))
432 (content (nskk-candidate-window--format-page window)))
433 (nskk-candidate-window--update-overlay overlay content)))
434
435 window))

438(defun nskk-select-candidate (index)
439 &quot;指定インデックスの候補を選択する。
440
441INDEX: 候補インデックス（0始まり）&quot;
442 (interactive)
443 (unless nskk-candidate-window--current
444 (error &quot;候補ウィンドウが表示されていません&quot;))
445
446 (let* ((window nskk-candidate-window--current)
447 (candidates (nskk-candidate-window-candidates window))
448 (page-size (nskk-candidate-window-page-size window)))

450 (when (and (&gt;= index 0) (&lt; index (length candidates)))
451 (let ((new-page (/ index page-size)))
452 (setf (nskk-candidate-window-selected-index window) index)
453 (setf (nskk-candidate-window-current-page window) new-page)
454
455 (let* ((overlay (nskk-candidate-window-overlay window))
456 (content (nskk-candidate-window--format-page window)))
457 (nskk-candidate-window--update-overlay overlay content))))
