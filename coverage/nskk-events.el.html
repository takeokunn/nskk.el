139(defun nskk-events-add-listener (event-type listener &optional global)
140 &quot;EVENT-TYPE に対して LISTENER を登録する。
141
142GLOBAL が非nilの場合、グローバルリスナーとして登録。
143nilの場合、現在のバッファのローカルリスナーとして登録。
144
145LISTENER は (lambda (event) ...) 形式の関数で、
146EVENT は `nskk-event' 構造体。
147
148戻り値: t（成功）、nil（失敗）&quot;
149 (unless (memq event-type nskk-events-types)
150 (error &quot;Unknown event type: %s&quot; event-type))
151 (unless (functionp listener)
152 (error &quot;Listener must be a function: %s&quot; listener))
153
154 (let ((table (if global
155 nskk-events--global-listeners
156 ;; バッファローカル変数を確実に初期化
157 (progn
158 (unless (hash-table-p nskk-events--listeners)
159 (setq-local nskk-events--listeners (make-hash-table :test 'eq)))
160 nskk-events--listeners))))
161 (let ((listeners (gethash event-type table)))
162 (unless (memq listener listeners)
163 (puthash event-type
164 (append listeners (list listener))
165 table)))
166 t))
167
168(defun nskk-events-remove-listener (event-type listener &optional global)
169 &quot;EVENT-TYPE から LISTENER を削除する。
170
171GLOBAL が非nilの場合、グローバルリスナーから削除。
172nilの場合、現在のバッファのローカルリスナーから削除。
173
174戻り値: t（削除成功）、nil（リスナーが見つからない）&quot;
175 (unless (memq event-type nskk-events-types)
176 (error &quot;Unknown event type: %s&quot; event-type))
177
178 (let ((table (if global
179 nskk-events--global-listeners
180 (when (hash-table-p nskk-events--listeners)
181 nskk-events--listeners))))
182 (when table
183 (let ((listeners (gethash event-type table)))
184 (when (memq listener listeners)
185 (puthash event-type
186 (delq listener listeners)
187 table)
188 t)))))
189
190(defun nskk-events-clear-listeners (&optional event-type global)
191 &quot;リスナーをクリアする。
192
193EVENT-TYPE が指定された場合、そのイベントタイプのリスナーのみクリア。
194nilの場合、全てのリスナーをクリア。
195
196GLOBAL が非nilの場合、グローバルリスナーをクリア。
197nilの場合、現在のバッファのローカルリスナーをクリア。&quot;
198 (let ((table (if global
199 nskk-events--global-listeners
200 (when (hash-table-p nskk-events--listeners)
201 nskk-events--listeners))))
202 (when table
203 (if event-type
204 (puthash event-type nil table)
205 (clrhash table)))))

207(defun nskk-events-get-listeners (event-type &optional include-global)
208 &quot;EVENT-TYPE のリスナーリストを取得する。
209
210INCLUDE-GLOBAL が非nilの場合、グローバルリスナーも含める。
211
212戻り値: リスナー関数のリスト&quot;
213 (let ((local-listeners
214 (when (hash-table-p nskk-events--listeners)
215 (gethash event-type nskk-events--listeners)))
216 (global-listeners (when include-global
217 (gethash event-type nskk-events--global-listeners))))
218 (append local-listeners global-listeners)))

222(defun nskk-events-emit (event-type &rest data)
223 &quot;EVENT-TYPE イベントを発行する。
224
225DATA はイベントデータで、plist形式で指定する。
226例: (nskk-events-emit :mode-switched :from 'hiragana :to 'katakana)
227
228戻り値: 実行されたリスナー数&quot;
229 (unless (memq event-type nskk-events-types)
230 (error &quot;Unknown event type: %s&quot; event-type))
231
232 ;; 再帰的なイベント発行を検出
233 (when nskk-events--emitting
234 (nskk-events--log event-type
235 (append data (list :recursive t))
236 'warning)
237 (when nskk-events-enable-logging
238 (message &quot;Warning: Recursive event emission detected for %s&quot; event-type)))
239
240 (let ((nskk-events--emitting t)
241 (event (nskk-event-create
242 :type event-type
243 :timestamp (current-time)
244 :buffer (current-buffer)
245 :data data))
246 (count 0))
247
248 (nskk-events--log event-type data)
249
250 ;; ローカルリスナーとグローバルリスナーを取得
251 (let ((listeners (nskk-events-get-listeners event-type t)))
252 (dolist (listener listeners)
253 (condition-case err
254 (progn
255 (funcall listener event)
256 (setq count (1+ count)))
257 (error
258 (nskk-events--handle-error event-type listener err)))))
259
260 count))

264(defun nskk-events--handle-error (event-type listener error-data)
265 &quot;イベントハンドラー内のエラーを処理する。
266
267EVENT-TYPE: エラーが発生したイベントタイプ
268LISTENER: エラーが発生したリスナー関数
269ERROR-DATA: エラーデータ&quot;
270 (let ((error-info (list :event-type event-type
271 :listener listener
272 :error error-data)))
273 (nskk-events--log :error error-info 'error)
274
275 (if nskk-events-error-handler
276 (condition-case err
277 (funcall nskk-events-error-handler event-type error-info)
278 (error
279 (message &quot;Error in error handler: %s&quot; err)))
280 (message &quot;NSKK event handler error [%s]: %s&quot;
281 event-type (error-message-string error-data)))))

285(defun nskk-events--log (event-type data &optional level)
286 &quot;イベントをログに記録する。
287
288EVENT-TYPE: イベントタイプ
289DATA: イベントデータ
290LEVEL: ログレベル（'info, 'warning, 'error）デフォルトは'info&quot;
291 (when nskk-events-enable-logging
292 (let ((entry (list :timestamp (current-time)
293 :level (or level 'info)
294 :event-type event-type
295 :data data)))
296 (push entry nskk-events--log-entries)
297
298 ;; 最大エントリ数を超えた場合、古いエントリを削除
299 (when (&gt; (length nskk-events--log-entries)
300 nskk-events-max-log-entries)
301 (setq nskk-events--log-entries
302 (seq-take nskk-events--log-entries
303 nskk-events-max-log-entries)))
304
305 ;; ログバッファに出力
306 (nskk-events--write-log-buffer entry))))
307
308(defun nskk-events--write-log-buffer (entry)
309 &quot;ログエントリをログバッファに書き込む。
310
311ENTRY: ログエントリ（plist形式）&quot;
312 (with-current-buffer (get-buffer-create nskk-events-log-buffer-name)
313 (goto-char (point-max))
314 (let ((inhibit-read-only t))
315 (insert (format &quot;[%s] [%s] %s: %s\n&quot;
316 (format-time-string &quot;%Y-%m-%d %H:%M:%S&quot;
317 (plist-get entry :timestamp))
318 (upcase (symbol-name (plist-get entry :level)))
319 (plist-get entry :event-type)
320 (plist-get entry :data))))))

349(defun nskk-events-stats ()
350 &quot;イベントシステムの統計情報を返す。
351
352戻り値: 統計情報のplist
353 :local-listeners-count - ローカルリスナー総数
354 :global-listeners-count - グローバルリスナー総数
355 :log-entries-count - ログエントリ総数
356 :event-types-count - 登録されているイベントタイプ数&quot;
357 (let ((local-count 0)
358 (global-count 0))
359 (when (hash-table-p nskk-events--listeners)
360 (maphash (lambda (_type listeners)
361 (setq local-count (+ local-count (length listeners))))
362 nskk-events--listeners))
363 (maphash (lambda (_type listeners)
364 (setq global-count (+ global-count (length listeners))))
365 nskk-events--global-listeners)
366
367 (list :local-listeners-count local-count
368 :global-listeners-count global-count
369 :log-entries-count (length nskk-events--log-entries)
370 :event-types-count (length nskk-events-types))))
