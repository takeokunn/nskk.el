25;;; Commentary:

27;; このファイルは高速な前方一致検索を実現するトライ木（Trie Tree）を実装します。
28;;
29;; トライ木は文字列検索に最適化されたデータ構造で、以下の特徴があります:
30;; - 挿入/検索/削除の計算量: O(m) (m=キー長)
31;; - 前方一致検索: O(k + n) (k=prefix長, n=結果数)
32;; - メモリ効率的な実装（ハッシュテーブルベース）
33;;
34;; 使用例:
35;;
36;; (require 'nskk-trie)
37;;
38;; ;; トライ木の作成
39;; (let ((trie (nskk-trie-create)))
40;; ;; エントリの挿入
41;; (nskk-trie-insert trie &quot;かんじ&quot; entry1)
42;; (nskk-trie-insert trie &quot;かんたん&quot; entry2)
43;; (nskk-trie-insert trie &quot;かん&quot; entry3)
44;;
45;; ;; 完全一致検索
46;; (nskk-trie-lookup trie &quot;かんじ&quot;)
47;; ;; =&gt; (entry1 . t)
48;;
49;; ;; 前方一致検索
50;; (nskk-trie-prefix-search trie &quot;かん&quot;)
51;; ;; =&gt; ((&quot;かん&quot; . entry3) (&quot;かんじ&quot; . entry1) (&quot;かんたん&quot; . entry2))
52;;
53;; ;; 削除
54;; (nskk-trie-delete trie &quot;かんじ&quot;)
55;;
56;; ;; シリアライズ
57;; (nskk-trie-save-to-file trie &quot;/tmp/trie.dat&quot;))
58;;
59;; パフォーマンス目標:
60;; - 10万キー挿入: &lt; 3秒
61;; - 検索: &lt; 1ms
62;; - 前方一致検索（100件）: &lt; 10ms
63;; - メモリ使用量: 10万キーで &lt; 30MB
64

127(defun nskk-trie-insert (trie key value)
128 &quot;トライ木TRIEにKEYとVALUEを挿入する。
129
130引数:
131 TRIE - nskk-trie構造体
132 KEY - キー文字列
133 VALUE - 関連付ける値
134
135処理:
136 1. ルートから開始
137 2. KEYの各文字について:
138 - 対応する子ノードがあればそれをたどる
139 - なければ新しいノードを作成
140 3. 最後のノードにVALUEを設定
141 4. サイズをインクリメント
142
143戻り値:
144 TRIE（更新後）&quot;
145 (unless (stringp key)
146 (error &quot;Key must be a string: %s&quot; key))
147 (when (zerop (length key))
148 (error &quot;Key cannot be empty&quot;))
149
150 (let ((node (nskk-trie-root trie))
151 (key-len (length key))
152 (was-new nil))
153
154 ;; KEYの各文字をたどる
155 (dotimes (i key-len)
156 (let ((char (aref key i)))
157 ;; 子ノードテーブルがなければ作成
158 (unless (nskk-trie-node-children node)
159 (setf (nskk-trie-node-children node)
160 (make-hash-table :test 'eq :size nskk-trie-default-hash-size)))
161
162 ;; 子ノードを取得または作成
163 (let ((next-node (gethash char (nskk-trie-node-children node))))
164 (unless next-node
165 (setq next-node (nskk-trie-node--create :char char))
166 (puthash char next-node (nskk-trie-node-children node)))
167
168 ;; カウントをインクリメント
169 (cl-incf (nskk-trie-node-count next-node))
170
171 ;; 次のノードへ移動
172 (setq node next-node))))
173
174 ;; 最後のノードに値を設定
175 ;; 既に終端ノードであれば上書き、新規であればサイズをインクリメント
176 (setq was-new (not (nskk-trie-node-is-end node)))
177 (setf (nskk-trie-node-is-end node) t)
178 (setf (nskk-trie-node-value node) value)
179
180 ;; サイズ更新
181 (when was-new
182 (cl-incf (nskk-trie-size trie)))
183
184 trie))

218(defun nskk-trie--find-node (trie key)
219 &quot;トライ木TRIEからKEYに対応するノードを検索する（内部関数）。
220
221引数:
222 TRIE - nskk-trie構造体

240(defun nskk-trie-delete (trie key)
241 &quot;トライ木TRIEからKEYを削除する。
242
243引数:
244 TRIE - nskk-trie構造体
245 KEY - 削除するキー文字列
246
247処理:
248 1. KEYを検索
249 2. 終端ノードのis-endをnilに設定
250 3. 値をクリア
251 4. 子ノードがない場合、親方向にノードを削除
252 5. サイズをデクリメント
253
254戻り値:
255 削除成功の場合t、キーが存在しない場合nil&quot;
256 (unless (stringp key)
257 (error &quot;Key must be a string: %s&quot; key))
258
259 (let ((node (nskk-trie--find-node trie key)))
260 (when (and node (nskk-trie-node-is-end node))
261 ;; 終端フラグをクリア
262 (setf (nskk-trie-node-is-end node) nil)
263 (setf (nskk-trie-node-value node) nil)
264
265 ;; サイズをデクリメント
266 (cl-decf (nskk-trie-size trie))
267
268 ;; 不要なノードを削除（最適化）
269 ;; 子ノードがなく、終端でもない場合は親から削除
270 (nskk-trie--cleanup-path trie key)
271
272 t)))

274(defun nskk-trie--cleanup-path (trie key)
275 &quot;削除後の不要ノードをクリーンアップする（内部関数）。
276
277引数:
278 TRIE - nskk-trie構造体
279 KEY - 削除したキー文字列
280
281処理:
282 KEYのパスを逆順にたどり、子がなく終端でもないノードを削除&quot;
283 (let ((nodes-path nil)
284 (node (nskk-trie-root trie))
285 (key-len (length key)))
286
287 ;; パスを記録
288 (dotimes (i key-len)
289 (let ((char (aref key i)))
290 (when (and (nskk-trie-node-children node)
291 (setq node (gethash char (nskk-trie-node-children node))))
292 (push (cons char node) nodes-path))))
293
294 ;; 逆順にクリーンアップ
295 (let ((parent (nskk-trie-root trie))
296 (parent-stack nil))
297 (setq node (nskk-trie-root trie))
298
299 (dotimes (i key-len)
300 (let ((char (aref key i)))
301 (push parent parent-stack)
302 (setq node (gethash char (nskk-trie-node-children node)))))
303
304 ;; 末尾から親方向にたどる
305 (dotimes (i key-len)
306 (when (and node
307 (not (nskk-trie-node-is-end node))
308 (or (null (nskk-trie-node-children node))
309 (zerop (hash-table-count (nskk-trie-node-children node)))))
310 ;; このノードは削除可能
311 (setq parent (pop parent-stack))
312 (when (nskk-trie-node-children parent)
313 (remhash (nskk-trie-node-char node) (nskk-trie-node-children parent)))
314 (setq node parent))))))

319(defun nskk-trie-prefix-search (trie prefix &optional limit)
320 &quot;トライ木TRIEからPREFIXで始まるすべてのキーを検索する。
321
322引数:
323 TRIE - nskk-trie構造体
324 PREFIX - 前方一致させるプレフィックス
325 LIMIT - 結果の最大数（nilの場合無制限）
326
327戻り値:
328 ((key . value) ...) のリスト
329
330アルゴリズム:
331 1. PREFIXをたどって開始ノードを見つける
332 2. そのノードから深さ優先探索
333 3. 終端ノードに到達したらキーと値を収集
334 4. LIMITに達したら探索を中断&quot;
335 (unless (stringp prefix)
336 (error &quot;Prefix must be a string: %s&quot; prefix))
337
338 ;; PREFIXのノードを見つける
339 (let ((node (if (zerop (length prefix))
340 (nskk-trie-root trie)
341 (nskk-trie--find-node trie prefix))))
342
343 (if node
344 (nskk-trie--collect-all node prefix limit 0)
345 nil)))

347(defun nskk-trie--collect-all (node prefix limit collected-count)
348 &quot;ノードNODEから始まるすべてのキーと値を収集する（内部関数）。
349
350引数:
351 NODE - 開始ノード
352 PREFIX - 現在のプレフィックス
353 LIMIT - 収集する最大数（nilの場合無制限）
354 COLLECTED-COUNT - 既に収集したアイテム数
355
356戻り値:
357 ((key . value) ...) のリスト&quot;
358 (let ((results nil)
359 (count collected-count))
360
361 ;; 深さ優先探索で収集
362 (catch 'limit-reached
363 (cl-labels ((dfs-collect (node prefix)
364 ;; 現在のノードが終端なら収集
365 (when (nskk-trie-node-is-end node)
366 (push (cons prefix (nskk-trie-node-value node)) results)
367 (cl-incf count)
368 (when (and limit (&gt;= count limit))
369 (throw 'limit-reached nil)))
370
371 ;; 子ノードを探索
372 (when (nskk-trie-node-children node)
373 (maphash (lambda (char child-node)
374 (let ((new-prefix (concat prefix (char-to-string char))))
375 (dfs-collect child-node new-prefix)))
376 (nskk-trie-node-children node)))))
377 (dfs-collect node prefix)))
378
379 (nreverse results)))

425(defun nskk-trie-statistics (trie)
426 &quot;トライ木TRIEの統計情報を返す。
427
428引数:
429 TRIE - nskk-trie構造体
430
431戻り値:
432 plist形式の統計情報
433 :size - キー数
434 :node-count - ノード総数
435 :max-depth - 最大深度
436 :avg-depth - 平均深度
437 :memory-usage - 推定メモリ使用量&quot;
438 (let ((node-count 0)
439 (max-depth 0)
440 (total-depth 0)
441 (key-count (nskk-trie-size trie)))
442
443 ;; 統計収集用の再帰関数
444 (cl-labels ((count-nodes (node depth)
445 (cl-incf node-count)
446 (when (&gt; depth max-depth)
447 (setq max-depth depth))
448 (when (nskk-trie-node-is-end node)
449 (cl-incf total-depth depth))
450 (when (nskk-trie-node-children node)
451 (maphash (lambda (_ child)
452 (count-nodes child (1+ depth)))
453 (nskk-trie-node-children node)))))
454 (count-nodes (nskk-trie-root trie) 0))
455
456 ;; メモリ使用量の推定
457 ;; ノードごとに約200バイト + ハッシュテーブルオーバーヘッド
458 (let ((memory-usage (* node-count 250)))
459
460 (list :size key-count
461 :node-count node-count
462 :max-depth max-depth
463 :avg-depth (if (&gt; key-count 0)
464 (/ (float total-depth) key-count)
465 0.0)
466 :memory-usage memory-usage))))
