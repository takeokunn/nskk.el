62(defun nskk-input-hybrid-configure (&rest options)
63 &quot;ハイブリッド入力方式を設定する。
64
65OPTIONS はキーワード引数:
66 :methods LIST - 使用する入力方式のリスト
67 :priority-order LIST - 優先順位順のリスト
68 :fallback SYMBOL - フォールバック入力方式
69 :auto-select BOOL - 自動選択モードの有効化
70
71使用例:
72 (nskk-input-hybrid-configure
73 :methods '(azik act qwerty)
74 :priority-order '(azik act qwerty)
75 :fallback 'qwerty
76 :auto-select t)&quot;
77 (when-let ((methods (plist-get options :methods)))
78 (setq nskk-input-hybrid-methods methods))
79
80 (when-let ((fallback (plist-get options :fallback)))
81 (setq nskk-input-hybrid-fallback fallback))
82
83 (when (plist-member options :auto-select)
84 (setq nskk-input-hybrid-auto-select (plist-get options :auto-select))))

88(defun nskk-input-hybrid--get-lookup-function (method-name)
89 &quot;METHOD-NAMEに対応する検索関数を返す。&quot;
90 (cond
91 ((eq method-name 'azik)
92 #'nskk-input-azik-lookup)
93 ((eq method-name 'act)
94 #'nskk-input-act-lookup)
95 ((eq method-name 'tutcode)
96 (require 'nskk-input-tutcode)
97 #'nskk-input-tutcode-lookup)
98 ((eq method-name 'qwerty)
99 #'nskk-input-qwerty-lookup)
100 ((eq method-name 'nicola)
101 (require 'nskk-input-nicola)
102 #'nskk-input-nicola-lookup)
103 ((eq method-name 'kana)
104 (require 'nskk-input-kana)
105 #'nskk-input-kana-lookup)
106 ((eq method-name 'dvorak)
107 (require 'nskk-input-dvorak)
108 #'nskk-input-dvorak-lookup)
109 ((eq method-name 'colemak)
110 (require 'nskk-input-colemak)
111 #'nskk-input-colemak-lookup)
112 (t
113 (warn &quot;Unknown input method: %s&quot; method-name)
114 nil)))

118(defun nskk-input-hybrid-lookup (key)
119 &quot;ハイブリッド入力方式でKEYを変換する。
120
121自動選択モードの場合:
122 優先順位に従って各入力方式を試し、最初に成功したものを返す。
123
124手動選択モードの場合:
125 指定された入力方式のみを使用する。
126
127使用例:
128 (nskk-input-hybrid-lookup \&quot;kj\&quot;) ;; AZIKで \&quot;きゃ\&quot;
129 (nskk-input-hybrid-lookup \&quot;ka\&quot;) ;; AZIKまたはQWERTYで \&quot;か\&quot;&quot;
130 (cond
131 ;; 手動選択モード
132 ((and nskk-input-hybrid-current-method
133 (not nskk-input-hybrid-auto-select))
134 (when-let ((lookup-fn (nskk-input-hybrid--get-lookup-function
135 nskk-input-hybrid-current-method)))
136 (funcall lookup-fn key)))
137
138 ;; 自動選択モード
139 (t
140 (let ((result nil))
141 ;; 優先順位に従って各入力方式を試す
142 (cl-dolist (method nskk-input-hybrid-methods)
143 (when-let ((lookup-fn (nskk-input-hybrid--get-lookup-function method)))
144 (when-let ((converted (funcall lookup-fn key)))
145 (setq result converted)
146 (cl-return))))
147
148 ;; 結果が見つからない場合、フォールバックを使用
149 (unless result
150 (when-let ((fallback-fn (nskk-input-hybrid--get-lookup-function
151 nskk-input-hybrid-fallback)))
152 (setq result (funcall fallback-fn key))))
153
154 result))))

156(defun nskk-input-hybrid-get-candidates (prefix)
157 &quot;PREFIX で始まる全ての候補を取得する。
158全ての入力方式から候補を収集し、重複を除去して返す。&quot;
159 (let ((candidates nil)
160 (seen (make-hash-table :test 'equal)))
161
162 ;; 各入力方式から候補を取得
163 (dolist (method nskk-input-hybrid-methods)
164 (let ((get-candidates-fn
165 (cond
166 ((eq method 'azik) #'nskk-input-azik-get-candidates)
167 ((eq method 'act) #'nskk-input-act-get-candidates)
168 ((eq method 'qwerty) #'nskk-input-qwerty-get-candidates)
169 (t nil))))
170 (when get-candidates-fn
171 (dolist (candidate (funcall get-candidates-fn prefix))
172 (unless (gethash candidate seen)
173 (puthash candidate t seen)
174 (push candidate candidates))))))
175
176 (nreverse candidates)))

180(defun nskk-input-hybrid-switch-method (method)
181 &quot;現在の入力方式をMETHODに切り替える。
182nilを指定すると自動選択モードに戻る。&quot;
183 (interactive
184 (list (intern-soft
185 (completing-read
186 &quot;Switch to input method (empty for auto): &quot;
187 (cons &quot;auto&quot; (mapcar #'symbol-name nskk-input-hybrid-methods))
188 nil t))))
189 (setq nskk-input-hybrid-current-method
190 (if (eq method 'auto) nil method))
191 (message &quot;Hybrid input method: %s&quot;
192 (or method &quot;auto&quot;)))
