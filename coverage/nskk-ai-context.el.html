180;;;###autoload
181(defun nskk-ai-context-parse-sentence (text)
182 &quot;テキスト TEXT を解析して単語リストに変換する。
183日本語の場合は文字単位で分割（簡易版）。
184戻り値: 単語リスト&quot;
185 (when (stringp text)
186 ;; 簡易的な単語分割（文字単位＋英数字まとまり）
187 (let ((words nil)
188 (i 0)
189 (len (length text))
190 (current-word &quot;&quot;))
191 (while (&lt; i len)
192 (let ((char (aref text i)))
193 (cond
194 ;; 英数字はまとめる
195 ((or (and (&gt;= char ?a) (&lt;= char ?z))
196 (and (&gt;= char ?A) (&lt;= char ?Z))
197 (and (&gt;= char ?0) (&lt;= char ?9)))
198 (setq current-word (concat current-word (char-to-string char))))
199 ;; スペース・句読点は区切り
200 ((memq char '(?\s ?\t ?\n ?、 ?。 ?！ ?？ ?, ?. ?! ??))
201 (when (&gt; (length current-word) 0)
202 (push current-word words)
203 (setq current-word &quot;&quot;))
204 (unless (memq char '(?\s ?\t ?\n))
205 (push (char-to-string char) words)))
206 ;; その他の文字は1文字ずつ
207 (t
208 (when (&gt; (length current-word) 0)
209 (push current-word words)
210 (setq current-word &quot;&quot;))
211 (push (char-to-string char) words))))
212 (setq i (1+ i)))
213 (when (&gt; (length current-word) 0)
214 (push current-word words))
215 (nreverse words))))

230(defun nskk-ai-context--update-ngram (model words)
231 &quot;N-gramモデル MODEL を単語リスト WORDS で更新する。&quot;
232 (let ((n (nskk-ai-ngram-model-n model))
233 (model-table (nskk-ai-ngram-model-model model))
234 (freq-table (nskk-ai-ngram-model-frequencies model)))
235 (when (&gt;= (length words) n)
236 (dotimes (i (- (length words) n -1))
237 (let ((ngram (cl-subseq words i (+ i n))))
238 ;; N-gram全体をキーとして保存
239 (let ((key (mapconcat #'identity ngram &quot; &quot;))
240 (prefix (mapconcat #'identity (butlast ngram) &quot; &quot;))
241 (word (car (last ngram))))
242 ;; prefix -&gt; word のマッピング
243 (let ((candidates (gethash prefix model-table nil)))
244 (puthash prefix
245 (if (member word candidates)
246 candidates
247 (cons word candidates))
248 model-table))
249 ;; 頻度カウント
250 (puthash key (1+ (gethash key freq-table 0)) freq-table)
251 (cl-incf (nskk-ai-ngram-model-total-count model))))))))

283(defun nskk-ai-context--update-tfidf (text words)
284 &quot;TF-IDFモデルを更新する。TEXT は文書、WORDS は単語リスト。&quot;
285 (let ((model nskk-ai-context-tfidf-model)
286 (word-set (make-hash-table :test 'equal)))
287 ;; 文書内の単語をセットに追加
288 (dolist (word words)
289 (puthash word t word-set)
290 ;; 単語頻度を更新
291 (puthash word (1+ (gethash word (nskk-ai-tfidf-model-term-freq model) 0))
292 (nskk-ai-tfidf-model-term-freq model)))
293
294 ;; 文書頻度を更新（この文書に含まれる各単語）
295 (maphash
296 (lambda (word _)
297 (puthash word (1+ (gethash word (nskk-ai-tfidf-model-doc-freq model) 0))
298 (nskk-ai-tfidf-model-doc-freq model)))
299 word-set)
300
301 ;; 文書数を増やす
302 (cl-incf (nskk-ai-tfidf-model-num-docs model))
303
304 ;; IDFキャッシュをクリア
305 (clrhash (nskk-ai-tfidf-model-idf-cache model))))

307(defsubst nskk-ai-context--calc-idf (word)
308 &quot;単語 WORD のIDF（逆文書頻度）を計算する。&quot;
309 (let* ((model nskk-ai-context-tfidf-model)
310 (cached (gethash word (nskk-ai-tfidf-model-idf-cache model))))
311 (if cached
312 cached
313 (let* ((df (gethash word (nskk-ai-tfidf-model-doc-freq model) 0))
314 (n (max 1 (nskk-ai-tfidf-model-num-docs model)))
315 (idf (if (&gt; df 0)
316 (log (/ (float n) df))
317 0.0)))
318 (puthash word idf (nskk-ai-tfidf-model-idf-cache model))
319 idf))))

321(defsubst nskk-ai-context--calc-tf (word words)
322 &quot;単語リスト WORDS における単語 WORD のTF（単語頻度）を計算する。&quot;
323 (let ((count 0))
324 (dolist (w words)
325 (when (equal w word)
326 (cl-incf count)))
327 (/ (float count) (max 1 (length words)))))

339(defun nskk-ai-context--update-embeddings (words)
340 &quot;共起情報から単語埋め込みを更新する（簡易版）。
341WORDS は単語リスト。&quot;
342 (let ((model nskk-ai-context-embedding-model)
343 (window 2)) ; 共起ウィンドウサイズ
344 ;; 語彙に追加
345 (dolist (word words)
346 (unless (gethash word (nskk-ai-embedding-model-vocab model))
347 (let ((id (hash-table-count (nskk-ai-embedding-model-vocab model))))
348 (puthash word id (nskk-ai-embedding-model-vocab model)))))
349
350 ;; 共起行列を更新
351 (dotimes (i (length words))
352 (let ((center-word (nth i words)))
353 (dotimes (j (1+ (* 2 window)))
354 (let ((offset (- j window)))
355 (when (and (/= offset 0)
356 (&gt;= (+ i offset) 0)
357 (&lt; (+ i offset) (length words)))
358 (let* ((context-word (nth (+ i offset) words))
359 (key (cons center-word context-word)))
360 (puthash key (1+ (gethash key (nskk-ai-embedding-model-co-occurrence model) 0))
361 (nskk-ai-embedding-model-co-occurrence model))))))))))

364 &quot;単語 WORD の埋め込みベクトルを取得する（簡易版）。
365実際のベクトルではなく、共起情報ベースのスパースベクトル。&quot;

367 (vector (make-hash-table :test 'equal)))
368 (maphash
369 (lambda (key count)

371 (puthash (cdr key) (float count) vector)))
372 (nskk-ai-embedding-model-co-occurrence model))

375(defsubst nskk-ai-context--cosine-similarity (vec1 vec2)
376 &quot;2つのスパースベクトル VEC1 と VEC2 のコサイン類似度を計算する。&quot;
377 (let ((dot-product 0.0)
378 (norm1 0.0)
379 (norm2 0.0))
380 ;; VEC1のノルムと内積計算
381 (maphash
382 (lambda (word val1)
383 (setq norm1 (+ norm1 (* val1 val1)))
384 (let ((val2 (gethash word vec2 0.0)))
385 (setq dot-product (+ dot-product (* val1 val2)))))
386 vec1)
387 ;; VEC2のノルム計算
388 (maphash
389 (lambda (_word val2)
390 (setq norm2 (+ norm2 (* val2 val2))))
391 vec2)
392
393 (if (and (&gt; norm1 0) (&gt; norm2 0))
394 (/ dot-product (* (sqrt norm1) (sqrt norm2)))
395 0.0)))

439(defun nskk-ai-context--calc-ngram-score (word context-words)
440 &quot;N-gramモデルに基づいて単語 WORD のスコアを計算する。
441CONTEXT-WORDS は文脈の単語リスト。&quot;
442 (let ((score 0.0)
443 (count 0))
444 ;; 直前の単語とのバイグラムスコア
445 (when (&gt;= (length context-words) 1)
446 (let ((prev (car (last context-words))))
447 (let ((prob (nskk-ai-context--ngram-probability
448 nskk-ai-context-bigram-model
449 (list prev) word)))
450 (setq score (+ score prob))
451 (cl-incf count))))
452
453 ;; 直前2単語とのトライグラムスコア
454 (when (&gt;= (length context-words) 2)
455 (let ((prev-2 (nth (- (length context-words) 2) context-words))
456 (prev-1 (car (last context-words))))
457 (let ((prob (nskk-ai-context--ngram-probability
458 nskk-ai-context-trigram-model
459 (list prev-2 prev-1) word)))
460 (setq score (+ score (* 2.0 prob))) ; トライグラムに高重み
461 (cl-incf count 2))))
462
463 (if (&gt; count 0)
464 (/ score count)
465 0.0)))

467(defsubst nskk-ai-context--ngram-probability (model prefix-words word)
468 &quot;N-gramモデル MODEL において PREFIX-WORDS の後に WORD が続く確率。&quot;
469 (let* ((prefix (mapconcat #'identity prefix-words &quot; &quot;))
470 (key (concat prefix &quot; &quot; word))
471 (freq-table (nskk-ai-ngram-model-frequencies model))
472 (ngram-count (gethash key freq-table 0))
473 (prefix-count 0))
474 ;; prefix の出現回数を計算
475 (maphash
476 (lambda (k v)
477 (when (string-prefix-p prefix k)
478 (setq prefix-count (+ prefix-count v))))
479 freq-table)
480
481 (if (&gt; prefix-count 0)
482 (/ (float ngram-count) prefix-count)
483 0.0)))

486 &quot;単語 WORD と文脈単語 CONTEXT-WORDS の意味的類似度スコアを計算する。&quot;

530 :word-count (length words)
531 :unique-words (hash-table-count
532 (let ((ht (make-hash-table :test 'equal)))
533 (dolist (w words ht) (puthash w t ht))))
534 :topics (let ((scores (make-hash-table :test 'equal)))
535 (dolist (word words)
536 (when (&gt; (length word) 1)
537 (let ((score (or (nskk-ai-context-calc-tfidf-score word words) 0.0)))
538 (puthash word score scores))))
539 (let ((topics nil))
540 (maphash (lambda (w s) (push (cons w s) topics)) scores)
541 (seq-take (sort topics (lambda (a b) (&gt; (cdr a) (cdr b)))) 3)))
542 :avg-word-length (if (&gt; (length words) 0)
543 (/ (apply #'+ (mapcar #'length words))
544 (float (length words)))
545 0.0))))

549(defun nskk-ai-context--check-limits ()
550 &quot;モデルのエントリ数が上限を超えた場合、低頻度エントリを削除する。&quot;
551 ;; N-gramモデルの制限
552 (dolist (model (list nskk-ai-context-bigram-model
553 nskk-ai-context-trigram-model
554 nskk-ai-context-4gram-model))
555 (when (&gt; (hash-table-count (nskk-ai-ngram-model-frequencies model))
556 nskk-ai-context-max-ngrams)
557 (nskk-ai-context--prune-ngram model)))
558
559 ;; 共起行列の制限
560 (when (&gt; (hash-table-count (nskk-ai-embedding-model-co-occurrence
561 nskk-ai-context-embedding-model))
562 (* 2 nskk-ai-context-max-ngrams))
563 (nskk-ai-context--prune-cooccurrence)))

565(defun nskk-ai-context--prune-ngram (model)
566 &quot;N-gramモデル MODEL から低頻度エントリを削除する。&quot;
567 (let* ((freq-table (nskk-ai-ngram-model-frequencies model))
568 (entries nil)
569 (keep-count (floor (* nskk-ai-context-max-ngrams 0.9))))
570 ;; 全エントリをリストに変換
571 (maphash (lambda (k v) (push (cons k v) entries)) freq-table)
572 ;; 頻度でソート
573 (setq entries (sort entries (lambda (a b) (&gt; (cdr a) (cdr b)))))
574 ;; 上位エントリのみ保持
575 (clrhash freq-table)
576 (dolist (entry (seq-take entries keep-count))
577 (puthash (car entry) (cdr entry) freq-table))))

579(defun nskk-ai-context--prune-cooccurrence ()
580 &quot;共起行列から低頻度エントリを削除する。&quot;
581 (let* ((co-occ (nskk-ai-embedding-model-co-occurrence nskk-ai-context-embedding-model))
582 (entries nil)
583 (keep-count (floor (* nskk-ai-context-max-ngrams 1.8))))
584 (maphash (lambda (k v) (push (cons k v) entries)) co-occ)
585 (setq entries (sort entries (lambda (a b) (&gt; (cdr a) (cdr b)))))
586 (clrhash co-occ)
587 (dolist (entry (seq-take entries keep-count))
588 (puthash (car entry) (cdr entry) co-occ))))
