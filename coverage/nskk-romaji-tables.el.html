428;;; 統合テーブル

430(defconst nskk-romaji-table
431 (append nskk-romaji-youon-table ; 拗音（最長一致優先のため先頭）
432 nskk-romaji-foreign-table ; 外来音
433 nskk-romaji-small-table ; 小書き文字
434 nskk-romaji-voiced-table ; 濁音・半濁音
435 nskk-romaji-base-table) ; 基本五十音
436 &quot;統合されたローマ字→ひらがな変換テーブル。
437最長一致優先のため、長いパターンを先に配置している。
438
439使用例:
440 (assoc \&quot;kya\&quot; nskk-romaji-table) ;; =&gt; (\&quot;kya\&quot; . \&quot;きゃ\&quot;)
441 (assoc \&quot;ka\&quot; nskk-romaji-table) ;; =&gt; (\&quot;ka\&quot; . \&quot;か\&quot;)
442
443注意:
444 このテーブルは最長一致アルゴリズムでの使用を想定している。
445 例えば \&quot;kya\&quot; を変換する際、\&quot;k\&quot; -&gt; \&quot;y\&quot; -&gt; \&quot;a\&quot; ではなく
446 \&quot;kya\&quot; として一括で検索すべき。&quot;)

454(defun nskk-romaji-init-hash-table ()
455 &quot;ローマ字変換用のハッシュテーブルを初期化する。
456`nskk-romaji-table' の内容をハッシュテーブルに変換し、

523(defvar nskk-romaji-table-extend-functions nil
524 &quot;ローマ字テーブルを拡張するフック。

535(defun nskk-romaji-get-extended-table ()
536 &quot;拡張テーブルを含む完全なローマ字変換テーブルを返す。

539 (let ((extended-table nskk-romaji-table))
540 (dolist (func nskk-romaji-table-extend-functions)
