128(defun nskk-core-convert-romaji (input)
129 &quot;ローマ字をかなに変換する。
130INPUTはローマ字文字列。
131戻り値は (確定文字列 . 未確定文字列) のコンスセル。&quot;
132 ;; キャッシュチェック
133 (or (gethash input nskk-core--conversion-cache)
134 (let ((result (nskk-core--convert-romaji-uncached input)))
135 ;; キャッシュに保存
136 (when (&lt; (hash-table-count nskk-core--conversion-cache)
137 nskk-core-cache-size)
138 (puthash input result nskk-core--conversion-cache))
139 result)))

158 &quot;ひらがなをカタカナに変換する。
159HIRAGANAはひらがな文字列。&quot;
160 (mapconcat
161 (lambda (char)

288(defun nskk-core-measure-performance (function &rest args)
289 &quot;関数のパフォーマンスを測定する。
290FUNCTIONは測定する関数、ARGSは引数。&quot;
291 (let ((start-time (float-time))
292 (result (apply function args))
293 (end-time (float-time)))
294 (push (list :function function
295 :duration (- end-time start-time)
296 :timestamp start-time)
297 nskk-core--performance-log)
298 result))
299
300(defun nskk-core-get-performance-statistics ()
301 &quot;パフォーマンス統計を取得する。&quot;
302 (interactive)
303 (if nskk-core--performance-log
304 (let* ((durations (mapcar (lambda (log) (plist-get log :duration))
305 nskk-core--performance-log))
306 (avg (/ (apply #'+ durations) (length durations)))
307 (max-duration (apply #'max durations))
308 (min-duration (apply #'min durations)))
309 (message &quot;Performance: avg=%.3fms max=%.3fms min=%.3fms&quot;
310 (* avg 1000) (* max-duration 1000) (* min-duration 1000)))
311 (message &quot;No performance data available&quot;)))
