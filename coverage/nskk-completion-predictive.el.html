145(defun nskk-completion-predictive-search (index model &rest args)
146 &quot;予測補完検索を実行する。
147
148引数:
149 INDEX - nskk-dict-index構造体
150 MODEL - 予測モデル
151
152キーワード引数:
153 :context - 文脈（直前の単語のリスト）
154 :limit - 最大結果数
155 :okuri-type - 送り仮名タイプ
156
157戻り値:
158 nskk-completion-predictive-result構造体のリスト（確率順）&quot;
159 (let* ((context (plist-get args :context))
160 (limit (or (plist-get args :limit)
161 nskk-completion-predictive-default-limit))
162 (okuri-type (plist-get args :okuri-type))
163 (predictions (nskk-completion-predictive-predict-next
164 model context :n limit)))
165
166 ;; 予測結果を辞書エントリと結合
167 (let ((results nil)
168 (rank 1))
169 (dolist (pred predictions)
170 (let* ((word (car pred))
171 (prob (cdr pred))
172 (entry (nskk-dict-struct-lookup index word okuri-type)))
173 (when entry
174 (push (nskk-completion-predictive-result--create
175 :midashi word
176 :entry entry
177 :probability prob
178 :confidence (nskk-completion-predictive--calculate-confidence
179 model context word)
180 :rank rank)
181 results)
182 (setq rank (1+ rank)))))
183
184 (nreverse results))))

189(defun nskk-completion-predictive-predict-next (model context &rest args)
190 &quot;次の単語を予測する。
191
192引数:
193 MODEL - 予測モデル
194 CONTEXT - 文脈（直前の単語のリスト）
195
196キーワード引数:
197 :n - 予測候補数（デフォルト: nskk-completion-predictive-default-limit）
198
199戻り値:
200 ((word . probability) ...) のリスト（確率の高い順）&quot;
201 (unless model
202 (error &quot;Predictive model is required&quot;))
203
204 (let* ((n (or (plist-get args :n) nskk-completion-predictive-default-limit))
205 (order (nskk-completion-predictive-model-order model))
206 (state (nskk-completion-predictive--make-state context order))
207 (transitions (gethash state
208 (nskk-completion-predictive-model-transition-table model)))
209 (predictions nil))
210
211 (if (null transitions)
212 ;; 遷移データがない場合は空リスト
213 nil
214 ;; 確率順にソート
215 (setq predictions (sort (copy-sequence transitions)
216 (lambda (a b) (&gt; (cdr a) (cdr b)))))
217
218 ;; 最小信頼度でフィルタ
219 (setq predictions (seq-filter
220 (lambda (pred)
221 (&gt;= (cdr pred) nskk-completion-predictive-min-confidence))
222 predictions))
223
224 ;; 上位N件を返す
225 (seq-take predictions n))))

227(defun nskk-completion-predictive--make-state (context order)
228 &quot;文脈から状態を作成する（内部関数）。
229
230引数:
231 CONTEXT - 文脈（直前の単語のリスト、新しい順）
232 ORDER - マルコフ連鎖の次数
233
234戻り値:
235 状態（文字列またはリスト）&quot;
236 (cond
237 ((= order 1)
238 ;; 1次マルコフ連鎖: 直前の1単語
239 (if context (car context) &quot;&quot;))
240 ((= order 2)
241 ;; 2次マルコフ連鎖: 直前の2単語
242 (if (&gt;= (length context) 2)
243 (list (nth 1 context) (nth 0 context))
244 (if context (list (car context)) (list &quot;&quot;))))
245 (t
246 &quot;&quot;)))

248(defun nskk-completion-predictive--calculate-confidence (model context word)
249 &quot;予測の信頼度を計算する（内部関数）。
250
251引数:
252 MODEL - 予測モデル
253 CONTEXT - 文脈
254 WORD - 予測単語
255
256処理:
257 状態の出現回数に基づく信頼度
258 confidence = min(1.0, state-count / 100)
259
260戻り値:
261 信頼度（0.0-1.0）&quot;
262 (let* ((order (nskk-completion-predictive-model-order model))
263 (state (nskk-completion-predictive--make-state context order))
264 (count (gethash state (nskk-completion-predictive-model-state-count model) 0)))
265 (min 1.0 (/ (float count) 100.0))))

270(defun nskk-completion-predictive-learn (model word-sequence)
271 &quot;予測モデルを更新する（学習）。
272
273引数:
274 MODEL - 予測モデル
275 WORD-SEQUENCE - 単語シーケンス（リスト）
276
277処理:
278 マルコフ連鎖の遷移確率を更新
279
280戻り値:
281 更新された予測モデル&quot;
282 (unless (listp word-sequence)
283 (error &quot;Word sequence must be a list: %s&quot; word-sequence))
284
285 (let ((order (nskk-completion-predictive-model-order model))
286 (seq-len (length word-sequence)))
287
288 (when (&gt; seq-len order)
289 ;; 各遷移を学習
290 (dotimes (i (- seq-len order))
291 (let* ((context-words (if (= order 1)
292 (list (nth i word-sequence))
293 (list (nth i word-sequence) (nth (1+ i) word-sequence))))
294 (next-word (nth (+ i order) word-sequence))
295 (state (nskk-completion-predictive--make-state
296 (nreverse context-words) order)))
297
298 ;; 状態カウント更新
299 (let ((count (gethash state (nskk-completion-predictive-model-state-count model) 0)))
300 (puthash state (1+ count) (nskk-completion-predictive-model-state-count model)))
301
302 ;; 遷移テーブル更新
303 (nskk-completion-predictive--update-transition
304 model state next-word))))
305
306 ;; 確率の正規化
307 (nskk-completion-predictive--normalize-probabilities model)
308
309 model))

311(defun nskk-completion-predictive--update-transition (model state next-word)
312 &quot;遷移テーブルを更新する（内部関数）。
313
314引数:
315 MODEL - 予測モデル
316 STATE - 状態
317 NEXT-WORD - 次の単語&quot;
318 (let* ((transitions (gethash state (nskk-completion-predictive-model-transition-table model)))
319 (current-count (or (cdr (assoc next-word transitions)) 0))
320 (new-count (+ current-count nskk-completion-predictive-learning-rate)))
321
322 ;; 遷移リストを更新
323 (if transitions
324 (let ((updated nil))
325 (setq transitions
326 (mapcar (lambda (pair)
327 (if (equal (car pair) next-word)
328 (progn
329 (setq updated t)
330 (cons next-word new-count))
331 pair))
332 transitions))
333 (unless updated
334 (push (cons next-word new-count) transitions)))
335 (setq transitions (list (cons next-word new-count))))
336
337 (puthash state transitions (nskk-completion-predictive-model-transition-table model))
338 (cl-incf (nskk-completion-predictive-model-total-transitions model))))

340(defun nskk-completion-predictive--normalize-probabilities (model)
341 &quot;遷移確率を正規化する（内部関数）。
342
343引数:
344 MODEL - 予測モデル
345
346処理:
347 各状態の遷移確率の合計が1.0になるように正規化&quot;
348 (maphash
349 (lambda (state transitions)
350 (let ((total (apply #'+ (mapcar #'cdr transitions))))
351 (when (&gt; total 0)
352 (let ((normalized (mapcar (lambda (pair)
353 (cons (car pair)
354 (/ (cdr pair) (float total))))
355 transitions)))
356 (puthash state normalized
357 (nskk-completion-predictive-model-transition-table model))))))
358 (nskk-completion-predictive-model-transition-table model)))
