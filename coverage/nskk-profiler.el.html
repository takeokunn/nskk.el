108;;;###autoload
109(defun nskk-profile-start ()
110 &quot;プロファイリングを開始する。&quot;
111 (interactive)
112 (when nskk-profiler--active
113 (error &quot;Profiler is already running. Call `nskk-profile-stop' first&quot;))
114
115 ;; 初期化
116 (setq nskk-profiler--active t
117 nskk-profiler--start-time (current-time)
118 nskk-profiler--memory-snapshots nil
119 nskk-profiler--gc-events nil
120 nskk-profiler--function-stats nil)
121
122 ;; GCフック追加
123 (when (and nskk-profiler-gc-tracking
124 (not nskk-profiler--gc-hook-added))
125 (add-hook 'post-gc-hook #'nskk-profiler--record-gc-event)
126 (setq nskk-profiler--gc-hook-added t))
127
128 ;; 初期メモリスナップショット
129 (when nskk-profiler-memory-tracking
130 (nskk-profiler--take-memory-snapshot))
131
132 ;; Emacsビルトインプロファイラ起動
133 (profiler-start 'cpu+mem)
134
135 (message &quot;NSKK profiler started&quot;))
136
137;;;###autoload
138(defun nskk-profile-stop ()
139 &quot;プロファイリングを停止する。&quot;
140 (interactive)
141 (unless nskk-profiler--active
142 (error &quot;Profiler is not running. Call `nskk-profile-start' first&quot;))
143
144 ;; ビルトインプロファイラ停止
145 (profiler-stop)
146
147 ;; 最終メモリスナップショット
148 (when nskk-profiler-memory-tracking
149 (nskk-profiler--take-memory-snapshot))
150
151 ;; GCフック削除
152 (when nskk-profiler--gc-hook-added
153 (remove-hook 'post-gc-hook #'nskk-profiler--record-gc-event)
154 (setq nskk-profiler--gc-hook-added nil))
155
156 (setq nskk-profiler--active nil)
157
158 (message &quot;NSKK profiler stopped&quot;)
159
160 ;; 自動レポート表示
161 (when nskk-profiler-auto-report
162 (nskk-profile-report)))

164;;;###autoload
165(defun nskk-profile-report ()
166 &quot;プロファイリングレポートを表示する。&quot;
167 (interactive)
168 (unless nskk-profiler--start-time
169 (error &quot;No profiling data available. Run `nskk-profile-start' first&quot;))
170
171 (let ((elapsed (if nskk-profiler--active
172 (float-time (time-subtract (current-time)
173 nskk-profiler--start-time))
174 (if (and nskk-profiler--memory-snapshots
175 (car (last nskk-profiler--memory-snapshots)))
176 (float-time (time-subtract (car (car (last nskk-profiler--memory-snapshots)))
177 nskk-profiler--start-time))
178 0.0))))
179 (with-output-to-temp-buffer &quot;*NSKK Profiler Report*&quot;
180 ;; ヘッダー
181 (princ &quot;=======================================================\n&quot;)
182 (princ &quot; NSKK Performance Profiler Report\n&quot;)
183 (princ &quot;=======================================================\n\n&quot;)
184
185 (princ (format &quot;Profiling Duration: %.3f seconds\n&quot; elapsed))
186 (princ (format &quot;Status: %s\n\n&quot; (if nskk-profiler--active &quot;RUNNING&quot; &quot;STOPPED&quot;)))
187
188 ;; メモリ統計
189 (when nskk-profiler-memory-tracking
190 (nskk-profiler--report-memory-stats))
191
192 ;; GC統計
193 (when nskk-profiler-gc-tracking
194 (nskk-profiler--report-gc-stats))
195
196 ;; 関数統計
197 (when nskk-profiler--function-stats
198 (nskk-profiler--report-function-stats))
199
200 ;; フッター
201 (princ &quot;\n=======================================================\n&quot;)
202 (princ &quot;Use `profiler-report' for detailed CPU/Memory report\n&quot;)
203 (princ &quot;=======================================================\n&quot;))))

207(defun nskk-profiler--take-memory-snapshot ()
208 &quot;現在のメモリ使用量のスナップショットを取得する。&quot;
209 (let* ((timestamp (current-time))
210 (mem-counts (memory-use-counts))
211 (cons-cells (nth 0 mem-counts))
212 (floats (nth 1 mem-counts))
213 (vectors (nth 2 mem-counts))
214 (symbols (nth 3 mem-counts))
215 (strings (nth 4 mem-counts))
216 (intervals (nth 5 mem-counts))
217 (buffers (nth 6 mem-counts)))
218 (push (list timestamp
219 :cons-cells cons-cells
220 :floats floats
221 :vectors vectors
222 :symbols symbols
223 :strings strings
224 :intervals intervals
225 :buffers buffers)
226 nskk-profiler--memory-snapshots)))
227
228(defun nskk-profiler--report-memory-stats ()
229 &quot;メモリ統計をレポートバッファに出力する。&quot;
230 (princ &quot;--- Memory Statistics ---\n\n&quot;)
231
232 (when (&gt;= (length nskk-profiler--memory-snapshots) 2)
233 (let* ((snapshots (reverse nskk-profiler--memory-snapshots))
234 (initial (cdr (car snapshots)))
235 (final (cdr (car (last snapshots)))))
236
237 (princ (format &quot;%-20s %15s %15s %15s\n&quot;
238 &quot;Resource&quot; &quot;Initial&quot; &quot;Final&quot; &quot;Delta&quot;))
239 (princ (make-string 65 ?-))
240 (princ &quot;\n&quot;)
241
242 ;; 各リソースの変化を表示
243 (dolist (key '(:cons-cells :floats :vectors :symbols :strings :intervals :buffers))
244 (let* ((init-val (plist-get initial key))
245 (final-val (plist-get final key))
246 (delta (- final-val init-val)))
247 (princ (format &quot;%-20s %15d %15d %15d\n&quot;
248 (substring (symbol-name key) 1)
249 init-val
250 final-val
251 delta))))
252
253 (princ &quot;\n&quot;))))

257(defun nskk-profiler--record-gc-event ()
258 &quot;GCイベントを記録する（post-gc-hookから呼ばれる）。&quot;
259 (when nskk-profiler--active
260 (push (list (current-time)
261 :gc-elapsed gc-elapsed
262 :gcs-done gcs-done)
263 nskk-profiler--gc-events)))
264
265(defun nskk-profiler--report-gc-stats ()
266 &quot;GC統計をレポートバッファに出力する。&quot;
267 (princ &quot;--- Garbage Collection Statistics ---\n\n&quot;)
268
269 (let ((gc-count (length nskk-profiler--gc-events)))
270 (princ (format &quot;Total GC count: %d\n&quot; gc-count))
271 (princ (format &quot;Current GC elapsed time: %.3f seconds\n&quot; gc-elapsed))
272 (princ (format &quot;Total GCs done: %d\n&quot; gcs-done))
273
274 (when (&gt; gc-count 0)
275 (let* ((events (reverse nskk-profiler--gc-events))
276 (first-gc (plist-get (cadr (car events)) :gc-elapsed))
277 (last-gc (plist-get (cadr (car (last events))) :gc-elapsed))
278 (gc-time-delta (- last-gc first-gc)))
279 (princ (format &quot;GC time during profiling: %.3f seconds\n&quot; gc-time-delta))))
280
281 (princ &quot;\n&quot;)))

285(defmacro nskk-profile-function (name &rest body)
286 &quot;関数の実行時間をプロファイリングする。
287
288NAME: 関数名（シンボルまたは文字列）
289BODY: 実行するコード
290
291返り値: BODYの返り値
292
293例:
294 (nskk-profile-function \&quot;romaji-conversion\&quot;
295 (nskk-convert-romaji \&quot;konnnichiha\&quot;))&quot;
296 (declare (indent 1) (debug t))
297 (let ((func-name (if (stringp name) name (symbol-name name)))
298 (start-time (make-symbol &quot;start-time&quot;))
299 (end-time (make-symbol &quot;end-time&quot;))
300 (result (make-symbol &quot;result&quot;))
301 (elapsed (make-symbol &quot;elapsed&quot;)))
302 `(if nskk-profiler--active
303 (let* ((,start-time (current-time))
304 (,result (progn ,@body))
305 (,end-time (current-time))
306 (,elapsed (float-time (time-subtract ,end-time ,start-time))))
307 (nskk-profiler--record-function-stats ,func-name ,elapsed)
308 ,result)
309 (progn ,@body))))
310
311(defun nskk-profiler--record-function-stats (func-name elapsed-time)
312 &quot;関数統計を記録する。
313
314FUNC-NAME: 関数名
315ELAPSED-TIME: 経過時間（秒）&quot;
316 (let* ((entry (assoc func-name nskk-profiler--function-stats))
317 (count (if entry (nth 1 entry) 0))
318 (total (if entry (nth 2 entry) 0.0))
319 (min-time (if entry (min (nth 3 entry) elapsed-time) elapsed-time))
320 (max-time (if entry (max (nth 4 entry) elapsed-time) elapsed-time)))
321 (setf (alist-get func-name nskk-profiler--function-stats nil nil #'string=)
322 (list (1+ count) (+ total elapsed-time) min-time max-time))))
323
324(defun nskk-profiler--report-function-stats ()
325 &quot;関数統計をレポートバッファに出力する。&quot;
326 (princ &quot;--- Function Statistics ---\n\n&quot;)
327 (princ (format &quot;%-30s %10s %15s %15s %15s %15s\n&quot;
328 &quot;Function&quot; &quot;Calls&quot; &quot;Total(s)&quot; &quot;Avg(s)&quot; &quot;Min(s)&quot; &quot;Max(s)&quot;))
329 (princ (make-string 100 ?-))
330 (princ &quot;\n&quot;)
331
332 (dolist (entry (sort (copy-sequence nskk-profiler--function-stats)
333 (lambda (a b) (&gt; (nth 2 (cdr a)) (nth 2 (cdr b))))))
334 (let* ((name (car entry))
335 (count (nth 1 entry))
336 (total (nth 2 entry))
337 (min-time (nth 3 entry))
338 (max-time (nth 4 entry))
339 (avg (/ total count)))
340 (princ (format &quot;%-30s %10d %15.6f %15.6f %15.6f %15.6f\n&quot;
341 name count total avg min-time max-time))))
342 (princ &quot;\n&quot;))

347 &quot;現在のスレッド情報を取得する。
