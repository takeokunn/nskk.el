192;;; ハッシュテーブル
193
194(defvar nskk-input-tutcode-hash-table nil
195 &quot;TUT-code入力方式用のハッシュテーブル。&quot;)
196
197(defun nskk-input-tutcode-init-hash-table ()
198 &quot;TUT-code入力方式用のハッシュテーブルを初期化する。&quot;
199 (setq nskk-input-tutcode-hash-table
200 (make-hash-table :test 'equal :size 300))
201 (dolist (entry nskk-input-tutcode-table)
202 (puthash (car entry) (cdr entry) nskk-input-tutcode-hash-table)))
203
204;;; 検索関数
205
206(defun nskk-input-tutcode-lookup (code)
207 &quot;TUT-codeでCODEをひらがなに変換する。
208
209使用例:
210 (nskk-input-tutcode-lookup \&quot;kj\&quot;) ;; =&gt; \&quot;か\&quot;
211 (nskk-input-tutcode-lookup \&quot;kja\&quot;) ;; =&gt; \&quot;きゃ\&quot;&quot;
212 (unless nskk-input-tutcode-hash-table
213 (nskk-input-tutcode-init-hash-table))
214 (gethash code nskk-input-tutcode-hash-table))
215
216(defun nskk-input-tutcode-get-candidates (prefix)
217 &quot;PREFIX で始まる全てのTUT-code候補を取得する。&quot;
218 (let ((candidates nil))
219 (dolist (entry nskk-input-tutcode-table)
220 (when (string-prefix-p prefix (car entry))
221 (push (car entry) candidates)))
222 (nreverse candidates)))
223
224;;; 入力方式登録
225
226(defun nskk-input-tutcode-register ()
227 &quot;TUT-code入力方式をNSKKに登録する。&quot;
228 (nskk-input-tutcode-init-hash-table))
229
230;;; 統計情報
231
232(defun nskk-input-tutcode-stats ()
233 &quot;TUT-code入力方式の統計情報を返す。&quot;
234 (list :total (length nskk-input-tutcode-table)
235 :two-stroke (cl-count-if (lambda (e) (= (length (car e)) 2))
236 nskk-input-tutcode-table)
237 :three-stroke (cl-count-if (lambda (e) (= (length (car e)) 3))
238 nskk-input-tutcode-table)))
239
240;;; パフォーマンス最適化
241
242(defsubst nskk-input-tutcode-lookup-fast (code)
243 &quot;TUT-codeでCODEを高速変換する。&quot;
244 (gethash code nskk-input-tutcode-hash-table))
